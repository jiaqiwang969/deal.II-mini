1367  const TrilinosScalar *values)
1368  {

1370  // writing to this vector at all.
1371  Assert(!has_ghost_elements(), ExcGhostsPresent());
1372 
1373  if (last_action != Add)
1374  {
1375  if (last_action == Insert)
1376  {
1377  const int ierr = vector->GlobalAssemble(Insert);
1378  AssertThrow(ierr == 0, ExcTrilinosError(ierr));
1379  }
1380  last_action = Add;
1381  }
1382 
1383  for (size_type i = 0; i < n_elements; ++i)
1384  {
1385  const size_type row = indices[i];
1386  const TrilinosWrappers::types::int_type local_row = vector->Map().LID(
1387  static_cast<TrilinosWrappers::types::int_type>(row));
1388  if (local_row != -1)
1389  (*vector)[0][local_row] += values[i];
1390  else if (nonlocal_vector.get() == nullptr)
1391  {
1392  const int ierr = vector->SumIntoGlobalValues(
1393  1,
1394  reinterpret_cast<const TrilinosWrappers::types::int_type *>(
1395  &row),
1396  &values[i]);
1397  AssertThrow(ierr == 0, ExcTrilinosError(ierr));
1398  compressed = false;
1399  }
1400  else
1401  {
1402  // use pre-allocated vector for non-local entries if it exists for
1403  // addition operation
1404  const TrilinosWrappers::types::int_type my_row =
1405  nonlocal_vector->Map().LID(
1406  static_cast<TrilinosWrappers::types::int_type>(row));
1407  Assert(my_row != -1,
