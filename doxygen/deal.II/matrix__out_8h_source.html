258 template <class Matrix>
259 inline double
260 MatrixOut::get_gridpoint_value(const Matrix & matrix,
261  const size_type i,
262  const size_type j,
263  const Options & options)
264 {
265  // special case if block size is
266  // one since we then don't need all
267  // that loop overhead
268  if (options.block_size == 1)
269  {
270  if (options.show_absolute_values == true)
271  return std::fabs(
272  internal::MatrixOutImplementation::get_element(matrix, i, j));
273  else
274  return internal::MatrixOutImplementation::get_element(matrix, i, j);
275  }
276 
277  // if blocksize greater than one,
278  // then compute average of elements
279  double average = 0;
280  size_type n_elements = 0;
281  for (size_type row = i * options.block_size;
282  row <
283  std::min(size_type(matrix.m()), size_type((i + 1) * options.block_size));
284  ++row)
285  for (size_type col = j * options.block_size;
286  col < std::min(size_type(matrix.m()),
287  size_type((j + 1) * options.block_size));
288  ++col, ++n_elements)
289  if (options.show_absolute_values == true)
290  average += std::fabs(
291  internal::MatrixOutImplementation::get_element(matrix, row, col));
292  else
293  average +=
294  internal::MatrixOutImplementation::get_element(matrix, row, col);
295  average /= n_elements;
296  return average;
297 }

301 template <class Matrix>
302 void
303 MatrixOut::build_patches(const Matrix & matrix,
304  const std::string &name,
305  const Options options)
306 {
307  size_type gridpoints_x = (matrix.n() / options.block_size +
308  (matrix.n() % options.block_size != 0 ? 1 : 0)),
309  gridpoints_y = (matrix.m() / options.block_size +
310  (matrix.m() % options.block_size != 0 ? 1 : 0));
311 
312  // If continuous, the number of
313  // plotted patches is matrix size-1
314  if (!options.discontinuous)
315  {
316  --gridpoints_x;
317  --gridpoints_y;
318  }
319 
320  // first clear old data and set it
321  // to virgin state
322  patches.clear();
323  patches.resize((gridpoints_x) * (gridpoints_y));
324 
325  // now build the patches
326  size_type index = 0;
327  for (size_type i = 0; i < gridpoints_y; ++i)
328  for (size_type j = 0; j < gridpoints_x; ++j, ++index)
329  {
330  // within each patch, order the points in such a way that if some
331  // graphical output program (such as gnuplot) plots the quadrilaterals
332  // as two triangles, then the diagonal of the quadrilateral which cuts
333  // it into the two printed triangles is parallel to the diagonal of the
334  // matrix, rather than perpendicular to it. this has the advantage that,
335  // for example, the unit matrix is plotted as a straight rim, rather
336  // than as a series of bumps and valleys along the diagonal
337  patches[index].vertices[0](0) = j;
338  patches[index].vertices[0](1) = -static_cast<signed int>(i);
339  patches[index].vertices[1](0) = j;
340  patches[index].vertices[1](1) = -static_cast<signed int>(i + 1);
341  patches[index].vertices[2](0) = j + 1;
342  patches[index].vertices[2](1) = -static_cast<signed int>(i);
343  patches[index].vertices[3](0) = j + 1;
344  patches[index].vertices[3](1) = -static_cast<signed int>(i + 1);
345  // next scale all the patch
346  // coordinates by the block
347  // size, to get original
348  // coordinates
349  for (auto &vertex : patches[index].vertices)
350  vertex *= options.block_size;
351 
352  patches[index].n_subdivisions = 1;
353 
354  patches[index].data.reinit(1, 4);
355  if (options.discontinuous)
356  {
357  patches[index].data(0, 0) =
358  get_gridpoint_value(matrix, i, j, options);
359  patches[index].data(0, 1) =
360  get_gridpoint_value(matrix, i, j, options);
361  patches[index].data(0, 2) =
362  get_gridpoint_value(matrix, i, j, options);
363  patches[index].data(0, 3) =
364  get_gridpoint_value(matrix, i, j, options);
365  }
366  else
367  {
368  patches[index].data(0, 0) =
369  get_gridpoint_value(matrix, i, j, options);
370  patches[index].data(0, 1) =
371  get_gridpoint_value(matrix, i + 1, j, options);
372  patches[index].data(0, 2) =
373  get_gridpoint_value(matrix, i, j + 1, options);
374  patches[index].data(0, 3) =
375  get_gridpoint_value(matrix, i + 1, j + 1, options);
376  }
377  };
378 
379  // finally set the name
380  this->name = name;
381 }
