137 template <typename number>
138 void
139 DataOutStack<dim, spacedim, void>::add_data_vector(const Vector<number> &vec,
140  const std::string & name)
141 {
142  const unsigned int n_components = dof_handler->get_fe(0).n_components();
143 
144  std::vector<std::string> names;
145  // if only one component or vector
146  // is cell vector: we only need one
147  // name
148  if ((n_components == 1) ||
149  (vec.size() == dof_handler->get_triangulation().n_active_cells()))
150  {
151  names.resize(1, name);
152  }
153  else
154  // otherwise append _i to the
155  // given name
156  {
157  names.resize(n_components);
158  for (unsigned int i = 0; i < n_components; ++i)
159  {
160  std::ostringstream namebuf;
161  namebuf << '_' << i;
162  names[i] = name + namebuf.str();
163  }
164  }
165 
166  add_data_vector(vec, names);
167 }

170 template <int dim, int spacedim>
171 template <typename number>
172 void
173 DataOutStack<dim, spacedim, void>::add_data_vector(
174  const Vector<number> & vec,
175  const std::vector<std::string> &names)
176 {
177  Assert(dof_handler != nullptr,
178  Exceptions::DataOutImplementation::ExcNoDoFHandlerSelected());
179  // either cell data and one name,
180  // or dof data and n_components names
181  Assert(((vec.size() == dof_handler->get_triangulation().n_active_cells()) &&
182  (names.size() == 1)) ||
183  ((vec.size() == dof_handler->n_dofs()) &&
184  (names.size() == dof_handler->get_fe(0).n_components())),
185  Exceptions::DataOutImplementation::ExcInvalidNumberOfNames(
186  names.size(), dof_handler->get_fe(0).n_components()));
187  for (const auto &name : names)
188  {
189  (void)name;
190  Assert(name.find_first_not_of("abcdefghijklmnopqrstuvwxyz"
191  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
192  "0123456789_<>()") == std::string::npos,
193  Exceptions::DataOutImplementation::ExcInvalidCharacter(
194  name,
195  name.find_first_not_of("abcdefghijklmnopqrstuvwxyz"
196  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
197  "0123456789_<>()")));
198  }
199 
200  if (vec.size() == dof_handler->n_dofs())
201  {
202  typename std::vector<DataVector>::iterator data_vector = dof_data.begin();
203  for (; data_vector != dof_data.end(); ++data_vector)
204  if (data_vector->names == names)
205  {
206  data_vector->data.reinit(vec.size());
207  std::copy(vec.begin(), vec.end(), data_vector->data.begin());
208  return;
209  }
210 
211  // ok. not found. there is a
212  // slight chance that
213  // n_dofs==n_cells, so only
214  // bomb out if the next if
215  // statement will not be run
216  if (dof_handler->n_dofs() !=
217  dof_handler->get_triangulation().n_active_cells())
218  Assert(false, ExcVectorNotDeclared(names[0]));
219  }
220 
221  // search cell data
222  if ((vec.size() != dof_handler->n_dofs()) ||
223  (dof_handler->n_dofs() ==
224  dof_handler->get_triangulation().n_active_cells()))
225  {
226  typename std::vector<DataVector>::iterator data_vector =
227  cell_data.begin();
228  for (; data_vector != cell_data.end(); ++data_vector)
229  if (data_vector->names == names)
230  {
231  data_vector->data.reinit(vec.size());
232  std::copy(vec.begin(), vec.end(), data_vector->data.begin());
233  return;
234  }
235  Assert(false, ExcVectorNotDeclared(names[0]));
236  }
237 
238  // we have either return or Assert
239  // statements above, so shouldn't
240  // get here!
241  Assert(false, ExcInternalError());
242 }

245 template <int dim, int spacedim>
246 void
247 DataOutStack<dim, spacedim, void>::build_patches(
248  const unsigned int nnnn_subdivisions)
249 {
250  // this is mostly copied from the
251  // DataOut class
252  unsigned int n_subdivisions =
253  (nnnn_subdivisions != 0) ? nnnn_subdivisions : this->default_subdivisions;
254 
255  Assert(n_subdivisions >= 1,
256  Exceptions::DataOutImplementation::ExcInvalidNumberOfSubdivisions(
257  n_subdivisions));
258  Assert(dof_handler != nullptr,
259  Exceptions::DataOutImplementation::ExcNoDoFHandlerSelected());
260 
261  this->validate_dataset_names();
262 
263  const unsigned int n_components = dof_handler->get_fe(0).n_components();
264  const unsigned int n_datasets =
265  dof_data.size() * n_components + cell_data.size();
266 
267  // first count the cells we want to
268  // create patches of and make sure
269  // there is enough memory for that
270  unsigned int n_patches = 0;
271  for (typename DoFHandler<dim, spacedim>::active_cell_iterator cell =
272  dof_handler->begin_active();
273  cell != dof_handler->end();
274  ++cell)
275  ++n_patches;
276 
277 
278  // before we start the loop:
279  // create a quadrature rule that
280  // actually has the points on this
281  // patch, and an object that
282  // extracts the data on each
283  // cell to these points
284  QTrapezoid<1> q_trapez;
285  QIterated<dim> patch_points(q_trapez, n_subdivisions);
286 
287  // create collection objects from
288  // single quadratures,
289  // and finite elements. if we have
290  // an hp-DoFHandler,
291  // dof_handler.get_fe() returns a
292  // collection of which we do a
293  // shallow copy instead
294  const hp::QCollection<dim> q_collection(patch_points);
295  const hp::FECollection<dim> &fe_collection = dof_handler->get_fe_collection();
296 
297  hp::FEValues<dim> x_fe_patch_values(fe_collection,
298  q_collection,
299  update_values);
300 
301  const unsigned int n_q_points = patch_points.size();
302  std::vector<double> patch_values(n_q_points);
303  std::vector<Vector<double>> patch_values_system(n_q_points,
304  Vector<double>(n_components));
305 
306  // add the required number of
307  // patches. first initialize a template
308  // patch with n_q_points (in the plane
309  // of the cells) times n_subdivisions+1 (in
310  // the time direction) points
311  ::DataOutBase::Patch<patch_dim, patch_spacedim> default_patch;
312  default_patch.n_subdivisions = n_subdivisions;
313  default_patch.data.reinit(n_datasets, n_q_points * (n_subdivisions + 1));
314  patches.insert(patches.end(), n_patches, default_patch);
315 
316  // now loop over all cells and
317  // actually create the patches
318  typename std::vector<
319  ::DataOutBase::Patch<patch_dim, patch_spacedim>>::iterator patch =
320  patches.begin() + (patches.size() - n_patches);
321  unsigned int cell_number = 0;
322  for (typename DoFHandler<dim, spacedim>::active_cell_iterator cell =
323  dof_handler->begin_active();
324  cell != dof_handler->end();
325  ++cell, ++patch, ++cell_number)
326  {
327  Assert(cell->is_locally_owned(), ExcNotImplemented());
328 
329  Assert(patch != patches.end(), ExcInternalError());
330 
331  // first fill in the vertices of the patch
332 
333  // Patches are organized such
334  // that the parameter direction
335  // is the last
336  // coordinate. Thus, vertices
337  // are two copies of the space
338  // patch, one at parameter-step
339  // and one at parameter.
340  switch (dim)
341  {
342  case 1:
343  patch->vertices[0] =
344  Point<dim + 1>(cell->vertex(0)(0), parameter - parameter_step);
345  patch->vertices[1] =
346  Point<dim + 1>(cell->vertex(1)(0), parameter - parameter_step);
347  patch->vertices[2] = Point<dim + 1>(cell->vertex(0)(0), parameter);
348  patch->vertices[3] = Point<dim + 1>(cell->vertex(1)(0), parameter);
349  break;
350 
351  case 2:
352  patch->vertices[0] = Point<dim + 1>(cell->vertex(0)(0),
353  cell->vertex(0)(1),
354  parameter - parameter_step);
355  patch->vertices[1] = Point<dim + 1>(cell->vertex(1)(0),
356  cell->vertex(1)(1),
357  parameter - parameter_step);
358  patch->vertices[2] = Point<dim + 1>(cell->vertex(2)(0),
359  cell->vertex(2)(1),
360  parameter - parameter_step);
361  patch->vertices[3] = Point<dim + 1>(cell->vertex(3)(0),
362  cell->vertex(3)(1),
363  parameter - parameter_step);
364  patch->vertices[4] =
365  Point<dim + 1>(cell->vertex(0)(0), cell->vertex(0)(1), parameter);
366  patch->vertices[5] =
367  Point<dim + 1>(cell->vertex(1)(0), cell->vertex(1)(1), parameter);
368  patch->vertices[6] =
369  Point<dim + 1>(cell->vertex(2)(0), cell->vertex(2)(1), parameter);
370  patch->vertices[7] =
371  Point<dim + 1>(cell->vertex(3)(0), cell->vertex(3)(1), parameter);
372  break;
373 
374  default:
375  Assert(false, ExcNotImplemented());
376  }
377 
378 
379  // now fill in the data values.
380  // note that the required order is
381  // with highest coordinate running
382  // fastest, we need to enter each
383  // value (n_subdivisions+1) times
384  // in succession
385  if (n_datasets > 0)
386  {
387  x_fe_patch_values.reinit(cell);
388  const FEValues<dim> &fe_patch_values =
389  x_fe_patch_values.get_present_fe_values();
390 
391  // first fill dof_data
392  for (unsigned int dataset = 0; dataset < dof_data.size(); ++dataset)
393  {
394  if (n_components == 1)
395  {
396  fe_patch_values.get_function_values(dof_data[dataset].data,
397  patch_values);
398  for (unsigned int i = 0; i < n_subdivisions + 1; ++i)
399  for (unsigned int q = 0; q < n_q_points; ++q)
400  patch->data(dataset, q + n_q_points * i) =
401  patch_values[q];
402  }
403  else
404  // system of components
405  {
406  fe_patch_values.get_function_values(dof_data[dataset].data,
407  patch_values_system);
408  for (unsigned int component = 0; component < n_components;
409  ++component)
410  for (unsigned int i = 0; i < n_subdivisions + 1; ++i)
411  for (unsigned int q = 0; q < n_q_points; ++q)
412  patch->data(dataset * n_components + component,
413  q + n_q_points * i) =
414  patch_values_system[q](component);
415  }
416  }
417 
418  // then do the cell data
419  for (unsigned int dataset = 0; dataset < cell_data.size(); ++dataset)
420  {
421  const double value = cell_data[dataset].data(cell_number);
422  for (unsigned int q = 0; q < n_q_points; ++q)
423  for (unsigned int i = 0; i < n_subdivisions + 1; ++i)
424  patch->data(dataset + dof_data.size() * n_components,
425  q * (n_subdivisions + 1) + i) = value;
426  }
427  }
428  }
429 }
