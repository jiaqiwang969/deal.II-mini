233  template <typename T>
234  ThreadLocalStorage<T>::ThreadLocalStorage(const ThreadLocalStorage<T> &t)
235  : exemplar(t.exemplar)
236  {
237  // Raise a reader lock while we are populating our own data in order to

245  template <typename T>
246  ThreadLocalStorage<T>::ThreadLocalStorage(ThreadLocalStorage<T> &&t) noexcept
247  : exemplar(std::move(t.exemplar))
248  {
249  // We are nice and raise the writer lock before copying over internal
250  // data structures from the argument.
251  //
252  // The point is a bit moot, though: Users of ThreadLocalStorage
253  // typically obtain their thread's thread-local object through the
254  // get() function. That function also acquires the lock, but
255  // whether or not we do that here really doesn't make any
256  // difference in terms of correctness: If another thread manages
257  // to call get() just before we get here, then the result of that
258  // get() function immediately becomes invalid; if it manages to
259  // call get() at the same time as this function if there were no
260  // locking here, it might access undefined state; and if it
261  // manages to call get() just after we moved away the state --
262  // well, then it just got lucky to escape the race condition, but
263  // the race condition is still there.
264  //
265  // On the other hand, there is no harm in doing at least
266  // conceptually the right thing, so ask for that lock:
267  std::unique_lock<decltype(insertion_mutex)> lock(t.insertion_mutex);
268  data = std::move(t.data);
269  }

360  template <typename T>
361  inline T &
362  ThreadLocalStorage<T>::get(bool &exists)
363  {
364  const std::thread::id my_id = std::this_thread::get_id();
365 
366  // Note that std::map<..>::emplace guarantees that no iterators or
367  // references to stored objects are invalidated. We thus only have to
368  // ensure that we do not perform a lookup while writing, and that we
369  // do not write concurrently. This is precisely the "reader-writer
370  // lock" paradigm supported by C++14 by means of the std::shared_lock
371  // and the std::unique_lock.
372 
373  {
374  // Take a shared ("reader") lock for lookup and record the fact
375  // whether we could find an entry in the boolean exists.
376  std::shared_lock<decltype(insertion_mutex)> lock(insertion_mutex);
377 
378  const auto it = data.find(my_id);
379  if (it != data.end())
380  {
381  exists = true;
382  return it->second;
383  }
384  else
385  {
386  exists = false;
387  }
388  }
389 
390  {
391  // Take a unique ("writer") lock for manipulating the std::map. This
392  // lock ensures that no other threat does a lookup at the same time.
393  std::unique_lock<decltype(insertion_mutex)> lock(insertion_mutex);
394 
395  return internal::construct_element(data, my_id, exemplar);
396  }
397  }
