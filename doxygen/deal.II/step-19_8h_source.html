801  * {
802  * cell_matrix = 0;
803  * cell_rhs = 0;
804  *

808  * for (const unsigned int i : fe_values.dof_indices())
809  * {
810  * for (const unsigned int j : fe_values.dof_indices())
811  * cell_matrix(i, j) +=
812  * (fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
813  * fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
814  * fe_values.JxW(q_index)); // dx
815  * }
816  *
817  * @endcode
818  *
819  * 这个函数唯一有趣的部分是它是如何形成线性系统的右手边的。回顾一下，PDE的右边是
820  * @f[
821  * \sum_p (N e)\delta(\mathbf x-\mathbf x_p),
822  * @f]
823  * ，在这里我们用 @f$p@f$ 来索引粒子，以避免与形状函数 @f$\varphi_i@f$ 混淆； @f$\mathbf x_p@f$ 是第 @f$p@f$ 个粒子的位置。
824  *
825 
826  *
827  * 当与测试函数 @f$\varphi_i@f$ 相乘并在域上积分时，会得到一个右手边的向量
828  * @f{align*}{
829  * F_i &= \int_\Omega \varphi_i (\mathbf x)\left[
830  * \sum_p (N e)\delta(\mathbf x-\mathbf x_p) \right] dx
831  * \\ &= \sum_p (N e) \varphi_i(\mathbf x_p).
832  * @f}
833  * 注意最后一行不再包含一个积分，因此也没有出现 @f$dx@f$ ，这需要在我们的代码中出现`JxW`符号。
834  *
835  *
836  * 对于一个给定的单元 @f$K@f$ ，这个单元对右边的贡献是
837  * @f{align*}{
838  * F_i^K &= \sum_{p, \mathbf x_p\in K} (N e) \varphi_i(\mathbf x_p),
839  * @f}，
840  * 也就是说，我们只需要担心那些实际位于当前单元 @f$K@f$ 上的粒子。
841  *
842 
843  *
844  * 在实践中，我们在这里所做的是以下几点。如果当前单元格上有任何粒子，那么我们首先获得一个迭代器范围，指向该单元格的第一个粒子以及该单元格上最后一个粒子之后的粒子（或结束迭代器）--即C++函数中常见的半开放范围。现在知道了粒子的列表，我们查询它们的参考位置（相对于参考单元），评估这些参考位置的形状函数，并根据上面的公式计算力（没有任何 FEValues::JxW). ）。
845  * @note 值得指出的是，调用 Particles::ParticleHandler::particles_in_cell() 和 Particles::ParticleHandler::n_particles_in_cell() 函数在有大量粒子的问题上不是很有效。但是它说明了写这个算法的最简单的方法，所以我们愿意为了说明问题而暂时承担这个代价。 我们在下面的<a href="#extensions">"possibilities for extensions" section</a>中更详细地讨论了这个问题，并在 @ref step_70 "step-70" 中使用了一个更好的方法，例如：。

1578 <a name="Avoidingaperformancebottleneckwithparticles"></a><h4> Avoiding a performance bottleneck with particles </h4>

1580 
1581 `assemble_system()`、`move_particles()`和`update_timestep_size()`函数都调用 Particles::ParticleHandler::particles_in_cell() 和 Particles::ParticleHandler::n_particles_in_cell() ，查询位于当前单元上的粒子信息。虽然这很方便，但也很低效。为了理解为什么会这样，我们需要知道粒子是如何存储在 Particles::ParticleHandler: 中的，即在一个数据结构中，粒子是以某种线性方式按它们所在的单元进行排序的。因此，为了找到与给定单元相关的粒子，这些函数需要搜索给定单元上的第一个（也可能是最后一个）粒子--这种努力需要花费 @f${\cal O}(\log N)@f$ 次操作，其中 @f$N@f$ 是粒子的数量。但这是在每个单元上重复的；假设对于大型计算来说，单元和粒子的数量大致成正比，那么这些函数调用的累积成本是 @f${\cal O}(N \log N)@f$ ，因此大于我们应该对程序的所有部分进行的 @f${\cal O}(N)@f$ 成本。
1582

1590  particles_in_cell.end());
1591 @endcode

1599  for (const auto &cell : dof_handler.active_cell_iterators())
1600  {

1602  auto end_particle_on_cell = begin_particle_on_cell;
1603  while (end_particle_on_cell->get_surrounding_cell(triangulation)
1604  == cell)

1606  ++n_particles_on_cell;
1607  ++end_particle_on_cell;
1608  }
1609 
1610

1612  to end_particle_on_cell, all of which are known to be on the current
1613  cell...;
1614 
1615 
1616  // Move the begin iterator forward so that it points to the first
1617  // particle on the next cell
1618  begin_particle_on_cell = end_particle_on_cell;
1619  }
