246 template <typename VectorType>
247 template <typename PreconditionerType>
248 void
249 SolverFIRE<VectorType>::solve(
250  const std::function<double(VectorType &, const VectorType &)> &compute,
251  VectorType & x,
252  const PreconditionerType &inverse_mass_matrix)
253 {
254  LogStream::Prefix prefix("FIRE");
255 
256  // FIRE algorithm constants
257  const double DELAYSTEP = 5;
258  const double TIMESTEP_GROW = 1.1;
259  const double TIMESTEP_SHRINK = 0.5;
260  const double ALPHA_0 = 0.1;
261  const double ALPHA_SHRINK = 0.99;
262 
263  using real_type = typename VectorType::real_type;
264 
265  typename VectorMemory<VectorType>::Pointer v(this->memory);
266  typename VectorMemory<VectorType>::Pointer g(this->memory);
267 
268  // Set velocities to zero but not gradients
269  // as we are going to compute them soon.
270  v->reinit(x, false);
271  g->reinit(x, true);
272 
273  // Refer to v and g with some readable names.
274  VectorType &velocities = *v;
275  VectorType &gradients = *g;
276 
277  // Update gradients for the new x.
278  compute(gradients, x);
279 
280  unsigned int iter = 0;
281 
282  SolverControl::State conv = SolverControl::iterate;
283  conv = this->iteration_status(iter, gradients * gradients, x);
284  if (conv != SolverControl::iterate)
285  return;
286 
287  // Refer to additional data members with some readable names.
288  const auto &maximum_timestep = additional_data.maximum_timestep;
289  double timestep = additional_data.initial_timestep;
290 
291  // First scaling factor.
292  double alpha = ALPHA_0;
293 
294  unsigned int previous_iter_with_positive_v_dot_g = 0;
295 
296  while (conv == SolverControl::iterate)
297  {
298  ++iter;
299  // Euler integration step.
300  x.add(timestep, velocities); // x += dt * v
301  inverse_mass_matrix.vmult(gradients, gradients); // g = M^{-1} * g
302  velocities.add(-timestep, gradients); // v -= dt * h
303 
304  // Compute gradients for the new x.
305  compute(gradients, x);
306 
307  const real_type gradient_norm_squared = gradients * gradients;
308  conv = this->iteration_status(iter, gradient_norm_squared, x);
309  if (conv != SolverControl::iterate)
310  break;
311 
312  // v_dot_g = V * G
313  const real_type v_dot_g = velocities * gradients;
314 
315  if (v_dot_g < 0.)
316  {
317  const real_type velocities_norm_squared = velocities * velocities;
318 
319  // Check if we divide by zero in DEBUG mode.
320  Assert(gradient_norm_squared > 0., ExcInternalError());
321 
322  // beta = - alpha |V|/|G|
323  const real_type beta =
324  -alpha * std::sqrt(velocities_norm_squared / gradient_norm_squared);
325 
326  // V = (1-alpha) V + beta G.
327  velocities.sadd(1. - alpha, beta, gradients);
328 
329  if (iter - previous_iter_with_positive_v_dot_g > DELAYSTEP)
330  {
331  // Increase timestep and decrease alpha.
332  timestep = std::min(timestep * TIMESTEP_GROW, maximum_timestep);
333  alpha *= ALPHA_SHRINK;
334  }
335  }
336  else
337  {
338  // Decrease timestep, reset alpha and set V = 0.
339  previous_iter_with_positive_v_dot_g = iter;
340  timestep *= TIMESTEP_SHRINK;
341  alpha = ALPHA_0;
342  velocities = 0.;
343  }
344 
345  real_type vmax = velocities.linfty_norm();
346 
347  // Change timestep if any dof would move more than maximum_linfty_norm.
348  if (vmax > 0.)
349  {
350  const double minimal_timestep =
351  additional_data.maximum_linfty_norm / vmax;
352  if (minimal_timestep < timestep)
353  timestep = minimal_timestep;
354  }
355 
356  print_vectors(iter, x, velocities, gradients);
357 
358  } // While we need to iterate.
359 
360  // In the case of failure: throw exception.
361  if (conv != SolverControl::success)
362  AssertThrow(false,
363  SolverControl::NoConvergence(iter, gradients * gradients));
364 }

368 template <typename VectorType>
369 template <typename MatrixType, typename PreconditionerType>
370 void
371 SolverFIRE<VectorType>::solve(const MatrixType & A,
372  VectorType & x,
373  const VectorType & b,
374  const PreconditionerType &preconditioner)
375 {
376  std::function<double(VectorType &, const VectorType &)> compute_func =
377  [&](VectorType &g, const VectorType &x) -> double {
378  // Residual of the quadratic form @f$ \frac{1}{2} xAx - xb @f$.
379  // G = b - Ax
380  A.residual(g, x, b);
381 
382  // Gradient G = Ax -b.
383  g *= -1.;
384 
385  // The quadratic form @f$\frac{1}{2} xAx - xb @f$.
386  return 0.5 * A.matrix_norm_square(x) - x * b;
387  };
388 
389  this->solve(compute_func, x, preconditioner);
390 }
