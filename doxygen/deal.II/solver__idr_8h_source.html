269 template <class VectorType>
270 template <typename MatrixType, typename PreconditionerType>
271 void
272 SolverIDR<VectorType>::solve(const MatrixType & A,
273  VectorType & x,
274  const VectorType & b,
275  const PreconditionerType &preconditioner)
276 {
277  LogStream::Prefix prefix("IDR(s)");
278 
279  SolverControl::State iteration_state = SolverControl::iterate;
280  unsigned int step = 0;
281 
282  const unsigned int s = additional_data.s;
283 
284  // Define temporary vectors which do not do not depend on s
285  typename VectorMemory<VectorType>::Pointer r_pointer(this->memory);
286  typename VectorMemory<VectorType>::Pointer v_pointer(this->memory);
287  typename VectorMemory<VectorType>::Pointer vhat_pointer(this->memory);
288  typename VectorMemory<VectorType>::Pointer uhat_pointer(this->memory);
289  typename VectorMemory<VectorType>::Pointer ghat_pointer(this->memory);
290 
291  VectorType &r = *r_pointer;
292  VectorType &v = *v_pointer;
293  VectorType &vhat = *vhat_pointer;
294  VectorType &uhat = *uhat_pointer;
295  VectorType &ghat = *ghat_pointer;
296 
297  r.reinit(x, true);
298  v.reinit(x, true);
299  vhat.reinit(x, true);
300  uhat.reinit(x, true);
301  ghat.reinit(x, true);
302 
303  // Initial residual
304  A.vmult(r, x);
305  r.sadd(-1.0, 1.0, b);
306 
307  // Check for convergent initial guess
308  double res = r.l2_norm();
309  iteration_state = this->iteration_status(step, res, x);
310  if (iteration_state == SolverControl::success)
311  return;
312 
313  // Initialize sets of vectors/matrices whose size dependent on s
314  internal::SolverIDRImplementation::TmpVectors<VectorType> G(s, this->memory);
315  internal::SolverIDRImplementation::TmpVectors<VectorType> U(s, this->memory);
316  internal::SolverIDRImplementation::TmpVectors<VectorType> Q(s, this->memory);
317  FullMatrix<double> M(s, s);
318 
319  // Random number generator for vector entries of
320  // Q (normal distribution, mean=0 sigma=1)
321  std::mt19937 rng;
322  std::normal_distribution<> normal_distribution(0.0, 1.0);
323  for (unsigned int i = 0; i < s; ++i)
324  {
325  VectorType &tmp_g = G(i, x);
326  VectorType &tmp_u = U(i, x);
327  tmp_g = 0;
328  tmp_u = 0;
329 
330  // Compute random set of s orthonormalized vectors Q
331  // Note: the first vector is chosen to be the initial
332  // residual to match BiCGStab (as is done in comparisons
333  // with BiCGStab in the papers listed in the documentation
334  // of this function)
335  VectorType &tmp_q = Q(i, x);
336  if (i != 0)
337  {
338  for (auto indx : tmp_q.locally_owned_elements())
339  tmp_q(indx) = normal_distribution(rng);
340  tmp_q.compress(VectorOperation::insert);
341  }
342  else
343  tmp_q = r;
344 
345  for (unsigned int j = 0; j < i; ++j)
346  {
347  v = Q[j];
348  v *= (v * tmp_q) / (tmp_q * tmp_q);
349  tmp_q.add(-1.0, v);
350  }
351 
352  if (i != 0)
353  tmp_q *= 1.0 / tmp_q.l2_norm();
354 
355  M(i, i) = 1.;
356  }
357 
358  double omega = 1.;
359 
360  bool early_exit = false;
361 
362  // Outer iteration
363  while (iteration_state == SolverControl::iterate)
364  {
365  ++step;
366 
367  // Compute phi
368  Vector<double> phi(s);
369  for (unsigned int i = 0; i < s; ++i)
370  phi(i) = Q[i] * r;
371 
372  // Inner iteration over s
373  for (unsigned int k = 0; k < s; ++k)
374  {
375  // Solve M(k:s)*gamma = phi(k:s)
376  Vector<double> gamma(s - k);
377  {
378  Vector<double> phik(s - k);
379  FullMatrix<double> Mk(s - k, s - k);
380  std::vector<unsigned int> indices;
381  unsigned int j = 0;
382  for (unsigned int i = k; i < s; ++i, ++j)
383  {
384  indices.push_back(i);
385  phik(j) = phi(i);
386  }
387  Mk.extract_submatrix_from(M, indices, indices);
388 
389  FullMatrix<double> Mk_inv(s - k, s - k);
390  Mk_inv.invert(Mk);
391  Mk_inv.vmult(gamma, phik);
392  }
393 
394  {
395  v = r;
396 
397  unsigned int j = 0;
398  for (unsigned int i = k; i < s; ++i, ++j)
399  v.add(-1.0 * gamma(j), G[i]);
400  preconditioner.vmult(vhat, v);
401 
402  uhat = vhat;
403  uhat *= omega;
404  j = 0;
405  for (unsigned int i = k; i < s; ++i, ++j)
406  uhat.add(gamma(j), U[i]);
407  A.vmult(ghat, uhat);
408  }
409 
410  // Update G and U
411  // Orthogonalize ghat to Q0,..,Q_{k-1}
412  // and update uhat
413  for (unsigned int i = 0; i < k; ++i)
414  {
415  double alpha = (Q[i] * ghat) / M(i, i);
416  ghat.add(-alpha, G[i]);
417  uhat.add(-alpha, U[i]);
418  }
419  G[k] = ghat;
420  U[k] = uhat;
421 
422  // Update kth column of M
423  for (unsigned int i = k; i < s; ++i)
424  M(i, k) = Q[i] * G[k];
425 
426  // Orthogonalize r to Q0,...,Qk,
427  // update x
428  {
429  double beta = phi(k) / M(k, k);
430  r.add(-1.0 * beta, G[k]);
431  x.add(beta, U[k]);
432 
433  print_vectors(step, x, r, U[k]);
434 
435  // Check for early convergence. If so, store
436  // information in early_exit so that outer iteration
437  // is broken before recomputing the residual
438  res = r.l2_norm();
439  iteration_state = this->iteration_status(step, res, x);
440  if (iteration_state != SolverControl::iterate)
441  {
442  early_exit = true;
443  break;
444  }
445 
446  // Update phi
447  if (k + 1 < s)
448  {
449  for (unsigned int i = 0; i < k + 1; ++i)
450  phi(i) = 0.0;
451  for (unsigned int i = k + 1; i < s; ++i)
452  phi(i) -= beta * M(i, k);
453  }
454  }
455  }
456  if (early_exit == true)
457  break;
458 
459  // Update r and x
460  preconditioner.vmult(vhat, r);
461  A.vmult(v, vhat);
462 
463  omega = (v * r) / (v * v);
464 
465  r.add(-1.0 * omega, v);
466  x.add(omega, vhat);
467 
468  print_vectors(step, x, r, vhat);
469 
470  // Check for convergence
471  res = r.l2_norm();
472  iteration_state = this->iteration_status(step, res, x);
473  if (iteration_state != SolverControl::iterate)
474  break;
475  }
476 
477  if (iteration_state != SolverControl::success)
478  AssertThrow(false, SolverControl::NoConvergence(step, res));
479 }
