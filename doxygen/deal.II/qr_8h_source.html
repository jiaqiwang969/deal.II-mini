583 template <typename VectorType>
584 bool
585 ImplicitQR<VectorType>::append_column(const VectorType &column)
586 {
587  if (this->current_size == 0)
588  {
589  this->R.grow_or_shrink(this->current_size + 1);
590  this->columns.push_back(std::make_unique<VectorType>(column));
591  this->R(0, 0) = column.l2_norm();
592  ++this->current_size;
593  }
594  else
595  {
596  // first get scalar products with A^T
597  Vector<Number> u(this->current_size);
598  this->multiply_with_AT(u, column);
599 
600  // now solve R^T x = (A^T * column)
601  const int lda = this->current_size;
602  const int ldb = this->current_size;
603  const int N = this->current_size;
604  const int n_rhs = 1;
605  int info = 0;
606  internal::QRImplementation::call_trtrs(
607  'U', 'T', 'N', N, n_rhs, &this->R(0, 0), lda, &u(0), ldb, &info);
608 
609  // finally get the diagonal element:
610  // rho2 = |column|^2 - |u|^2
611  const Number column_norm_sqr = column.norm_sqr();
612  const Number rho2 = column_norm_sqr - u.norm_sqr();
613  const bool linearly_independent =
614  column_signal.empty() ? rho2 > 0 :
615  column_signal(u, rho2, column_norm_sqr).get();
616 
617  // bail out if it turns out to be linearly dependent
618  if (!linearly_independent)
619  return false;
620 
621  // at this point we update is successful and we can enlarge R
622  // and store the column:
623  this->columns.push_back(std::make_unique<VectorType>(column));
624  this->R.grow_or_shrink(this->current_size + 1);
625  this->R(this->current_size, this->current_size) = std::sqrt(rho2);
626  for (unsigned int i = 0; i < this->current_size; ++i)
627  this->R(i, this->current_size) = u(i);
628 
629  this->current_size++;
630  }
631 
632  return true;
633 }

637 template <typename VectorType>
638 void
639 ImplicitQR<VectorType>::apply_givens_rotation(const unsigned int i,
640  const unsigned int k)
641 {
642  AssertIndexRange(i, k);
643  AssertIndexRange(k, this->current_size);
644  const std::array<Number, 3> csr =
645  ::Utilities::LinearAlgebra::givens_rotation<Number>(this->R(i, k),
646  this->R(k, k));
647 
648  // first, set k'th column:
649  this->R(i, k) = csr[2];
650  this->R(k, k) = 0.;
651  // now do the rest:
652  for (unsigned int j = 0; j < this->R.n(); ++j)
653  if (j != k)
654  {
655  const Number t = this->R(i, j);
656  this->R(i, j) = csr[0] * this->R(i, j) + csr[1] * this->R(k, j);
657  this->R(k, j) = -csr[1] * t + csr[0] * this->R(k, j);
658  }
659 
660  if (!this->givens_signal.empty())
661  this->givens_signal(i, k, csr);
662 }

666 template <typename VectorType>
667 void
668 ImplicitQR<VectorType>::remove_column(const unsigned int k)
669 {
670  // before actually removing a column from Q and resizing R,
671  // apply givens rotations to bring H into upper triangular form:
672  for (unsigned int j = k + 1; j < this->R.n(); ++j)
673  {
674  const unsigned int i = j - 1;
675  apply_givens_rotation(i, j);
676  }
677 
678  // remove last row and k-th column
679  --this->current_size;
680  this->R.remove_row_and_column(this->current_size, k);
681 
682  // Finally remove the column from A
683  this->columns.erase(this->columns.begin() + k);
684 }

743 template <typename VectorType>
744 bool
745 QR<VectorType>::append_column(const VectorType &column)
746 {
747  // resize R:
748  this->R.grow_or_shrink(this->current_size + 1);
749  this->columns.push_back(std::make_unique<VectorType>(column));
750 
751  // now a Gram-Schmidt part: orthonormalize the new column
752  // against everything we have so far:
753  auto &last_col = *this->columns.back();
754  for (unsigned int i = 0; i < this->current_size; ++i)
755  {
756  const auto &i_col = *this->columns[i];
757  this->R(i, this->current_size) = i_col * last_col;
758  last_col.add(-this->R(i, this->current_size), i_col);
759  }
760 
761  this->R(this->current_size, this->current_size) = last_col.l2_norm();
762 
763  Assert(this->R(this->current_size, this->current_size) > 0.,
764  ExcDivideByZero());
765  last_col *= 1. / this->R(this->current_size, this->current_size);
766 
767  ++this->current_size;
768  return true;
769 }

773 template <typename VectorType>
774 void
775 QR<VectorType>::apply_givens_rotation(const unsigned int i,
776  const unsigned int k)
777 {
778  AssertIndexRange(i, k);
779  AssertIndexRange(k, this->current_size);
780  const std::array<Number, 3> csr =
781  ::Utilities::LinearAlgebra::givens_rotation<Number>(this->R(i, k),
782  this->R(k, k));
783 
784  // first, set k'th column:
785  this->R(i, k) = csr[2];
786  this->R(k, k) = 0.;
787  // now do the rest:
788  for (unsigned int j = 0; j < this->R.n(); ++j)
789  if (j != k)
790  {
791  const Number t = this->R(i, j);
792  this->R(i, j) = csr[0] * this->R(i, j) + csr[1] * this->R(k, j);
793  this->R(k, j) = -csr[1] * t + csr[0] * this->R(k, j);
794  }
795 
796  // now adjust i,k columns due to multiplication with the
797  // transpose Givens matrix from right:
798  auto &col_i = *this->columns[i];
799  auto &col_k = *this->columns[k];
800  // save column i:
801  tmp = col_i;
802  col_i.sadd(csr[0], csr[1], col_k);
803  col_k.sadd(csr[0], -csr[1], tmp);
804 
805  if (!this->givens_signal.empty())
806  this->givens_signal(i, k, csr);
807 }

811 template <typename VectorType>
812 void
813 QR<VectorType>::remove_column(const unsigned int k)
814 {
815  AssertIndexRange(k, this->current_size);
816  Assert(this->current_size > 0,
817  ExcMessage("Can not remove a column if QR is empty"));
818  // apply a sequence of Givens rotations
819  // see section 6.5 "Updating matrix factorizations" in Golub 2013, Matrix
820  // computations
821 
822  // So we want to have QR for \tilde A \in R^{m*(n-1)}
823  // if we remove the column k, we end up with upper Hessenberg matrix
824  // x x x x x
825  // x x x x
826  // H = x x x
827  // x x x
828  // x x
829  // x
830  // where k = 2 (3rd column), m = 7, n = 6
831  //
832  // before actually removing a column from Q and resizing R,
833  // apply givens rotations to bring H into upper triangular form:
834  for (unsigned int j = k + 1; j < this->R.n(); ++j)
835  {
836  const unsigned int i = j - 1;
837  apply_givens_rotation(i, j);
838  }
839 
840  // now we can throw away the column from Q and adjust R
841  // since we do thin-QR, after Givens rotations we need to throw
842  // away the last column:
843  const unsigned int size_minus_1 = this->columns.size() - 1;
844  this->columns.erase(this->columns.begin() + size_minus_1);
845 
846  // remove last row and k-th column
847  --this->current_size;
848  this->R.remove_row_and_column(this->current_size, k);
849 }
