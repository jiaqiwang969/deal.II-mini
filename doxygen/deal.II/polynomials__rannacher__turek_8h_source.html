165  template <int order>
166  inline Tensor<order, 2>
167  compute_derivative(const unsigned int i, const Point<2> &p)
168  {
169  const unsigned int dim = 2;
170 
171  Tensor<order, dim> derivative;
172  switch (order)
173  {
174  case 1:
175  {
176  Tensor<1, dim> &grad =
177  *reinterpret_cast<Tensor<1, dim> *>(&derivative);
178  if (i == 0)
179  {
180  grad[0] = -2.5 + 3 * p(0);
181  grad[1] = 1.5 - 3 * p(1);
182  }
183  else if (i == 1)
184  {
185  grad[0] = -0.5 + 3.0 * p(0);
186  grad[1] = 1.5 - 3.0 * p(1);
187  }
188  else if (i == 2)
189  {
190  grad[0] = 1.5 - 3.0 * p(0);
191  grad[1] = -2.5 + 3.0 * p(1);
192  }
193  else if (i == 3)
194  {
195  grad[0] = 1.5 - 3.0 * p(0);
196  grad[1] = -0.5 + 3.0 * p(1);
197  }
198  else
199  {
200  Assert(false, ExcNotImplemented());
201  }
202  return derivative;
203  }
204  case 2:
205  {
206  Tensor<2, dim> &grad_grad =
207  *reinterpret_cast<Tensor<2, dim> *>(&derivative);
208  if (i == 0)
209  {
210  grad_grad[0][0] = 3;
211  grad_grad[0][1] = 0;
212  grad_grad[1][0] = 0;
213  grad_grad[1][1] = -3;
214  }
215  else if (i == 1)
216  {
217  grad_grad[0][0] = 3;
218  grad_grad[0][1] = 0;
219  grad_grad[1][0] = 0;
220  grad_grad[1][1] = -3;
221  }
222  else if (i == 2)
223  {
224  grad_grad[0][0] = -3;
225  grad_grad[0][1] = 0;
226  grad_grad[1][0] = 0;
227  grad_grad[1][1] = 3;
228  }
229  else if (i == 3)
230  {
231  grad_grad[0][0] = -3;
232  grad_grad[0][1] = 0;
233  grad_grad[1][0] = 0;
234  grad_grad[1][1] = 3;
235  }
236  return derivative;
237  }
238  default:
239  {
240  // higher derivatives are all zero
241  return Tensor<order, dim>();
242  }
243  }
244  }
