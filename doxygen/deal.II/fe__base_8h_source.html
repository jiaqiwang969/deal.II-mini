633  inline Domination operator&(const Domination d1, const Domination d2)
634  {
635  // go through the entire list of possibilities. note that if we were into
636  // speed, obfuscation and cared enough, we could implement this operator
637  // by doing a bitwise & (and) if we gave these values to the enum values:
638  // neither_element_dominates=0, this_element_dominates=1,
639  // other_element_dominates=2, either_element_can_dominate=3
640  // =this_element_dominates|other_element_dominates
641  switch (d1)
642  {
643  case this_element_dominates:
644  if ((d2 == this_element_dominates) ||
645  (d2 == either_element_can_dominate) || (d2 == no_requirements))
646  return this_element_dominates;
647  else
648  return neither_element_dominates;
649 
650  case other_element_dominates:
651  if ((d2 == other_element_dominates) ||
652  (d2 == either_element_can_dominate) || (d2 == no_requirements))
653  return other_element_dominates;
654  else
655  return neither_element_dominates;
656 
657  case neither_element_dominates:
658  return neither_element_dominates;
659 
660  case either_element_can_dominate:
661  if (d2 == no_requirements)
662  return either_element_can_dominate;
663  else
664  return d2;
665 
666  case no_requirements:
667  return d2;
668 
669  default:
670  // shouldn't get here
671  Assert(false, ExcInternalError());
672  }
673 
674  return neither_element_dominates;
675  }
676 } // namespace FiniteElementDomination
