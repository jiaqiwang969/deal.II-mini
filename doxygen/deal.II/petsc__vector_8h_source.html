364  template <typename number>
365  inline Vector &
366  Vector::operator=(const ::Vector<number> &v)
367  {
368  Assert(size() == v.size(), ExcDimensionMismatch(size(), v.size()));
369 
370  // FIXME: the following isn't necessarily fast, but this is due to
371  // the fact that PETSc doesn't offer an inlined access operator.
372  //
373  // if someone wants to contribute some code: to make this code
374  // faster, one could either first convert all values to PetscScalar,
375  // and then set them all at once using VecSetValues. This has the
376  // drawback that it could take quite some memory, if the vector is
377  // large, and it would in addition allocate memory on the heap, which
378  // is expensive. an alternative would be to split the vector into
379  // chunks of, say, 128 elements, convert a chunk at a time and set it
380  // in the output vector using VecSetValues. since 128 elements is
381  // small enough, this could easily be allocated on the stack (as a
382  // local variable) which would make the whole thing much more
383  // efficient.
384  //
385  // a second way to make things faster is for the special case that
386  // number==PetscScalar. we could then declare a specialization of
387  // this template, and omit the conversion. the problem with this is
388  // that the best we can do is to use VecSetValues, but this isn't
389  // very efficient either: it wants to see an array of indices, which
390  // in this case a) again takes up a whole lot of memory on the heap,
391  // and b) is totally dumb since its content would simply be the
392  // sequence 0,1,2,3,...,n. the best of all worlds would probably be a
393  // function in Petsc that would take a pointer to an array of
394  // PetscScalar values and simply copy n elements verbatim into the
395  // vector...
396  for (size_type i = 0; i < v.size(); ++i)
397  (*this)(i) = v(i);
398 
399  compress(::VectorOperation::insert);
400 
401  return *this;
402  }
