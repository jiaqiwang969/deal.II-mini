1205  struct Inverse<4, 2, Number>
1206  {
1207  constexpr static inline ::SymmetricTensor<4, 2, Number>
1208  value(const ::SymmetricTensor<4, 2, Number> &t)
1209  {
1210  ::SymmetricTensor<4, 2, Number> tmp;
1211 
1212  // Inverting this tensor is a little more complicated than necessary,
1213  // since we store the data of 't' as a 3x3 matrix t.data, but the
1214  // product between a rank-4 and a rank-2 tensor is really not the
1215  // product between this matrix and the 3-vector of a rhs, but rather
1216  //
1217  // B.vec = t.data * mult * A.vec
1218  //
1219  // where mult is a 3x3 matrix with entries [[1,0,0],[0,1,0],[0,0,2]] to
1220  // capture the fact that we need to add up both the c_ij12*a_12 and the
1221  // c_ij21*a_21 terms.
1222  //
1223  // In addition, in this scheme, the identity tensor has the matrix
1224  // representation mult^-1.
1225  //
1226  // The inverse of 't' therefore has the matrix representation
1227  //
1228  // inv.data = mult^-1 * t.data^-1 * mult^-1
1229  //
1230  // in order to compute it, let's first compute the inverse of t.data and
1231  // put it into tmp.data; at the end of the function we then scale the
1232  // last row and column of the inverse by 1/2, corresponding to the left
1233  // and right multiplication with mult^-1.
1234  const Number t4 = t.data[0][0] * t.data[1][1],
1235  t6 = t.data[0][0] * t.data[1][2],
1236  t8 = t.data[0][1] * t.data[1][0],
1237  t00 = t.data[0][2] * t.data[1][0],
1238  t01 = t.data[0][1] * t.data[2][0],
1239  t04 = t.data[0][2] * t.data[2][0],
1240  t07 = 1.0 / (t4 * t.data[2][2] - t6 * t.data[2][1] -
1241  t8 * t.data[2][2] + t00 * t.data[2][1] +
1242  t01 * t.data[1][2] - t04 * t.data[1][1]);
1243  tmp.data[0][0] =
1244  (t.data[1][1] * t.data[2][2] - t.data[1][2] * t.data[2][1]) * t07;
1245  tmp.data[0][1] =
1246  -(t.data[0][1] * t.data[2][2] - t.data[0][2] * t.data[2][1]) * t07;
1247  tmp.data[0][2] =
1248  -(-t.data[0][1] * t.data[1][2] + t.data[0][2] * t.data[1][1]) * t07;
1249  tmp.data[1][0] =
1250  -(t.data[1][0] * t.data[2][2] - t.data[1][2] * t.data[2][0]) * t07;
1251  tmp.data[1][1] = (t.data[0][0] * t.data[2][2] - t04) * t07;
1252  tmp.data[1][2] = -(t6 - t00) * t07;
1253  tmp.data[2][0] =
1254  -(-t.data[1][0] * t.data[2][1] + t.data[1][1] * t.data[2][0]) * t07;
1255  tmp.data[2][1] = -(t.data[0][0] * t.data[2][1] - t01) * t07;
1256  tmp.data[2][2] = (t4 - t8) * t07;
1257 
1258  // scale last row and column as mentioned
1259  // above
1260  tmp.data[2][0] /= 2;
1261  tmp.data[2][1] /= 2;
1262  tmp.data[0][2] /= 2;
1263  tmp.data[1][2] /= 2;
1264  tmp.data[2][2] /= 4;
1265 
1266  return tmp;
1267  }

1271  template <typename Number>
1272  struct Inverse<4, 3, Number>
1273  {
1274  static ::SymmetricTensor<4, 3, Number>
1275  value(const ::SymmetricTensor<4, 3, Number> &t)
1276  {
1277  ::SymmetricTensor<4, 3, Number> tmp = t;
1278 
1279  // This function follows the exact same scheme as the 2d case, except
1280  // that hardcoding the inverse of a 6x6 matrix is pretty wasteful.
1281  // Instead, we use the Gauss-Jordan algorithm implemented for
1282  // FullMatrix. For historical reasons the following code is copied from
1283  // there, with the tangential benefit that we do not need to copy the
1284  // tensor entries to and from the FullMatrix.
1285  const unsigned int N = 6;
1286 
1287  // First get an estimate of the size of the elements of this matrix,
1288  // for later checks whether the pivot element is large enough, or
1289  // whether we have to fear that the matrix is not regular.
1290  Number diagonal_sum = internal::NumberType<Number>::value(0.0);
1291  for (unsigned int i = 0; i < N; ++i)
1292  diagonal_sum += std::fabs(tmp.data[i][i]);
1293  const Number typical_diagonal_element =
1294  diagonal_sum / static_cast<double>(N);
1295  (void)typical_diagonal_element;
1296 
1297  unsigned int p[N];
1298  for (unsigned int i = 0; i < N; ++i)
1299  p[i] = i;
1300 
1301  for (unsigned int j = 0; j < N; ++j)
1302  {
1303  // Pivot search: search that part of the line on and right of the
1304  // diagonal for the largest element.
1305  Number max = std::fabs(tmp.data[j][j]);
1306  unsigned int r = j;
1307  for (unsigned int i = j + 1; i < N; ++i)
1308  if (std::fabs(tmp.data[i][j]) > max)
1309  {
1310  max = std::fabs(tmp.data[i][j]);
1311  r = i;
1312  }
1313 
1314  // Check whether the pivot is too small
1315  Assert(max > 1.e-16 * typical_diagonal_element,
1316  ExcMessage("This tensor seems to be noninvertible"));
1317 
1318  // Row interchange
1319  if (r > j)
1320  {
1321  for (unsigned int k = 0; k < N; ++k)
1322  std::swap(tmp.data[j][k], tmp.data[r][k]);
1323 
1324  std::swap(p[j], p[r]);
1325  }
1326 
1327  // Transformation
1328  const Number hr = 1. / tmp.data[j][j];
1329  tmp.data[j][j] = hr;
1330  for (unsigned int k = 0; k < N; ++k)
1331  {
1332  if (k == j)
1333  continue;
1334  for (unsigned int i = 0; i < N; ++i)
1335  {
1336  if (i == j)
1337  continue;
1338  tmp.data[i][k] -= tmp.data[i][j] * tmp.data[j][k] * hr;
1339  }
1340  }
1341  for (unsigned int i = 0; i < N; ++i)
1342  {
1343  tmp.data[i][j] *= hr;
1344  tmp.data[j][i] *= -hr;
1345  }
1346  tmp.data[j][j] = hr;
1347  }
1348 
1349  // Column interchange
1350  Number hv[N];
1351  for (unsigned int i = 0; i < N; ++i)
1352  {
1353  for (unsigned int k = 0; k < N; ++k)
1354  hv[p[k]] = tmp.data[i][k];
1355  for (unsigned int k = 0; k < N; ++k)
1356  tmp.data[i][k] = hv[k];
1357  }
1358 
1359  // Scale rows and columns. The mult matrix
1360  // here is diag[1, 1, 1, 1/2, 1/2, 1/2].
1361  for (unsigned int i = 3; i < 6; ++i)
1362  for (unsigned int j = 0; j < 3; ++j)
1363  tmp.data[i][j] /= 2;
1364 
1365  for (unsigned int i = 0; i < 3; ++i)
1366  for (unsigned int j = 3; j < 6; ++j)
1367  tmp.data[i][j] /= 2;
1368 
1369  for (unsigned int i = 3; i < 6; ++i)
1370  for (unsigned int j = 3; j < 6; ++j)
1371  tmp.data[i][j] /= 4;
1372 
1373  return tmp;
1374  }
