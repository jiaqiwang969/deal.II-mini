1364  MatrixBase::end(const size_type r) const
1365  {
1366  Assert(in_local_range(r),
1367  ExcIndexRange(r, local_range().first, local_range().second));
1368 
1369  // place the iterator on the first entry past this line, or at the
1370  // end of the matrix
1371  //
1372  // in the parallel case, we need to put it on the first entry of
1373  // the first row after the locally owned range. this of course
1374  // doesn't exist, but we can nevertheless create such an
1375  // iterator. we need to check whether 'i' is past the locally
1376  // owned range of rows first, before we ask for the length of the
1377  // row since the latter query leads to an exception in case we ask
1378  // for a row that is not locally owned
1379  for (size_type i = r + 1; i < m(); ++i)
1380  if (i == local_range().second || (row_length(i) > 0))
1381  return const_iterator(this, i, 0);
1382 
1383  // if there is no such line, then take the
1384  // end iterator of the matrix
1385  // we don't allow calling end() directly for distributed matrices so we need
1386  // to copy the code without the assertion.
1387  return {this, m(), 0};
1388  }
