849  template <typename MatrixType>
850  inline void
851  MGMatrixSimple<MatrixType>::assemble(
852  MatrixType & G,
853  const FullMatrix<double> & M,
854  const std::vector<types::global_dof_index> &i1,
855  const std::vector<types::global_dof_index> &i2,
856  const unsigned int level)
857  {
858  AssertDimension(M.m(), i1.size());
859  AssertDimension(M.n(), i2.size());
860 
861  if (mg_constrained_dofs == nullptr)
862  {
863  for (unsigned int j = 0; j < i1.size(); ++j)
864  for (unsigned int k = 0; k < i2.size(); ++k)
865  if (std::fabs(M(j, k)) >= threshold)
866  G.add(i1[j], i2[k], M(j, k));
867  }
868  else
869  {
870  for (unsigned int j = 0; j < i1.size(); ++j)
871  for (unsigned int k = 0; k < i2.size(); ++k)
872  {
873  // Only enter the local values into the global matrix,
874  // if the value is larger than the threshold
875  if (std::fabs(M(j, k)) < threshold)
876  continue;
877 
878  // Do not enter, if either the row or the column
879  // corresponds to an index on the refinement edge. The
880  // level problems are solved with homogeneous
881  // Dirichlet boundary conditions, therefore we
882  // eliminate these rows and columns. The corresponding
883  // matrix entries are entered by assemble_in() and
884  // assemble_out().
885  if (mg_constrained_dofs->at_refinement_edge(level, i1[j]) ||
886  mg_constrained_dofs->at_refinement_edge(level, i2[k]))
887  continue;
888 
889  // At the boundary, only enter the term on the
890  // diagonal, but not the coupling terms
891  if ((mg_constrained_dofs->is_boundary_index(level, i1[j]) ||
892  mg_constrained_dofs->is_boundary_index(level, i2[k])) &&
893  (i1[j] != i2[k]))
894  continue;
895 
896  G.add(i1[j], i2[k], M(j, k));
897  }
898  }
899  }
