635  { // 如果 x 为左子节点
636  auto brother = xp->right;
637  if (rb_tree_is_red(brother))
638  { // case 1
639  rb_tree_set_black(brother);
640  rb_tree_set_red(xp);
641  rb_tree_rotate_left(xp, root);
642  brother = xp->right;
643  }
644  // case 1 转为为了 case 2、3、4 中的一种
645  if ((brother->left == nullptr || !rb_tree_is_red(brother->left)) &&
646  (brother->right == nullptr || !rb_tree_is_red(brother->right)))
647  { // case 2
648  rb_tree_set_red(brother);
649  x = xp;
650  xp = xp->parent;
651  }
652  else
653  {
654  if (brother->right == nullptr || !rb_tree_is_red(brother->right))
655  { // case 3
656  if (brother->left != nullptr)
657  rb_tree_set_black(brother->left);
658  rb_tree_set_red(brother);
659  rb_tree_rotate_right(brother, root);
660  brother = xp->right;
661  }
662  // 转为 case 4
663  brother->color = xp->color;
664  rb_tree_set_black(xp);
665  if (brother->right != nullptr)
666  rb_tree_set_black(brother->right);
667  rb_tree_rotate_left(xp, root);
668  break;
669  }
670  }
671  else // x 为右子节点，对称处理
672  {
673  auto brother = xp->left;
674  if (rb_tree_is_red(brother))
675  { // case 1
676  rb_tree_set_black(brother);
677  rb_tree_set_red(xp);
678  rb_tree_rotate_right(xp, root);
679  brother = xp->left;
680  }
681  if ((brother->left == nullptr || !rb_tree_is_red(brother->left)) &&
682  (brother->right == nullptr || !rb_tree_is_red(brother->right)))
683  { // case 2
684  rb_tree_set_red(brother);
685  x = xp;
686  xp = xp->parent;
687  }
688  else
689  {
690  if (brother->left == nullptr || !rb_tree_is_red(brother->left))
691  { // case 3
692  if (brother->right != nullptr)
693  rb_tree_set_black(brother->right);
694  rb_tree_set_red(brother);
695  rb_tree_rotate_left(brother, root);
696  brother = xp->left;
697  }
698  // 转为 case 4
699  brother->color = xp->color;
700  rb_tree_set_black(xp);
701  if (brother->left != nullptr)
702  rb_tree_set_black(brother->left);
703  rb_tree_rotate_right(xp, root);
704  break;
705  }
706  }
707  }
708  if (x != nullptr)
709  rb_tree_set_black(x);
710  }
711  return y;
712 }
