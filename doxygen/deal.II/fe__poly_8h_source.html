223  virtual std::unique_ptr<
224  typename FiniteElement<dim, spacedim>::InternalDataBase>
225  get_data(
226  const UpdateFlags update_flags,
227  const Mapping<dim, spacedim> &mapping,
228  const Quadrature<dim> & quadrature,
229  ::internal::FEValuesImplementation::FiniteElementRelatedData<dim,
230  spacedim>
231  &output_data) const override
232  {
233  (void)mapping;
234 
235  // generate a new data object and
236  // initialize some fields
237  std::unique_ptr<typename FiniteElement<dim, spacedim>::InternalDataBase>
238  data_ptr = std::make_unique<InternalData>();
239  auto &data = dynamic_cast<InternalData &>(*data_ptr);
240  data.update_each = requires_update_flags(update_flags);
241 
242  const unsigned int n_q_points = quadrature.size();
243 
244  // initialize some scratch arrays. we need them for the underlying
245  // polynomial to put the values and derivatives of shape functions
246  // to put there, depending on what the user requested
247  std::vector<double> values(
248  update_flags & update_values ? this->n_dofs_per_cell() : 0);
249  std::vector<Tensor<1, dim>> grads(
250  update_flags & update_gradients ? this->n_dofs_per_cell() : 0);
251  std::vector<Tensor<2, dim>> grad_grads(
252  update_flags & update_hessians ? this->n_dofs_per_cell() : 0);
253  std::vector<Tensor<3, dim>> third_derivatives(
254  update_flags & update_3rd_derivatives ? this->n_dofs_per_cell() : 0);
255  std::vector<Tensor<4, dim>>
256  fourth_derivatives; // won't be needed, so leave empty
257 
258  // now also initialize fields the fields of this class's own
259  // temporary storage, depending on what we need for the given
260  // update flags.
261  //
262  // there is one exception from the rule: if we are dealing with
263  // cells (i.e., if this function is not called via
264  // get_(sub)face_data()), then we can already store things in the
265  // final location where FEValues::reinit() later wants to see
266  // things. we then don't need the intermediate space. we determine
267  // whether we are on a cell by asking whether the number of
268  // elements in the output array equals the number of quadrature
269  // points (yes, it's a cell) or not (because in that case the
270  // number of quadrature points we use here equals the number of
271  // quadrature points summed over *all* faces or subfaces, whereas
272  // the number of output slots equals the number of quadrature
273  // points on only *one* face)
274  if ((update_flags & update_values) &&
275  !((output_data.shape_values.n_rows() > 0) &&
276  (output_data.shape_values.n_cols() == n_q_points)))
277  data.shape_values.reinit(this->n_dofs_per_cell(), n_q_points);
278 
279  if (update_flags & update_gradients)
280  data.shape_gradients.reinit(this->n_dofs_per_cell(), n_q_points);
281 
282  if (update_flags & update_hessians)
283  data.shape_hessians.reinit(this->n_dofs_per_cell(), n_q_points);
284 
285  if (update_flags & update_3rd_derivatives)
286  data.shape_3rd_derivatives.reinit(this->n_dofs_per_cell(), n_q_points);
287 
288  // next already fill those fields of which we have information by
289  // now. note that the shape gradients are only those on the unit
290  // cell, and need to be transformed when visiting an actual cell
291  if (update_flags & (update_values | update_gradients | update_hessians |
292  update_3rd_derivatives))
293  for (unsigned int i = 0; i < n_q_points; ++i)
294  {
295  poly_space->evaluate(quadrature.point(i),
296  values,
297  grads,
298  grad_grads,
299  third_derivatives,
300  fourth_derivatives);
301 
302  // the values of shape functions at quadrature points don't change.
303  // consequently, write these values right into the output array if
304  // we can, i.e., if the output array has the correct size. this is
305  // the case on cells. on faces, we already precompute data on *all*
306  // faces and subfaces, but we later on copy only a portion of it
307  // into the output object; in that case, copy the data from all
308  // faces into the scratch object
309  if (update_flags & update_values)
310  if (output_data.shape_values.n_rows() > 0)
311  {
312  if (output_data.shape_values.n_cols() == n_q_points)
313  for (unsigned int k = 0; k < this->n_dofs_per_cell(); ++k)
314  output_data.shape_values[k][i] = values[k];
315  else
316  for (unsigned int k = 0; k < this->n_dofs_per_cell(); ++k)
317  data.shape_values[k][i] = values[k];
318  }
319 
320  // for everything else, derivatives need to be transformed,
321  // so we write them into our scratch space and only later
322  // copy stuff into where FEValues wants it
323  if (update_flags & update_gradients)
324  for (unsigned int k = 0; k < this->n_dofs_per_cell(); ++k)
325  data.shape_gradients[k][i] = grads[k];
326 
327  if (update_flags & update_hessians)
328  for (unsigned int k = 0; k < this->n_dofs_per_cell(); ++k)
329  data.shape_hessians[k][i] = grad_grads[k];
330 
331  if (update_flags & update_3rd_derivatives)
332  for (unsigned int k = 0; k < this->n_dofs_per_cell(); ++k)
333  data.shape_3rd_derivatives[k][i] = third_derivatives[k];
334  }
335  return data_ptr;
336  }
