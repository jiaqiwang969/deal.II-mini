444  // lexicographical_compare
445  // 以字典序排列对两个序列进行比较，当在某个位置发现第一组不相等元素时，有下列几种情况：
446  // (1)如果第一序列的元素较小，返回 true ，否则返回 false
447  // (2)如果到达 last1 而尚未到达 last2 返回 true
448  // (3)如果到达 last2 而尚未到达 last1 返回 false
449  // (4)如果同时到达 last1 和 last2 返回 false
450  /*****************************************************************************************/
451  template <class InputIter1, class InputIter2>
452  bool lexicographical_compare(InputIter1 first1, InputIter1 last1,
453  InputIter2 first2, InputIter2 last2)
454  {
455  for (; first1 != last1 && first2 != last2; ++first1, ++first2)
456  {
457  if (*first1 < *first2)

461  }
