970  Iterator::operator++()
971  {
972  Assert(accessor.a_row < accessor.sparsity_pattern->n_rows(),
973  ExcIteratorPastEnd());
974 
975  ++accessor.a_index;
976 
977  // If at end of line: do one step, then cycle until we find a row with a
978  // nonzero number of entries that is stored locally.
979  if (accessor.a_index >= accessor.colnum_cache->size())
980  {
981  accessor.a_index = 0;
982  ++accessor.a_row;
983 
984  while (accessor.a_row < accessor.sparsity_pattern->n_rows())
985  {
986  const auto row_length =

990  accessor.a_row))

1146  template <typename ForwardIterator>
1147  inline void
1148  SparsityPattern::add_entries(const size_type row,
1149  ForwardIterator begin,
1150  ForwardIterator end,
1151  const bool /*indices_are_sorted*/ )
1152  {
1153  if (begin == end)
1154  return;
1155 
1156  // verify that the size of the data type Trilinos expects matches that the
1157  // iterator points to. we allow for some slippage between signed and
1158  // unsigned and only compare that they are both either 32 or 64 bit. to
1159  // write this test properly, not that we cannot compare the size of
1160  // '*begin' because 'begin' may be an iterator and '*begin' may be an
1161  // accessor class. consequently, we need to somehow get an actual value
1162  // from it which we can by evaluating an expression such as when
1163  // multiplying the value produced by 2
1164  Assert(sizeof(TrilinosWrappers::types::int_type) == sizeof((*begin) * 2),
1165  ExcNotImplemented());
1166 
1167  TrilinosWrappers::types::int_type *col_index_ptr =
1168  reinterpret_cast<TrilinosWrappers::types::int_type *>(
1169  const_cast<typename std::decay<decltype(*begin)>::type *>(&*begin));
1170  // Check at least for the first index that the conversion actually works
1171  AssertDimension(*col_index_ptr, *begin);
1172  TrilinosWrappers::types::int_type trilinos_row_index = row;
1173  const int n_cols = static_cast<int>(end - begin);
1174 
1175  int ierr;
1176  if (row_is_stored_locally(row))
1177  ierr =
1178  graph->InsertGlobalIndices(trilinos_row_index, n_cols, col_index_ptr);
1179  else if (nonlocal_graph.get() != nullptr)
1180  {
1181  // this is the case when we have explicitly set the off-processor rows
1182  // and want to create a separate matrix object for them (to retain
1183  // thread-safety)
1184  Assert(nonlocal_graph->RowMap().LID(
1185  static_cast<TrilinosWrappers::types::int_type>(row)) != -1,
1186  ExcMessage("Attempted to write into off-processor matrix row "
1187  "that has not be specified as being writable upon "
1188  "initialization"));
1189  ierr = nonlocal_graph->InsertGlobalIndices(trilinos_row_index,
1190  n_cols,
1191  col_index_ptr);
1192  }
1193  else
1194  ierr = graph->InsertGlobalIndices(1,
1195  &trilinos_row_index,
1196  n_cols,
1197  col_index_ptr);
1198 
1199  AssertThrow(ierr >= 0, ExcTrilinosError(ierr));
1200  }
