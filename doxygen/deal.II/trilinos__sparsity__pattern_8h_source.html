1250  template <typename ForwardIterator>
1251  inline void
1252  SparsityPattern::add_entries(const size_type row,
1253  ForwardIterator begin,
1254  ForwardIterator end,
1255  const bool /*indices_are_sorted*/)
1256  {
1257  if (begin == end)
1258  return;
1259 
1260  // verify that the size of the data type Trilinos expects matches that the
1261  // iterator points to. we allow for some slippage between signed and
1262  // unsigned and only compare that they are both either 32 or 64 bit. to
1263  // write this test properly, not that we cannot compare the size of
1264  // '*begin' because 'begin' may be an iterator and '*begin' may be an
1265  // accessor class. consequently, we need to somehow get an actual value
1266  // from it which we can by evaluating an expression such as when
1267  // multiplying the value produced by 2
1268  Assert(sizeof(TrilinosWrappers::types::int_type) == sizeof((*begin) * 2),
1269  ExcNotImplemented());
1270 
1271  TrilinosWrappers::types::int_type *col_index_ptr =
1272  reinterpret_cast<TrilinosWrappers::types::int_type *>(
1273  const_cast<typename std::decay<decltype(*begin)>::type *>(&*begin));
1274  // Check at least for the first index that the conversion actually works
1275  AssertDimension(*col_index_ptr, *begin);
1276  TrilinosWrappers::types::int_type trilinos_row_index = row;
1277  const int n_cols = static_cast<int>(end - begin);
1278 
1279  int ierr;
1280  if (row_is_stored_locally(row))
1281  ierr =
1282  graph->InsertGlobalIndices(trilinos_row_index, n_cols, col_index_ptr);
1283  else if (nonlocal_graph.get() != nullptr)
1284  {
1285  // this is the case when we have explicitly set the off-processor rows
1286  // and want to create a separate matrix object for them (to retain
1287  // thread-safety)
1288  Assert(nonlocal_graph->RowMap().LID(
1289  static_cast<TrilinosWrappers::types::int_type>(row)) != -1,
1290  ExcMessage("Attempted to write into off-processor matrix row "
1291  "that has not be specified as being writable upon "
1292  "initialization"));
1293  ierr = nonlocal_graph->InsertGlobalIndices(trilinos_row_index,
1294  n_cols,
1295  col_index_ptr);
1296  }
1297  else
1298  ierr = graph->InsertGlobalIndices(1,
1299  &trilinos_row_index,
1300  n_cols,
1301  col_index_ptr);
1302 
1303  AssertThrow(ierr >= 0, ExcTrilinosError(ierr));
1304  }
