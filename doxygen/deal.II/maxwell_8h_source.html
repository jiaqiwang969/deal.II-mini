146  template <int dim>
147  Tensor<1, dim>
148  curl_curl(const Tensor<2, dim> &h0,
149  const Tensor<2, dim> &h1,
150  const Tensor<2, dim> &h2)
151  {
152  Tensor<1, dim> result;
153  switch (dim)
154  {
155  case 2:
156  result[0] = h1[0][1] - h0[1][1];
157  result[1] = h0[0][1] - h1[0][0];
158  break;
159  case 3:
160  result[0] = h1[0][1] + h2[0][2] - h0[1][1] - h0[2][2];
161  result[1] = h2[1][2] + h0[1][0] - h1[2][2] - h1[0][0];
162  result[2] = h0[2][0] + h1[2][1] - h2[0][0] - h2[1][1];
163  break;
164  default:
165  Assert(false, ExcNotImplemented());
166  }
167  return result;
168  }

178  template <int dim>
179  Tensor<1, dim>
180  tangential_curl(const Tensor<1, dim> &g0,
181  const Tensor<1, dim> &g1,
182  const Tensor<1, dim> &g2,
183  const Tensor<1, dim> &normal)
184  {
185  Tensor<1, dim> result;
186 
187  switch (dim)
188  {
189  case 2:
190  result[0] = normal[1] * (g1[0] - g0[1]);
191  result[1] = -normal[0] * (g1[0] - g0[1]);
192  break;
193  case 3:
194  result[0] =
195  normal[2] * (g2[1] - g0[2]) + normal[1] * (g1[0] - g0[1]);
196  result[1] =
197  normal[0] * (g0[2] - g1[0]) + normal[2] * (g2[1] - g1[2]);
198  result[2] =
199  normal[1] * (g1[0] - g2[1]) + normal[0] * (g0[2] - g2[0]);
200  break;
201  default:
202  Assert(false, ExcNotImplemented());
203  }
204  return result;

214  template <int dim>
215  void
216  curl_curl_matrix(FullMatrix<double> & M,
217  const FEValuesBase<dim> &fe,
218  const double factor = 1.)
219  {
220  const unsigned int n_dofs = fe.dofs_per_cell;
221 
222  AssertDimension(fe.get_fe().n_components(), dim);
223  AssertDimension(M.m(), n_dofs);
224  AssertDimension(M.n(), n_dofs);
225 
226  // Depending on the dimension,
227  // the cross product is either
228  // a scalar (2d) or a vector
229  // (3d). Accordingly, in the
230  // latter case we have to sum
231  // up three bilinear forms, but
232  // in 2d, we don't. Thus, we
233  // need to adapt the loop over
234  // all dimensions
235  const unsigned int d_max = (dim == 2) ? 1 : dim;
236 
237  for (unsigned int k = 0; k < fe.n_quadrature_points; ++k)
238  {
239  const double dx = factor * fe.JxW(k);
240  for (unsigned int i = 0; i < n_dofs; ++i)
241  for (unsigned int j = 0; j < n_dofs; ++j)
242  for (unsigned int d = 0; d < d_max; ++d)
243  {
244  const unsigned int d1 = (d + 1) % dim;
245  const unsigned int d2 = (d + 2) % dim;
246 
247  const double cv = fe.shape_grad_component(i, k, d2)[d1] -
248  fe.shape_grad_component(i, k, d1)[d2];
249  const double cu = fe.shape_grad_component(j, k, d2)[d1] -
250  fe.shape_grad_component(j, k, d1)[d2];
251 
252  M(i, j) += dx * cu * cv;
253  }
254  }
255  }

264  void
265  curl_matrix(FullMatrix<double> & M,
266  const FEValuesBase<dim> &fe,
267  const FEValuesBase<dim> &fetest,
268  double factor = 1.)
269  {
270  const unsigned int n_dofs = fe.dofs_per_cell;
271  const unsigned int t_dofs = fetest.dofs_per_cell;
272  AssertDimension(fe.get_fe().n_components(), dim);
273  // There should be the right number of components (3 in 3D, otherwise 1)
274  // for the curl.
275  AssertDimension(fetest.get_fe().n_components(), (dim == 3) ? dim : 1);
276  AssertDimension(M.m(), t_dofs);
277  AssertDimension(M.n(), n_dofs);
278 
279  const unsigned int d_max = (dim == 2) ? 1 : dim;
280 
281  for (unsigned int k = 0; k < fe.n_quadrature_points; ++k)
282  {
283  const double dx = fe.JxW(k) * factor;
284  for (unsigned int i = 0; i < t_dofs; ++i)
285  for (unsigned int j = 0; j < n_dofs; ++j)
286  for (unsigned int d = 0; d < d_max; ++d)
287  {
288  const unsigned int d1 = (d + 1) % dim;
289  const unsigned int d2 = (d + 2) % dim;
290 
291  const double vv = fetest.shape_value_component(i, k, d);
292  const double cu = fe.shape_grad_component(j, k, d2)[d1] -
293  fe.shape_grad_component(j, k, d1)[d2];
294  M(i, j) += dx * cu * vv;
295  }
296  }

344  for (unsigned int i = 0; i < n_dofs; ++i)
345  for (unsigned int j = 0; j < n_dofs; ++j)
346  if (fe.get_fe().has_support_on_face(i, face_no) &&
347  fe.get_fe().has_support_on_face(j, face_no))
348  {
349  for (unsigned int d = 0; d < d_max; ++d)
350  {
351  const unsigned int d1 = (d + 1) % dim;
352  const unsigned int d2 = (d + 2) % dim;
353 
354  const double cv = fe.shape_grad_component(i, k, d2)[d1] -
355  fe.shape_grad_component(i, k, d1)[d2];
356  const double cu = fe.shape_grad_component(j, k, d2)[d1] -
357  fe.shape_grad_component(j, k, d1)[d2];
358  const double v =
359  fe.shape_value_component(i, k, d1) * n[d2] -
360  fe.shape_value_component(i, k, d2) * n[d1];
361  const double u =
362  fe.shape_value_component(j, k, d1) * n[d2] -
363  fe.shape_value_component(j, k, d2) * n[d1];
364 
365  M(i, j) += dx * (2. * penalty * u * v - cv * u - cu * v);
366  }
367  }

437  ip_curl_matrix(FullMatrix<double> & M11,
438  FullMatrix<double> & M12,
439  FullMatrix<double> & M21,
440  FullMatrix<double> & M22,
441  const FEValuesBase<dim> &fe1,
442  const FEValuesBase<dim> &fe2,
443  const double pen,
444  const double factor1 = 1.,
445  const double factor2 = -1.)
446  {
447  const unsigned int n_dofs = fe1.n_dofs_per_cell();
448 
449  AssertDimension(fe1.get_fe().n_components(), dim);
450  AssertDimension(fe2.get_fe().n_components(), dim);
451  AssertDimension(M11.m(), n_dofs);
452  AssertDimension(M11.n(), n_dofs);
453  AssertDimension(M12.m(), n_dofs);
454  AssertDimension(M12.n(), n_dofs);
455  AssertDimension(M21.m(), n_dofs);
456  AssertDimension(M21.n(), n_dofs);
457  AssertDimension(M22.m(), n_dofs);
458  AssertDimension(M22.n(), n_dofs);
459 
460  const double nu1 = factor1;
461  const double nu2 = (factor2 < 0) ? factor1 : factor2;
462  const double penalty = .5 * pen * (nu1 + nu2);
463 
464  // Depending on the
465  // dimension, the cross
466  // product is either a scalar
467  // (2d) or a vector
468  // (3d). Accordingly, in the
469  // latter case we have to sum
470  // up three bilinear forms,
471  // but in 2d, we don't. Thus,
472  // we need to adapt the loop
473  // over all dimensions
474  const unsigned int d_max = (dim == 2) ? 1 : dim;
475 
476  for (unsigned int k = 0; k < fe1.n_quadrature_points; ++k)
477  {
478  const double dx = fe1.JxW(k);
479  const Tensor<1, dim> n = fe1.normal_vector(k);
480  for (unsigned int i = 0; i < n_dofs; ++i)
481  for (unsigned int j = 0; j < n_dofs; ++j)
482  for (unsigned int d = 0; d < d_max; ++d)
483  {
484  const unsigned int d1 = (d + 1) % dim;
485  const unsigned int d2 = (d + 2) % dim;
486  // curl u, curl v
487  const double cv1 =
488  nu1 * fe1.shape_grad_component(i, k, d2)[d1] -
489  fe1.shape_grad_component(i, k, d1)[d2];
490  const double cv2 =
491  nu2 * fe2.shape_grad_component(i, k, d2)[d1] -
492  fe2.shape_grad_component(i, k, d1)[d2];
493  const double cu1 =
494  nu1 * fe1.shape_grad_component(j, k, d2)[d1] -
495  fe1.shape_grad_component(j, k, d1)[d2];
496  const double cu2 =
497  nu2 * fe2.shape_grad_component(j, k, d2)[d1] -
498  fe2.shape_grad_component(j, k, d1)[d2];
499 
500  // u x n, v x n
501  const double u1 =
502  fe1.shape_value_component(j, k, d1) * n(d2) -
503  fe1.shape_value_component(j, k, d2) * n(d1);
504  const double u2 =
505  -fe2.shape_value_component(j, k, d1) * n(d2) +
506  fe2.shape_value_component(j, k, d2) * n(d1);
507  const double v1 =
508  fe1.shape_value_component(i, k, d1) * n(d2) -
509  fe1.shape_value_component(i, k, d2) * n(d1);
510  const double v2 =
511  -fe2.shape_value_component(i, k, d1) * n(d2) +
512  fe2.shape_value_component(i, k, d2) * n(d1);
513 
514  M11(i, j) +=
515  .5 * dx * (2. * penalty * u1 * v1 - cv1 * u1 - cu1 * v1);
516  M12(i, j) +=
517  .5 * dx * (2. * penalty * v1 * u2 - cv1 * u2 - cu2 * v1);
518  M21(i, j) +=
519  .5 * dx * (2. * penalty * u1 * v2 - cv2 * u1 - cu1 * v2);
520  M22(i, j) +=
521  .5 * dx * (2. * penalty * u2 * v2 - cv2 * u2 - cu2 * v2);
522  }
523  }
