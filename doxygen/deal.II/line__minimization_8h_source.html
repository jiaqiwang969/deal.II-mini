350  template <typename NumberType>
351  std_cxx17::optional<NumberType>
352  quadratic_fit(const NumberType x1,
353  const NumberType f1,
354  const NumberType g1,
355  const NumberType x2,
356  const NumberType f2)
357  {
358  Assert(x1 != x2, ExcMessage("Point are the same"));
359  const NumberType denom = (2. * g1 * x2 - 2. * g1 * x1 - 2. * f2 + 2. * f1);
360  if (denom == 0)
361  return {};
362  else
363  return (g1 * (x2 * x2 - x1 * x1) + 2. * (f1 - f2) * x1) / denom;
364  }

368  template <typename NumberType>
369  std_cxx17::optional<NumberType>
370  cubic_fit(const NumberType x1,
371  const NumberType f1,
372  const NumberType g1,
373  const NumberType x2,
374  const NumberType f2,
375  const NumberType g2)
376  {
377  Assert(x1 != x2, ExcMessage("Points are the same"));
378  const NumberType beta1 = g1 + g2 - 3. * (f1 - f2) / (x1 - x2);
379  const NumberType s = beta1 * beta1 - g1 * g2;
380  if (s < 0)
381  return {};
382 
383  const NumberType beta2 = std::sqrt(s);
384  const NumberType denom =
385  x1 < x2 ? g2 - g1 + 2. * beta2 : g1 - g2 + 2. * beta2;
386  if (denom == 0.)
387  return {};
388 
389  return x1 < x2 ? x2 - (x2 - x1) * (g2 + beta2 - beta1) / denom :
390  x1 - (x1 - x2) * (g1 + beta2 - beta1) / denom;
391  }

395  template <typename NumberType>
396  std_cxx17::optional<NumberType>
397  cubic_fit_three_points(const NumberType x1,
398  const NumberType f1,
399  const NumberType g1,
400  const NumberType x2,
401  const NumberType f2,
402  const NumberType x3,
403  const NumberType f3)
404  {
405  Assert(x1 != x2, ExcMessage("Points are the same"));
406  Assert(x1 != x3, ExcMessage("Points are the same"));
407  // f(x) = A *(x-x1)^3 + B*(x-x1)^2 + C*(x-x1) + D
408  // =>
409  // D = f1
410  // C = g1
411 
412  // the rest is a system of 2 equations:
413 
414  const NumberType x2_shift = x2 - x1;
415  const NumberType x3_shift = x3 - x1;
416  const NumberType r1 = f2 - f1 - g1 * x2_shift;
417  const NumberType r2 = f3 - f1 - g1 * x3_shift;
418  const NumberType denom =
419  std::pow(x2_shift * x3_shift, 2) * (x2_shift - x3_shift);
420  if (denom == 0.)
421  return {};
422 
423  const NumberType A =
424  (r1 * std::pow(x3_shift, 2) - r2 * std::pow(x2_shift, 2)) / denom;
425  const NumberType B =
426  (r2 * std::pow(x2_shift, 3) - r1 * std::pow(x3_shift, 3)) / denom;
427  const NumberType &C = g1;
428 
429  // now get the minimizer:
430  const NumberType radical = B * B - A * C * 3;
431  if (radical < 0)
432  return {};
433 
434  return x1 + (-B + std::sqrt(radical)) / (A * 3);
435  }

439  template <typename NumberType>
440  NumberType
441  poly_fit(const NumberType x1,
442  const NumberType f1,
443  const NumberType g1,
444  const NumberType x2,
445  const NumberType f2,
446  const NumberType g2,
447  const FiniteSizeHistory<NumberType> &,
448  const FiniteSizeHistory<NumberType> &,
449  const FiniteSizeHistory<NumberType> &,
450  const std::pair<NumberType, NumberType> bounds)
451  {
452  Assert(bounds.first < bounds.second, ExcMessage("Incorrect bounds"));
453 
454  // Similar to scipy implementation but we fit based on two points
455  // with their gradients and do bisection on bounds.
456  // https://github.com/scipy/scipy/blob/v1.0.0/scipy/optimize/linesearch.py#L555-L563
457 
458  // First try cubic interpolation
459  std_cxx17::optional<NumberType> res = cubic_fit(x1, f1, g1, x2, f2, g2);
460  if (res && *res >= bounds.first && *res <= bounds.second)
461  return *res;
462 
463  // cubic either fails or outside of safe region, do quadratic:
464  res = quadratic_fit(x1, f1, g1, x2, f2);
465  if (res && *res >= bounds.first && *res <= bounds.second)
466  return *res;
467 
468  // quadratic either failed or outside of safe region. Do bisection
469  // on safe region
470  return (bounds.first + bounds.second) * 0.5;
471  }

514  template <typename NumberType>
515  std::pair<NumberType, unsigned int>
516  line_search(
517  const std::function<std::pair<NumberType, NumberType>(const NumberType x)>
518  & func,
519  const NumberType f0,
520  const NumberType g0,
521  const std::function<
522  NumberType(const NumberType x_low,
523  const NumberType f_low,
524  const NumberType g_low,
525  const NumberType x_hi,
526  const NumberType f_hi,
527  const NumberType g_hi,
528  const FiniteSizeHistory<NumberType> & x_rec,
529  const FiniteSizeHistory<NumberType> & f_rec,
530  const FiniteSizeHistory<NumberType> & g_rec,
531  const std::pair<NumberType, NumberType> bounds)> &choose,
532  const NumberType a1,
533  const NumberType eta,
534  const NumberType mu,
535  const NumberType a_max,
536  const unsigned int max_evaluations,
537  const bool debug_output)
538  {
539  // Note that scipy use dcsrch() from Minpack2 Fortran lib for line search
540  Assert(mu < 0.5 && mu > 0, ExcMessage("mu is not in (0,1/2)."));
541  Assert(eta < 1. && eta > mu, ExcMessage("eta is not in (mu,1)."));
542  Assert(a_max > 0, ExcMessage("max is not positive."));
543  Assert(a1 > 0 && a1 <= a_max, ExcMessage("a1 is not in (0,max]."));
544  Assert(g0 < 0, ExcMessage("Initial slope is not negative"));
545 
546  // Growth parameter for bracketing phase:
547  // 1 < tau1
548  const NumberType tau1 = 9.;
549  // shrink parameters for sectioning phase to prevent ai from being
550  // arbitrary close to the extremes of the interval.
551  // 0 < tau2 < tau3 <= 1/2
552  // tau2 <= eta is advisable
553  const NumberType tau2 = 0.1; // bound for closeness to a_lo
554  const NumberType tau3 = 0.5; // bound for closeness to a_hi
555 
556  const NumberType g0_abs = std::abs(g0);
557  const NumberType f_min = f0 + a_max * mu * g0;
558 
559  // return True if the first Wolfe condition (sufficient decrease) is
560  // satisfied
561  const auto w1 = [&](const NumberType a, const NumberType f) {
562  return f <= f0 + a * mu * g0;
563  };
564 
565  // return True if the second Wolfe condition (curvature condition) is
566  // satisfied
567  const auto w2 = [&](const NumberType g) {
568  return std::abs(g) <= eta * g0_abs;
569  };
570 
571  // Bracketing phase (Algorithm 2.6.2): look for a non-trivial interval
572  // which is known to contain an interval of acceptable points.
573  // We adopt notation of Noceal.
574  const NumberType x = std::numeric_limits<NumberType>::signaling_NaN();
575  NumberType a_lo = x, f_lo = x, g_lo = x;
576  NumberType a_hi = x, f_hi = x, g_hi = x;
577  NumberType ai = x, fi = x, gi = x;
578 
579  // count function calls in i:
580  unsigned int i = 0;
581  {
582  NumberType f_prev, g_prev, a_prev;
583  ai = a1;
584  f_prev = f0;
585  g_prev = g0;
586  a_prev = 0;
587 
588  while (i < max_evaluations)
589  {
590  const auto fgi = func(ai);
591  fi = fgi.first;
592  gi = fgi.second;
593  i++;
594 
595  if (debug_output)
596  deallog << "Bracketing phase: " << i << std::endl
597  << ai << " " << fi << " " << gi << " " << w1(ai, fi) << " "
598  << w2(gi) << " " << f_min << std::endl;
599 
600  // first check if we can stop bracketing or the whole line search:
601  if (fi <= f_min || ai == a_max)
602  {
603  if (debug_output)
604  deallog << "Reached the maximum step size." << std::endl;
605  return std::make_pair(ai, i);
606  }
607 
608  if (!w1(ai, fi) ||
609  (fi >= f_prev && i > 1)) // violate first Wolfe or not descending
610  {
611  a_lo = a_prev;
612  f_lo = f_prev;
613  g_lo = g_prev;
614 
615  a_hi = ai;
616  f_hi = fi;
617  g_hi = gi;
618  break; // end bracketing
619  }
620 
621  if (w2(gi)) // satisfies both Wolfe conditions
622  {
623  if (debug_output)
624  deallog << "Satisfied both Wolfe conditions during Bracketing."
625  << std::endl;
626 
627  Assert(w1(ai, fi), ExcInternalError());
628  return std::make_pair(ai, i);
629  }
630 
631  if (gi >= 0) // not descending
632  {
633  a_lo = ai;
634  f_lo = fi;
635  g_lo = gi;
636 
637  a_hi = a_prev;
638  f_hi = f_prev;
639  g_hi = g_prev;
640  break; // end bracketing
641  }
642 
643  // extrapolation step with the bounds
644  const auto bounds =
645  std::make_pair(2. * ai - a_prev,
646  std::min(a_max, ai + tau1 * (ai - a_prev)));
647 
648  a_prev = ai;
649  f_prev = fi;
650  g_prev = gi;
651 
652  // NOTE: Fletcher's 2.6.2 includes optional extrapolation, we
653  // simply take the upper bound
654  // Scipy increases by factor of two:
655  // https://github.com/scipy/scipy/blob/v1.0.0/scipy/optimize/linesearch.py#L447
656  ai = bounds.second;
657  }
658  }
659 
660  AssertThrow(
661  i < max_evaluations,
662  ExcMessage(
663  "Could not find the initial bracket within the given number of iterations."));
664 
665  // Check properties of the bracket (Theorem 3.2 in More and Thuente, 94
666  // and Eq. 2.6.3 in Fletcher 2013
667 
668  // FIXME: these conditions are actually violated for Fig3 and a1=10^3 in
669  // More and Thorenton, 94.
670 
671  /* Assert((f_lo < f_hi) && w1(a_lo, f_lo), ExcInternalError()) ; Assert((a_hi
672 *
673 * - a_lo) g_lo < 0) && !w2(g_lo), ExcInternalError()); Assert((w1(a_hi, f_hi) || f_hi >= f_lo), ExcInternalError()。
674 *
675 */
676 
677  // keep short history of last points to improve interpolation
678  FiniteSizeHistory<NumberType> a_rec(5), f_rec(5), g_rec(5);
679  // if neither a_lo nor a_hi are zero:
680  if (std::abs(a_lo) > std::numeric_limits<NumberType>::epsilon() &&
681  std::abs(a_hi) > std::numeric_limits<NumberType>::epsilon())
682  {
683  a_rec.add(0);
684  f_rec.add(f0);
685  g_rec.add(g0);
686  }
687 
688  // Now sectioning phase: we allow both [a_lo, a_hi] and [a_hi, a_lo]
689  while (i < max_evaluations)
690  {
691  const NumberType a_lo_safe = a_lo + tau2 * (a_hi - a_lo);
692  const NumberType a_hi_safe = a_hi - tau3 * (a_hi - a_lo);
693  const auto bounds = std::minmax(a_lo_safe, a_hi_safe);
694 
695  ai = choose(
696  a_lo, f_lo, g_lo, a_hi, f_hi, g_hi, a_rec, f_rec, g_rec, bounds);
697 
698  const std::pair<NumberType, NumberType> fgi = func(ai);
699  fi = fgi.first;
700  gi = fgi.second;
701  i++;
702 
703  if (debug_output)
704  deallog << "Sectioning phase: " << i << std::endl
705  << a_lo << " " << f_lo << " " << g_lo << " " << w1(a_lo, f_lo)
706  << " " << w2(g_lo) << std::endl
707  << a_hi << " " << f_hi << " " << g_hi << " " << w1(a_hi, f_hi)
708  << " " << w2(g_hi) << std::endl
709  << ai << " " << fi << " " << gi << " " << w1(ai, fi) << " "
710  << w2(gi) << std::endl;
711 
712  if (!w1(ai, fi) || fi >= f_lo)
713  // take [a_lo, ai]
714  {
715  a_rec.add(a_hi);
716  f_rec.add(f_hi);
717  g_rec.add(g_hi);
718 
719  a_hi = ai;
720  f_hi = fi;
721  g_hi = gi;
722  }
723  else
724  {
725  if (w2(gi)) // satisfies both wolf
726  {
727  if (debug_output)
728  deallog << "Satisfied both Wolfe conditions." << std::endl;
729  Assert(w1(ai, fi), ExcInternalError());
730  return std::make_pair(ai, i);
731  }
732 
733  if (gi * (a_hi - a_lo) >= 0)
734  // take [ai, a_lo]
735  {
736  a_rec.add(a_hi);
737  f_rec.add(f_hi);
738  g_rec.add(g_hi);
739 
740  a_hi = a_lo;
741  f_hi = f_lo;
742  g_hi = g_lo;
743  }
744  else
745  // take [ai, a_hi]
746  {
747  a_rec.add(a_lo);
748  f_rec.add(f_lo);
749  g_rec.add(g_lo);
750  }
751 
752  a_lo = ai;
753  f_lo = fi;
754  g_lo = gi;
755  }
756  }
757 
758  // if we got here, we could not find the solution
759  AssertThrow(
760  false,
761  ExcMessage(
762  "Could not could complete the sectioning phase within the given number of iterations."));
763  return std::make_pair(std::numeric_limits<NumberType>::signaling_NaN(), i);
764  }
