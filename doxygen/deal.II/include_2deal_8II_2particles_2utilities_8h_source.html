182  template <int dim,
183  int spacedim,
184  typename InputVectorType,

186  void
187  interpolate_field_on_particles(
188  const DoFHandler<dim, spacedim> & field_dh,
189  const Particles::ParticleHandler<dim, spacedim> &particle_handler,
190  const InputVectorType & field_vector,
191  OutputVectorType & interpolated_field,
192  const ComponentMask &field_comps = ComponentMask())
193  {
194  if (particle_handler.n_locally_owned_particles() == 0)
195  {
196  interpolated_field.compress(VectorOperation::add);
197  return; // nothing else to do here
198  }
199 
200  const auto &fe = field_dh.get_fe();
201  auto particle = particle_handler.begin();
202 
203  // Take care of components
204  const ComponentMask comps =
205  (field_comps.size() == 0 ? ComponentMask(fe.n_components(), true) :
206  field_comps);
207  AssertDimension(comps.size(), fe.n_components());
208  const auto n_comps = comps.n_selected_components();
209 
210  AssertDimension(field_vector.size(), field_dh.n_dofs());
211  AssertDimension(interpolated_field.size(),
212  particle_handler.get_next_free_particle_index() *
213  n_comps);
214 
215  // Global to local indices
216  std::vector<unsigned int> space_gtl(fe.n_components(),
217  numbers::invalid_unsigned_int);
218  for (unsigned int i = 0, j = 0; i < space_gtl.size(); ++i)
219  if (comps[i])
220  space_gtl[i] = j++;
221 
222  std::vector<types::global_dof_index> dof_indices(fe.n_dofs_per_cell());
223 
224  while (particle != particle_handler.end())
225  {
226  const auto &cell = particle->get_surrounding_cell();
227  const auto &dh_cell =
228  typename DoFHandler<dim, spacedim>::cell_iterator(*cell, &field_dh);
229  dh_cell->get_dof_indices(dof_indices);
230  const auto pic = particle_handler.particles_in_cell(cell);
231 
232  Assert(pic.begin() == particle, ExcInternalError());
233  for (unsigned int i = 0; particle != pic.end(); ++particle, ++i)
234  {
235  const Point<dim> reference_location =
236  particle->get_reference_location();
237 
238  const auto id = particle->get_id();
239 
240  for (unsigned int j = 0; j < fe.n_dofs_per_cell(); ++j)
241  {
242  const auto comp_j =
243  space_gtl[fe.system_to_component_index(j).first];
244  if (comp_j != numbers::invalid_unsigned_int)
245  interpolated_field[id * n_comps + comp_j] +=
246  fe.shape_value(j, reference_location) *
247  field_vector(dof_indices[j]);
248  }
249  }
250  }
251  interpolated_field.compress(VectorOperation::add);
252  }
