135  template <int dim,
136  int spacedim,
137  typename InputVectorType,
138  typename OutputVectorType>
139  void
140  interpolate_field_on_particles(
141  const DoFHandler<dim, spacedim> & field_dh,
142  const Particles::ParticleHandler<dim, spacedim> &particle_handler,
143  const InputVectorType & field_vector,
144  OutputVectorType & interpolated_field,
145  const ComponentMask &field_comps = ComponentMask())
146  {
147  if (particle_handler.n_locally_owned_particles() == 0)
148  {
149  interpolated_field.compress(VectorOperation::add);
150  return; // nothing else to do here
151  }
152 
153  const auto &tria = field_dh.get_triangulation();
154  const auto &fe = field_dh.get_fe();
155  auto particle = particle_handler.begin();
156 
157  // Take care of components
158  const ComponentMask comps =
159  (field_comps.size() == 0 ? ComponentMask(fe.n_components(), true) :
160  field_comps);
161  AssertDimension(comps.size(), fe.n_components());
162  const auto n_comps = comps.n_selected_components();
163 
164  AssertDimension(field_vector.size(), field_dh.n_dofs());
165  AssertDimension(interpolated_field.size(),
166  particle_handler.get_next_free_particle_index() *
167  n_comps);
168 
169  // Global to local indices
170  std::vector<unsigned int> space_gtl(fe.n_components(),
171  numbers::invalid_unsigned_int);
172  for (unsigned int i = 0, j = 0; i < space_gtl.size(); ++i)
173  if (comps[i])
174  space_gtl[i] = j++;
175 
176  std::vector<types::global_dof_index> dof_indices(fe.n_dofs_per_cell());
177 
178  while (particle != particle_handler.end())
179  {
180  const auto &cell = particle->get_surrounding_cell(tria);
181  const auto &dh_cell =
182  typename DoFHandler<dim, spacedim>::cell_iterator(*cell, &field_dh);
183  dh_cell->get_dof_indices(dof_indices);
184  const auto pic = particle_handler.particles_in_cell(cell);
185 
186  Assert(pic.begin() == particle, ExcInternalError());
187  for (unsigned int i = 0; particle != pic.end(); ++particle, ++i)
188  {
189  const Point<dim> reference_location =
190  particle->get_reference_location();
191 
192  const auto id = particle->get_id();
193 
194  for (unsigned int j = 0; j < fe.n_dofs_per_cell(); ++j)
195  {
196  const auto comp_j =
197  space_gtl[fe.system_to_component_index(j).first];
198  if (comp_j != numbers::invalid_unsigned_int)
199  interpolated_field[id * n_comps + comp_j] +=
200  fe.shape_value(j, reference_location) *
201  field_vector(dof_indices[j]);
202  }
203  }
204  }
205  interpolated_field.compress(VectorOperation::add);
206  }
