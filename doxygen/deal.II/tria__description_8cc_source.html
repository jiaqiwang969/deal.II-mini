100  template <int dim, int spacedim>
101  Description<dim, spacedim>
102  create_description_from_triangulation(
103  const ::Triangulation<dim, spacedim> &tria,
104  const MPI_Comm & comm,
105  const TriangulationDescription::Settings settings,
106  const unsigned int my_rank_in)
107  {
108  if (auto tria_pdt = dynamic_cast<
109  const parallel::distributed::Triangulation<dim, spacedim> *>(&tria))
110  Assert(comm == tria_pdt->get_communicator(),
111  ExcMessage("MPI communicators do not match."));
112 
113  // First, figure out for what rank we are supposed to build the
114  // TriangulationDescription::Description object
115  unsigned int my_rank = my_rank_in;
116  Assert(my_rank == numbers::invalid_unsigned_int ||
117  my_rank < ::Utilities::MPI::n_mpi_processes(comm),
118  ExcMessage("Rank has to be smaller than available processes."));
119 
120  if (auto tria_pdt = dynamic_cast<
121  const parallel::distributed::Triangulation<dim, spacedim> *>(&tria))
122  {
123  Assert(my_rank == numbers::invalid_unsigned_int ||
124  my_rank == ::Utilities::MPI::this_mpi_process(comm),
125  ExcMessage(
126  "For creation from a parallel::distributed::Triangulation, "
127  "my_rank has to equal global rank."));
128 
129  my_rank = ::Utilities::MPI::this_mpi_process(comm);
130  }
131  else if (auto tria_serial =
132  dynamic_cast<const ::Triangulation<dim, spacedim> *>(
133  &tria))
134  {
135  if (my_rank == numbers::invalid_unsigned_int)
136  my_rank = ::Utilities::MPI::this_mpi_process(comm);
137  }
138  else
139  {
140  Assert(false,
141  ExcMessage("This type of triangulation is not supported!"));
142  }
143 
144  Description<dim, spacedim> construction_data;
145 
146  // store the communicator
147  construction_data.comm = comm;
148 
149  std::map<unsigned int, std::vector<unsigned int>>
150  coinciding_vertex_groups;
151  std::map<unsigned int, unsigned int> vertex_to_coinciding_vertex_group;
152 
153  GridTools::collect_coinciding_vertices(tria,
154  coinciding_vertex_groups,
155  vertex_to_coinciding_vertex_group);
156 
157  // helper function, which collects all vertices belonging to a cell
158  // (also taking into account periodicity)
159  const auto add_vertices_of_cell_to_vertices_owned_by_locally_owned_cells =
160  [&coinciding_vertex_groups, &vertex_to_coinciding_vertex_group](
161  const auto & cell,
162  std::vector<bool> &vertices_owned_by_locally_owned_cells) {
163  // add local vertices
164  for (const auto v : cell->vertex_indices())
165  {
166  vertices_owned_by_locally_owned_cells[cell->vertex_index(v)] =
167  true;
168  const auto coinciding_vertex_group =
169  vertex_to_coinciding_vertex_group.find(cell->vertex_index(v));
170  if (coinciding_vertex_group !=
171  vertex_to_coinciding_vertex_group.end())
172  for (const auto &co_vertex : coinciding_vertex_groups.at(
173  coinciding_vertex_group->second))
174  vertices_owned_by_locally_owned_cells[co_vertex] = true;
175  }
176  };
177 
178  construction_data.smoothing = tria.get_mesh_smoothing();
179  construction_data.settings = settings;
180 
181  const bool construct_multigrid =
182  settings &
183  TriangulationDescription::Settings::construct_multigrid_hierarchy;
184 
185  Assert(
186  !(settings &
187  TriangulationDescription::Settings::construct_multigrid_hierarchy) ||
188  (tria.get_mesh_smoothing() &
189  Triangulation<dim, spacedim>::limit_level_difference_at_vertices),
190  ExcMessage(
191  "Source triangulation has to be set up with "
192  "limit_level_difference_at_vertices if the construction of the "
193  "multigrid hierarchy is requested!"));
194 
195  // 1) collect locally relevant cells (set user_flag)
196  std::vector<bool> old_user_flags;
197  tria.save_user_flags(old_user_flags);
198 
199  // 1a) clear user_flags
200  const_cast<::Triangulation<dim, spacedim> &>(tria)
201  .clear_user_flags();
202 
203  // 1b) loop over levels (from fine to coarse) and mark on each level
204  // the locally relevant cells
205  for (int level = tria.get_triangulation().n_global_levels() - 1;
206  level >= 0;
207  --level)
208  {
209  // collect vertices connected to a (on any level) locally owned
210  // cell
211  std::vector<bool> vertices_owned_by_locally_owned_cells_on_level(
212  tria.n_vertices());
213  for (const auto &cell : tria.cell_iterators_on_level(level))
214  if ((construct_multigrid &&
215  (cell->level_subdomain_id() == my_rank)) ||
216  (cell->is_active() && cell->subdomain_id() == my_rank))
217  add_vertices_of_cell_to_vertices_owned_by_locally_owned_cells(
218  cell, vertices_owned_by_locally_owned_cells_on_level);
219 
220  for (const auto &cell : tria.active_cell_iterators())
221  if (cell->subdomain_id() == my_rank)
222  add_vertices_of_cell_to_vertices_owned_by_locally_owned_cells(
223  cell, vertices_owned_by_locally_owned_cells_on_level);
224 
225  // helper function to determine if cell is locally relevant
226  // (i.e. a cell which is connected to a vertex via a locally
227  // owned cell)
228  const auto is_locally_relevant_on_level = [&](const auto &cell) {
229  for (const auto v : cell->vertex_indices())
230  if (vertices_owned_by_locally_owned_cells_on_level
231  [cell->vertex_index(v)])
232  return true;
233  return false;
234  };
235 
236  // mark all locally relevant cells
237  for (const auto &cell : tria.cell_iterators_on_level(level))
238  if (is_locally_relevant_on_level(cell))
239  set_user_flag_and_of_its_parents(cell);
240  }
241 
242  // 2) set_up coarse-grid triangulation
243  {
244  std::map<unsigned int, unsigned int> vertices_locally_relevant;
245 
246  // a) loop over all cells
247  for (const auto &cell : tria.cell_iterators_on_level(0))
248  {
249  if (!cell->user_flag_set())
250  continue;
251 
252  // extract cell definition (with old numbering of vertices)
253  ::CellData<dim> cell_data(cell->n_vertices());
254  cell_data.material_id = cell->material_id();
255  cell_data.manifold_id = cell->manifold_id();
256  for (const auto v : cell->vertex_indices())
257  cell_data.vertices[v] = cell->vertex_index(v);
258  construction_data.coarse_cells.push_back(cell_data);
259 
260  // save indices of each vertex of this cell
261  for (const auto v : cell->vertex_indices())
262  vertices_locally_relevant[cell->vertex_index(v)] =
263  numbers::invalid_unsigned_int;
264 
265  // save translation for corase grid: lid -> gid
266  construction_data.coarse_cell_index_to_coarse_cell_id.push_back(
267  cell->id().get_coarse_cell_id());
268  }
269 
270  // b) enumerate locally relevant vertices
271  unsigned int vertex_counter = 0;
272  for (auto &vertex : vertices_locally_relevant)
273  {
274  construction_data.coarse_cell_vertices.push_back(
275  tria.get_vertices()[vertex.first]);
276  vertex.second = vertex_counter++;
277  }
278 
279  // c) correct vertices of cells (make them local)
280  for (auto &cell : construction_data.coarse_cells)
281  for (unsigned int v = 0; v < cell.vertices.size(); ++v)
282  cell.vertices[v] = vertices_locally_relevant[cell.vertices[v]];
283  }
284 
285 
286  // 3) collect info of each cell
287  construction_data.cell_infos.resize(
288  tria.get_triangulation().n_global_levels());
289 
290  // collect local vertices on active level
291  std::vector<bool> vertices_owned_by_locally_owned_active_cells(
292  tria.n_vertices());
293  for (const auto &cell : tria.active_cell_iterators())
294  if (cell->subdomain_id() == my_rank)
295  add_vertices_of_cell_to_vertices_owned_by_locally_owned_cells(
296  cell, vertices_owned_by_locally_owned_active_cells);
297 
298  // helper function to determine if cell is locally relevant
299  // on active level
300  const auto is_locally_relevant_on_active_level = [&](const auto &cell) {
301  if (cell->is_active())
302  for (const auto v : cell->vertex_indices())
303  if (vertices_owned_by_locally_owned_active_cells[cell->vertex_index(
304  v)])
305  return true;
306  return false;
307  };
308 
309  for (unsigned int level = 0;
310  level < tria.get_triangulation().n_global_levels();
311  ++level)
312  {
313  // collect local vertices on level
314  std::vector<bool> vertices_owned_by_locally_owned_cells_on_level(
315  tria.n_vertices());
316  for (const auto &cell : tria.cell_iterators_on_level(level))
317  if ((construct_multigrid &&
318  (cell->level_subdomain_id() == my_rank)) ||
319  (cell->is_active() && cell->subdomain_id() == my_rank))
320  add_vertices_of_cell_to_vertices_owned_by_locally_owned_cells(
321  cell, vertices_owned_by_locally_owned_cells_on_level);
322 
323  // helper function to determine if cell is locally relevant
324  // on level
325  const auto is_locally_relevant_on_level = [&](const auto &cell) {
326  for (const auto v : cell->vertex_indices())
327  if (vertices_owned_by_locally_owned_cells_on_level
328  [cell->vertex_index(v)])
329  return true;
330  return false;
331  };
332 
333  auto &level_cell_infos = construction_data.cell_infos[level];
334  for (const auto &cell : tria.cell_iterators_on_level(level))
335  {
336  // check if cell is locally relevant
337  if (!(cell->user_flag_set()))
338  continue;
339 
340  CellData<dim> cell_info;
341 
342  // save coarse-cell id
343  cell_info.id = cell->id().template to_binary<dim>();
344 
345  // save boundary_ids of each face of this cell
346  for (const auto f : cell->face_indices())
347  {
348  types::boundary_id boundary_ind =
349  cell->face(f)->boundary_id();
350  if (boundary_ind != numbers::internal_face_boundary_id)
351  cell_info.boundary_ids.emplace_back(f, boundary_ind);
352  }
353 
354  // save manifold id
355  {
356  // ... of cell
357  cell_info.manifold_id = cell->manifold_id();
358 
359  // ... of lines
360  if (dim >= 2)
361  for (const auto line : cell->line_indices())
362  cell_info.manifold_line_ids[line] =
363  cell->line(line)->manifold_id();
364 
365  // ... of quads
366  if (dim == 3)
367  for (const auto f : cell->face_indices())
368  cell_info.manifold_quad_ids[f] =
369  cell->quad(f)->manifold_id();
370  }
371 
372  // subdomain and level subdomain id
373  cell_info.subdomain_id = numbers::artificial_subdomain_id;
374  cell_info.level_subdomain_id = numbers::artificial_subdomain_id;
375 
376  if (is_locally_relevant_on_active_level(cell))
377  {
378  cell_info.level_subdomain_id = cell->level_subdomain_id();
379  cell_info.subdomain_id = cell->subdomain_id();
380  }
381  else if (is_locally_relevant_on_level(cell))
382  {
383  cell_info.level_subdomain_id = cell->level_subdomain_id();
384  }
385  // else: cell is locally relevant but an artificial cell
386 
387  level_cell_infos.emplace_back(cell_info);
388  }
389  }
390 
391  const_cast<::Triangulation<dim, spacedim> &>(tria).load_user_flags(
392  old_user_flags);
393 
394  return construction_data;
395  }

399  template <int dim, int spacedim>
400  Description<dim, spacedim>
401  create_description_from_triangulation_in_groups(
402  const std::function<void(::Triangulation<dim, spacedim> &)>
403  & serial_grid_generator,
404  const std::function<void(::Triangulation<dim, spacedim> &,
405  const MPI_Comm &,
406  const unsigned int)> &serial_grid_partitioner,
407  const MPI_Comm & comm,
408  const int group_size,
409  const typename Triangulation<dim, spacedim>::MeshSmoothing smoothing,
410  const TriangulationDescription::Settings settings)
411  {
412 #ifndef DEAL_II_WITH_MPI
413  (void)serial_grid_generator;
414  (void)serial_grid_partitioner;
415  (void)comm;
416  (void)group_size;
417  (void)smoothing;
418  (void)settings;
419 
420  return Description<dim, spacedim>();
421 #else
422  const unsigned int my_rank =
423  ::Utilities::MPI::this_mpi_process(comm);
424  const unsigned int group_root = (my_rank / group_size) * group_size;
425 
426  const int mpi_tag =
427  ::Utilities::MPI::internal::Tags::fully_distributed_create;
428 
429  // check if process is root of the group
430  if (my_rank == group_root)
431  {
432  // Step 1: create serial triangulation
433  ::Triangulation<dim, spacedim> tria(
434  (settings & TriangulationDescription::Settings::
435  construct_multigrid_hierarchy) ?
436  static_cast<
437  typename ::Triangulation<dim, spacedim>::MeshSmoothing>(
438  smoothing |
439  Triangulation<dim,
440  spacedim>::limit_level_difference_at_vertices) :
441  smoothing);
442  serial_grid_generator(tria);
443 
444  // Step 2: partition active cells and ...
445  serial_grid_partitioner(tria, comm, group_size);
446 
447  // ... cells on the levels if multigrid is required
448  if (settings &
449  TriangulationDescription::Settings::construct_multigrid_hierarchy)
450  GridTools::partition_multigrid_levels(tria);
451 
452  const unsigned int end_group =
453  std::min(group_root + group_size,
454  ::Utilities::MPI::n_mpi_processes(comm));
455 
456  // 3) create Description for the other processes in group; since
457  // we expect that this function is called for huge meshes, one
458  // Description is created at a time and send away; only once the
459  // Description has been sent away, the next rank is processed.
460  for (unsigned int other_rank = group_root + 1; other_rank < end_group;
461  ++other_rank)
462  {
463  // 3a) create construction data for other ranks
464  const auto construction_data =
465  create_description_from_triangulation(tria,
466  comm,
467  settings,
468  other_rank);
469  // 3b) pack
470  std::vector<char> buffer;
471  ::Utilities::pack(construction_data, buffer, false);
472 
473  // 3c) send TriangulationDescription::Description
474  const auto ierr = MPI_Send(buffer.data(),
475  buffer.size(),
476  MPI_CHAR,
477  other_rank,
478  mpi_tag,
479  comm);
480  AssertThrowMPI(ierr);
481  }
482 
483  // 4) create TriangulationDescription::Description for this process
484  // (root of group)
485  return create_description_from_triangulation(tria,
486  comm,
487  settings,
488  my_rank);
489  }
490  else
491  {
492  // 3a) recv packed TriangulationDescription::Description from
493  // group-root process
494  // (counter-part of 3c of root process)
495  MPI_Status status;
496  auto ierr = MPI_Probe(group_root, mpi_tag, comm, &status);
497  AssertThrowMPI(ierr);
498 
499  int len;
500  MPI_Get_count(&status, MPI_CHAR, &len);
501 
502  std::vector<char> buf(len);
503  ierr = MPI_Recv(buf.data(),
504  len,
505  MPI_CHAR,
506  status.MPI_SOURCE,
507  mpi_tag,
508  comm,
509  &status);
510  AssertThrowMPI(ierr);
511 
512  // 3b) unpack TriangulationDescription::Description (counter-part of
513  // 3b of root process)
514  auto construction_data =
515  ::Utilities::template unpack<Description<dim, spacedim>>(
516  buf, false);
517 
518  // WARNING: serialization cannot handle the MPI communicator
519  // which is the reason why we have to set it here explicitly
520  construction_data.comm = comm;
521 
522  return construction_data;
523  }
524 #endif
525  }

563  void
564  reduce()
565  {
566  // make coarse cells unique
567  {
568  std::vector<std::tuple<types::coarse_cell_id,
569  ::CellData<dim>,
570  unsigned int>>
571  temp;
572 
573  for (unsigned int i = 0; i < this->coarse_cells.size(); ++i)
574  temp.emplace_back(this->coarse_cell_index_to_coarse_cell_id[i],
575  this->coarse_cells[i],
576  i);
577 
578  std::sort(temp.begin(), temp.end(), [](const auto &a, const auto &b) {
579  return std::get<0>(a) < std::get<0>(b);
580  });
581  temp.erase(std::unique(temp.begin(),
582  temp.end(),
583  [](const auto &a, const auto &b) {
584  return std::get<0>(a) == std::get<0>(b);
585  }),
586  temp.end());
587  std::sort(temp.begin(), temp.end(), [](const auto &a, const auto &b) {
588  return std::get<2>(a) < std::get<2>(b);
589  });
590 
591  this->coarse_cell_index_to_coarse_cell_id.resize(temp.size());
592  this->coarse_cells.resize(temp.size());
593 
594  for (unsigned int i = 0; i < temp.size(); ++i)
595  {
596  this->coarse_cell_index_to_coarse_cell_id[i] =
597  std::get<0>(temp[i]);
598  this->coarse_cells[i] = std::get<1>(temp[i]);
599  }
600  }
601 
602  // make coarse cell vertices unique
603  {
604  std::sort(this->coarse_cell_vertices.begin(),
605  this->coarse_cell_vertices.end(),
606  [](const auto &a, const auto &b) {
607  return a.first < b.first;
608  });
609  this->coarse_cell_vertices.erase(
610  std::unique(this->coarse_cell_vertices.begin(),
611  this->coarse_cell_vertices.end(),
612  [](const auto &a, const auto &b) {
613  return a.first == b.first;
614  }),
615  this->coarse_cell_vertices.end());
616  }
617 
618  // make cells unique
619  for (unsigned int i = 0; i < this->cell_infos.size(); ++i)
620  {
621  std::sort(this->cell_infos[i].begin(),
622  this->cell_infos[i].end(),
623  [](const auto &a, const auto &b) { return a.id < b.id; });
624  this->cell_infos[i].erase(std::unique(this->cell_infos[i].begin(),
625  this->cell_infos[i].end(),
626  [](const auto &a,
627  const auto &b) {
628  return a.id == b.id;
629  }),
630  this->cell_infos[i].end());
631  }
632  }

634  Description<dim, spacedim>
635  convert(const MPI_Comm comm,
636  const typename Triangulation<dim, spacedim>::MeshSmoothing
637  mesh_smoothing)
638  {
639  Description<dim, spacedim> description;
640 
641  // copy communicator
642  description.comm = comm;
643 
644  // set settings (no multigrid levels are set for now)
645  description.settings =
646  TriangulationDescription::Settings::default_setting;
647 
648  // use mesh smoothing from base triangulation
649  description.smoothing = mesh_smoothing;
650 
651  std::map<unsigned int, unsigned int> map;
652 
653  for (unsigned int i = 0; i < this->coarse_cell_vertices.size(); ++i)
654  {

668  description.cell_infos = this->cell_infos;

685  template <int dim, int spacedim>
686  void
687  fill_cell_infos(const TriaIterator<CellAccessor<dim, spacedim>> &cell,
688  std::vector<std::vector<CellData<dim>>> & cell_infos,
689  const LinearAlgebra::distributed::Vector<double> &partition)
690  {
691  if (cell->user_flag_set())
692  return; // cell has been already added -> nothing to do
693 
694  cell->set_user_flag();
695 
696  CellData<dim> cell_info;
697 
698  // save coarse-cell id
699  cell_info.id = cell->id().template to_binary<dim>();
700 
701  // save boundary_ids of each face of this cell
702  for (const auto f : cell->face_indices())
703  {
704  types::boundary_id boundary_ind = cell->face(f)->boundary_id();
705  if (boundary_ind != numbers::internal_face_boundary_id)
706  cell_info.boundary_ids.emplace_back(f, boundary_ind);
707  }
708 
709  // save manifold id
710  {
711  // ... of cell
712  cell_info.manifold_id = cell->manifold_id();
713 
714  // ... of lines
715  if (dim >= 2)
716  for (const auto line : cell->line_indices())
717  cell_info.manifold_line_ids[line] = cell->line(line)->manifold_id();
718 
719  // ... of quads
720  if (dim == 3)
721  for (const auto f : cell->face_indices())
722  cell_info.manifold_quad_ids[f] = cell->quad(f)->manifold_id();
723  }
724 
725  // subdomain and level subdomain id
726  if (cell->is_active())
727  cell_info.subdomain_id = static_cast<unsigned int>(
728  partition[cell->global_active_cell_index()]);
729  else
730  cell_info.subdomain_id = numbers::artificial_subdomain_id;
731 
732  cell_info.level_subdomain_id = numbers::artificial_subdomain_id;
733 
734  cell_infos[cell->level()].emplace_back(cell_info);
735 
736  if (cell->level() != 0) // proceed with parent
737  fill_cell_infos(cell->parent(), cell_infos, partition);
738  }

742  template <int dim, int spacedim>
743  Description<dim, spacedim>
744  create_description_from_triangulation(
745  const Triangulation<dim, spacedim> & tria,
746  const LinearAlgebra::distributed::Vector<double> &partition)
747  {
748 #ifdef DEAL_II_WITH_MPI
749  if (tria.get_communicator() == MPI_COMM_NULL)
750  AssertDimension(partition.local_size(), 0);
751 #endif
752 
753  if (partition.size() == 0)
754  return create_description_from_triangulation(tria,
755  tria.get_communicator());
756 
757  partition.update_ghost_values();
758 
759  // 1) determine processes owning locally owned cells
760  const std::vector<unsigned int> relevant_processes = [&]() {
761  std::set<unsigned int> relevant_processes;
762 
763  for (unsigned int i = 0; i < partition.local_size(); ++i)
764  relevant_processes.insert(
765  static_cast<unsigned int>(partition.local_element(i)));
766 
767  return std::vector<unsigned int>(relevant_processes.begin(),
768  relevant_processes.end());
769  }();
770 
771  // 2) determine coinciding vertices (important for periodicity)
772  std::map<unsigned int, std::vector<unsigned int>>
773  coinciding_vertex_groups;
774  std::map<unsigned int, unsigned int> vertex_to_coinciding_vertex_group;
775 
776  GridTools::collect_coinciding_vertices(tria,
777  coinciding_vertex_groups,
778  vertex_to_coinciding_vertex_group);
779 
780  const auto add_vertices_of_cell_to_vertices_owned_by_locally_owned_cells =
781  [&coinciding_vertex_groups, &vertex_to_coinciding_vertex_group](
782  const auto & cell,
783  std::vector<bool> &vertices_owned_by_locally_owned_cells) {
784  for (const auto v : cell->vertex_indices())
785  {
786  vertices_owned_by_locally_owned_cells[cell->vertex_index(v)] =
787  true;
788  const auto coinciding_vertex_group =
789  vertex_to_coinciding_vertex_group.find(cell->vertex_index(v));
790  if (coinciding_vertex_group !=
791  vertex_to_coinciding_vertex_group.end())
792  for (const auto &co_vertex : coinciding_vertex_groups.at(
793  coinciding_vertex_group->second))
794  vertices_owned_by_locally_owned_cells[co_vertex] = true;
795  }
796  };
797 
798  // 3) create a description (locally owned cell and a layer of ghost cells
799  // and all their parents)
800  std::vector<DescriptionTemp<dim, spacedim>> description_temp(
801  relevant_processes.size());
802 
803  for (unsigned int i = 0; i < description_temp.size(); ++i)
804  {
805  const unsigned int proc = relevant_processes[i];
806  auto & description_temp_i = description_temp[i];
807  description_temp_i.cell_infos.resize(
808  tria.get_triangulation().n_global_levels());
809 
810  // clear user_flags
811  std::vector<bool> old_user_flags;
812  tria.save_user_flags(old_user_flags);
813 
814  const_cast<::Triangulation<dim, spacedim> &>(tria)
815  .clear_user_flags();
816 
817  // mark all vertices attached to locally owned cells
818  std::vector<bool> vertices_owned_by_locally_owned_cells(
819  tria.n_vertices());
820  for (const auto &cell : tria.active_cell_iterators())
821  if (cell->is_locally_owned() &&
822  static_cast<unsigned int>(
823  partition[cell->global_active_cell_index()]) == proc)
824  add_vertices_of_cell_to_vertices_owned_by_locally_owned_cells(
825  cell, vertices_owned_by_locally_owned_cells);
826 
827  // helper function if a cell is locally relevant (active and
828  // connected to cell via a vertex)
829  const auto is_locally_relevant_on_level = [&](const auto &cell) {
830  for (const auto v : cell->vertex_indices())
831  if (vertices_owned_by_locally_owned_cells[cell->vertex_index(v)])
832  return true;
833  return false;
834  };
835 
836  // collect locally relevant cells (including their parents)
837  for (const auto &cell : tria.active_cell_iterators())
838  if (is_locally_relevant_on_level(cell))
839  fill_cell_infos(cell, description_temp_i.cell_infos, partition);
840 
841  // collect coarse-grid cells
842  std::vector<bool> vertices_locally_relevant(tria.n_vertices(), false);
843 
844  for (const auto &cell : tria.cell_iterators_on_level(0))
845  {
846  if (!cell->user_flag_set())
847  continue;
848 
849  // extract cell definition
850  ::CellData<dim> cell_data(cell->n_vertices());
851  cell_data.material_id = cell->material_id();
852  cell_data.manifold_id = cell->manifold_id();
853  for (const auto v : cell->vertex_indices())
854  cell_data.vertices[v] = cell->vertex_index(v);
855  description_temp_i.coarse_cells.push_back(cell_data);
856 
857  // mark cell vertices as relevant
858  for (const auto v : cell->vertex_indices())
859  vertices_locally_relevant[cell->vertex_index(v)] = true;
860 
861  // save translation for corase grid: lid -> gid
862  description_temp_i.coarse_cell_index_to_coarse_cell_id.push_back(
863  cell->id().get_coarse_cell_id());
864  }
865 
866  // collect coarse-grid vertices
867  for (unsigned int i = 0; i < vertices_locally_relevant.size(); ++i)
868  if (vertices_locally_relevant[i])
869  description_temp_i.coarse_cell_vertices.emplace_back(
870  i, tria.get_vertices()[i]);
871 
872  // restore flags
873  const_cast<::Triangulation<dim, spacedim> &>(tria)
874  .load_user_flags(old_user_flags);
875  }
876 
877  // collect description from all processes that used to own locally-owned
878  // active cells of this process in a single description
879  DescriptionTemp<dim, spacedim> description_merged;
880 
881  ::Utilities::MPI::ConsensusAlgorithms::AnonymousProcess<char, char>
882  process([&]() { return relevant_processes; },
883  [&](const unsigned int other_rank,
884  std::vector<char> &send_buffer) {
885  const auto ptr = std::find(relevant_processes.begin(),
886  relevant_processes.end(),
887  other_rank);
888 
889  Assert(ptr != relevant_processes.end(), ExcInternalError());
890 
891  const auto other_rank_index =
892  std::distance(relevant_processes.begin(), ptr);
893 
894  send_buffer =
895  ::Utilities::pack(description_temp[other_rank_index],
896  false);
897  },
898  [&](const unsigned int &,
899  const std::vector<char> &recv_buffer,
900  std::vector<char> &) {
901  description_merged.merge(
902  ::Utilities::unpack<DescriptionTemp<dim, spacedim>>(
903  recv_buffer, false));
904  });
905 
906  ::Utilities::MPI::ConsensusAlgorithms::Selector<char, char>(
907  process, partition.get_mpi_communicator())
908  .run();
909 
910  // remove redundant entries
911  description_merged.reduce();
912 
913  // convert to actual description
914  return description_merged.convert(partition.get_mpi_communicator(),
915  tria.get_mesh_smoothing());
916  }
917 
918  } // namespace Utilities
919 } // namespace TriangulationDescription
