209  // 从内存池中取空间给 free list 使用，条件不允许时，会调整 nblock
210  char *alloc::M_chunk_alloc(size_t size, size_t &nblock)
211  {
212  char *result;
213  size_t need_bytes = size * nblock;
214  size_t pool_bytes = end_free - start_free;
215 
216  // 如果内存池剩余大小完全满足需求量，返回它
217  if (pool_bytes >= need_bytes)
218  {
219  result = start_free;
220  start_free += need_bytes;
221  return result;
222  }
223 
224  // 如果内存池剩余大小不能完全满足需求量，但至少可以分配一个或一个以上的区块，就返回它
225  else if (pool_bytes >= size)
226  {
227  nblock = pool_bytes / size;
228  need_bytes = size * nblock;
229  result = start_free;
230  start_free += need_bytes;
231  return result;
232  }
233 
234  // 如果内存池剩余大小连一个区块都无法满足
235  else
236  {
237  if (pool_bytes > 0)
238  { // 如果内存池还有剩余，把剩余的空间加入到 free list 中
239  FreeList *my_free_list = free_list[M_freelist_index(pool_bytes)];
240  ((FreeList *)start_free)->next = my_free_list;
241  my_free_list = (FreeList *)start_free;
242  }
243  // 申请 heap 空间
244  size_t bytes_to_get = (need_bytes << 1) + M_round_up(heap_size >> 4);
245  start_free = (char *)std::malloc(bytes_to_get);
246  if (!start_free)
247  { // heap 空间也不够
248  FreeList *my_free_list, *p;
249  // 试着查找有无未用区块，且区块足够大的 free list
250  for (size_t i = size; i <= ESmallObjectBytes; i += M_align(i))
251  {
252  my_free_list = free_list[M_freelist_index(i)];
253  p = my_free_list;
254  if (p)
255  {
256  my_free_list = p->next;
257  start_free = (char *)p;
258  end_free = start_free + i;
259  return M_chunk_alloc(size, nblock);
260  }
261  }
262  std::printf("out of memory");
263  end_free = nullptr;
264  throw std::bad_alloc();
265  }
266  end_free = start_free + bytes_to_get;
267  heap_size += bytes_to_get;
268  return M_chunk_alloc(size, nblock);
269  }
