1447  template <typename MeshIteratorType>
1448  std::pair<std::array<Point<MeshIteratorType::AccessorType::space_dimension>,
1449  n_default_points_per_cell<MeshIteratorType>()>,
1450  std::array<double, n_default_points_per_cell<MeshIteratorType>()>>
1451  get_default_points_and_weights(const MeshIteratorType &iterator,
1452  const bool with_interpolation)
1453  {
1454  const int dim = MeshIteratorType::AccessorType::structure_dimension;
1455  const int spacedim = MeshIteratorType::AccessorType::space_dimension;
1456  constexpr std::size_t points_per_cell =
1457  n_default_points_per_cell<MeshIteratorType>();
1458 
1459  std::pair<std::array<Point<spacedim>, points_per_cell>,
1460  std::array<double, points_per_cell>>
1461  points_weights;
1462 
1463 
1464  // note that the exact weights are chosen such as to minimize the
1465  // distortion of the four new quads from the optimal shape; their
1466  // derivation and values is copied over from the
1467  // interpolation function in the mapping
1468  switch (dim)
1469  {
1470  case 1:
1471  Assert(points_weights.first.size() == 2, ExcInternalError());
1472  Assert(points_weights.second.size() == 2, ExcInternalError());
1473  points_weights.first[0] = iterator->vertex(0);
1474  points_weights.second[0] = .5;
1475  points_weights.first[1] = iterator->vertex(1);
1476  points_weights.second[1] = .5;
1477  break;
1478  case 2:
1479  Assert(points_weights.first.size() == 8, ExcInternalError());
1480  Assert(points_weights.second.size() == 8, ExcInternalError());
1481 
1482  for (unsigned int i = 0; i < 4; ++i)
1483  {
1484  points_weights.first[i] = iterator->vertex(i);
1485  points_weights.first[4 + i] =
1486  (iterator->line(i)->has_children() ?
1487  iterator->line(i)->child(0)->vertex(1) :
1488  iterator->line(i)->get_manifold().get_new_point_on_line(
1489  iterator->line(i)));
1490  }
1491 
1492  if (with_interpolation)
1493  {
1494  std::fill(points_weights.second.begin(),
1495  points_weights.second.begin() + 4,
1496  -0.25);
1497  std::fill(points_weights.second.begin() + 4,
1498  points_weights.second.end(),
1499  0.5);
1500  }
1501  else
1502  std::fill(points_weights.second.begin(),
1503  points_weights.second.end(),
1504  1.0 / 8.0);
1505  break;
1506  case 3:
1507  {
1508  TriaIterator<TriaAccessor<3, 3, 3>> hex =
1509  static_cast<TriaIterator<TriaAccessor<3, 3, 3>>>(iterator);
1510  const unsigned int np = GeometryInfo<dim>::vertices_per_cell +
1511  GeometryInfo<dim>::lines_per_cell +
1512  GeometryInfo<dim>::faces_per_cell;
1513  Assert(points_weights.first.size() == np, ExcInternalError());
1514  Assert(points_weights.second.size() == np, ExcInternalError());
1515  auto *sp3 = reinterpret_cast<
1516  std::array<Point<3>, n_default_points_per_cell<decltype(hex)>()>
1517  *>(&points_weights.first);
1518 
1519  unsigned int j = 0;
1520 
1521  // note that the exact weights are chosen such as to minimize the
1522  // distortion of the eight new hexes from the optimal shape through
1523  // transfinite interpolation from the faces and vertices, see
1524  // TransfiniteInterpolationManifold for a deeper explanation of the
1525  // mechanisms
1526  if (with_interpolation)
1527  {
1528  for (unsigned int i = 0;
1529  i < GeometryInfo<dim>::vertices_per_cell;
1530  ++i, ++j)
1531  {
1532  (*sp3)[j] = hex->vertex(i);
1533  points_weights.second[j] = 1.0 / 8.0;
1534  }
1535  for (unsigned int i = 0; i < GeometryInfo<dim>::lines_per_cell;
1536  ++i, ++j)
1537  {
1538  (*sp3)[j] =
1539  (hex->line(i)->has_children() ?
1540  hex->line(i)->child(0)->vertex(1) :
1541  hex->line(i)->get_manifold().get_new_point_on_line(
1542  hex->line(i)));
1543  points_weights.second[j] = -1.0 / 4.0;
1544  }
1545  for (unsigned int i = 0; i < GeometryInfo<dim>::faces_per_cell;
1546  ++i, ++j)
1547  {
1548  (*sp3)[j] =
1549  (hex->quad(i)->has_children() ?
1550  hex->quad(i)->isotropic_child(0)->vertex(3) :
1551  hex->quad(i)->get_manifold().get_new_point_on_quad(
1552  hex->quad(i)));
1553  points_weights.second[j] = 1.0 / 2.0;
1554  }
1555  }
1556  else
1557  // Overwrite the weights with 1/np if we don't want to use
1558  // interpolation.
1559  std::fill(points_weights.second.begin(),
1560  points_weights.second.end(),
1561  1.0 / np);
1562  }
1563  break;
1564  default:
1565  Assert(false, ExcInternalError());
1566  break;
1567  }
1568  return points_weights;
1569  }
