127  nitsche_matrix(FullMatrix<double> & M,
128  const FEValuesBase<dim> &fe,
129  double penalty,
130  double factor = 1.)
131  {
132  const unsigned int n_dofs = fe.dofs_per_cell;
133 
134  AssertDimension(fe.get_fe().n_components(), dim);
135  AssertDimension(M.m(), n_dofs);
136  AssertDimension(M.n(), n_dofs);
137 
138  for (unsigned int k = 0; k < fe.n_quadrature_points; ++k)
139  {
140  const double dx = factor * fe.JxW(k);
141  const Tensor<1, dim> n = fe.normal_vector(k);
142  for (unsigned int i = 0; i < n_dofs; ++i)
143  for (unsigned int j = 0; j < n_dofs; ++j)
144  for (unsigned int d1 = 0; d1 < dim; ++d1)
145  {
146  const double u = fe.shape_value_component(j, k, d1);
147  const double v = fe.shape_value_component(i, k, d1);
148  M(i, j) += dx * 2. * penalty * u * v;
149  for (unsigned int d2 = 0; d2 < dim; ++d2)
150  {
151  // v . nabla u n
152  M(i, j) -= .5 * dx *
153  fe.shape_grad_component(j, k, d1)[d2] * n[d2] *
154  v;
155  // v (nabla u)^T n
156  M(i, j) -= .5 * dx *
157  fe.shape_grad_component(j, k, d2)[d1] * n[d2] *
158  v;
159  // u nabla v n
160  M(i, j) -= .5 * dx *
161  fe.shape_grad_component(i, k, d1)[d2] * n[d2] *
162  u;
163  // u (nabla v)^T n
164  M(i, j) -= .5 * dx *
165  fe.shape_grad_component(i, k, d2)[d1] * n[d2] *

183  template <int dim>
184  inline void
185  nitsche_tangential_matrix(FullMatrix<double> & M,
186  const FEValuesBase<dim> &fe,
187  double penalty,
188  double factor = 1.)
189  {
190  const unsigned int n_dofs = fe.dofs_per_cell;
191 
192  AssertDimension(fe.get_fe().n_components(), dim);
193  AssertDimension(M.m(), n_dofs);
194  AssertDimension(M.n(), n_dofs);
195 
196  for (unsigned int k = 0; k < fe.n_quadrature_points; ++k)
197  {
198  const double dx = factor * fe.JxW(k);
199  const Tensor<1, dim> n = fe.normal_vector(k);
200  for (unsigned int i = 0; i < n_dofs; ++i)
201  for (unsigned int j = 0; j < n_dofs; ++j)
202  {
203  double udotn = 0.;
204  double vdotn = 0.;
205  double ngradun = 0.;
206  double ngradvn = 0.;
207 
208  for (unsigned int d = 0; d < dim; ++d)
209  {
210  udotn += n[d] * fe.shape_value_component(j, k, d);
211  vdotn += n[d] * fe.shape_value_component(i, k, d);
212  ngradun += n * fe.shape_grad_component(j, k, d) * n[d];
213  ngradvn += n * fe.shape_grad_component(i, k, d) * n[d];
214  }
215  for (unsigned int d1 = 0; d1 < dim; ++d1)
216  {
217  const double u =
218  fe.shape_value_component(j, k, d1) - udotn * n[d1];
219  const double v =
220  fe.shape_value_component(i, k, d1) - vdotn * n[d1];
221  M(i, j) += dx * 2. * penalty * u * v;
222  // Correct the gradients below and subtract normal component
223  M(i, j) += dx * (ngradun * v + ngradvn * u);
224  for (unsigned int d2 = 0; d2 < dim; ++d2)
225  {
226  // v . nabla u n
227  M(i, j) -= .5 * dx *
228  fe.shape_grad_component(j, k, d1)[d2] *
229  n[d2] * v;
230  // v (nabla u)^T n
231  M(i, j) -= .5 * dx *
232  fe.shape_grad_component(j, k, d2)[d1] *
233  n[d2] * v;
234  // u nabla v n
235  M(i, j) -= .5 * dx *
236  fe.shape_grad_component(i, k, d1)[d2] *
237  n[d2] * u;
238  // u (nabla v)^T n
239  M(i, j) -= .5 * dx *
240  fe.shape_grad_component(i, k, d2)[d1] *
241  n[d2] * u;
242  }
243  }
244  }

337  const FEValuesBase<dim> & fe,

347  AssertVectorVectorDimension(data, dim, fe.n_quadrature_points);
348

356  double gdotn = 0.;
357  double vdotn = 0.;
358  double ngradun = 0.;
359  double ngradvn = 0.;
360 
361  for (unsigned int d = 0; d < dim; ++d)
362  {
363  udotn += n[d] * input[d][k];
364  gdotn += n[d] * data[d][k];
365  vdotn += n[d] * fe.shape_value_component(i, k, d);
366  ngradun += n * Dinput[d][k] * n[d];
367  ngradvn += n * fe.shape_grad_component(i, k, d) * n[d];
368  }

370  {
371  const double u = input[d1][k] - udotn * n[d1];
372  const double v =
373  fe.shape_value_component(i, k, d1) - vdotn * n[d1];
374  const double g = data[d1][k] - gdotn * n[d1];
375  result(i) += dx * 2. * penalty * (u - g) * v;
376  // Correct the gradients below and subtract normal component
377  result(i) += dx * (ngradun * v + ngradvn * (u - g));
378  for (unsigned int d2 = 0; d2 < dim; ++d2)
379  {
380  // v . nabla u n
381  result(i) -= .5 * dx * Dinput[d1][k][d2] * n[d2] * v;
382  // v (nabla u)^T n
383  result(i) -= .5 * dx * Dinput[d2][k][d1] * n[d2] * v;
384  // u nabla v n
385  result(i) -= .5 * dx * (u - g) *
386  fe.shape_grad_component(i, k, d1)[d2] *
387  n[d2];
388  // u (nabla v)^T n
389  result(i) -= .5 * dx * (u - g) *
390  fe.shape_grad_component(i, k, d2)[d1] *
391  n[d2];
392  }

564  template <int dim, typename number>
565  void
566  ip_residual(Vector<number> & result1,
567  Vector<number> & result2,
568  const FEValuesBase<dim> & fe1,
569  const FEValuesBase<dim> & fe2,
570  const ArrayView<const std::vector<double>> & input1,
571  const ArrayView<const std::vector<Tensor<1, dim>>> &Dinput1,
572  const ArrayView<const std::vector<double>> & input2,
573  const ArrayView<const std::vector<Tensor<1, dim>>> &Dinput2,
574  double pen,
575  double int_factor = 1.,
576  double ext_factor = -1.)
577  {
578  const unsigned int n1 = fe1.dofs_per_cell;
579 
580  AssertDimension(fe1.get_fe().n_components(), dim);
581  AssertDimension(fe2.get_fe().n_components(), dim);
582  AssertVectorVectorDimension(input1, dim, fe1.n_quadrature_points);
583  AssertVectorVectorDimension(Dinput1, dim, fe1.n_quadrature_points);
584  AssertVectorVectorDimension(input2, dim, fe2.n_quadrature_points);
585  AssertVectorVectorDimension(Dinput2, dim, fe2.n_quadrature_points);
586 
587  const double nu1 = int_factor;
588  const double nu2 = (ext_factor < 0) ? int_factor : ext_factor;
589  const double penalty = .5 * pen * (nu1 + nu2);
590 
591 
592  for (unsigned int k = 0; k < fe1.n_quadrature_points; ++k)
593  {
594  const double dx = fe1.JxW(k);
595  const Tensor<1, dim> n = fe1.normal_vector(k);
596 
597  for (unsigned int i = 0; i < n1; ++i)
598  for (unsigned int d1 = 0; d1 < dim; ++d1)
599  {
600  const double v1 = fe1.shape_value_component(i, k, d1);
601  const double v2 = fe2.shape_value_component(i, k, d1);
602  const double u1 = input1[d1][k];
603  const double u2 = input2[d1][k];
604 
605  result1(i) += dx * penalty * u1 * v1;
606  result1(i) -= dx * penalty * u2 * v1;
607  result2(i) -= dx * penalty * u1 * v2;
608  result2(i) += dx * penalty * u2 * v2;
609 
610  for (unsigned int d2 = 0; d2 < dim; ++d2)
611  {
612  // v . nabla u n
613  result1(i) -=
614  .25 * dx *
615  (nu1 * Dinput1[d1][k][d2] + nu2 * Dinput2[d1][k][d2]) *
616  n[d2] * v1;
617  result2(i) +=
618  .25 * dx *
619  (nu1 * Dinput1[d1][k][d2] + nu2 * Dinput2[d1][k][d2]) *
620  n[d2] * v2;
621  // v . (nabla u)^T n
622  result1(i) -=
623  .25 * dx *
624  (nu1 * Dinput1[d2][k][d1] + nu2 * Dinput2[d2][k][d1]) *
625  n[d2] * v1;
626  result2(i) +=
627  .25 * dx *
628  (nu1 * Dinput1[d2][k][d1] + nu2 * Dinput2[d2][k][d1]) *
629  n[d2] * v2;
630  // u nabla v n
631  result1(i) -= .25 * dx * nu1 *
