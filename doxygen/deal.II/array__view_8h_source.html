340  template <typename MemorySpaceType>
341  inline bool
342  is_in_correct_memory_space(const void *const ptr)
343  {
344 #ifndef DEAL_II_COMPILER_CUDA_AWARE
345  (void)ptr;
346  static_assert(std::is_same<MemorySpaceType, MemorySpace::Host>::value,
347  "If the compiler doesn't understand CUDA code, "
348  "the only possible memory space is 'MemorySpace::Host'!");
349  return true;
350 #else
351  cudaPointerAttributes attributes;
352  const cudaError_t cuda_error = cudaPointerGetAttributes(&attributes, ptr);
353  if (cuda_error != cudaErrorInvalidValue)
354  {
355  AssertCuda(cuda_error);
356  if (std::is_same<MemorySpaceType, MemorySpace::Host>::value)
357  return (attributes.type == cudaMemoryTypeHost) ||
358  (attributes.type == cudaMemoryTypeUnregistered);
359  else
360  return attributes.type == cudaMemoryTypeDevice;
361  }
362  else
363  {
364  // ignore and reset the error since host pointers produce an error
365  cudaGetLastError();
366  return std::is_same<MemorySpaceType, MemorySpace::Host>::value;
367  }
368 #endif
369  }

674 template <typename Iterator, typename MemorySpaceType = MemorySpace::Host>
675 ArrayView<typename std::remove_reference<
676  typename std::iterator_traits<Iterator>::reference>::type,
677  MemorySpaceType>
678 make_array_view(const Iterator begin, const Iterator end)
679 {
680  static_assert(
681  std::is_same<typename std::iterator_traits<Iterator>::iterator_category,
682  typename std::random_access_iterator_tag>::value,
683  "The provided iterator should be a random access iterator.");
684  Assert(begin <= end,
685  ExcMessage(
686  "The beginning of the array view should be before the end."));
687  Assert(internal::ArrayViewHelper::is_contiguous(begin, end),
688  ExcMessage("The provided range isn't contiguous in memory!"));
689  // the reference type, not the value type, knows the constness of the iterator
690  return ArrayView<typename std::remove_reference<
691  typename std::iterator_traits<Iterator>::reference>::type,
692  MemorySpaceType>(std::addressof(*begin), end - begin);
693 }
