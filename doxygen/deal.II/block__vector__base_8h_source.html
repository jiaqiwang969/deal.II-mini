1304  (parent.block_indices.local_to_global(current_block, 0) +
1305  parent.block_indices.block_size(current_block) - 1);

1309  // value for this

1311  this->global_index = parent.size();
1312  current_block = parent.n_blocks();

1326  ++index_within_block;
1327  else
1328  {
1329  // ok, we traverse a boundary
1330  // between blocks:
1331  index_within_block = 0;
1332  ++current_block;
1333

1335  // break forward

1341  parent->block_indices.block_size(current_block);
1342  else
1343  // if we are beyond the end,
1344  // then move the next
1345  // boundary arbitrarily far
1346  // away

1348  };
