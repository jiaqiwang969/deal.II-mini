1439 inline bool
1440 SparsityPatternBase::operator==(const SparsityPatternBase &sp2) const
1441 {
1442  // it isn't quite necessary to compare *all* member variables. by only
1443  // comparing the essential ones, we can say that two sparsity patterns are
1444  // equal even if one is compressed and the other is not (in which case some
1445  // of the member variables are not yet set correctly)
1446  if (rows != sp2.rows || cols != sp2.cols || compressed != sp2.compressed)
1447  return false;
1448 
1449  for (size_type i = 0; i < rows + 1; ++i)
1450  if (rowstart[i] != sp2.rowstart[i])
1451  return false;
1452 
1453  for (size_type i = 0; i < rowstart[rows]; ++i)
1454  if (colnums[i] != sp2.colnums[i])
1455  return false;
1456 
1457  return true;
1458 }
