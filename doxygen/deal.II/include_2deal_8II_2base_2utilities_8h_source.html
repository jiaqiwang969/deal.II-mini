408  template <typename T>
409  constexpr T
410  pow(const T base, const int iexp)
411  {
412 #if defined(DEBUG) && !defined(DEAL_II_CXX14_CONSTEXPR_BUG)
413  // Up to __builtin_expect this is the same code as in the 'Assert' macro.
414  // The call to __builtin_expect turns out to be problematic.
415  if (!(iexp >= 0))
416  ::deal_II_exceptions::internals::issue_error_noreturn(
417  ::deal_II_exceptions::internals::abort_or_throw_on_exception,
418  __FILE__,
419  __LINE__,
420  __PRETTY_FUNCTION__,
421  "iexp>=0",
422  "ExcMessage(\"The exponent must not be negative!\")",
423  ExcMessage("The exponent must not be negative!"));
424 #endif
425  // The "exponentiation by squaring" algorithm used below has to be
426  // compressed to one statement due to C++11's restrictions on constexpr
427  // functions. A more descriptive version would be:
428  //
429  // <code>
430  // if (iexp <= 0)
431  // return 1;
432  //
433  // // avoid overflow of one additional recursion with pow(base * base, 0)
434  // if (iexp == 1)
435  // return base;
436  //
437  // // if the current exponent is not divisible by two,
438  // // we need to account for that.
439  // const unsigned int prefactor = (iexp % 2 == 1) ? base : 1;
440  //
441  // // a^b = (a*a)^(b/2) for b even
442  // // a^b = a*(a*a)^((b-1)/2 for b odd
443  // return prefactor * ::Utilities::pow(base*base, iexp/2);
444  // </code>
445 
446  static_assert(std::is_integral<T>::value, "Only integral types supported");
447 
448  return iexp <= 0 ?
449  1 :
450  (iexp == 1 ? base :
451  (((iexp % 2 == 1) ? base : 1) *
452  ::Utilities::pow(base * base, iexp / 2)));
453  }

929  fixed_power(const T x)
930  {
931  Assert(
932  !std::is_integral<T>::value || (N >= 0),

966  template <typename Iterator, typename T, typename Comp>
967  inline Iterator
968  lower_bound(Iterator first, Iterator last, const T &val, const Comp comp)
969  {
970  // verify that the two iterators are properly ordered. since
971  // we need operator- for the iterator type anyway, do the
972  // test as follows, rather than via 'last >= first'
973  Assert(last - first >= 0,
974  ExcMessage(
975  "The given iterators do not satisfy the proper ordering."));
976 
977  unsigned int len = static_cast<unsigned int>(last - first);
978 
979  if (len == 0)
980  return first;
981 
982  while (true)
983  {
984  // if length equals 8 or less,
985  // then do a rolled out
986  // search. use a switch without
987  // breaks for that and roll-out
988  // the loop somehow
989  if (len < 8)
990  {
991  switch (len)
992  {
993  case 7:
994  if (!comp(*first, val))
995  return first;
996  ++first;
997  DEAL_II_FALLTHROUGH;
998  case 6:
999  if (!comp(*first, val))
1000  return first;
1001  ++first;
1002  DEAL_II_FALLTHROUGH;
1003  case 5:
1004  if (!comp(*first, val))
1005  return first;
1006  ++first;
1007  DEAL_II_FALLTHROUGH;
1008  case 4:
1009  if (!comp(*first, val))
1010  return first;
1011  ++first;
1012  DEAL_II_FALLTHROUGH;
1013  case 3:
1014  if (!comp(*first, val))
1015  return first;
1016  ++first;
1017  DEAL_II_FALLTHROUGH;
1018  case 2:
1019  if (!comp(*first, val))
1020  return first;
1021  ++first;
1022  DEAL_II_FALLTHROUGH;
1023  case 1:
1024  if (!comp(*first, val))
1025  return first;
1026  return first + 1;
1027  default:
1028  // indices seem
1029  // to not be
1030  // sorted
1031  // correctly!? or
1032  // did len
1033  // become==0
1034  // somehow? that
1035  // shouldn't have
1036  // happened
1037  Assert(false, ExcInternalError());
1038  }
1039  }
1040 
1041 
1042 
1043  const unsigned int half = len >> 1;
1044  const Iterator middle = first + half;
1045 
1046  // if the value is larger than
1047  // that pointed to by the
1048  // middle pointer, then the
1049  // insertion point must be
1050  // right of it
1051  if (comp(*middle, val))
1052  {
1053  first = middle + 1;
1054  len -= half + 1;
1055  }
1056  else
1057  len = half;
1058  }
1059  }

1064  template <typename T>
1065  size_t
1066  pack(const T & object,
1067  std::vector<char> &dest_buffer,
1068  const bool allow_compression)
1069  {
1070  std::size_t size = 0;
1071 
1072  // see if the object is small and copyable via memcpy. if so, use
1073  // this fast path. otherwise, we have to go through the BOOST
1074  // serialization machinery
1075 #ifdef DEAL_II_HAVE_CXX17
1076  if constexpr (std::is_trivially_copyable<T>() && sizeof(T) < 256)
1077 #else
1078  if (std::is_trivially_copyable<T>() && sizeof(T) < 256)
1079 #endif
1080  {
1081  (void)allow_compression;
1082  const std::size_t previous_size = dest_buffer.size();
1083  dest_buffer.resize(previous_size + sizeof(T));
1084 
1085  std::memcpy(dest_buffer.data() + previous_size, &object, sizeof(T));
1086 
1087  size = sizeof(T);
1088  }
1089  else
1090  {
1091  // use buffer as the target of a compressing
1092  // stream into which we serialize the current object
1093  const std::size_t previous_size = dest_buffer.size();
1094  {
1095  boost::iostreams::filtering_ostreambuf fosb;
1096 #ifdef DEAL_II_WITH_ZLIB
1097  if (allow_compression)
1098  fosb.push(boost::iostreams::gzip_compressor());
1099 #else
1100  (void)allow_compression;
1101 #endif
1102  fosb.push(boost::iostreams::back_inserter(dest_buffer));
1103 
1104  boost::archive::binary_oarchive boa(fosb);
1105  boa << object;
1106  // the stream object has to be destroyed before the return statement
1107  // to ensure that all data has been written in the buffer
1108  }
1109  size = dest_buffer.size() - previous_size;
1110  }
1111 
1112  return size;
1113  }
