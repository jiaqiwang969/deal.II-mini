206  template <class INFOBOX, class DOFINFO, int dim, int spacedim, class ITERATOR>
207  void
208  cell_action(
209  ITERATOR cell,
210  DoFInfoBox<dim, DOFINFO> &dof_info,
211  INFOBOX & info,
212  const std::function<void(DOFINFO &, typename INFOBOX::CellInfo &)>
213  &cell_worker,
214  const std::function<void(DOFINFO &, typename INFOBOX::CellInfo &)>
215  & boundary_worker,
216  const std::function<void(DOFINFO &,
217  DOFINFO &,
218  typename INFOBOX::CellInfo &,
219  typename INFOBOX::CellInfo &)> &face_worker,
220  const LoopControl & loop_control)
221  {
222  const bool ignore_subdomain =
223  (cell->get_triangulation().locally_owned_subdomain() ==
224  numbers::invalid_subdomain_id);
225 
226  types::subdomain_id csid = (cell->is_level_cell()) ?
227  cell->level_subdomain_id() :
228  cell->subdomain_id();
229 
230  const bool own_cell =
231  ignore_subdomain ||
232  (csid == cell->get_triangulation().locally_owned_subdomain());
233 
234  dof_info.reset();
235 
236  if ((!ignore_subdomain) && (csid == numbers::artificial_subdomain_id))
237  return;
238 
239  dof_info.cell.reinit(cell);
240  dof_info.cell_valid = true;
241 
242  const bool integrate_cell = (cell_worker != nullptr);
243  const bool integrate_boundary = (boundary_worker != nullptr);
244  const bool integrate_interior_face = (face_worker != nullptr);
245 
246  if (integrate_cell)
247  info.cell.reinit(dof_info.cell);
248  // Execute this, if cells
249  // have to be dealt with
250  // before faces
251  if (integrate_cell && loop_control.cells_first &&
252  ((loop_control.own_cells && own_cell) ||
253  (loop_control.ghost_cells && !own_cell)))
254  cell_worker(dof_info.cell, info.cell);
255 
256  // Call the callback function in
257  // the info box to do
258  // computations between cell and
259  // face action.
260  info.post_cell(dof_info);
261 
262  if (integrate_interior_face || integrate_boundary)
263  for (const unsigned int face_no : cell->face_indices())
264  {
265  typename ITERATOR::AccessorType::Container::face_iterator face =
266  cell->face(face_no);
267  if (cell->at_boundary(face_no) &&
268  !cell->has_periodic_neighbor(face_no))
269  {
270  // only integrate boundary faces of own cells
271  if (integrate_boundary && own_cell)
272  {
273  dof_info.interior_face_available[face_no] = true;
274  dof_info.interior[face_no].reinit(cell, face, face_no);
275  info.boundary.reinit(dof_info.interior[face_no]);
276  boundary_worker(dof_info.interior[face_no], info.boundary);
277  }
278  }
279  else if (integrate_interior_face)
280  {
281  // Interior face
282  TriaIterator<typename ITERATOR::AccessorType> neighbor =
283  cell->neighbor_or_periodic_neighbor(face_no);
284 
285  types::subdomain_id neighbid = numbers::artificial_subdomain_id;
286  if (neighbor->is_level_cell())
287  neighbid = neighbor->level_subdomain_id();
288  // subdomain id is only valid for active cells
289  else if (neighbor->is_active())
290  neighbid = neighbor->subdomain_id();
291 
292  const bool own_neighbor =
293  ignore_subdomain ||
294  (neighbid ==
295  cell->get_triangulation().locally_owned_subdomain());
296 
297  // skip all faces between two ghost cells
298  if (!own_cell && !own_neighbor)
299  continue;
300 
301  // skip if the user doesn't want faces between own cells
302  if (own_cell && own_neighbor &&
303  loop_control.own_faces == LoopControl::never)
304  continue;
305 
306  // skip face to ghost
307  if (own_cell != own_neighbor &&
308  loop_control.faces_to_ghost == LoopControl::never)
309  continue;
310 
311  // Deal with refinement edges from the refined side. Assuming
312  // one-irregular meshes, this situation should only occur if both
313  // cells are active.
314  const bool periodic_neighbor =
315  cell->has_periodic_neighbor(face_no);
316 
317  if ((!periodic_neighbor && cell->neighbor_is_coarser(face_no)) ||
318  (periodic_neighbor &&
319  cell->periodic_neighbor_is_coarser(face_no)))
320  {
321  Assert(cell->is_active(), ExcInternalError());
322  Assert(neighbor->is_active(), ExcInternalError());
323 
324  // skip if only one processor needs to assemble the face
325  // to a ghost cell and the fine cell is not ours.
326  if (!own_cell &&
327  loop_control.faces_to_ghost == LoopControl::one)
328  continue;
329 
330  const std::pair<unsigned int, unsigned int> neighbor_face_no =
331  periodic_neighbor ?
332  cell->periodic_neighbor_of_coarser_periodic_neighbor(
333  face_no) :
334  cell->neighbor_of_coarser_neighbor(face_no);
335  const typename ITERATOR::AccessorType::Container::
336  face_iterator nface =
337  neighbor->face(neighbor_face_no.first);
338 
339  dof_info.interior_face_available[face_no] = true;
340  dof_info.exterior_face_available[face_no] = true;
341  dof_info.interior[face_no].reinit(cell, face, face_no);
342  info.face.reinit(dof_info.interior[face_no]);
343  dof_info.exterior[face_no].reinit(neighbor,
344  nface,
345  neighbor_face_no.first,
346  neighbor_face_no.second);
347  info.subface.reinit(dof_info.exterior[face_no]);
348 
349  face_worker(dof_info.interior[face_no],
350  dof_info.exterior[face_no],
351  info.face,
352  info.subface);
353  }
354  else
355  {
356  // If iterator is active and neighbor is refined, skip
357  // internal face.
358  if (::internal::is_active_iterator(cell) &&
359  neighbor->has_children())
360  {
361  Assert(
362  loop_control.own_faces != LoopControl::both,
363  ExcMessage(
364  "Assembling from both sides for own_faces is not "
365  "supported with hanging nodes!"));
366  continue;
367  }
368 
369  // Now neighbor is on same level, double-check this:
370  Assert(cell->level() == neighbor->level(),
371  ExcInternalError());
372 
373  // If we own both cells only do faces from one side (unless
374  // LoopControl says otherwise). Here, we rely on cell
375  // comparison that will look at cell->index().
376  if (own_cell && own_neighbor &&
377  loop_control.own_faces == LoopControl::one &&
378  (neighbor < cell))
379  continue;
380 
381  // independent of loop_control.faces_to_ghost,
382  // we only look at faces to ghost on the same level once
383  // (only where own_cell=true and own_neighbor=false)
384  if (!own_cell)
385  continue;
386 
387  // now only one processor assembles faces_to_ghost. We let the
388  // processor with the smaller (level-)subdomain id assemble
389  // the face.
390  if (own_cell && !own_neighbor &&
391  loop_control.faces_to_ghost == LoopControl::one &&
392  (neighbid < csid))
393  continue;
394 
395  const unsigned int neighbor_face_no =
396  periodic_neighbor ?
397  cell->periodic_neighbor_face_no(face_no) :
398  cell->neighbor_face_no(face_no);
399  Assert(periodic_neighbor ||
400  neighbor->face(neighbor_face_no) == face,
401  ExcInternalError());
402  // Regular interior face
403  dof_info.interior_face_available[face_no] = true;
404  dof_info.exterior_face_available[face_no] = true;
405  dof_info.interior[face_no].reinit(cell, face, face_no);
406  info.face.reinit(dof_info.interior[face_no]);
407  dof_info.exterior[face_no].reinit(neighbor,
408  neighbor->face(
409  neighbor_face_no),
410  neighbor_face_no);
411  info.neighbor.reinit(dof_info.exterior[face_no]);
412 
413  face_worker(dof_info.interior[face_no],
414  dof_info.exterior[face_no],
415  info.face,
416  info.neighbor);
417  }
418  }
419  } // faces
420  // Call the callback function in
421  // the info box to do
422  // computations between face and
423  // cell action.
424  info.post_faces(dof_info);
425 
426  // Execute this, if faces
427  // have to be handled first
428  if (integrate_cell && !loop_control.cells_first &&
429  ((loop_control.own_cells && own_cell) ||
430  (loop_control.ghost_cells && !own_cell)))
431  cell_worker(dof_info.cell, info.cell);
432  }
