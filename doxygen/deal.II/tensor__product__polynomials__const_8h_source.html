280 template <int order>
281 Tensor<order, dim>
282 TensorProductPolynomialsConst<dim>::compute_derivative(
283  const unsigned int i,
284  const Point<dim> & p) const
285 {
286  const unsigned int max_indices = tensor_polys.n();
287  Assert(i <= max_indices, ExcInternalError());
288 
289  // treat the regular basis functions
290  if (i < max_indices)
291  return tensor_polys.template compute_derivative<order>(i, p);
292  else
293  // this is for the constant function
294  return Tensor<order, dim>();
295 }
