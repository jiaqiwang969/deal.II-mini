895  inline void
896  Accessor::advance()
897  {
898  const auto chunk_size = sparsity_pattern->get_chunk_size();
899  Assert(chunk_row < chunk_size && chunk_col < chunk_size,
900  ExcIteratorPastEnd());
901  Assert(reduced_accessor.row() * chunk_size + chunk_row <
902  sparsity_pattern->n_rows() &&
903  reduced_accessor.column() * chunk_size + chunk_col <
904  sparsity_pattern->n_cols(),
905  ExcIteratorPastEnd());
906  if (chunk_size == 1)
907  {
908  reduced_accessor.advance();
909  return;
910  }
911 
912  ++chunk_col;
913 
914  // end of chunk
915  if (chunk_col == chunk_size ||
916  reduced_accessor.column() * chunk_size + chunk_col ==
917  sparsity_pattern->n_cols())
918  {
919  const auto reduced_row = reduced_accessor.row();
920  // end of row
921  if (reduced_accessor.linear_index + 1 ==
922  reduced_accessor.container->rowstart[reduced_row + 1])
923  {
924  ++chunk_row;
925 
926  chunk_col = 0;
927 
928  // end of chunk rows or end of matrix
929  if (chunk_row == chunk_size ||
930  (reduced_row * chunk_size + chunk_row ==
931  sparsity_pattern->n_rows()))
932  {
933  chunk_row = 0;
934  reduced_accessor.advance();
935  }
936  // go back to the beginning of the same reduced row but with
937  // chunk_row increased by one
938  else
939  reduced_accessor.linear_index =
940  reduced_accessor.container->rowstart[reduced_row];
941  }
942  // advance within chunk
943  else
944  {
945  reduced_accessor.advance();
946  chunk_col = 0;
947  }
948  }
949  }

1081 template <typename ForwardIterator>
1082 void
1083 ChunkSparsityPattern::copy_from(const size_type n_rows,
1084  const size_type n_cols,
1085  const ForwardIterator begin,
1086  const ForwardIterator end,
1087  const size_type chunk_size)
1088 {
1089  Assert(static_cast<size_type>(std::distance(begin, end)) == n_rows,
1090  ExcIteratorRange(std::distance(begin, end), n_rows));
1091 
1092  // first determine row lengths for each row. if the matrix is quadratic,
1093  // then we might have to add an additional entry for the diagonal, if that
1094  // is not yet present. as we have to call compress anyway later on, don't
1095  // bother to check whether that diagonal entry is in a certain row or not
1096  const bool is_square = (n_rows == n_cols);
1097  std::vector<size_type> row_lengths;
1098  row_lengths.reserve(n_rows);
1099  for (ForwardIterator i = begin; i != end; ++i)
1100  row_lengths.push_back(std::distance(i->begin(), i->end()) +
1101  (is_square ? 1 : 0));
1102  reinit(n_rows, n_cols, row_lengths, chunk_size);
1103 
1104  // now enter all the elements into the matrix
1105  size_type row = 0;
1106  using inner_iterator =
1107  typename std::iterator_traits<ForwardIterator>::value_type::const_iterator;
1108  for (ForwardIterator i = begin; i != end; ++i, ++row)
1109  {
1110  const inner_iterator end_of_row = i->end();
1111  for (inner_iterator j = i->begin(); j != end_of_row; ++j)
1112  {
1113  const size_type col =
1114  internal::SparsityPatternTools::get_column_index_from_iterator(*j);
1115  Assert(col < n_cols, ExcInvalidIndex(col, n_cols));
1116 
1117  add(row, col);
1118  }
1119  }
1120 
1121  // finally compress everything. this also sorts the entries within each row
1122  compress();
1123 }
