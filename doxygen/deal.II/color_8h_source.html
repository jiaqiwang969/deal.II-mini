156 // Determines whether the terminal color of this system can be modified.
157 inline bool is_modifiable()
158 {
159 #if defined(REDBUD_LINUX) || defined(REDBUD_OSX)
160  static constexpr const char* terms[] = {
161  "ansi", "color", "console", "cygwin", "gnome", "konsole", "kterm",
162  "linux", "msys", "putty", "rxvt", "screen", "vt100", "xterm"
163  };
164  const char *penv = std::getenv("TERM");
165  if (penv == nullptr)
166  {
167  return false;
168  }
169  bool result = false;
170  for (const auto& t : terms)
171  {
172  if (std::strstr(penv, t) != nullptr)
173  {
174  result = true;
175  break;
176  }
177  }
178 
179 #elif defined(REDBUD_WIN)
180  static constexpr bool result = true;
181 #endif
182  return result;
183 }

186 inline bool is_terminal(const std::streambuf* buf)
187 {
188  if (buf == get_coutbuf())
189  {
190 #if defined(REDBUD_LINUX) || defined(REDBUD_OSX)
191  return isatty(fileno(stdout)) ? true : false;
192 #elif defined(REDBUD_WIN)
193  return _isatty(_fileno(stdout)) ? true : false;
194 #endif
195  }
196 
197  if (buf == get_cerrbuf() || buf == get_clogbuf())
198  {
199 #if defined(REDBUD_LINUX) || defined(REDBUD_OSX)
200  return isatty(fileno(stderr)) ? true : false;
201 #elif defined(REDBUD_WIN)
202  return _isatty(_fileno(stderr)) ? true : false;
203 #endif
204  }
205  return false;
206 }

308 template <typename T>
309 inline color_return_t<T> set_color(std::ostream& os, const T& value)
310 {
311  HANDLE h = get_console_handle();
312  if (h && is_terminal(os.rdbuf()))

327 template <typename T>
328 inline details::color_return_t<T>
329 operator<<(std::ostream& os, const T& value)
330 {
331  return (os.iword(details::get_iword()) ||
332  (details::is_modifiable() &&
333  details::is_terminal(os.rdbuf())))
334  ? details::set_color(os, value)
335  : os;

338 template <typename T>
339 inline details::state_return_t<T>
340 operator<<(std::ostream& os, const T& value)
341 {
342  if (value == redbud::io::state::automatic)
343  {
344  os.iword(details::get_iword()) = 0;
345  }
346  else if (value == redbud::io::state::manual)
347  {
348  os.iword(details::get_iword()) = 1;
349  }
