814  // fill_insert 函数
815  template <class T>
816  typename vector<T>::iterator
817  vector<T>::
818  fill_insert(iterator pos, size_type n, const value_type &value)
819  {
820  if (n == 0)
821  return pos;
822  const size_type xpos = pos - begin_;
823  const value_type value_copy = value; // 避免被覆盖
824  if (static_cast<size_type>(cap_ - end_) >= n)
825  { // 如果备用空间大于等于增加的空间
826  const size_type after_elems = end_ - pos;
827  auto old_end = end_;
828  if (after_elems > n)
829  {
830  mystl::uninitialized_copy(end_ - n, end_, end_);
831  end_ += n;
832  mystl::move_backward(pos, old_end - n, old_end);
833  mystl::uninitialized_fill_n(pos, n, value_copy);
834  }
835  else
836  {
837  end_ = mystl::uninitialized_fill_n(end_, n - after_elems, value_copy);
838  end_ = mystl::uninitialized_move(pos, old_end, end_);
839  mystl::uninitialized_fill_n(pos, after_elems, value_copy);
840  }
841  }
842  else
843  { // 如果备用空间不足
844  const auto new_size = get_new_cap(n);
845  auto new_begin = data_allocator::allocate(new_size);
846  auto new_end = new_begin;
847  try
848  {
849  new_end = mystl::uninitialized_move(begin_, pos, new_begin);
850  new_end = mystl::uninitialized_fill_n(new_end, n, value);
851  new_end = mystl::uninitialized_move(pos, end_, new_end);
852  }
853  catch (...)
854  {
855  destroy_and_recover(new_begin, new_end, new_size);
856  throw;
857  }
858  data_allocator::deallocate(begin_, cap_ - begin_);
859  begin_ = new_begin;
860  end_ = new_end;
861  cap_ = begin_ + new_size;
862  }
863  return begin_ + xpos;
