552 template <typename CellIteratorType, typename DataType>
553 template <typename T>
554 inline void
555 CellDataStorage<CellIteratorType, DataType>::initialize(
556  const CellIteratorType &cell,
557  const unsigned int n_q_points)
558 {
559  static_assert(std::is_base_of<DataType, T>::value,
560  "User's T class should be derived from user's DataType class");
561  // The first time this method is called, it has to initialize the reference
562  // to the triangulation object
563  if (!tria)
564  tria = &cell->get_triangulation();
565  Assert(&cell->get_triangulation() == tria, ExcTriangulationMismatch());
566 
567  const auto key = cell->id();
568  if (map.find(key) == map.end())
569  {
570  map[key] = std::vector<std::shared_ptr<DataType>>(n_q_points);
571  // we need to initialize one-by-one as the std::vector<>(q, T())
572  // will end with a single same T object stored in each element of the
573  // vector:
574  const auto it = map.find(key);
575  for (unsigned int q = 0; q < n_q_points; q++)
576  it->second[q] = std::make_shared<T>();
577  }
578 }

598 inline bool
599 CellDataStorage<CellIteratorType, DataType>::erase(const CellIteratorType &cell)
600 {
601  const auto key = cell->id();
602  const auto it = map.find(key);
603  if (it == map.end())
604  return false;
605  Assert(&cell->get_triangulation() == tria, ExcTriangulationMismatch());
606  for (unsigned int i = 0; i < it->second.size(); i++)
607  {
608  Assert(
609  it->second[i].unique(),
610  ExcMessage(
611  "Can not erase the cell data multiple objects reference its data."));
612  }

645 inline std::vector<std::shared_ptr<T>>
646 CellDataStorage<CellIteratorType, DataType>::get_data(
647  const CellIteratorType &cell)
648 {
649  static_assert(std::is_base_of<DataType, T>::value,
650  "User's T class should be derived from user's DataType class");
651  Assert(&cell->get_triangulation() == tria, ExcTriangulationMismatch());
652 
653  const auto it = map.find(cell->id());
654  Assert(it != map.end(), ExcMessage("Could not find data for the cell"));
655 
656  // It would be nice to have a specialized version of this function for
657  // T==DataType. However explicit (i.e full) specialization of a member
658  // template is only allowed when the enclosing class is also explicitly (i.e
659  // fully) specialized. Thus, stick with copying of shared pointers even when
660  // the T==DataType:
661  std::vector<std::shared_ptr<T>> res(it->second.size());
662  for (unsigned int q = 0; q < res.size(); q++)
663  {
664  res[q] = std::dynamic_pointer_cast<T>(it->second[q]);
665  Assert(res[q], ExcCellDataTypeMismatch());
666  }
667  return res;

770 inline void
771 pack_cell_data(const CellIteratorType & cell,
772  const CellDataStorage<CellIteratorType, DataType> *data_storage,
773  FullMatrix<double> & matrix_data)
774 {
775  static_assert(
776  std::is_base_of<TransferableQuadraturePointData, DataType>::value,
777  "User's DataType class should be derived from QPData");
778 
779  if (const auto qpd = data_storage->try_get_data(cell))
780  {
781  const unsigned int m = qpd->size();
782  Assert(m > 0, ExcInternalError());
783  const unsigned int n = (*qpd)[0]->number_of_values();
784  matrix_data.reinit(m, n);
785 
786  std::vector<double> single_qp_data(n);
787  for (unsigned int q = 0; q < m; ++q)
788  {
789  (*qpd)[q]->pack_values(single_qp_data);
790  AssertDimension(single_qp_data.size(), n);
791 
792  for (unsigned int i = 0; i < n; ++i)
793  matrix_data(q, i) = single_qp_data[i];
794  }
795  }
796  else
797  {
798  matrix_data.reinit({0, 0});
799  }
800 }

940  template <int dim, typename DataType>
941  inline void
942  ContinuousQuadratureDataTransfer<dim, DataType>::unpack_function(
943  const typename parallel::distributed::Triangulation<dim>::cell_iterator
944  &cell,
945  const typename parallel::distributed::Triangulation<dim>::CellStatus
946  status,
947  const boost::iterator_range<std::vector<char>::const_iterator>
948  &data_range)
949  {
950  Assert((status !=
951  parallel::distributed::Triangulation<dim, dim>::CELL_COARSEN),
952  ExcNotImplemented());
953  (void)status;
954 
955  matrix_dofs =
956  Utilities::unpack<FullMatrix<double>>(data_range.begin(),
957  data_range.end(),
958  /*allow_compression=*/ false);
959  const unsigned int number_of_values = matrix_dofs.n();
960  if (number_of_values == 0)
961  return;
962 
963  matrix_quadrature.reinit(n_q_points, number_of_values);
964 
965  if (cell->has_children())
966  {
967  // we need to first use prolongation matrix to get dofvalues on child
968  // cells based on dofvalues stored in the parent's data_store
969  matrix_dofs_child.reinit(projection_fe->n_dofs_per_cell(),
970  number_of_values);
971  for (unsigned int child = 0; child < cell->n_children(); ++child)
972  if (cell->child(child)->is_locally_owned())
973  {
974  projection_fe
975  ->get_prolongation_matrix(child, cell->refinement_case())
976  .mmult(matrix_dofs_child, matrix_dofs);
977 
978  // now we do the usual business of evaluating FE on quadrature
979  // points:
980  project_to_qp_matrix.mmult(matrix_quadrature,
981  matrix_dofs_child);
982 
983  // finally, put back into the map:
984  unpack_to_cell_data(cell->child(child),
985  matrix_quadrature,
986  data_storage);
987  }
988  }
989  else
990  {
991  // if there are no children, evaluate FE field at
992  // rhs_quadrature points.
993  project_to_qp_matrix.mmult(matrix_quadrature, matrix_dofs);
994 
995  // finally, put back into the map:
996  unpack_to_cell_data(cell, matrix_quadrature, data_storage);
997  }
998  }
