315 template <int dim>
316 template <int order>
317 Tensor<order, dim>
318 PolynomialSpace<dim>::compute_derivative(const unsigned int i,
319  const Point<dim> & p) const
320 {
321  const std::array<unsigned int, dim> indices = compute_index(i);
322 
323  ndarray<double, dim, order + 1> v;
324  {
325  std::vector<double> tmp(order + 1);
326  for (unsigned int d = 0; d < dim; ++d)
327  {
328  polynomials[indices[d]].value(p(d), tmp);
329  for (unsigned int j = 0; j < order + 1; ++j)
330  v[d][j] = tmp[j];
331  }
332  }
333 
334  Tensor<order, dim> derivative;
335  switch (order)
336  {
337  case 1:
338  {
339  Tensor<1, dim> &derivative_1 =
340  *reinterpret_cast<Tensor<1, dim> *>(&derivative);
341  for (unsigned int d = 0; d < dim; ++d)
342  {
343  derivative_1[d] = 1.;
344  for (unsigned int x = 0; x < dim; ++x)
345  {
346  unsigned int x_order = 0;
347  if (d == x)
348  ++x_order;
349 
350  derivative_1[d] *= v[x][x_order];
351  }
352  }
353 
354  return derivative;
355  }
356  case 2:
357  {
358  Tensor<2, dim> &derivative_2 =
359  *reinterpret_cast<Tensor<2, dim> *>(&derivative);
360  for (unsigned int d1 = 0; d1 < dim; ++d1)
361  for (unsigned int d2 = 0; d2 < dim; ++d2)
362  {
363  derivative_2[d1][d2] = 1.;
364  for (unsigned int x = 0; x < dim; ++x)
365  {
366  unsigned int x_order = 0;
367  if (d1 == x)
368  ++x_order;
369  if (d2 == x)
370  ++x_order;
371 
372  derivative_2[d1][d2] *= v[x][x_order];
373  }
374  }
375 
376  return derivative;
377  }
378  case 3:
379  {
380  Tensor<3, dim> &derivative_3 =
381  *reinterpret_cast<Tensor<3, dim> *>(&derivative);
382  for (unsigned int d1 = 0; d1 < dim; ++d1)
383  for (unsigned int d2 = 0; d2 < dim; ++d2)
384  for (unsigned int d3 = 0; d3 < dim; ++d3)
385  {
386  derivative_3[d1][d2][d3] = 1.;
387  for (unsigned int x = 0; x < dim; ++x)
388  {
389  unsigned int x_order = 0;
390  if (d1 == x)
391  ++x_order;
392  if (d2 == x)
393  ++x_order;
394  if (d3 == x)
395  ++x_order;
396 
397  derivative_3[d1][d2][d3] *= v[x][x_order];
398  }
399  }
400 
401  return derivative;
402  }
403  case 4:
404  {
405  Tensor<4, dim> &derivative_4 =
406  *reinterpret_cast<Tensor<4, dim> *>(&derivative);
407  for (unsigned int d1 = 0; d1 < dim; ++d1)
408  for (unsigned int d2 = 0; d2 < dim; ++d2)
409  for (unsigned int d3 = 0; d3 < dim; ++d3)
410  for (unsigned int d4 = 0; d4 < dim; ++d4)
411  {
412  derivative_4[d1][d2][d3][d4] = 1.;
413  for (unsigned int x = 0; x < dim; ++x)
414  {
415  unsigned int x_order = 0;
416  if (d1 == x)
417  ++x_order;
418  if (d2 == x)
419  ++x_order;
420  if (d3 == x)
421  ++x_order;
422  if (d4 == x)
423  ++x_order;
424 
425  derivative_4[d1][d2][d3][d4] *= v[x][x_order];
426  }
427  }
428 
429  return derivative;
430  }
431  default:
432  {
433  Assert(false, ExcNotImplemented());
434  return derivative;
435  }
436  }
