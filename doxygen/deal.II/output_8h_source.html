191  template <int dim>
192  inline void
193  GnuplotPatch::assemble(const DoFInfo<dim> &info)
194  {
195  const unsigned int np = info.n_quadrature_points();
196  const unsigned int nv = info.n_quadrature_values();
197  const unsigned int patch_dim =
198  (info.face_number == numbers::invalid_unsigned_int) ? dim : (dim - 1);
199  const unsigned int row_length = n_points;
200  // If patches are 1D, end the
201  // patch after a row, else end
202  // it after a square
203  const unsigned int row_length2 =
204  (patch_dim == 1) ? row_length : (row_length * row_length);
205 
206  // AssertDimension(np, Utilities::fixed_power<dim>(n_points));
207  AssertDimension(nv, n_vectors + dim);
208 
209 
210  for (unsigned int k = 0; k < np; ++k)
211  {
212  if (k % row_length == 0)
213  write_endl();
214  if (k % row_length2 == 0)
215  write_endl();
216 
217  for (unsigned int i = 0; i < nv; ++i)
218  {
219  write(info.quadrature_value(k, i));
220  write('\t');
221  }
222  write_endl();
223  }
224  }
