146  AssertIndexRange(ind, data.size());
147  auto el = std::move(data[ind]);
148  data.erase(data.begin() + ind);

159 template <typename T>
160 void
161 FiniteSizeHistory<T>::add(const T &element)
162 {
163  std::unique_ptr<T> new_el;
164  if (data.size() < max_n_elements)
165  // have not reached the maximum number of elements yet
166  {
167  if (cache.size() == 0)
168  // nothing is cached, just copy a given element
169  {
170  new_el = std::make_unique<T>(element);
171  }
172  else
173  // something is cached, take one element and copy
174  // the user provided one there.
175  {
176  new_el = std::move(cache.back());
177  (*new_el) = element;
178 
179  cache.pop_back(); // removes a pointer that is now a nullptr anyway
180  }
181  }
182  else
183  // we reached the maximum number of elements and
184  // thus have to re-order/cycle elements currently stored
185  {
186  new_el = std::move(data.back());
187  (*new_el) = element;
188 
189  data.pop_back(); // removes a pointer that is now a nullptr anyway
190  }
191 
192  // finally insert the new one where appropriate
193  data.push_front(std::move(new_el));
194 
195  // whatever we do, we shall not store more than the maximum number of
196  // elements
197  Assert(data.size() + cache.size() <= max_n_elements, ExcInternalError());
198 }
