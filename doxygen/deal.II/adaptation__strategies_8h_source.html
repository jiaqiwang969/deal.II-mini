238  value_type
239  check_equality(
240  const typename ::Triangulation<dim, spacedim>::cell_iterator &,
241  const std::vector<value_type> &children_values)
242  {
243  Assert(!children_values.empty(), ExcInternalError());
244 
245  const auto first_child = children_values.cbegin();
246  for (auto other_child = first_child + 1;
247  other_child != children_values.cend();
248  ++other_child)
249  Assert(*first_child == *other_child,
250  ExcMessage(
251  "Values on cells that will be coarsened are not equal!"));
252 
253  return *first_child;
254  }
