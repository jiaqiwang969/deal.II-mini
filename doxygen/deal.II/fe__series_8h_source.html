401  template <int dim, typename CoefficientType>
402  void
403  fill_map_index(
404  const Table<dim, CoefficientType> &coefficients,
405  const TableIndices<dim> & ind,
406  const std::function<
407  std::pair<bool, unsigned int>(const TableIndices<dim> &)> &predicate,
408  std::map<unsigned int, std::vector<CoefficientType>> &pred_to_values)
409  {
410  const std::pair<bool, unsigned int> pred_pair = predicate(ind);
411  // don't add a value if predicate is false
412  if (pred_pair.first == false)
413  return;
414 
415  const unsigned int pred_value = pred_pair.second;
416  const CoefficientType &coeff_value = coefficients(ind);
417  // If pred_value is not in the pred_to_values map, the element will be
418  // created. Otherwise a reference to the existing element is returned.
419  pred_to_values[pred_value].push_back(coeff_value);
420  }

502 template <int dim, typename CoefficientType>
503 std::pair<std::vector<unsigned int>, std::vector<double>>
504 FESeries::process_coefficients(
505  const Table<dim, CoefficientType> &coefficients,
506  const std::function<std::pair<bool, unsigned int>(const TableIndices<dim> &)>
507  & predicate,
508  const VectorTools::NormType norm_type,
509  const double smallest_abs_coefficient)
510 {
511  Assert(smallest_abs_coefficient >= 0.,
512  ExcMessage("smallest_abs_coefficient should be non-negative."));
513 
514  std::vector<unsigned int> predicate_values;
515  std::vector<double> norm_values;
516 
517  // first, parse all table elements into a map of predicate values and
518  // coefficients. We could have stored (predicate values ->TableIndicies) map,
519  // but its processing would have been much harder later on.
520  std::map<unsigned int, std::vector<CoefficientType>> pred_to_values;
521  internal::FESeriesImplementation::fill_map(coefficients,
522  predicate,
523  pred_to_values);
524 
525  // now go through the map and populate the @p norm_values based on @p norm:
526  for (const auto &pred_to_value : pred_to_values)
527  {
528  Vector<CoefficientType> values(pred_to_value.second.cbegin(),
529  pred_to_value.second.cend());
530 
531  double norm_value = 0;
532  switch (norm_type)
533  {
534  case VectorTools::L2_norm:
535  {
536  norm_value = values.l2_norm();
537  break;
538  }
539  case VectorTools::L1_norm:
540  {
541  norm_value = values.l1_norm();
542  break;
543  }
544  case VectorTools::Linfty_norm:
545  {
546  norm_value = values.linfty_norm();
547  break;
548  }
549  case VectorTools::mean:
550  {
551  norm_value = internal::FESeriesImplementation::complex_mean_value(
552  values.mean_value());
553  break;
554  }
555  default:
556  AssertThrow(false, ExcNotImplemented());
557  break;
558  }
559 
560  // will use all non-zero coefficients
561  if (std::abs(norm_value) > smallest_abs_coefficient)
562  {
563  predicate_values.push_back(pred_to_value.first);
564  norm_values.push_back(norm_value);
565  }
566  }
567 
568  return std::make_pair(predicate_values, norm_values);
569 }

603 template <int dim, int spacedim>
604 template <class Archive>
605 inline void
606 FESeries::Fourier<dim, spacedim>::load_transformation_matrices(
607  Archive &ar,
608  const unsigned int /*version*/ )
609 {
610  // Check whether the currently registered resources are compatible with
611  // the transformation matrices to load.
612  // mode vector
613  std::vector<unsigned int> compare_coefficients;
614  ar & compare_coefficients;
615  Assert(compare_coefficients == n_coefficients_per_direction,
616  ExcMessage("A different number of coefficients vector has been used "
617  "to generate the transformation matrices you are about "
618  "to load!"));
619 
620  // finite element collection
621  unsigned int size;
622  ar & size;
623  AssertDimension(size, fe_collection->size());
624  std::string name;
625  for (unsigned int i = 0; i < size; ++i)
626  {
627  ar &name;
628  Assert(name.compare((*fe_collection)[i].get_name()) == 0,
629  ExcMessage("A different FECollection has been used to generate "
630  "the transformation matrices you are about to load!"));
631  }
632 
633  // quadrature collection
634  ar &size;
635  AssertDimension(size, q_collection.size());
636  Quadrature<dim> quadrature;
637  for (unsigned int i = 0; i < size; ++i)
638  {
639  ar &quadrature;
640  Assert(quadrature == q_collection[i],
641  ExcMessage("A different QCollection has been used to generate "
642  "the transformation matrices you are about to load!"));
643  }
644 
645  // Restore the transform matrices since all prerequisites are fulfilled.
646  ar &fourier_transform_matrices;
647 }
