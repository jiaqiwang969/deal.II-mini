107  template <int dim, int spacedim>
108  LinearAlgebra::distributed::Vector<double>
109  FirstChildPolicy<dim, spacedim>::partition(
110  const Triangulation<dim, spacedim> &tria_coarse_in) const
111  {
112  const auto communicator = tria_coarse_in.get_communicator();
113 
114  const internal::CellIDTranslator<dim> cell_id_translator(n_coarse_cells,
115  n_global_levels);
116 
117  IndexSet is_coarse(cell_id_translator.size());
118 
119  for (const auto &cell : tria_coarse_in.active_cell_iterators())
120  if (cell->is_locally_owned())
121  is_coarse.add_index(cell_id_translator.translate(cell));
122 
123  std::vector<unsigned int> owning_ranks_of_coarse_cells(
124  is_coarse.n_elements());
125  {
126  Utilities::MPI::internal::ComputeIndexOwner::ConsensusAlgorithmsPayload
127  process(is_level_partitions,
128  is_coarse,
129  communicator,
130  owning_ranks_of_coarse_cells,
131  false);
132 
133  Utilities::MPI::ConsensusAlgorithms::Selector<
134  std::pair<types::global_cell_index, types::global_cell_index>,
135  unsigned int>
136  consensus_algorithm(process, communicator);
137  consensus_algorithm.run();
138  }
139 
140  const auto tria =
141  dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
142  &tria_coarse_in);
143 
144  Assert(tria, ExcNotImplemented());
145 
146  LinearAlgebra::distributed::Vector<double> partition(
147  tria->global_active_cell_index_partitioner().lock());
148 
149  for (const auto &cell : tria_coarse_in.active_cell_iterators())
150  if (cell->is_locally_owned())
151  partition[cell->global_active_cell_index()] =
152  owning_ranks_of_coarse_cells[is_coarse.index_within_set(
153  cell_id_translator.translate(cell))];
154 
155  return partition;
156  }

167  template <int dim, int spacedim>
168  LinearAlgebra::distributed::Vector<double>
169  MinimalGranularityPolicy<dim, spacedim>::partition(
170  const Triangulation<dim, spacedim> &tria_in) const
171  {
172  const auto tria =
173  dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
174  &tria_in);
175 
176  Assert(tria, ExcNotImplemented());
177 
178  LinearAlgebra::distributed::Vector<double> partition(
179  tria->global_active_cell_index_partitioner().lock());
180 
181  // step 1) check if all processes have enough cells
182 
183  unsigned int n_locally_owned_active_cells = 0;
184  for (const auto &cell : tria_in.active_cell_iterators())
185  if (cell->is_locally_owned())
186  ++n_locally_owned_active_cells;
187 
188  const auto comm = tria_in.get_communicator();
189 
190  if (Utilities::MPI::min(n_locally_owned_active_cells, comm) >= n_min_cells)
191  return {}; // all processes have enough cells
192 
193  // step 2) there are processes which do not have enough cells so that
194  // a repartitioning kicks in with the aim that all processes that own
195  // cells have at least the specified number of cells
196 
197  const unsigned int n_global_active_cells = tria_in.n_global_active_cells();
198 
199  const unsigned int n_partitions =
200  std::max<unsigned int>(1,
201  std::min(n_global_active_cells / n_min_cells,
202  Utilities::MPI::n_mpi_processes(comm)));
203 
204  const unsigned int min_cells = n_global_active_cells / n_partitions;
205 
206  const auto convert = [&](const unsigned int i) {
207  // determine the owner of a given index: we try to assign each process
208  // the same number of cells; if there is a remainder, we assign the
209  // first processes an additional cell (so that the difference of number of
210  // locally owned cells is never larger than one between processes).
211  if (i < (i / min_cells) * min_cells + (n_global_active_cells % min_cells))
212  return i / (min_cells + 1);
213  else
214  return (i - (n_global_active_cells % min_cells)) / min_cells;
215  };
216 
217  for (const auto i : partition.locally_owned_elements())
218  partition[i] = convert(i);
219 
220  return partition;
221  }
