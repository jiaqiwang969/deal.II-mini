658  template <typename number>
659  inline number
660  Polynomial<number>::value(const number x) const
661  {
662  if (in_lagrange_product_form == false)
663  {
664  Assert(coefficients.size() > 0, ExcEmptyObject());
665 
666  // Horner scheme
667  const unsigned int m = coefficients.size();
668  number value = coefficients.back();
669  for (int k = m - 2; k >= 0; --k)
670  value = value * x + coefficients[k];
671  return value;
672  }
673  else
674  {
675  // direct evaluation of Lagrange polynomial
676  const unsigned int m = lagrange_support_points.size();
677  number value = 1.;
678  for (unsigned int j = 0; j < m; ++j)
679  value *= x - lagrange_support_points[j];
680  value *= lagrange_weight;
681  return value;
682  }
683  }

687  template <typename number>
688  template <typename Number2>
689  inline void
690  Polynomial<number>::value(const Number2 x,
691  const unsigned int n_derivatives,
692  Number2 * values) const
693  {
694  // evaluate Lagrange polynomial and derivatives
695  if (in_lagrange_product_form == true)
696  {
697  // to compute the value and all derivatives of a polynomial of the
698  // form (x-x_1)*(x-x_2)*...*(x-x_n), expand the derivatives like
699  // automatic differentiation does.
700  const unsigned int n_supp = lagrange_support_points.size();
701  const number weight = lagrange_weight;
702  switch (n_derivatives)
703  {
704  default:
705  values[0] = 1.;
706  for (unsigned int d = 1; d <= n_derivatives; ++d)
707  values[d] = 0.;
708  for (unsigned int i = 0; i < n_supp; ++i)
709  {
710  const Number2 v = x - lagrange_support_points[i];
711 
712  // multiply by (x-x_i) and compute action on all derivatives,
713  // too (inspired from automatic differentiation: implement the
714  // product rule for the old value and the new variable 'v',
715  // i.e., expand value v and derivative one). since we reuse a
716  // value from the next lower derivative from the steps before,
717  // need to start from the highest derivative
718  for (unsigned int k = n_derivatives; k > 0; --k)
719  values[k] = (values[k] * v + values[k - 1]);
720  values[0] *= v;
721  }
722  // finally, multiply by the weight in the Lagrange
723  // denominator. Could be done instead of setting values[0] = 1
724  // above, but that gives different accumulation of round-off
725  // errors (multiplication is not associative) compared to when we
726  // computed the weight, and hence a basis function might not be
727  // exactly one at the center point, which is nice to have. We also
728  // multiply derivatives by k! to transform the product p_n =
729  // p^(n)(x)/k! into the actual form of the derivative
730  {
731  number k_factorial = 1;
732  for (unsigned int k = 0; k <= n_derivatives; ++k)
733  {
734  values[k] *= k_factorial * weight;
735  k_factorial *= static_cast<number>(k + 1);
736  }
737  }
738  break;
739 
740  // manually implement case 0 (values only), case 1 (value + first
741  // derivative), and case 2 (up to second derivative) since they
742  // might be called often. then, we can unroll the inner loop and
743  // keep the temporary results as local variables to help the
744  // compiler with the pointer aliasing analysis.
745  case 0:
746  {
747  Number2 value = 1.;
748  for (unsigned int i = 0; i < n_supp; ++i)
749  {
750  const Number2 v = x - lagrange_support_points[i];
751  value *= v;
752  }
753  values[0] = weight * value;
754  break;
755  }
756 
757  case 1:
758  {
759  Number2 value = 1.;
760  Number2 derivative = 0.;
761  for (unsigned int i = 0; i < n_supp; ++i)
762  {
763  const Number2 v = x - lagrange_support_points[i];
764  derivative = derivative * v + value;
765  value *= v;
766  }
767  values[0] = weight * value;
768  values[1] = weight * derivative;
769  break;
770  }
771 
772  case 2:
773  {
774  Number2 value = 1.;
775  Number2 derivative = 0.;
776  Number2 second = 0.;
777  for (unsigned int i = 0; i < n_supp; ++i)
778  {
779  const Number2 v = x - lagrange_support_points[i];
780  second = second * v + derivative;
781  derivative = derivative * v + value;
782  value *= v;
783  }
784  values[0] = weight * value;
785  values[1] = weight * derivative;
786  values[2] = static_cast<number>(2) * weight * second;
787  break;
788  }
789  }
790  return;
791  }
792 
793  Assert(coefficients.size() > 0, ExcEmptyObject());
794 
795  // if derivatives are needed, then do it properly by the full
796  // Horner scheme
797  const unsigned int m = coefficients.size();
798  std::vector<Number2> a(coefficients.size());
799  std::copy(coefficients.begin(), coefficients.end(), a.begin());
800  unsigned int j_factorial = 1;
801 
802  // loop over all requested derivatives. note that derivatives @p{j>m} are
803  // necessarily zero, as they differentiate the polynomial more often than
804  // the highest power is
805  const unsigned int min_valuessize_m = std::min(n_derivatives + 1, m);
806  for (unsigned int j = 0; j < min_valuessize_m; ++j)
807  {
808  for (int k = m - 2; k >= static_cast<int>(j); --k)
809  a[k] += x * a[k + 1];
810  values[j] = static_cast<number>(j_factorial) * a[j];
811 
812  j_factorial *= j + 1;
813  }
814 
815  // fill higher derivatives by zero
816  for (unsigned int j = min_valuessize_m; j <= n_derivatives; ++j)
817  values[j] = 0.;
818  }

837  template <typename Number>
838  Number
839  jacobi_polynomial_value(const unsigned int degree,
840  const int alpha,
841  const int beta,
842  const Number x)
843  {
844  Assert(alpha >= 0 && beta >= 0,
845  ExcNotImplemented("Negative alpha/beta coefficients not supported"));
846  // the Jacobi polynomial is evaluated using a recursion formula.
847  Number p0, p1;
848 
849  // The recursion formula is defined for the interval [-1, 1], so rescale
850  // to that interval here
851  const Number xeval = Number(-1) + 2. * x;
852 
853  // initial values P_0(x), P_1(x):
854  p0 = 1.0;
855  if (degree == 0)
856  return p0;
857  p1 = ((alpha + beta + 2) * xeval + (alpha - beta)) / 2;
858  if (degree == 1)
859  return p1;
860 
861  for (unsigned int i = 1; i < degree; ++i)
862  {
863  const Number v = 2 * i + (alpha + beta);
864  const Number a1 = 2 * (i + 1) * (i + (alpha + beta + 1)) * v;
865  const Number a2 = (v + 1) * (alpha * alpha - beta * beta);
866  const Number a3 = v * (v + 1) * (v + 2);
867  const Number a4 = 2 * (i + alpha) * (i + beta) * (v + 2);
868 
869  const Number pn = ((a2 + a3 * xeval) * p1 - a4 * p0) / a1;
870  p0 = p1;
871  p1 = pn;
872  }
873  return p1;
874  }

878  template <typename Number>
879  std::vector<Number>
880  jacobi_polynomial_roots(const unsigned int degree,
881  const int alpha,
882  const int beta)
883  {
884  std::vector<Number> x(degree, 0.5);
885 
886  // compute zeros with a Newton algorithm.
887 
888  // Set tolerance. For long double we might not always get the additional
889  // precision in a run time environment (e.g. with valgrind), so we must
890  // limit the tolerance to double. Since we do a Newton iteration, doing
891  // one more iteration after the residual has indicated convergence will be
892  // enough for all number types due to the quadratic convergence of
893  // Newton's method
894 
895  const Number tolerance =
896  4 * std::max(static_cast<Number>(std::numeric_limits<double>::epsilon()),
897  std::numeric_limits<Number>::epsilon());
898 
899  // The following implementation follows closely the one given in the
900  // appendix of the book by Karniadakis and Sherwin: Spectral/hp element
901  // methods for computational fluid dynamics (Oxford University Press,
902  // 2005)
903 
904  // If symmetric, we only need to compute the half of points
905  const unsigned int n_points = (alpha == beta ? degree / 2 : degree);
906  for (unsigned int k = 0; k < n_points; ++k)
907  {
908  // we take the zeros of the Chebyshev polynomial (alpha=beta=-0.5) as
909  // initial values, corrected by the initial value
910  Number r = 0.5 - 0.5 * std::cos(static_cast<Number>(2 * k + 1) /
911  (2 * degree) * numbers::PI);
912  if (k > 0)
913  r = (r + x[k - 1]) / 2;
914 
915  unsigned int converged = numbers::invalid_unsigned_int;
916  for (unsigned int it = 1; it < 1000; ++it)
917  {
918  Number s = 0.;
919  for (unsigned int i = 0; i < k; ++i)
920  s += 1. / (r - x[i]);
921 
922  // derivative of P_n^{alpha,beta}, rescaled to [0, 1]
923  const Number J_x =
924  (alpha + beta + degree + 1) *
925  jacobi_polynomial_value(degree - 1, alpha + 1, beta + 1, r);
926 
927  // value of P_n^{alpha,beta}
928  const Number f = jacobi_polynomial_value(degree, alpha, beta, r);
929  const Number delta = f / (f * s - J_x);
930  r += delta;
931  if (converged == numbers::invalid_unsigned_int &&
932  std::abs(delta) < tolerance)
933  converged = it;
934 
935  // do one more iteration to ensure accuracy also for tighter
936  // types than double (e.g. long double)
937  if (it == converged + 1)
938  break;
939  }
940 
941  Assert(converged != numbers::invalid_unsigned_int,
942  ExcMessage("Newton iteration for zero of Jacobi polynomial "
943  "did not converge."));
944 
945  x[k] = r;
946  }
947 
948  // in case we assumed symmetry, fill up the missing values
949  for (unsigned int k = n_points; k < degree; ++k)
950  x[k] = 1.0 - x[degree - k - 1];
951 
952  return x;
953  }
