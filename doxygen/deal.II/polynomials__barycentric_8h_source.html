459 template <int dim, typename Number>
460 void
461 BarycentricPolynomial<dim, Number>::print(std::ostream &out) const
462 {
463  const auto &coeffs = this->coefficients;
464  auto first = index_to_indices(0, coeffs.size());
465  bool print_plus = false;
466  if (coeffs(first) != Number())
467  {
468  out << coeffs(first);
469  print_plus = true;
470  }
471  for (std::size_t i = 1; i < coeffs.n_elements(); ++i)
472  {
473  const auto indices = index_to_indices(i, coeffs.size());
474  if (coeffs(indices) == Number())
475  continue;
476  if (print_plus)
477  out << " + ";
478  out << coeffs(indices);
479  for (unsigned int d = 0; d < dim + 1; ++d)
480  {
481  if (indices[d] != 0)
482  out << " * t" << d << '^' << indices[d];
483  }
484  print_plus = true;
485  }
486 
487  if (!print_plus)
488  out << Number();
489 }

608 template <int dim, typename Number>
609 BarycentricPolynomial<dim, Number> BarycentricPolynomial<dim, Number>::
610  operator*(const BarycentricPolynomial<dim, Number> &multiplicand) const
611 {
612  TableIndices<dim + 1> deg;
613  for (unsigned int d = 0; d < dim + 1; ++d)
614  {
615  deg[d] = multiplicand.degrees()[d] + degrees()[d];
616  }
617 
618  BarycentricPolynomial<dim, Number> result(deg, Number());
619 
620  const auto &coef_1 = this->coefficients;
621  const auto &coef_2 = multiplicand.coefficients;
622  auto & coef_out = result.coefficients;
623 
624  for (std::size_t i1 = 0; i1 < coef_1.n_elements(); ++i1)
625  {
626  const auto index_1 = index_to_indices(i1, coef_1.size());
627  for (std::size_t i2 = 0; i2 < coef_2.n_elements(); ++i2)
628  {
629  const auto index_2 = index_to_indices(i2, coef_2.size());
630 
631  TableIndices<dim + 1> index_out;
632  for (unsigned int d = 0; d < dim + 1; ++d)
633  index_out[d] = index_1[d] + index_2[d];
634  coef_out(index_out) += coef_1(index_1) * coef_2(index_2);
635  }
636  }
637 
638  return result;
639 }

643 template <int dim, typename Number>
644 BarycentricPolynomial<dim, Number>
645 BarycentricPolynomial<dim, Number>::barycentric_derivative(
646  const unsigned int coordinate) const
647 {
648  AssertIndexRange(coordinate, dim + 1);
649 
650  if (degrees()[coordinate] == 0)
651  return BarycentricPolynomial<dim, Number>();
652 
653  auto deg = degrees();
654  deg[coordinate] -= 1;
655  BarycentricPolynomial<dim, Number> result(deg,
656  std::numeric_limits<Number>::max());
657  const auto & coeffs_in = coefficients;
658  auto & coeffs_out = result.coefficients;
659  for (std::size_t i = 0; i < coeffs_out.n_elements(); ++i)
660  {
661  const auto out_index = index_to_indices(i, coeffs_out.size());
662  auto input_index = out_index;
663  input_index[coordinate] += 1;
664 
665  coeffs_out(out_index) = coeffs_in(input_index) * input_index[coordinate];
666  }
667 
668  return result;
669 }

684 template <int dim, typename Number>
685 Number
686 BarycentricPolynomial<dim, Number>::value(const Point<dim> &point) const
687 {
688  // TODO: this is probably not numerically stable for higher order.
689  // We really need some version of Horner's method.
690  Number result = {};
691 
692  // Begin by converting point (which is in Cartesian coordinates) to
693  // barycentric coordinates:
694  std::array<Number, dim + 1> b_point;
695  b_point[0] = 1.0;
696  for (unsigned int d = 0; d < dim; ++d)
697  {
698  b_point[0] -= point[d];
699  b_point[d + 1] = point[d];
700  }
701 
702  // Now evaluate the polynomial at the computed barycentric point:
703  for (std::size_t i = 0; i < coefficients.n_elements(); ++i)
704  {
705  const auto indices = index_to_indices(i, coefficients.size());
706  const auto coef = coefficients(indices);
707  if (coef == Number())
708  continue;
709 
710  auto temp = Number(1);
711  for (unsigned int d = 0; d < dim + 1; ++d)
712  temp *= std::pow(b_point[d], indices[d]);
713  result += coef * temp;
714  }
715 
716  return result;
717 }

728 BarycentricPolynomial<dim, Number>::index_to_indices(
