513 template <typename VectorType,
514  typename MatrixType1,
515  typename MatrixType2,
516  typename INVERSE>
517 inline void
518 ArpackSolver::solve(const MatrixType1 & /*system_matrix*/ ,
519  const MatrixType2 & mass_matrix,
520  const INVERSE & inverse,
521  std::vector<std::complex<double>> &eigenvalues,
522  std::vector<VectorType> & eigenvectors,
523  const unsigned int n_eigenvalues)
524 {
525  // Problem size
526  unsigned int n = eigenvectors[0].size();
527 
528  // Number of eigenvalues
529  const unsigned int nev_const =
530  (n_eigenvalues == 0) ? eigenvalues.size() : n_eigenvalues;
531  // nev for arpack, which might change by plus one during dneupd
532  unsigned int nev = nev_const;
533 
534  // check input sizes
535  if (additional_data.symmetric)
536  {
537  Assert(nev <= eigenvectors.size(),
538  ArpackExcInvalidEigenvectorSize(nev, eigenvectors.size()));
539  }
540  else
541  Assert(nev + 1 <= eigenvectors.size(),
542  ArpackExcInvalidEigenvectorSizeNonsymmetric(nev,
543  eigenvectors.size()));
544 
545  Assert(nev <= eigenvalues.size(),
546  ArpackExcInvalidEigenvalueSize(nev, eigenvalues.size()));
547 
548  // check large enough problem size
549  Assert(nev < n, ArpackExcInvalidNumberofEigenvalues(nev, n));
550 
551  Assert(additional_data.number_of_arnoldi_vectors < n,
552  ArpackExcInvalidNumberofArnoldiVectors(
553  additional_data.number_of_arnoldi_vectors, n));
554 
555  // check whether we have enough Arnoldi vectors
556  Assert(additional_data.number_of_arnoldi_vectors > 2 * nev + 1,
557  ArpackExcSmallNumberofArnoldiVectors(
558  additional_data.number_of_arnoldi_vectors, nev));
559 
560  // ARPACK mode for dsaupd/dnaupd, here only mode 3, i.e. shift-invert mode
561  int mode = 3;
562 
563  // reverse communication parameter
564  int ido = 0;
565 
566  // 'G' generalized eigenvalue problem 'I' standard eigenvalue problem
567  char bmat[2] = "G";
568 
569  // Specify the eigenvalues of interest, possible parameters "LA" algebraically
570  // largest "SA" algebraically smallest "LM" largest magnitude "SM" smallest
571  // magnitude "LR" largest real part "SR" smallest real part "LI" largest
572  // imaginary part "SI" smallest imaginary part "BE" both ends of spectrum
573  // simultaneous.
574  char which[3];
575  switch (additional_data.eigenvalue_of_interest)
576  {
577  case algebraically_largest:
578  std::strcpy(which, "LA");
579  break;
580  case algebraically_smallest:
581  std::strcpy(which, "SA");
582  break;
583  case largest_magnitude:
584  std::strcpy(which, "LM");
585  break;
586  case smallest_magnitude:
587  std::strcpy(which, "SM");
588  break;
589  case largest_real_part:
590  std::strcpy(which, "LR");
591  break;
592  case smallest_real_part:
593  std::strcpy(which, "SR");
594  break;
595  case largest_imaginary_part:
596  std::strcpy(which, "LI");
597  break;
598  case smallest_imaginary_part:
599  std::strcpy(which, "SI");
600  break;
601  case both_ends:
602  std::strcpy(which, "BE");
603  break;
604  }
605 
606  // tolerance for ARPACK
607  double tol = control().tolerance();
608 
609  // if the starting vector is used it has to be in resid
610  if (!initial_vector_provided || resid.size() != n)
611  resid.resize(n, 1.);
612 
613  // number of Arnoldi basis vectors specified
614  // in additional_data
615  int ncv = additional_data.number_of_arnoldi_vectors;
616 
617  int ldv = n;
618  std::vector<double> v(ldv * ncv, 0.0);
619 
620  // information to the routines
621  std::vector<int> iparam(11, 0);
622 
623  iparam[0] = 1; // shift strategy
624 
625  // maximum number of iterations
626  iparam[2] = control().max_steps();
627 
628  // Set the mode of dsaupd. 1 is exact shifting, 2 is user-supplied shifts,
629  // 3 is shift-invert mode, 4 is buckling mode, 5 is Cayley mode.
630 
631  iparam[6] = mode;
632  std::vector<int> ipntr(14, 0);
633 
634  // work arrays for ARPACK
635  std::vector<double> workd(3 * n, 0.);
636  int lworkl =
637  additional_data.symmetric ? ncv * ncv + 8 * ncv : 3 * ncv * ncv + 6 * ncv;
638  std::vector<double> workl(lworkl, 0.);
639 
640  // information out of the iteration
641  int info = 1;
642 
643  while (ido != 99)
644  {
645  // call of ARPACK dsaupd/dnaupd routine
646  if (additional_data.symmetric)
647  dsaupd_(&ido,
648  bmat,
649  &n,
650  which,
651  &nev,
652  &tol,
653  resid.data(),
654  &ncv,
655  v.data(),
656  &ldv,
657  iparam.data(),
658  ipntr.data(),
659  workd.data(),
660  workl.data(),
661  &lworkl,
662  &info);
663  else
664  dnaupd_(&ido,
665  bmat,
666  &n,
667  which,
668  &nev,
669  &tol,
670  resid.data(),
671  &ncv,
672  v.data(),
673  &ldv,
674  iparam.data(),
675  ipntr.data(),
676  workd.data(),
677  workl.data(),
678  &lworkl,
679  &info);
680 
681  if (ido == 99)
682  break;
683 
684  switch (mode)
685  {
686  case 3:
687  {
688  switch (ido)
689  {
690  case -1:
691  {
692  VectorType src, dst, tmp;
693  src.reinit(eigenvectors[0]);
694  dst.reinit(src);
695  tmp.reinit(src);
696 
697 
698  for (size_type i = 0; i < src.size(); ++i)
699  src(i) = workd[ipntr[0] - 1 + i];
700 
701  // multiplication with mass matrix M
702  mass_matrix.vmult(tmp, src);
703  // solving linear system
704  inverse.vmult(dst, tmp);
705 
706  for (size_type i = 0; i < dst.size(); ++i)
707  workd[ipntr[1] - 1 + i] = dst(i);
708  }
709  break;
710 
711  case 1:
712  {
713  VectorType src, dst, tmp, tmp2;
714  src.reinit(eigenvectors[0]);
715  dst.reinit(src);
716  tmp.reinit(src);
717  tmp2.reinit(src);
718 
719  for (size_type i = 0; i < src.size(); ++i)
720  {
721  src(i) = workd[ipntr[2] - 1 + i];
722  tmp(i) = workd[ipntr[0] - 1 + i];
723  }
724  // solving linear system
725  inverse.vmult(dst, src);
726 
727  for (size_type i = 0; i < dst.size(); ++i)
728  workd[ipntr[1] - 1 + i] = dst(i);
729  }
730  break;
731 
732  case 2:
733  {
734  VectorType src, dst;
735  src.reinit(eigenvectors[0]);
736  dst.reinit(src);
737 
738  for (size_type i = 0; i < src.size(); ++i)
739  src(i) = workd[ipntr[0] - 1 + i];
740 
741  // Multiplication with mass matrix M
742  mass_matrix.vmult(dst, src);
743 
744  for (size_type i = 0; i < dst.size(); ++i)
745  workd[ipntr[1] - 1 + i] = dst(i);
746  }
747  break;
748 
749  default:
750  Assert(false, ArpackExcArpackIdo(ido));
751  break;
752  }
753  }
754  break;
755  default:
756  Assert(false, ArpackExcArpackMode(mode));
757  break;
758  }
759  }
760 
761  // Set number of used iterations in SolverControl
762  control().check(iparam[2], 0.);
763 
764  if (info < 0)
765  {
766  if (additional_data.symmetric)
767  {
768  Assert(false, ArpackExcArpackInfodsaupd(info));
769  }
770  else
771  Assert(false, ArpackExcArpackInfodnaupd(info));
772  }
773  else
774  {
775  // 1 - compute eigenvectors, 0 - only eigenvalues
776  int rvec = 1;
777 
778  // which eigenvectors
779  char howmany = 'A';
780 
781  std::vector<int> select(ncv, 1);
782 
783  int ldz = n;
784 
785  std::vector<double> eigenvalues_real(nev + 1, 0.);
786  std::vector<double> eigenvalues_im(nev + 1, 0.);
787 
788  // call of ARPACK dseupd/dneupd routine
789  if (additional_data.symmetric)
790  {
791  std::vector<double> z(ldz * nev, 0.);
792  dseupd_(&rvec,
793  &howmany,
794  select.data(),
795  eigenvalues_real.data(),
796  z.data(),
797  &ldz,
798  &sigmar,
799  bmat,
800  &n,
801  which,
802  &nev,
803  &tol,
804  resid.data(),
805  &ncv,
806  v.data(),
807  &ldv,
808  iparam.data(),
809  ipntr.data(),
810  workd.data(),
811  workl.data(),
812  &lworkl,
813  &info);
814  }
815  else
816  {
817  std::vector<double> workev(3 * ncv, 0.);
818  dneupd_(&rvec,
819  &howmany,
820  select.data(),
821  eigenvalues_real.data(),
822  eigenvalues_im.data(),
823  v.data(),
824  &ldz,
825  &sigmar,
826  &sigmai,
827  workev.data(),
828  bmat,
829  &n,
830  which,
831  &nev,
832  &tol,
833  resid.data(),
834  &ncv,
835  v.data(),
836  &ldv,
837  iparam.data(),
838  ipntr.data(),
839  workd.data(),
840  workl.data(),
841  &lworkl,
842  &info);
843  }
844 
845  if (info == 1)
846  {
847  Assert(false, ArpackExcArpackInfoMaxIt(control().max_steps()));
848  }
849  else if (info == 3)
850  {
851  Assert(false, ArpackExcArpackNoShifts());
852  }
853  else if (info != 0)
854  {
855  if (additional_data.symmetric)
856  {
857  Assert(false, ArpackExcArpackInfodseupd(info));
858  }
859  else
860  Assert(false, ArpackExcArpackInfodneupd(info));
861  }
862 
863  for (unsigned int i = 0; i < nev; ++i)
864  for (unsigned int j = 0; j < n; ++j)
865  eigenvectors[i](j) = v[i * n + j];
866 
867  for (unsigned int i = 0; i < nev_const; ++i)
868  eigenvalues[i] =
869  std::complex<double>(eigenvalues_real[i], eigenvalues_im[i]);
870  }
871 }
