214  const Iterator & end,

216  const unsigned int chunk_size,

246  virtual void *
247  operator()(void *) override
248  {
249  // find first unused item. we know that there must be one
250  // because we have set the maximal number of tokens in flight
251  // and have set the ring buffer to have exactly this size. so
252  // if this function is called, we know that less than the
253  // maximal number of items in currently in flight
254  //
255  // note that we need not lock access to this array since
256  // the current stage is run sequentially and we can therefore
257  // enter the following block only once at any given time.
258  // thus, there can be no race condition between checking that
259  // a flag is false and setting it to true. (there may be
260  // another thread where we release items and set 'false'
261  // flags to 'true', but that too does not produce any
262  // problems)
263  ItemType *current_item = nullptr;
264  for (unsigned int i = 0; i < item_buffer.size(); ++i)
265  if (item_buffer[i].currently_in_use == false)
266  {
267  item_buffer[i].currently_in_use = true;
268  current_item = &item_buffer[i];
269  break;
270  }
271  Assert(current_item != nullptr,
272  ExcMessage("This can't be. There must be a free item!"));
273 
274  // initialize the next item. it may
275  // consist of at most chunk_size
276  // elements
277  current_item->n_items = 0;
278  while ((remaining_iterator_range.first !=
279  remaining_iterator_range.second) &&
280  (current_item->n_items < chunk_size))
281  {
282  current_item->work_items[current_item->n_items] =
283  remaining_iterator_range.first;
284 
285  ++remaining_iterator_range.first;
286  ++current_item->n_items;
287  }
288 
289  if (current_item->n_items == 0)
290  // there were no items
291  // left. terminate the pipeline
292  return nullptr;
293  else
294  return current_item;
295  }

364  void *
365  operator()(void *item) override
366  {
367  // first unpack the current item
368  using ItemType =
369  typename IteratorRangeToItemStream<Iterator,
370  ScratchData,
371  CopyData>::ItemType;
372 
373  ItemType *current_item = static_cast<ItemType *>(item);
374 
375  // we need to find an unused scratch data object in the list that
376  // corresponds to the current thread and then mark it as used. if
377  // we can't find one, create one
378  //
379  // as discussed in the discussion of the documentation of the
380  // IteratorRangeToItemStream::scratch_data variable, there is no
381  // need to synchronize access to this variable using a mutex
382  // as long as we have no yield-point in between. this means that
383  // we can't take an iterator into the list now and expect it to
384  // still be valid after calling the worker, but we at least do
385  // not have to lock the following section
386  ScratchData *scratch_data = nullptr;
387  {
388  typename ItemType::ScratchDataList &scratch_data_list =
389  current_item->scratch_data->get();
390 
391  // see if there is an unused object. if so, grab it and mark
392  // it as used
393  for (typename ItemType::ScratchDataList::iterator p =
394  scratch_data_list.begin();
395  p != scratch_data_list.end();
396  ++p)
397  if (p->currently_in_use == false)
398  {
399  scratch_data = p->scratch_data.get();
400  p->currently_in_use = true;
401  break;
402  }
403 
404  // if no object was found, create one and mark it as used
405  if (scratch_data == nullptr)
406  {
407  scratch_data =
408  new ScratchData(*current_item->sample_scratch_data);
409 
410  typename ItemType::ScratchDataList::value_type
411  new_scratch_object(scratch_data, true);
412  scratch_data_list.push_back(std::move(new_scratch_object));
413  }
414  }
415 
416  // then call the worker function on each element of the chunk we were
417  // given. since these worker functions are called on separate threads,
418  // nothing good can happen if they throw an exception and we are best
419  // off catching it and showing an error message
420  for (unsigned int i = 0; i < current_item->n_items; ++i)
421  {
422  try
423  {
424  if (worker)
425  worker(current_item->work_items[i],
426  *scratch_data,
427  current_item->copy_datas[i]);
428  }
429  catch (const std::exception &exc)
430  {
431  Threads::internal::handle_std_exception(exc);
432  }
433  catch (...)
434  {
435  Threads::internal::handle_unknown_exception();
436  }
437  }
438 
439  // finally mark the scratch object as unused again. as above, there
440  // is no need to lock anything here since the object we work on
441  // is thread-local
442  {
443  typename ItemType::ScratchDataList &scratch_data_list =
444  current_item->scratch_data->get();
445 
446  for (typename ItemType::ScratchDataList::iterator p =
447  scratch_data_list.begin();
448  p != scratch_data_list.end();
449  ++p)
450  if (p->scratch_data.get() == scratch_data)
451  {
452  Assert(p->currently_in_use == true, ExcInternalError());
453  p->currently_in_use = false;
454  }
455  }
456 
457  // if there is no copier, mark current item as usable again
458  if (copier_exist == false)
459  current_item->currently_in_use = false;
460 
461 
462  // then return the original pointer
463  // to the now modified object
464  return item;
465  }

771  {
772  // we need to find an unused scratch and corresponding copy
773  // data object in the list that corresponds to the current
774  // thread and then mark it as used. If we can't find one,
775  // create one as discussed in the discussion of the documentation
776  // of the IteratorRangeToItemStream::scratch_data variable,
777  // there is no need to synchronize access to this variable
778  // using a mutex as long as we have no yield-point in between.
779  // This means that we can't take an iterator into the list
780  // now and expect it to still be valid after calling the worker,
781  // but we at least do not have to lock the following section.
782  ScratchData *scratch_data = nullptr;
783  CopyData * copy_data = nullptr;
784  {
785  ScratchAndCopyDataList &scratch_and_copy_data_list = data.get();
786 
787  // see if there is an unused object. if so, grab it and mark
788  // it as used
789  for (typename ScratchAndCopyDataList::iterator p =
790  scratch_and_copy_data_list.begin();
791  p != scratch_and_copy_data_list.end();
792  ++p)
793  if (p->currently_in_use == false)
794  {
795  scratch_data = p->scratch_data.get();
796  copy_data = p->copy_data.get();
797  p->currently_in_use = true;
798  break;
799  }
800 
801  // if no element in the list was found, create one and mark it as
802  // used
803  if (scratch_data == nullptr)
804  {
805  Assert(copy_data == nullptr, ExcInternalError());
806 
807  scratch_and_copy_data_list.emplace_back(
808  std::make_unique<ScratchData>(sample_scratch_data),
809  std::make_unique<CopyData>(sample_copy_data),
810  true);
811  scratch_data =
812  scratch_and_copy_data_list.back().scratch_data.get();
813  copy_data = scratch_and_copy_data_list.back().copy_data.get();
814  }
815  }
816

818  // element of the chunk we were given.
819  for (typename std::vector<Iterator>::const_iterator p = range.begin();
820  p != range.end();
821  ++p)
822  {
823  try
824  {
825  if (worker)
826  worker(*p, *scratch_data, *copy_data);
827  if (copier)
828  copier(*copy_data);
829  }
830  catch (const std::exception &exc)

836  Threads::internal::handle_unknown_exception();

1014  Assert(chunk_size > 0, ExcMessage("The chunk_size must be at least one."));
1015  (void)chunk_size; // removes -Wunused-parameter warning in optimized mode

1040  // embarrassingly parallel problem where we can
1041  // essentially apply parallel_for. because parallel_for
1042  // requires subdividing the range for which operator- is
1043  // necessary between iterators, it is often inefficient to
1044  // apply it directly to cell ranges and similar iterator

1047  // iterators into a large array and use operator- on
1048  // iterators to this array of iterators.

1052  // just defer to that place

1054  for (Iterator p = begin; p != end; ++p)
1055  all_iterators[0].push_back(p);

1058  worker,
1059  copier,
1060  sample_scratch_data,
1061  sample_copy_data,
1062  queue_length,
