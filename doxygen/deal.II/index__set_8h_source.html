1385 inline std::ptrdiff_t
1386 IndexSet::ElementIterator::
1387 operator-(const IndexSet::ElementIterator &other) const
1388 {
1389  Assert(index_set == other.index_set,
1390  ExcMessage(
1391  "Can not compare iterators belonging to different IndexSets"));
1392  if (*this == other)
1393  return 0;
1394  if (!(*this < other))
1395  return -(other - *this);
1396 
1397  // only other can be equal to end() because of the checks above.
1398  Assert(is_valid(), ExcInternalError());
1399 
1400  // Note: we now compute how far advance *this in "*this < other" to get other,
1401  // so we need to return -c at the end.
1402 
1403  // first finish the current range:
1404  std::ptrdiff_t c = index_set->ranges[range_idx].end - idx;
1405 
1406  // now walk in steps of ranges (need to start one behind our current one):
1407  for (size_type range = range_idx + 1;
1408  range < index_set->ranges.size() && range <= other.range_idx;
1409  ++range)
1410  c += index_set->ranges[range].end - index_set->ranges[range].begin;
1411 
1412  Assert(
1413  other.range_idx < index_set->ranges.size() ||

1513 {

1516

1518  return end();
1519

1521  ranges.begin() + largest_range;
1522

1525  // the largest range first.

1527  if (global_index < main_range->begin)

1529  range_begin = ranges.begin();

1531  }
1532  else
1533  {
1534  range_begin = main_range;
1535  range_end = ranges.end();
1536  }
1537

1539  // a binary search

1541  Utilities::lower_bound(range_begin, range_end, r, Range::end_compare);
1542

1544  // global_index and also no range behind it, meaning we need to return end().
1545  if (p == ranges.end())
1546  return end();

1549  // which means we need to return an iterator to a because global_index, ...,
1550  // a-1 is not in the IndexSet (if branch). Alternatively, global_index is in
1551  // [a,b[ and we will return an iterator pointing directly at global_index
1552  // (else branch).
1553  if (global_index < p->begin)
1554  return {this, static_cast<size_type>(p - ranges.begin()), p->begin};
1555  else

1683 template <typename ForwardIterator>
1684 inline void
1685 IndexSet::add_indices(const ForwardIterator &begin, const ForwardIterator &end)
1686 {
1687  if (begin == end)
1688  return;
1689 
1690  // identify ranges in the given iterator range by checking whether some
1691  // indices happen to be consecutive. to avoid quadratic complexity when
1692  // calling add_range many times (as add_range() going into the middle of an
1693  // already existing range must shift entries around), we first collect a
1694  // vector of ranges.
1695  std::vector<std::pair<size_type, size_type>> tmp_ranges;
1696  bool ranges_are_sorted = true;
1697  for (ForwardIterator p = begin; p != end;)
1698  {
1699  const size_type begin_index = *p;
1700  size_type end_index = begin_index + 1;
1701  ForwardIterator q = p;
1702  ++q;
1703  while ((q != end) && (*q == end_index))
1704  {
1705  ++end_index;
1706  ++q;
1707  }
1708 
1709  tmp_ranges.emplace_back(begin_index, end_index);
1710  p = q;
1711 
1712  // if the starting index of the next go-around of the for loop is less
1713  // than the end index of the one just identified, then we will have at
1714  // least one pair of ranges that are not sorted, and consequently the
1715  // whole collection of ranges is not sorted.
1716  if (p != end && *p < end_index)
1717  ranges_are_sorted = false;
1718  }
1719 
1720  if (!ranges_are_sorted)
1721  std::sort(tmp_ranges.begin(), tmp_ranges.end());
1722 
1723  // if we have many ranges, we first construct a temporary index set (where
1724  // we add ranges in a consecutive way, so fast), otherwise, we work with
1725  // add_range(). the number 9 is chosen heuristically given the fact that
1726  // there are typically up to 8 independent ranges when adding the degrees of
1727  // freedom on a 3D cell or 9 when adding degrees of freedom of faces. if
1728  // doing cell-by-cell additions, we want to avoid repeated calls to
1729  // IndexSet::compress() which gets called upon merging two index sets, so we
1730  // want to be in the other branch then.
1731  if (tmp_ranges.size() > 9)
1732  {
1733  IndexSet tmp_set(size());
1734  tmp_set.ranges.reserve(tmp_ranges.size());
1735  for (const auto &i : tmp_ranges)
1736  tmp_set.add_range(i.first, i.second);
1737  this->add_indices(tmp_set);
1738  }
1739  else
1740  for (const auto &i : tmp_ranges)
1741  add_range(i.first, i.second);
1742 }

1746 inline bool
1747 IndexSet::is_element(const size_type index) const
1748 {
1749  if (ranges.empty() == false)
1750  {
1751  compress();
1752 
1753  // fast check whether the index is in the largest range
1754  Assert(largest_range < ranges.size(), ExcInternalError());
1755  if (index >= ranges[largest_range].begin &&
1756  index < ranges[largest_range].end)
1757  return true;
1758 
1759  // get the element after which we would have to insert a range that
1760  // consists of all elements from this element to the end of the index
1761  // range plus one. after this call we know that if p!=end() then
1762  // p->begin<=index unless there is no such range at all
1763  //
1764  // if the searched for element is an element of this range, then we're
1765  // done. otherwise, the element can't be in one of the following ranges
1766  // because otherwise p would be a different iterator
1767  //
1768  // since we already know the position relative to the largest range (we
1769  // called compress!), we can perform the binary search on ranges with
1770  // lower/higher number compared to the largest range
1771  std::vector<Range>::const_iterator p = std::upper_bound(
1772  ranges.begin() +
1773  (index < ranges[largest_range].begin ? 0 : largest_range + 1),
1774  index < ranges[largest_range].begin ? ranges.begin() + largest_range :
1775  ranges.end(),
1776  Range(index, size() + 1));
1777 
1778  if (p == ranges.begin())
1779  return ((index >= p->begin) && (index < p->end));
1780 
1781  Assert((p == ranges.end()) || (p->begin > index), ExcInternalError());
1782 
1783  // now move to that previous range
1784  --p;
1785  Assert(p->begin <= index, ExcInternalError());
1786 
1787  return (p->end > index);
1788  }
1789 
1790  // didn't find this index, so it's not in the set
1791  return false;
1792 }
