1284  template <typename T>
1285  std::map<unsigned int, T>
1286  some_to_some(const MPI_Comm & comm,
1287  const std::map<unsigned int, T> &objects_to_send)
1288  {
1289 # ifndef DEAL_II_WITH_MPI
1290  (void)comm;
1291  Assert(objects_to_send.size() < 2,
1292  ExcMessage("Cannot send to more than one processor."));
1293  Assert(objects_to_send.find(0) != objects_to_send.end() ||
1294  objects_to_send.size() == 0,
1295  ExcMessage("Can only send to myself or to nobody."));
1296  return objects_to_send;
1297 # else
1298  const auto my_proc = this_mpi_process(comm);
1299 
1300  std::map<unsigned int, T> received_objects;
1301 
1302  std::vector<unsigned int> send_to;
1303  send_to.reserve(objects_to_send.size());
1304  for (const auto &m : objects_to_send)
1305  if (m.first == my_proc)
1306  received_objects[my_proc] = m.second;
1307  else
1308  send_to.emplace_back(m.first);
1309 
1310  const unsigned int n_point_point_communications =
1311  Utilities::MPI::compute_n_point_to_point_communications(comm, send_to);
1312 
1313  // Protect the following communication:
1314  static CollectiveMutex mutex;
1315  CollectiveMutex::ScopedLock lock(mutex, comm);
1316 
1317  // If we have something to send, or we expect something from other
1318  // processors, we need to visit one of the two scopes below. Otherwise,
1319  // no other action is required by this mpi process, and we can safely
1320  // return.
1321  if (send_to.size() == 0 && n_point_point_communications == 0)
1322  return received_objects;
1323 
1324  const int mpi_tag =
1325  internal::Tags::compute_point_to_point_communication_pattern;
1326 
1327  // Sending buffers
1328  std::vector<std::vector<char>> buffers_to_send(send_to.size());
1329  std::vector<MPI_Request> buffer_send_requests(send_to.size());
1330  {
1331  unsigned int i = 0;
1332  for (const auto &rank_obj : objects_to_send)
1333  if (rank_obj.first != my_proc)
1334  {
1335  const auto &rank = rank_obj.first;
1336  buffers_to_send[i] = Utilities::pack(rank_obj.second,
1337  /*allow_compression=*/false);
1338  const int ierr = MPI_Isend(buffers_to_send[i].data(),
1339  buffers_to_send[i].size(),
1340  MPI_CHAR,
1341  rank,
1342  mpi_tag,
1343  comm,
1344  &buffer_send_requests[i]);
1345  AssertThrowMPI(ierr);
1346  ++i;
1347  }
1348  }
1349 
1350  // Fill the output map
1351  {
1352  std::vector<char> buffer;
1353  // We do this on a first come/first served basis
1354  for (unsigned int i = 0; i < n_point_point_communications; ++i)
1355  {
1356  // Probe what's going on. Take data from the first available sender
1357  MPI_Status status;
1358  int ierr = MPI_Probe(MPI_ANY_SOURCE, mpi_tag, comm, &status);
1359  AssertThrowMPI(ierr);
1360 
1361  // Length of the message
1362  int len;
1363  ierr = MPI_Get_count(&status, MPI_CHAR, &len);
1364  AssertThrowMPI(ierr);
1365  buffer.resize(len);
1366 
1367  // Source rank
1368  const unsigned int rank = status.MPI_SOURCE;
1369 
1370  // Actually receive the message
1371  ierr = MPI_Recv(buffer.data(),
1372  len,
1373  MPI_CHAR,
1374  status.MPI_SOURCE,
1375  status.MPI_TAG,
1376  comm,
1377  MPI_STATUS_IGNORE);
1378  AssertThrowMPI(ierr);
1379  Assert(received_objects.find(rank) == received_objects.end(),
1380  ExcInternalError(
1381  "I should not receive again from this rank"));
1382  received_objects[rank] =
1383  Utilities::unpack<T>(buffer,
1384  /*allow_compression=*/false);
1385  }
1386  }
1387 
1388  // Wait to have sent all objects.
1389  const int ierr = MPI_Waitall(send_to.size(),
1390  buffer_send_requests.data(),
1391  MPI_STATUSES_IGNORE);
1392  AssertThrowMPI(ierr);
1393 
1394  return received_objects;
1395 # endif // deal.II with MPI
1396  }

1398  template <typename T>
1399  std::vector<T>
1400  all_gather(const MPI_Comm &comm, const T &object)
1401  {
1402  if (job_supports_mpi() == false)
1403  return {object};
1404 
1405 # ifndef DEAL_II_WITH_MPI
1406  (void)comm;
1407  std::vector<T> v(1, object);
1408  return v;
1409 # else
1410  const auto n_procs = ::Utilities::MPI::n_mpi_processes(comm);
1411 
1412  std::vector<char> buffer = Utilities::pack(object);
1413 
1414  int n_local_data = buffer.size();
1415 
1416  // Vector to store the size of loc_data_array for every process
1417  std::vector<int> size_all_data(n_procs, 0);
1418 
1419  // Exchanging the size of each buffer
1420  int ierr = MPI_Allgather(
1421  &n_local_data, 1, MPI_INT, size_all_data.data(), 1, MPI_INT, comm);
1422  AssertThrowMPI(ierr);
1423 
1424  // Now computing the displacement, relative to recvbuf,
1425  // at which to store the incoming buffer
1426  std::vector<int> rdispls(n_procs);
1427  rdispls[0] = 0;
1428  for (unsigned int i = 1; i < n_procs; ++i)
1429  rdispls[i] = rdispls[i - 1] + size_all_data[i - 1];
1430 
1431  // Step 3: exchange the buffer:
1432  std::vector<char> received_unrolled_buffer(rdispls.back() +
1433  size_all_data.back());
1434 
1435  ierr = MPI_Allgatherv(buffer.data(),
1436  n_local_data,
1437  MPI_CHAR,
1438  received_unrolled_buffer.data(),
1439  size_all_data.data(),
1440  rdispls.data(),
1441  MPI_CHAR,
1442  comm);
1443  AssertThrowMPI(ierr);
1444 
1445  std::vector<T> received_objects(n_procs);
1446  for (unsigned int i = 0; i < n_procs; ++i)
1447  {
1448  std::vector<char> local_buffer(received_unrolled_buffer.begin() +
1449  rdispls[i],
1450  received_unrolled_buffer.begin() +
1451  rdispls[i] + size_all_data[i]);
1452  received_objects[i] = Utilities::unpack<T>(local_buffer);
1453  }
1454 
1455  return received_objects;
1456 # endif
1457  }

1459  template <typename T>
1460  std::vector<T>
1461  gather(const MPI_Comm & comm,
1462  const T & object_to_send,
1463  const unsigned int root_process)
1464  {
1465 # ifndef DEAL_II_WITH_MPI
1466  (void)comm;
1467  (void)root_process;
1468  std::vector<T> v(1, object_to_send);
1469  return v;
1470 # else
1471  const auto n_procs = ::Utilities::MPI::n_mpi_processes(comm);
1472  const auto my_rank = ::Utilities::MPI::this_mpi_process(comm);
1473 
1474  AssertIndexRange(root_process, n_procs);
1475 
1476  std::vector<char> buffer = Utilities::pack(object_to_send);
1477  int n_local_data = buffer.size();
1478 
1479  // Vector to store the size of loc_data_array for every process
1480  // only the root process needs to allocate memory for that purpose
1481  std::vector<int> size_all_data;
1482  if (my_rank == root_process)
1483  size_all_data.resize(n_procs, 0);
1484 
1485  // Exchanging the size of each buffer
1486  int ierr = MPI_Gather(&n_local_data,
1487  1,
1488  MPI_INT,
1489  size_all_data.data(),
1490  1,
1491  MPI_INT,
1492  root_process,
1493  comm);
1494  AssertThrowMPI(ierr);
1495 
1496  // Now computing the displacement, relative to recvbuf,
1497  // at which to store the incoming buffer; only for root
1498  std::vector<int> rdispls;
1499  if (my_rank == root_process)
1500  {
1501  rdispls.resize(n_procs, 0);
1502  for (unsigned int i = 1; i < n_procs; ++i)
1503  rdispls[i] = rdispls[i - 1] + size_all_data[i - 1];
1504  }
1505  // exchange the buffer:
1506  std::vector<char> received_unrolled_buffer;
1507  if (my_rank == root_process)
1508  received_unrolled_buffer.resize(rdispls.back() + size_all_data.back());
1509 
1510  ierr = MPI_Gatherv(buffer.data(),
1511  n_local_data,
1512  MPI_CHAR,
1513  received_unrolled_buffer.data(),
1514  size_all_data.data(),
1515  rdispls.data(),
1516  MPI_CHAR,
1517  root_process,
1518  comm);
1519  AssertThrowMPI(ierr);
1520 
1521  std::vector<T> received_objects;
1522 
1523  if (my_rank == root_process)
1524  {
1525  received_objects.resize(n_procs);
1526 
1527  for (unsigned int i = 0; i < n_procs; ++i)
1528  {
1529  const std::vector<char> local_buffer(
1530  received_unrolled_buffer.begin() + rdispls[i],
1531  received_unrolled_buffer.begin() + rdispls[i] +
1532  size_all_data[i]);
1533  received_objects[i] = Utilities::unpack<T>(local_buffer);
1534  }
1535  }
1536  return received_objects;
