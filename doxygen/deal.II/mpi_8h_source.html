1047  template <typename T>
1048  std::map<unsigned int, T>
1049  some_to_some(const MPI_Comm & comm,
1050  const std::map<unsigned int, T> &objects_to_send)
1051  {
1052 # ifndef DEAL_II_WITH_MPI
1053  (void)comm;
1054  Assert(objects_to_send.size() < 2,
1055  ExcMessage("Cannot send to more than one processor."));
1056  Assert(objects_to_send.find(0) != objects_to_send.end() ||
1057  objects_to_send.size() == 0,
1058  ExcMessage("Can only send to myself or to nobody."));
1059  return objects_to_send;
1060 # else
1061  const auto my_proc = this_mpi_process(comm);
1062 
1063  std::map<unsigned int, T> received_objects;
1064 
1065  std::vector<unsigned int> send_to;
1066  send_to.reserve(objects_to_send.size());
1067  for (const auto &m : objects_to_send)
1068  if (m.first == my_proc)
1069  received_objects[my_proc] = m.second;
1070  else
1071  send_to.emplace_back(m.first);
1072 
1073  const unsigned int n_point_point_communications =
1074  Utilities::MPI::compute_n_point_to_point_communications(comm, send_to);
1075 
1076  // Protect the following communication:
1077  static CollectiveMutex mutex;
1078  CollectiveMutex::ScopedLock lock(mutex, comm);
1079 
1080  // If we have something to send, or we expect something from other
1081  // processors, we need to visit one of the two scopes below. Otherwise,
1082  // no other action is required by this mpi process, and we can safely
1083  // return.
1084  if (send_to.size() == 0 && n_point_point_communications == 0)
1085  return received_objects;
1086 
1087  const int mpi_tag =
1088  internal::Tags::compute_point_to_point_communication_pattern;
1089 
1090  // Sending buffers
1091  std::vector<std::vector<char>> buffers_to_send(send_to.size());
1092  std::vector<MPI_Request> buffer_send_requests(send_to.size());
1093  {
1094  unsigned int i = 0;
1095  for (const auto &rank_obj : objects_to_send)
1096  if (rank_obj.first != my_proc)
1097  {
1098  const auto &rank = rank_obj.first;
1099  buffers_to_send[i] = Utilities::pack(rank_obj.second,
1100  /*allow_compression=*/ false);
1101  const int ierr = MPI_Isend(buffers_to_send[i].data(),
1102  buffers_to_send[i].size(),
1103  MPI_CHAR,
1104  rank,
1105  mpi_tag,
1106  comm,
1107  &buffer_send_requests[i]);
1108  AssertThrowMPI(ierr);
1109  ++i;
1110  }
1111  }
1112 
1113  // Fill the output map
1114  {
1115  std::vector<char> buffer;
1116  // We do this on a first come/first served basis
1117  for (unsigned int i = 0; i < n_point_point_communications; ++i)
1118  {
1119  // Probe what's going on. Take data from the first available sender
1120  MPI_Status status;
1121  int ierr = MPI_Probe(MPI_ANY_SOURCE, mpi_tag, comm, &status);
1122  AssertThrowMPI(ierr);
1123 
1124  // Length of the message
1125  int len;
1126  ierr = MPI_Get_count(&status, MPI_CHAR, &len);
1127  AssertThrowMPI(ierr);
1128  buffer.resize(len);
1129 
1130  // Source rank
1131  const unsigned int rank = status.MPI_SOURCE;
1132 
1133  // Actually receive the message
1134  ierr = MPI_Recv(buffer.data(),
1135  len,
1136  MPI_CHAR,
1137  status.MPI_SOURCE,
1138  status.MPI_TAG,
1139  comm,
1140  MPI_STATUS_IGNORE);
1141  AssertThrowMPI(ierr);
1142  Assert(received_objects.find(rank) == received_objects.end(),
1143  ExcInternalError(
1144  "I should not receive again from this rank"));
1145  received_objects[rank] =
1146  Utilities::unpack<T>(buffer,
1147  /*allow_compression=*/ false);
1148  }
1149  }
1150 
1151  // Wait to have sent all objects.
1152  const int ierr = MPI_Waitall(send_to.size(),
1153  buffer_send_requests.data(),
1154  MPI_STATUSES_IGNORE);
1155  AssertThrowMPI(ierr);
1156 
1157  return received_objects;
1158 # endif // deal.II with MPI
1159  }

1162  std::vector<T>
1163  all_gather(const MPI_Comm &comm, const T &object)
1164  {
1165  if (job_supports_mpi() == false)
1166  return {object};
1167 
1168 # ifndef DEAL_II_WITH_MPI
1169  (void)comm;
1170  std::vector<T> v(1, object);
1171  return v;
1172 # else
1173  const auto n_procs = ::Utilities::MPI::n_mpi_processes(comm);
1174 
1175  std::vector<char> buffer = Utilities::pack(object);
1176 
1177  int n_local_data = buffer.size();
1178 
1179  // Vector to store the size of loc_data_array for every process
1180  std::vector<int> size_all_data(n_procs, 0);
1181 
1182  // Exchanging the size of each buffer
1183  int ierr = MPI_Allgather(
1184  &n_local_data, 1, MPI_INT, size_all_data.data(), 1, MPI_INT, comm);
1185  AssertThrowMPI(ierr);
1186 
1187  // Now computing the displacement, relative to recvbuf,
1188  // at which to store the incoming buffer
1189  std::vector<int> rdispls(n_procs);
1190  rdispls[0] = 0;
1191  for (unsigned int i = 1; i < n_procs; ++i)
1192  rdispls[i] = rdispls[i - 1] + size_all_data[i - 1];
1193 
1194  // Step 3: exchange the buffer:
1195  std::vector<char> received_unrolled_buffer(rdispls.back() +
1196  size_all_data.back());
1197 
1198  ierr = MPI_Allgatherv(buffer.data(),
1199  n_local_data,
1200  MPI_CHAR,
1201  received_unrolled_buffer.data(),
1202  size_all_data.data(),
1203  rdispls.data(),
1204  MPI_CHAR,
1205  comm);
1206  AssertThrowMPI(ierr);
1207 
1208  std::vector<T> received_objects(n_procs);
1209  for (unsigned int i = 0; i < n_procs; ++i)
1210  {
1211  std::vector<char> local_buffer(received_unrolled_buffer.begin() +
1212  rdispls[i],
1213  received_unrolled_buffer.begin() +
1214  rdispls[i] + size_all_data[i]);
1215  received_objects[i] = Utilities::unpack<T>(local_buffer);
1216  }
1217 
1218  return received_objects;
1219 # endif
1220  }

1224  gather(const MPI_Comm & comm,

1226  const unsigned int root_process)
1227  {

1238 
1239  std::vector<char> buffer = Utilities::pack(object_to_send);
1240  int n_local_data = buffer.size();
1241 
1242  // Vector to store the size of loc_data_array for every process
1243  // only the root process needs to allocate memory for that purpose
1244  std::vector<int> size_all_data;
1245  if (my_rank == root_process)
1246  size_all_data.resize(n_procs, 0);
1247 
1248  // Exchanging the size of each buffer
1249  int ierr = MPI_Gather(&n_local_data,
1250  1,
1251  MPI_INT,
1252  size_all_data.data(),
1253  1,
1254  MPI_INT,
1255  root_process,
1256  comm);
1257  AssertThrowMPI(ierr);
1258 
1259  // Now computing the displacement, relative to recvbuf,
1260  // at which to store the incoming buffer; only for root
1261  std::vector<int> rdispls;
1262  if (my_rank == root_process)
1263  {
1264  rdispls.resize(n_procs, 0);
1265  for (unsigned int i = 1; i < n_procs; ++i)
1266  rdispls[i] = rdispls[i - 1] + size_all_data[i - 1];
1267  }
1268  // exchange the buffer:
1269  std::vector<char> received_unrolled_buffer;
1270  if (my_rank == root_process)
1271  received_unrolled_buffer.resize(rdispls.back() + size_all_data.back());

1274  n_local_data,
1275  MPI_CHAR,
1276  received_unrolled_buffer.data(),
1277  size_all_data.data(),
1278  rdispls.data(),
1279  MPI_CHAR,
1280  root_process,
1281  comm);
1282  AssertThrowMPI(ierr);
1283 
1284  std::vector<T> received_objects;
1285 
1286  if (my_rank == root_process)
1287  {
1288  received_objects.resize(n_procs);
1289 
1290  for (unsigned int i = 0; i < n_procs; ++i)
1291  {
1292  const std::vector<char> local_buffer(
1293  received_unrolled_buffer.begin() + rdispls[i],
1294  received_unrolled_buffer.begin() + rdispls[i] +
1295  size_all_data[i]);
1296  received_objects[i] = Utilities::unpack<T>(local_buffer);
1297  }
1298  }
1299  return received_objects;
1300 # endif
