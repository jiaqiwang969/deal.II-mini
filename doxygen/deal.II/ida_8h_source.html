348  prm.add_parameter(

351  "Indicate whether or not to suppress algebraic variables "

353  prm.leave_subsection();
354 
355  prm.enter_subsection("Initial condition correction parameters");
356  static std::string ic_type_str = "use_y_diff";
357  prm.add_parameter(
358  "Correction type at initial time",
359  ic_type_str,
360  "This is one of the following three options for the "
361  "initial condition calculation. \n"
362  " none: do not try to make initial conditions consistent. \n"
363  " use_y_diff: compute the algebraic components of y and differential\n"
364  " components of y_dot, given the differential components of y. \n"
365  " This option requires that the user specifies differential and \n"
366  " algebraic components in the function get_differential_components.\n"
367  " use_y_dot: compute all components of y, given y_dot.",
368  Patterns::Selection("none|use_y_diff|use_y_dot"));
369  prm.add_action("Correction type at initial time",
370  [&](const std::string &value) {
371  if (value == "use_y_diff")
372  ic_type = use_y_diff;
373  else if (value == "use_y_dot")
374  ic_type = use_y_dot;
375  else if (value == "none")
376  ic_type = none;
377  else
378  AssertThrow(false, ExcInternalError());
379  });
380 
381  static std::string reset_type_str = "use_y_diff";
382  prm.add_parameter(
383  "Correction type after restart",
384  reset_type_str,
385  "This is one of the following three options for the "
386  "initial condition calculation. \n"
387  " none: do not try to make initial conditions consistent. \n"
388  " use_y_diff: compute the algebraic components of y and differential\n"
389  " components of y_dot, given the differential components of y. \n"
390  " This option requires that the user specifies differential and \n"
391  " algebraic components in the function get_differential_components.\n"
392  " use_y_dot: compute all components of y, given y_dot.",
