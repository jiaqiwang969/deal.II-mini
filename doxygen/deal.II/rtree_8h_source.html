444  typename Box,

446 void
447 ExtractLevelVisitor<Value, Options, Translator, Box, Allocators>::
448 operator()(const ExtractLevelVisitor::InternalNode &node)
449 {
450  using ElmentsType =
451  typename boost::geometry::index::detail::rtree::elements_type<
452  InternalNode>::type;
453 
454  const auto &elements = boost::geometry::index::detail::rtree::elements(node);
455 
456  if (level == target_level)
457  {
458  const auto offset = boxes.size();
459  boxes.resize(offset + elements.size());
460 
461  unsigned int i = offset;
462  for (typename ElmentsType::const_iterator it = elements.begin();
463  it != elements.end();

473  ++level;

476  it != elements.end();
477  ++it)

479  boost::geometry::index::detail::rtree::apply_visitor(*this, *it->second);
480  }
481

483 }

497 template <typename Rtree>
498 inline std::vector<BoundingBox<
499  boost::geometry::dimension<typename Rtree::indexable_type>::value>>
500 extract_rtree_level(const Rtree &tree, const unsigned int level)
501 {
502  constexpr unsigned int dim =
503  boost::geometry::dimension<typename Rtree::indexable_type>::value;
504 
505  using RtreeView =
506  boost::geometry::index::detail::rtree::utilities::view<Rtree>;
507  RtreeView rtv(tree);
508 
509  std::vector<BoundingBox<dim>> boxes;
510 
511  if (rtv.depth() == 0)
512  {
513  // The below algorithm does not work for `rtv.depth()==0`, which might
514  // happen if the number entries in the tree is too small.
515  // In this case, simply return a single bounding box.
516  boxes.resize(1);
517  boost::geometry::convert(tree.bounds(), boxes[0]);
518  }
519  else
520  {
521  const unsigned int target_level =
522  std::min<unsigned int>(level, rtv.depth() - 1);
523 
524  ExtractLevelVisitor<typename RtreeView::value_type,
525  typename RtreeView::options_type,
526  typename RtreeView::translator_type,
527  typename RtreeView::box_type,
528  typename RtreeView::allocators_type>
529  extract_level_visitor(rtv.translator(), target_level, boxes);
530  rtv.apply_visitor(extract_level_visitor);
531  }
532 
533  return boxes;
534 }
