476 template <typename Range, typename Domain, typename Payload>
477 LinearOperator<Range, Domain, Payload>
478 operator*(typename Range::value_type number,
479  const LinearOperator<Range, Domain, Payload> &op)
480 {
481  static_assert(
482  std::is_convertible<typename Range::value_type,
483  typename Domain::value_type>::value,
484  "Range and Domain must have implicitly convertible 'value_type's");
485 
486  if (op.is_null_operator)
487  {
488  return op;
489  }
490  else if (number == 0.)
491  {
492  return null_operator(op);
493  }
494  else
495  {
496  LinearOperator<Range, Domain, Payload> return_op = op;
497 
498  // ensure to have valid computation objects by catching number and op by
499  // value
500 
501  return_op.vmult = [number, op](Range &v, const Domain &u) {
502  op.vmult(v, u);
503  v *= number;
504  };
505 
506  return_op.vmult_add = [number, op](Range &v, const Domain &u) {
507  v /= number;

509  v *= number;
510  };
