217  inline void
218  Matrix<VectorType>::initialize(const MGLevelObject<MatrixType> &p)
219  {
220  matrices.resize(p.min_level(), p.max_level());
221  for (unsigned int level = p.min_level(); level <= p.max_level(); ++level)
222  {
223  // Workaround: Unfortunately, not every "p[level]" object has a
224  // rich enough interface to populate reinit_(domain|range)_vector.
225  // Thus, apply an empty LinearOperator exemplar.
226  matrices[level] =
227  linear_operator<VectorType>(LinearOperator<VectorType>(),
228  Utilities::get_underlying_value(
229  p[level]));
230  }
231  }
