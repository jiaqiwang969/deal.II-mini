737 template <typename SparsityPatternType>
738 template <typename ForwardIterator>
739 void
740 BlockSparsityPatternBase<SparsityPatternType>::add_entries(
741  const size_type row,
742  ForwardIterator begin,
743  ForwardIterator end,
744  const bool indices_are_sorted)
745 {
746  // Resize scratch arrays
747  if (block_column_indices.size() < this->n_block_cols())
748  {
749  block_column_indices.resize(this->n_block_cols());
750  counter_within_block.resize(this->n_block_cols());
751  }
752 
753  const size_type n_cols = static_cast<size_type>(end - begin);
754 
755  // Resize sub-arrays to n_cols. This
756  // is a bit wasteful, but we resize
757  // only a few times (then the maximum
758  // row length won't increase that
759  // much any more). At least we know
760  // that all arrays are going to be of
761  // the same size, so we can check
762  // whether the size of one is large
763  // enough before actually going
764  // through all of them.
765  if (block_column_indices[0].size() < n_cols)
766  for (size_type i = 0; i < this->n_block_cols(); ++i)
767  block_column_indices[i].resize(n_cols);
768 
769  // Reset the number of added elements
770  // in each block to zero.
771  for (size_type i = 0; i < this->n_block_cols(); ++i)
772  counter_within_block[i] = 0;
773 
774  // Go through the column indices to
775  // find out which portions of the
776  // values should be set in which
777  // block of the matrix. We need to
778  // touch all the data, since we can't
779  // be sure that the data of one block
780  // is stored contiguously (in fact,
781  // indices will be intermixed when it
782  // comes from an element matrix).
783  for (ForwardIterator it = begin; it != end; ++it)
784  {
785  const size_type col = *it;
786 
787  const std::pair<size_type, size_type> col_index =
788  this->column_indices.global_to_local(col);
789 
790  const size_type local_index = counter_within_block[col_index.first]++;
791 
792  block_column_indices[col_index.first][local_index] = col_index.second;
793  }
794 
795 #ifdef DEBUG
796  // If in debug mode, do a check whether
797  // the right length has been obtained.
798  size_type length = 0;
799  for (size_type i = 0; i < this->n_block_cols(); ++i)
800  length += counter_within_block[i];
801  Assert(length == n_cols, ExcInternalError());
802 #endif
803 
804  // Now we found out about where the
805  // individual columns should start and
806  // where we should start reading out
807  // data. Now let's write the data into
808  // the individual blocks!
809  const std::pair<size_type, size_type> row_index =
810  this->row_indices.global_to_local(row);
811  for (size_type block_col = 0; block_col < n_block_cols(); ++block_col)
812  {
813  if (counter_within_block[block_col] == 0)
814  continue;
815  sub_objects[row_index.first][block_col]->add_entries(
816  row_index.second,
817  block_column_indices[block_col].begin(),
818  block_column_indices[block_col].begin() +
819  counter_within_block[block_col],
820  indices_are_sorted);
821  }
822 }

879 inline BlockDynamicSparsityPattern::size_type
880 BlockDynamicSparsityPattern::column_number(const size_type row,
881  const unsigned int index) const
882 {
883  // .first= ith block, .second = jth row in that block
884  const std::pair<size_type, size_type> row_index =
885  row_indices.global_to_local(row);
886 
887  AssertIndexRange(index, row_length(row));
888 
889  size_type c = 0;
890  size_type block_columns = 0; // sum of n_cols for all blocks to the left
891  for (unsigned int b = 0; b < columns; ++b)
892  {
893  unsigned int rowlen =
894  sub_objects[row_index.first][b]->row_length(row_index.second);
895  if (index < c + rowlen)
896  return block_columns +
897  sub_objects[row_index.first][b]->column_number(row_index.second,
898  index - c);
899  c += rowlen;
900  block_columns += sub_objects[row_index.first][b]->n_cols();
901  }
902 
903  Assert(false, ExcInternalError());
904  return 0;
905 }
