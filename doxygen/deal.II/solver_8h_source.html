495 template <class VectorType>
496 template <typename Iterator>
497 inline SolverControl::State
498 SolverBase<VectorType>::StateCombiner::operator()(const Iterator begin,
499  const Iterator end) const
500 {
501  Assert(begin != end,
502  ExcMessage("You can't combine iterator states if no state is given."));
503 
504  // combine the first with all of the following states
505  SolverControl::State state = *begin;
506  Iterator p = begin;
507  ++p;
508  for (; p != end; ++p)
509  state = this->operator()(state, *p);
510 
511  return state;
512 }
