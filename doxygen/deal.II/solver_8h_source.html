398 template <class VectorType>
399 template <typename Iterator>
400 inline SolverControl::State
401 SolverBase<VectorType>::StateCombiner::operator()(const Iterator begin,
402  const Iterator end) const
403 {
404  Assert(begin != end,
405  ExcMessage("You can't combine iterator states if no state is given."));
406 
407  // combine the first with all of the following states
408  SolverControl::State state = *begin;
409  Iterator p = begin;
410  ++p;
411  for (; p != end; ++p)
412  state = this->operator()(state, *p);
413 
414  return state;
415 }
