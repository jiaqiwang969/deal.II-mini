4371  template <int dim, int spacedim>
4372  inline typename SymmetricTensor<2, dim, spacedim>::divergence_type
4373  SymmetricTensor<2, dim, spacedim>::divergence(
4374  const unsigned int shape_function,
4375  const unsigned int q_point) const
4376  {
4377  AssertIndexRange(shape_function, fe_values->fe->n_dofs_per_cell());
4378  Assert(fe_values->update_flags & update_gradients,
4379  (typename FEValuesBase<dim, spacedim>::ExcAccessToUninitializedField(
4380  "update_gradients")));
4381 
4382  const int snc =
4383  shape_function_data[shape_function].single_nonzero_component;
4384 
4385  if (snc == -2)
4386  {
4387  // shape function is zero for the selected components
4388  return divergence_type();
4389  }
4390  else if (snc != -1)
4391  {
4392  // we have a single non-zero component when the symmetric tensor is
4393  // represented in unrolled form. this implies we potentially have
4394  // two non-zero components when represented in component form! we
4395  // will only have one non-zero entry if the non-zero component lies on
4396  // the diagonal of the tensor.
4397  //
4398  // the divergence of a second-order tensor is a first order tensor.
4399  //
4400  // assume the second-order tensor is A with components A_{ij}. then
4401  // A_{ij} = A_{ji} and there is only one (if diagonal) or two non-zero
4402  // entries in the tensorial representation. define the
4403  // divergence as:
4404  // b_i \dealcoloneq \dfrac{\partial phi_{ij}}{\partial x_j}.
4405  // (which is incidentally also
4406  // b_j \dealcoloneq \dfrac{\partial phi_{ij}}{\partial x_i}).
4407  // In both cases, a sum is implied.
4408  //
4409  // Now, we know the nonzero component in unrolled form: it is indicated
4410  // by 'snc'. we can figure out which tensor components belong to this:
4411  const unsigned int comp =
4412  shape_function_data[shape_function].single_nonzero_component_index;
4413  const unsigned int ii =
4414  value_type::unrolled_to_component_indices(comp)[0];
4415  const unsigned int jj =
4416  value_type::unrolled_to_component_indices(comp)[1];
4417 
4418  // given the form of the divergence above, if ii=jj there is only a
4419  // single nonzero component of the full tensor and the gradient
4420  // equals
4421  // b_ii \dealcoloneq \dfrac{\partial phi_{ii,ii}}{\partial x_ii}.
4422  // all other entries of 'b' are zero
4423  //
4424  // on the other hand, if ii!=jj, then there are two nonzero entries in
4425  // the full tensor and
4426  // b_ii \dealcoloneq \dfrac{\partial phi_{ii,jj}}{\partial x_ii}.
4427  // b_jj \dealcoloneq \dfrac{\partial phi_{ii,jj}}{\partial x_jj}.
4428  // again, all other entries of 'b' are zero
4429  const ::Tensor<1, spacedim> &phi_grad =
4430  fe_values->finite_element_output.shape_gradients[snc][q_point];
4431 
4432  divergence_type return_value;
4433  return_value[ii] = phi_grad[jj];
4434 
4435  if (ii != jj)
4436  return_value[jj] = phi_grad[ii];
4437 
4438  return return_value;
4439  }
4440  else
4441  {
4442  Assert(false, ExcNotImplemented());
4443  divergence_type return_value;
4444  return return_value;
4445  }
4446  }
