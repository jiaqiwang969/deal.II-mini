100  struct TBBForFunctor
101  {
102  TBBForFunctor(Functor & functor,
103  const size_type start,
104  const size_type end)
105  : functor(functor)
106  , start(start)
107  , end(end)
108  {
109  const size_type vec_size = end - start;
110  // set chunk size for sub-tasks
111  const unsigned int gs =
112  internal::VectorImplementation::minimum_parallel_grain_size;
113  n_chunks =
114  std::min(static_cast<size_type>(4 * MultithreadInfo::n_threads()),
115  vec_size / gs);
116  chunk_size = vec_size / n_chunks;
117 
118  // round to next multiple of 512 (or minimum grain size if that happens
119  // to be smaller). this is advantageous because our accumulation
120  // algorithms favor lengths of a power of 2 due to pairwise summation ->
121  // at most one 'oddly' sized chunk
122  if (chunk_size > 512)
123  chunk_size = ((chunk_size + 511) / 512) * 512;
124  n_chunks = (vec_size + chunk_size - 1) / chunk_size;
125  AssertIndexRange((n_chunks - 1) * chunk_size, vec_size);
126  AssertIndexRange(vec_size, n_chunks * chunk_size + 1);
127  }

146  void
147  parallel_for(
148  Functor & functor,
149  const size_type start,
150  const size_type end,
151  const std::shared_ptr<::parallel::internal::TBBPartitioner>
152  &partitioner)
153  {
154 #ifdef DEAL_II_WITH_TBB
155  const size_type vec_size = end - start;
156  // only go to the parallel function in case there are at least 4 parallel
157  // items, otherwise the overhead is too large
158  if (vec_size >=
159  4 * internal::VectorImplementation::minimum_parallel_grain_size &&
160  MultithreadInfo::n_threads() > 1)
161  {
162  Assert(partitioner.get() != nullptr,
163  ExcInternalError(
164  "Unexpected initialization of Vector that does "
165  "not set the TBB partitioner to a usable state."));
166  std::shared_ptr<tbb::affinity_partitioner> tbb_partitioner =
167  partitioner->acquire_one_partitioner();
168 
169  TBBForFunctor<Functor> generic_functor(functor, start, end);
170  // We use a minimum grain size of 1 here since the grains at this
171  // stage of dividing the work refer to the number of vector chunks
172  // that are processed by (possibly different) threads in the
173  // parallelized for loop (i.e., they do not refer to individual
174  // vector entries). The number of chunks here is calculated inside
175  // TBBForFunctor. See also GitHub issue #2496 for further discussion
176  // of this strategy.
177  ::parallel::internal::parallel_for(
178  static_cast<size_type>(0),
179  static_cast<size_type>(generic_functor.n_chunks),
180  generic_functor,
181  1,
182  tbb_partitioner);
183  partitioner->release_one_partitioner(tbb_partitioner);
184  }
185  else if (vec_size > 0)
186  functor(start, end);
187 #else
188  functor(start, end);
189  (void)partitioner;
190 #endif
191  }

943  // this is the main working loop for all vector sums using the templated
944  // operation above. it accumulates the sums using a block-wise summation
945  // algorithm with post-update. this blocked algorithm has been proposed in
946  // a similar form by Castaldo, Whaley and Chronopoulos (SIAM
947  // J. Sci. Comput. 31, 1156-1174, 2008) and we use the smallest possible
948  // block size, 2. Sometimes it is referred to as pairwise summation. The
949  // worst case error made by this algorithm is on the order O(eps *
950  // log2(vec_size)), whereas a naive summation is O(eps * vec_size). Even
951  // though the Kahan summation is even more accurate with an error O(eps)
952  // by carrying along remainders not captured by the main sum, that involves
953  // additional costs which are not worthwhile. See the Wikipedia article on
954  // the Kahan summation algorithm.
955 
956  // The algorithm implemented here has the additional benefit that it is
957  // easily parallelized without changing the order of how the elements are
958  // added (floating point addition is not associative). For the same vector
959  // size and minimum_parallel_grainsize, the blocks are always the
960  // same and added pairwise.
961 
962  // The depth of recursion is controlled by the 'magic' parameter
963  // vector_accumulation_recursion_threshold: If the length is below
964  // vector_accumulation_recursion_threshold * 32 (32 is the part of code we
965  // unroll), a straight loop instead of recursion will be used. At the
966  // innermost level, eight values are added consecutively in order to better
967  // balance multiplications and additions.
968 
969  // Loops are unrolled as follows: the range [first,last) is broken into
970  // @p n_chunks each of size 32 plus the @p remainder.
971  // accumulate_regular() does the work on 32*n_chunks elements employing SIMD
972  // if possible and stores the result of the operation for each chunk in @p outer_results.
973 
974  // The code returns the result as the last argument in order to make
975  // spawning tasks simpler and use automatic template deduction.
976 
977 
982  const unsigned int vector_accumulation_recursion_threshold = 128;
983 
984  template <typename Operation, typename ResultType>
985  void
986  accumulate_recursive(const Operation &op,
987  const size_type first,
988  const size_type last,
989  ResultType & result)
990  {
991  const size_type vec_size = last - first;
992  if (vec_size <= vector_accumulation_recursion_threshold * 32)
993  {
994  // the vector is short enough so we perform the summation. first
995  // work on the regular part. The innermost 32 values are expanded in
996  // order to obtain known loop bounds for most of the work.
997  size_type index = first;
998  ResultType outer_results[vector_accumulation_recursion_threshold];
999 
1000  // set the zeroth element to zero to correctly handle the case where
1001  // vec_size == 0
1002  outer_results[0] = ResultType();
1003 
1004  // the variable serves two purposes: (i) number of chunks (each 32
1005  // indices) for the given size; all results are stored in
1006  // outer_results[0,n_chunks) (ii) in the SIMD case n_chunks is also a
1007  // next free index in outer_results[] to which we can write after
1008  // accumulate_regular() is executed.
1009  size_type n_chunks = vec_size / 32;
1010  const size_type remainder = vec_size % 32;
1011  Assert(remainder == 0 ||
1012  n_chunks < vector_accumulation_recursion_threshold,
1013  ExcInternalError());
1014 
1015  // Select between the regular version and vectorized version based
1016  // on the number types we are given. To choose the vectorized
1017  // version often enough, we need to have all tasks but the last one
1018  // to be divisible by the vectorization length
1019  accumulate_regular(
1020  op,
1021  n_chunks,
1022  index,
1023  outer_results,
1024  std::integral_constant<bool, Operation::vectorizes>());
1025 
1026  // now work on the remainder, i.e., the last up to 32 values. Use
1027  // switch statement with fall-through to work on these values.
1028  if (remainder > 0)
1029  {
1030  // if we got here, it means that (vec_size <=
1031  // vector_accumulation_recursion_threshold * 32), which is to say
1032  // that the domain can be split into n_chunks <=
1033  // vector_accumulation_recursion_threshold:
1034  AssertIndexRange(n_chunks,
1035  vector_accumulation_recursion_threshold + 1);
1036  // split the remainder into chunks of 8, there could be up to 3
1037  // such chunks since remainder < 32.
1038  // Work on those chunks without any SIMD, that is we call
1039  // op(index).
1040  const size_type inner_chunks = remainder / 8;
1041  Assert(inner_chunks <= 3, ExcInternalError());
1042  const size_type remainder_inner = remainder % 8;
1043  ResultType r0 = ResultType(), r1 = ResultType(),
1044  r2 = ResultType();
1045  switch (inner_chunks)
1046  {
1047  case 3:
1048  r2 = op(index++);
1049  for (size_type j = 1; j < 8; ++j)
1050  r2 += op(index++);
1051  DEAL_II_FALLTHROUGH;
1052  case 2:
1053  r1 = op(index++);
1054  for (size_type j = 1; j < 8; ++j)
1055  r1 += op(index++);
1056  r1 += r2;
1057  DEAL_II_FALLTHROUGH;
1058  case 1:
1059  r2 = op(index++);
1060  for (size_type j = 1; j < 8; ++j)
1061  r2 += op(index++);
1062  DEAL_II_FALLTHROUGH;
1063  default:
1064  for (size_type j = 0; j < remainder_inner; ++j)
1065  r0 += op(index++);
1066  r0 += r2;
1067  r0 += r1;
1068  if (n_chunks == vector_accumulation_recursion_threshold)
1069  outer_results[vector_accumulation_recursion_threshold -
1070  1] += r0;
1071  else
1072  {
1073  outer_results[n_chunks] = r0;
1074  n_chunks++;
1075  }
1076  break;
1077  }
1078  }
1079  // make sure we worked through all indices
1080  AssertDimension(index, last);
1081 
1082  // now sum the results from the chunks stored in
1083  // outer_results[0,n_chunks) recursively
1084  while (n_chunks > 1)
1085  {
1086  if (n_chunks % 2 == 1)
1087  outer_results[n_chunks++] = ResultType();
1088  for (size_type i = 0; i < n_chunks; i += 2)
1089  outer_results[i / 2] = outer_results[i] + outer_results[i + 1];
1090  n_chunks /= 2;
1091  }
1092  result = outer_results[0];
1093  }
1094  else
1095  {
1096  // split vector into four pieces and work on the pieces
1097  // recursively. Make pieces (except last) divisible by one fourth the
1098  // recursion threshold.
1099  const size_type new_size =
1100  (vec_size / (vector_accumulation_recursion_threshold * 32)) *
1101  vector_accumulation_recursion_threshold * 8;
1102  Assert(first + 3 * new_size < last, ExcInternalError());
1103  ResultType r0, r1, r2, r3;
1104  accumulate_recursive(op, first, first + new_size, r0);
1105  accumulate_recursive(op, first + new_size, first + 2 * new_size, r1);
1106  accumulate_recursive(op,
1107  first + 2 * new_size,
1108  first + 3 * new_size,
1109  r2);
1110  accumulate_recursive(op, first + 3 * new_size, last, r3);
1111  r0 += r1;
1112  r2 += r3;
1113  result = r0 + r2;
1114  }
1115  }

1119  // below. This is the standard case where the loop bounds are known. We
1120  // pulled this function out of the regular accumulate routine because we
1121  // might do this thing vectorized (see specialized function below)
1122  template <typename Operation, typename ResultType>
1123  void
1124  accumulate_regular(
1125  const Operation &op,
1126  const size_type &n_chunks,
1127  size_type & index,
1128  ResultType (&outer_results)[vector_accumulation_recursion_threshold],
1129  std::integral_constant<bool, false>)
1130  {
1131  // note that each chunk is chosen to have a width of 32, thereby the index
1132  // is incremented by 4*8 for each @p i.
1133  for (size_type i = 0; i < n_chunks; ++i)
1134  {
1135  ResultType r0 = op(index);
1136  ResultType r1 = op(index + 1);
1137  ResultType r2 = op(index + 2);
1138  ResultType r3 = op(index + 3);
1139  index += 4;
1140  for (size_type j = 1; j < 8; ++j, index += 4)
1141  {
1142  r0 += op(index);
1143  r1 += op(index + 1);
1144  r2 += op(index + 2);
1145  r3 += op(index + 3);
1146  }
1147  r0 += r1;
1148  r2 += r3;
1149  outer_results[i] = r0 + r2;
1150  }
1151  }

1155  // this is the inner working routine for the accumulation loops
1156  // below. This is the specialized case where the loop bounds are known and
1157  // where we can vectorize. In that case, we request the 'do_vectorized'
1158  // routine of the operation instead of the regular one which does several
1159  // operations at once.
1160  template <typename Operation, typename Number>
1161  void
1162  accumulate_regular(
1163  const Operation &op,
1164  size_type & n_chunks,
1165  size_type & index,
1166  Number (&outer_results)[vector_accumulation_recursion_threshold],
1167  std::integral_constant<bool, true>)
1168  {
1169  // we start from @p index and workout @p n_chunks each of size 32.
1170  // in order employ SIMD and work on @p nvecs at a time, we split this
1171  // loop yet again:
1172  // First we work on (n_chunks/nvecs) chunks, where each chunk processes
1173  // nvecs*(4*8) elements.
1174 
1175  constexpr unsigned int nvecs = VectorizedArray<Number>::size();
1176  const size_type regular_chunks = n_chunks / nvecs;
1177  for (size_type i = 0; i < regular_chunks; ++i)
1178  {
1179  VectorizedArray<Number> r0 = op.do_vectorized(index);
1180  VectorizedArray<Number> r1 = op.do_vectorized(index + nvecs);
1181  VectorizedArray<Number> r2 = op.do_vectorized(index + 2 * nvecs);
1182  VectorizedArray<Number> r3 = op.do_vectorized(index + 3 * nvecs);
1183  index += nvecs * 4;
1184  for (size_type j = 1; j < 8; ++j, index += nvecs * 4)
1185  {
1186  r0 += op.do_vectorized(index);
1187  r1 += op.do_vectorized(index + nvecs);
1188  r2 += op.do_vectorized(index + 2 * nvecs);
1189  r3 += op.do_vectorized(index + 3 * nvecs);
1190  }
1191  r0 += r1;
1192  r2 += r3;
1193  r0 += r2;
1194  r0.store(&outer_results[i * nvecs]);
1195  }
1196 
1197  // If we are treating a case where the vector length is not divisible by
1198  // the vectorization length, need a cleanup loop
1199  // The remaining chunks are processed one by one starting from
1200  // regular_chunks * nvecs; We do as much as possible with 2 SIMD
1201  // operations within each chunk. Here we assume that nvecs < 32/2 = 16 as
1202  // well as 16%nvecs==0.
1203  static_assert(
1204  VectorizedArray<Number>::size() <= 16 &&
1205  16 % VectorizedArray<Number>::size() == 0,
1206  "VectorizedArray::size() must be a power of 2 and not more than 16");
1207  Assert(16 % nvecs == 0, ExcInternalError());
1208  if (n_chunks % nvecs != 0)
1209  {
1210  VectorizedArray<Number> r0 = VectorizedArray<Number>(),
1211  r1 = VectorizedArray<Number>();
1212  const size_type start_irreg = regular_chunks * nvecs;
1213  for (size_type c = start_irreg; c < n_chunks; ++c)
1214  for (size_type j = 0; j < 32; j += 2 * nvecs, index += 2 * nvecs)
1215  {
1216  r0 += op.do_vectorized(index);
1217  r1 += op.do_vectorized(index + nvecs);
1218  }
1219  r0 += r1;
1220  r0.store(&outer_results[start_irreg]);
1221  // update n_chunks to denote unused element in outer_results[] from
1222  // which we can keep writing.
1223  n_chunks = start_irreg + VectorizedArray<Number>::size();
1224  }
1225  }
