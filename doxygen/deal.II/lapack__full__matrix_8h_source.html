942 template <typename number>
943 template <typename MatrixType>
944 inline void
945 LAPACKFullMatrix<number>::copy_from(const MatrixType &M)
946 {
947  this->reinit(M.m(), M.n());
948 
949  // loop over the elements of the argument matrix row by row, as suggested
950  // in the documentation of the sparse matrix iterator class, and
951  // copy them into the current object
952  for (size_type row = 0; row < M.m(); ++row)
953  {
954  const typename MatrixType::const_iterator end_row = M.end(row);
955  for (typename MatrixType::const_iterator entry = M.begin(row);
956  entry != end_row;
957  ++entry)
958  this->el(row, entry->column()) = entry->value();
959  }
960 
961  state = LAPACKSupport::matrix;
962 }

966 template <typename number>
967 template <typename MatrixType>
968 inline void
969 LAPACKFullMatrix<number>::fill(const MatrixType &M,
970  const size_type dst_offset_i,
971  const size_type dst_offset_j,
972  const size_type src_offset_i,
973  const size_type src_offset_j,
974  const number factor,
975  const bool transpose)
976 {
977  // loop over the elements of the argument matrix row by row, as suggested
978  // in the documentation of the sparse matrix iterator class
979  for (size_type row = src_offset_i; row < M.m(); ++row)
980  {
981  const typename MatrixType::const_iterator end_row = M.end(row);
982  for (typename MatrixType::const_iterator entry = M.begin(row);
983  entry != end_row;
984  ++entry)
985  {
986  const size_type i = transpose ? entry->column() : row;
987  const size_type j = transpose ? row : entry->column();
988 
989  const size_type dst_i = dst_offset_i + i - src_offset_i;
990  const size_type dst_j = dst_offset_j + j - src_offset_j;
991  if (dst_i < this->n_rows() && dst_j < this->n_cols())
992  (*this)(dst_i, dst_j) = factor * entry->value();
993  }
994  }
995 
996  state = LAPACKSupport::matrix;
997 }
