640 template <int n_components, int dim, int spacedim, typename Number>
641 FEPointEvaluation<n_components, dim, spacedim, Number>::FEPointEvaluation(
642  const Mapping<dim> & mapping,
643  const FiniteElement<dim> &fe,
644  const UpdateFlags update_flags,
645  const unsigned int first_selected_component)
646  : mapping(&mapping)
647  , mapping_q_generic(
648  dynamic_cast<const MappingQGeneric<dim, spacedim> *>(&mapping))
649  , fe(&fe)
650  , update_flags(update_flags)
651  , update_flags_mapping(update_default)
652 {
653  bool same_base_element = true;
654  unsigned int base_element_number = 0;
655  unsigned int component = 0;
656  for (; base_element_number < fe.n_base_elements(); ++base_element_number)
657  if (component + fe.element_multiplicity(base_element_number) >
658  first_selected_component)
659  {
660  if (first_selected_component + n_components >
661  component + fe.element_multiplicity(base_element_number))
662  same_base_element = false;
663  break;
664  }
665  else
666  component += fe.element_multiplicity(base_element_number);
667  if (mapping_q_generic != nullptr &&
668  internal::FEPointEvaluation::is_fast_path_supported(
669  fe, base_element_number) &&
670  same_base_element)
671  {
672  internal::MatrixFreeFunctions::ShapeInfo<double> shape_info;
673 
674  shape_info.reinit(QMidpoint<1>(), fe, base_element_number);
675  renumber = shape_info.lexicographic_numbering;
676  dofs_per_component = shape_info.dofs_per_component_on_cell;
677  poly = internal::FEPointEvaluation::get_polynomial_space(
678  fe.base_element(base_element_number));
679 
680  polynomials_are_hat_functions =
681  (poly.size() == 2 && poly[0].value(0.) == 1. &&
682  poly[0].value(1.) == 0. && poly[1].value(0.) == 0. &&
683  poly[1].value(1.) == 1.);
684  }
685  else
686  {
687  nonzero_shape_function_component.resize(fe.n_dofs_per_cell());
688  for (unsigned int d = 0; d < n_components; ++d)
689  {
690  const unsigned int component = first_selected_component + d;
691  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
692  {
693  const bool is_primitive = fe.is_primitive() || fe.is_primitive(i);
694  if (is_primitive)
695  nonzero_shape_function_component[i][d] =
696  (component == fe.system_to_component_index(i).first);
697  else
698  nonzero_shape_function_component[i][d] =
699  (fe.get_nonzero_components(i)[component] == true);
700  }
701  }
702  }
703 
704  // translate update flags
705  if (update_flags & update_jacobians)
706  update_flags_mapping |= update_jacobians;
707  if (update_flags & update_gradients ||
708  update_flags & update_inverse_jacobians)
709  update_flags_mapping |= update_inverse_jacobians;
710  if (update_flags & update_quadrature_points)
711  update_flags_mapping |= update_quadrature_points;
712 }

717 void
718 FEPointEvaluation<n_components, dim, spacedim, Number>::reinit(

720  const ArrayView<const Point<dim>> & unit_points)
721 {

724 
725  if (!poly.empty())

727  cell, unit_points, update_flags_mapping, mapping_data);
728  else
729  {
730  fe_values = std::make_shared<FEValues<dim, spacedim>>(
731  *mapping,
732  *fe,
733  Quadrature<dim>(
734  std::vector<Point<dim>>(unit_points.begin(), unit_points.end())),
735  update_flags | update_flags_mapping);

738  if (update_flags_mapping & update_jacobians)
739  for (unsigned int q = 0; q < unit_points.size(); ++q)
740  mapping_data.jacobians[q] = fe_values->jacobian(q);
741  if (update_flags_mapping & update_inverse_jacobians)
742  for (unsigned int q = 0; q < unit_points.size(); ++q)
743  mapping_data.inverse_jacobians[q] = fe_values->inverse_jacobian(q);
744  if (update_flags_mapping & update_quadrature_points)
745  for (unsigned int q = 0; q < unit_points.size(); ++q)
746  mapping_data.quadrature_points[q] = fe_values->quadrature_point(q);
747  }
748 
749  if (update_flags & update_values)
750  values.resize(unit_points.size(), numbers::signaling_nan<value_type>());
751  if (update_flags & update_gradients)
752  gradients.resize(unit_points.size(),
753  numbers::signaling_nan<gradient_type>());

758 template <int n_components, int dim, int spacedim, typename Number>
759 void
760 FEPointEvaluation<n_components, dim, spacedim, Number>::evaluate(
761  const ArrayView<const Number> & solution_values,
762  const EvaluationFlags::EvaluationFlags &evaluation_flag)
763 {
764  if (unit_points.empty())
765  return;
766 
767  AssertDimension(solution_values.size(), fe->dofs_per_cell);
768  if (((evaluation_flag & EvaluationFlags::values) ||
769  (evaluation_flag & EvaluationFlags::gradients)) &&
770  !poly.empty())
771  {
772  // fast path with tensor product evaluation
773  if (solution_renumbered.size() != dofs_per_component)
774  solution_renumbered.resize(dofs_per_component);
775  for (unsigned int comp = 0; comp < n_components; ++comp)
776  for (unsigned int i = 0; i < dofs_per_component; ++i)
777  internal::FEPointEvaluation::
778  EvaluatorTypeTraits<dim, n_components, Number>::read_value(
779  solution_values[renumber[comp * dofs_per_component + i]],
780  comp,
781  solution_renumbered[i]);
782 
783  // unit gradients are currently only implemented with the fast tensor
784  // path
785  unit_gradients.resize(unit_points.size(),
786  numbers::signaling_nan<gradient_type>());
787 
788  const std::size_t n_points = unit_points.size();
789  const std::size_t n_lanes = VectorizedArray<Number>::size();
790  for (unsigned int i = 0; i < n_points; i += n_lanes)
791  {
792  // convert to vectorized format
793  Point<dim, VectorizedArray<Number>> vectorized_points;
794  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
795  for (unsigned int d = 0; d < dim; ++d)
796  vectorized_points[d][j] = unit_points[i + j][d];
797 
798  // compute
799  const auto val_and_grad =
800  internal::evaluate_tensor_product_value_and_gradient(
801  poly,
802  solution_renumbered,
803  vectorized_points,
804  polynomials_are_hat_functions);
805 
806  // convert back to standard format
807  if (evaluation_flag & EvaluationFlags::values)
808  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
809  internal::FEPointEvaluation::
810  EvaluatorTypeTraits<dim, n_components, Number>::set_value(
811  val_and_grad.first, j, values[i + j]);
812  if (evaluation_flag & EvaluationFlags::gradients)
813  {
814  Assert(update_flags & update_gradients ||
815  update_flags & update_inverse_jacobians,
816  ExcNotInitialized());
817  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
818  {
819  Assert(update_flags_mapping & update_inverse_jacobians,
820  ExcNotInitialized());
821  internal::FEPointEvaluation::EvaluatorTypeTraits<
822  dim,
823  n_components,
824  Number>::set_gradient(val_and_grad.second,
825  j,
826  unit_gradients[i + j]);
827  gradients[i + j] = apply_transformation(
828  mapping_data.inverse_jacobians[i + j].transpose(),
829  unit_gradients[i + j]);
830  }
831  }
832  }
833  }
834  else if ((evaluation_flag & EvaluationFlags::values) ||
835  (evaluation_flag & EvaluationFlags::gradients))
836  {
837  // slow path with FEValues
838  Assert(fe_values.get() != nullptr,
839  ExcMessage(
840  "Not initialized. Please call FEPointEvaluation::reinit()!"));
841 
842  if (evaluation_flag & EvaluationFlags::values)
843  {
844  values.resize(unit_points.size());
845  std::fill(values.begin(), values.end(), value_type());
846  for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
847  {
848  const Number value = solution_values[i];
849  for (unsigned int d = 0; d < n_components; ++d)
850  if (nonzero_shape_function_component[i][d] &&
851  (fe->is_primitive(i) || fe->is_primitive()))
852  for (unsigned int q = 0; q < unit_points.size(); ++q)
853  internal::FEPointEvaluation::
854  EvaluatorTypeTraits<dim, n_components, Number>::access(
855  values[q], d) += fe_values->shape_value(i, q) * value;
856  else if (nonzero_shape_function_component[i][d])
857  for (unsigned int q = 0; q < unit_points.size(); ++q)
858  internal::FEPointEvaluation::
859  EvaluatorTypeTraits<dim, n_components, Number>::access(
860  values[q], d) +=
861  fe_values->shape_value_component(i, q, d) * value;
862  }
863  }
864 
865  if (evaluation_flag & EvaluationFlags::gradients)
866  {
867  gradients.resize(unit_points.size());
868  std::fill(gradients.begin(), gradients.end(), gradient_type());
869  for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
870  {
871  const Number value = solution_values[i];
872  for (unsigned int d = 0; d < n_components; ++d)

892 template <int n_components, int dim, int spacedim, typename Number>
893 void
894 FEPointEvaluation<n_components, dim, spacedim, Number>::integrate(
895  const ArrayView<Number> & solution_values,
896  const EvaluationFlags::EvaluationFlags &integration_flags)
897 {
898  if (unit_points.size() == 0) // no evaluation points provided
899  {
900  std::fill(solution_values.begin(), solution_values.end(), 0.0);
901  return;
902  }
903 
904  AssertDimension(solution_values.size(), fe->dofs_per_cell);
905  if (((integration_flags & EvaluationFlags::values) ||
906  (integration_flags & EvaluationFlags::gradients)) &&
907  !poly.empty())
908  {
909  // fast path with tensor product integration
910 
911  if (integration_flags & EvaluationFlags::values)
912  AssertIndexRange(unit_points.size(), values.size() + 1);
913  if (integration_flags & EvaluationFlags::gradients)
914  AssertIndexRange(unit_points.size(), gradients.size() + 1);
915 
916  if (solution_renumbered_vectorized.size() != dofs_per_component)
917  solution_renumbered_vectorized.resize(dofs_per_component);
918  // zero content
919  solution_renumbered_vectorized.fill(
920  typename internal::FEPointEvaluation::EvaluatorTypeTraits<
921  dim,
922  n_components,
923  VectorizedArray<Number>>::value_type());
924 
925  const std::size_t n_points = unit_points.size();
926  const std::size_t n_lanes = VectorizedArray<Number>::size();
927  for (unsigned int i = 0; i < n_points; i += n_lanes)
928  {
929  // convert to vectorized format
930  Point<dim, VectorizedArray<Number>> vectorized_points;
931  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
932  for (unsigned int d = 0; d < dim; ++d)
933  vectorized_points[d][j] = unit_points[i + j][d];
934 
935  typename internal::ProductTypeNoPoint<value_type,
936  VectorizedArray<Number>>::type
937  value = {};
938  Tensor<1,
939  dim,
940  typename internal::ProductTypeNoPoint<
941  value_type,
942  VectorizedArray<Number>>::type>
943  gradient;
944 
945  if (integration_flags & EvaluationFlags::values)
946  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
947  internal::FEPointEvaluation::
948  EvaluatorTypeTraits<dim, n_components, Number>::get_value(
949  value, j, values[i + j]);
950  if (integration_flags & EvaluationFlags::gradients)
951  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
952  {
953  Assert(update_flags_mapping & update_inverse_jacobians,
954  ExcNotInitialized());
955  gradients[i + j] =
956  apply_transformation(mapping_data.inverse_jacobians[i + j],
957  gradients[i + j]);
958  internal::FEPointEvaluation::
959  EvaluatorTypeTraits<dim, n_components, Number>::get_gradient(
960  gradient, j, gradients[i + j]);
961  }
962 
963  // compute
964  internal::integrate_add_tensor_product_value_and_gradient(
965  poly,
966  value,
967  gradient,
968  vectorized_points,
969  solution_renumbered_vectorized);
970  }
971 
972  // add between the lanes and write into the result
973  std::fill(solution_values.begin(), solution_values.end(), Number());
974  for (unsigned int comp = 0; comp < n_components; ++comp)
975  for (unsigned int i = 0; i < dofs_per_component; ++i)
976  {
977  VectorizedArray<Number> result;
978  internal::FEPointEvaluation::
979  EvaluatorTypeTraits<dim, n_components, VectorizedArray<Number>>::
980  write_value(result, comp, solution_renumbered_vectorized[i]);
981  for (unsigned int lane = n_lanes / 2; lane > 0; lane /= 2)
982  for (unsigned int j = 0; j < lane; ++j)
983  result[j] += result[lane + j];
984  solution_values[renumber[comp * dofs_per_component + i]] =
985  result[0];
986  }
987  }
988  else if ((integration_flags & EvaluationFlags::values) ||
989  (integration_flags & EvaluationFlags::gradients))
990  {
991  // slow path with FEValues
992 
993  Assert(fe_values.get() != nullptr,
994  ExcMessage(
995  "Not initialized. Please call FEPointEvaluation::reinit()!"));
996  std::fill(solution_values.begin(), solution_values.end(), 0.0);
997 
998  if (integration_flags & EvaluationFlags::values)
999  {
1000  AssertIndexRange(unit_points.size(), values.size() + 1);
1001  for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
1002  {
1003  for (unsigned int d = 0; d < n_components; ++d)
1004  if (nonzero_shape_function_component[i][d] &&
1005  (fe->is_primitive(i) || fe->is_primitive()))
1006  for (unsigned int q = 0; q < unit_points.size(); ++q)
1007  solution_values[i] +=
1008  fe_values->shape_value(i, q) *
1009  internal::FEPointEvaluation::
1010  EvaluatorTypeTraits<dim, n_components, Number>::access(
1011  values[q], d);
1012  else if (nonzero_shape_function_component[i][d])
1013  for (unsigned int q = 0; q < unit_points.size(); ++q)
1014  solution_values[i] +=
1015  fe_values->shape_value_component(i, q, d) *
1016  internal::FEPointEvaluation::
1017  EvaluatorTypeTraits<dim, n_components, Number>::access(
1018  values[q], d);
1019  }
1020  }
1021 
1022  if (integration_flags & EvaluationFlags::gradients)
1023  {
1024  AssertIndexRange(unit_points.size(), gradients.size() + 1);
1025  for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
1026  {
1027  for (unsigned int d = 0; d < n_components; ++d)
1028  if (nonzero_shape_function_component[i][d] &&
1029  (fe->is_primitive(i) || fe->is_primitive()))
1030  for (unsigned int q = 0; q < unit_points.size(); ++q)
1031  solution_values[i] +=
1032  fe_values->shape_grad(i, q) *
1033  internal::FEPointEvaluation::
1034  EvaluatorTypeTraits<dim, n_components, Number>::access(
1035  gradients[q], d);
1036  else if (nonzero_shape_function_component[i][d])
1037  for (unsigned int q = 0; q < unit_points.size(); ++q)
1038  solution_values[i] +=
1039  fe_values->shape_grad_component(i, q, d) *
1040  internal::FEPointEvaluation::
1041  EvaluatorTypeTraits<dim, n_components, Number>::access(
1042  gradients[q], d);
1043  }
1044  }
1045  }
1046 }
