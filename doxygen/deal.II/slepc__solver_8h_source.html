634  template <typename OutputVector>
635  void
636  SolverBase::solve(const PETScWrappers::MatrixBase &A,
637  std::vector<PetscScalar> & eigenvalues,
638  std::vector<OutputVector> & eigenvectors,
639  const unsigned int n_eigenpairs)
640  {
641  // Panic if the number of eigenpairs wanted is out of bounds.
642  AssertThrow((n_eigenpairs > 0) && (n_eigenpairs <= A.m()),
643  ExcSLEPcWrappersUsageError());
644 
645  // Set the matrices of the problem
646  set_matrices(A);
647 
648  // and solve
649  unsigned int n_converged = 0;
650  solve(n_eigenpairs, &n_converged);
651 
652  if (n_converged > n_eigenpairs)
653  n_converged = n_eigenpairs;
654  AssertThrow(n_converged == n_eigenpairs,
655  ExcSLEPcEigenvectorConvergenceMismatchError(n_converged,
656  n_eigenpairs));
657 
658  AssertThrow(eigenvectors.size() != 0, ExcSLEPcWrappersUsageError());
659  eigenvectors.resize(n_converged, eigenvectors.front());
660  eigenvalues.resize(n_converged);
661 
662  for (unsigned int index = 0; index < n_converged; ++index)
663  get_eigenpair(index, eigenvalues[index], eigenvectors[index]);
664  }

706  SolverBase::solve(const PETScWrappers::MatrixBase &A,
707  const PETScWrappers::MatrixBase &B,
708  std::vector<double> & real_eigenvalues,
709  std::vector<double> & imag_eigenvalues,
710  std::vector<OutputVector> & real_eigenvectors,
711  std::vector<OutputVector> & imag_eigenvectors,
712  const unsigned int n_eigenpairs)
713  {
714  // Guard against incompatible matrix sizes:
715  AssertThrow(A.m() == B.m(), ExcDimensionMismatch(A.m(), B.m()));
716  AssertThrow(A.n() == B.n(), ExcDimensionMismatch(A.n(), B.n()));
717 
718  // and incompatible eigenvalue/eigenvector sizes
719  AssertThrow(real_eigenvalues.size() == imag_eigenvalues.size(),
720  ExcDimensionMismatch(real_eigenvalues.size(),
721  imag_eigenvalues.size()));
722  AssertThrow(real_eigenvectors.size() == imag_eigenvectors.size(),
723  ExcDimensionMismatch(real_eigenvectors.size(),
724  imag_eigenvectors.size()));
725 
726  // Panic if the number of eigenpairs wanted is out of bounds.
727  AssertThrow((n_eigenpairs > 0) && (n_eigenpairs <= A.m()),
728  ExcSLEPcWrappersUsageError());
729 
730  // Set the matrices of the problem
731  set_matrices(A, B);
732 
733  // and solve
734  unsigned int n_converged = 0;
735  solve(n_eigenpairs, &n_converged);
736 
737  if (n_converged >= n_eigenpairs)
738  n_converged = n_eigenpairs;
739 
740  AssertThrow(n_converged == n_eigenpairs,
741  ExcSLEPcEigenvectorConvergenceMismatchError(n_converged,
742  n_eigenpairs));
743  AssertThrow((real_eigenvectors.size() != 0) &&
744  (imag_eigenvectors.size() != 0),
745  ExcSLEPcWrappersUsageError());
746 
747  real_eigenvectors.resize(n_converged, real_eigenvectors.front());
748  imag_eigenvectors.resize(n_converged, imag_eigenvectors.front());
749  real_eigenvalues.resize(n_converged);
750  imag_eigenvalues.resize(n_converged);
751 
752  for (unsigned int index = 0; index < n_converged; ++index)
753  get_eigenpair(index,
754  real_eigenvalues[index],
755  imag_eigenvalues[index],
756  real_eigenvectors[index],
757  imag_eigenvectors[index]);

760  template <typename Vector>
761  void
762  SolverBase::set_initial_space(const std::vector<Vector> &this_initial_space)
763  {
764  std::vector<Vec> vecs(this_initial_space.size());
765 
766  for (unsigned int i = 0; i < this_initial_space.size(); i++)
767  {
768  Assert(this_initial_space[i].l2_norm() > 0.0,
769  ExcMessage("Initial vectors should be nonzero."));
770  vecs[i] = this_initial_space[i];
771  }
772 
773  // if the eigensolver supports only a single initial vector, but several
774  // guesses are provided, then all except the first one will be discarded.
775  // One could still build a vector that is rich in the directions of all
776  // guesses, by taking a linear combination of them. (TODO: make function
777  // virtual?)
778 
779  const PetscErrorCode ierr =
780  EPSSetInitialSpace(eps, vecs.size(), vecs.data());
781  AssertThrow(ierr == 0, ExcSLEPcError(ierr));
782  }
