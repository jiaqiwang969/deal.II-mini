210 template <typename MatrixType>
211 void
212 EigenPower<VectorType>::solve(double &value, const MatrixType &A, VectorType &x)
213 {
214  SolverControl::State conv = SolverControl::iterate;
215 
216  LogStream::Prefix prefix("Power method");
217 
218  typename VectorMemory<VectorType>::Pointer Vy(this->memory);
219  VectorType & y = *Vy;
220  y.reinit(x);
221  typename VectorMemory<VectorType>::Pointer Vr(this->memory);
222  VectorType & r = *Vr;
223  r.reinit(x);
224 
225  double length = x.l2_norm();
226  double old_length = 0.;
227  x *= 1. / length;
228 
229  A.vmult(y, x);
230 
231  // Main loop
232  int iter = 0;
233  for (; conv == SolverControl::iterate; iter++)
234  {
235  y.add(additional_data.shift, x);
236 
237  // Compute absolute value of eigenvalue
238  old_length = length;
239  length = y.l2_norm();
240 
241  // do a little trick to compute the sign
242  // with not too much effect of round-off errors.
243  double entry = 0.;
244  size_type i = 0;
245  double thresh = length / x.size();
246  do
247  {
248  Assert(i < x.size(), ExcInternalError());
249  entry = y(i++);
250  }
251  while (std::fabs(entry) < thresh);
252 
253  --i;
254 
255  // Compute unshifted eigenvalue
256  value = (entry * x(i) < 0.) ? -length : length;
257  value -= additional_data.shift;
258 
259  // Update normalized eigenvector
260  x.equ(1 / length, y);
261 
262  // Compute residual
263  A.vmult(y, x);
264 
265  // Check the change of the eigenvalue
266  // Brrr, this is not really a good criterion
267  conv = this->iteration_status(iter,
268  std::fabs(1. / length - 1. / old_length),
269  x);
270  }
271 
272  // in case of failure: throw exception
273  AssertThrow(conv == SolverControl::success,
274  SolverControl::NoConvergence(
275  iter, std::fabs(1. / length - 1. / old_length)));
276 
277  // otherwise exit as normal
278 }

293 template <typename MatrixType>
294 void
295 EigenInverse<VectorType>::solve(double & value,
296  const MatrixType &A,
297  VectorType & x)
298 {
299  LogStream::Prefix prefix("Wielandt");
300 
301  SolverControl::State conv = SolverControl::iterate;
302 
303  // Prepare matrix for solver
304  auto A_op = linear_operator(A);
305  double current_shift = -value;
306  auto A_s = A_op + current_shift * identity_operator(A_op);
307 
308  // Define solver
309  ReductionControl inner_control(5000, 1.e-16, 1.e-5, false, false);
310  PreconditionIdentity prec;
311  SolverGMRES<VectorType> solver(inner_control, this->memory);
312 
313  // Next step for recomputing the shift
314  unsigned int goal = additional_data.start_adaption;
315 
316  // Auxiliary vector
317  typename VectorMemory<VectorType>::Pointer Vy(this->memory);
318  VectorType & y = *Vy;
319  y.reinit(x);
320  typename VectorMemory<VectorType>::Pointer Vr(this->memory);
321  VectorType & r = *Vr;
322  r.reinit(x);
323 
324  double length = x.l2_norm();
325  double old_value = value;
326 
327  x *= 1. / length;
328 
329  // Main loop
330  double res = -std::numeric_limits<double>::max();
331  size_type iter = 0;
332  for (; conv == SolverControl::iterate; iter++)
333  {
334  solver.solve(A_s, y, x, prec);
335 
336  // Compute absolute value of eigenvalue
337  length = y.l2_norm();
338 
339  // do a little trick to compute the sign
340  // with not too much effect of round-off errors.
341  double entry = 0.;
342  size_type i = 0;
343  double thresh = length / x.size();
344  do
345  {
346  Assert(i < x.size(), ExcInternalError());
347  entry = y(i++);
348  }
349  while (std::fabs(entry) < thresh);
350 
351  --i;
352 
353  // Compute unshifted eigenvalue
354  value = (entry * x(i) < 0. ? -1. : 1.) / length - current_shift;
355 
356  if (iter == goal)
357  {
358  const auto & relaxation = additional_data.relaxation;
359  const double new_shift =
360  relaxation * (-value) + (1. - relaxation) * current_shift;
361 
362  A_s = A_op + new_shift * identity_operator(A_op);
363  current_shift = new_shift;
364 
365  ++goal;
366  }
367 
368  // Update normalized eigenvector
369  x.equ(1. / length, y);
370  // Compute residual
371  if (additional_data.use_residual)
372  {
373  y.equ(value, x);
374  A.vmult(r, x);
375  r.sadd(-1., value, x);
376  res = r.l2_norm();
377  // Check the residual
378  conv = this->iteration_status(iter, res, x);
379  }
380  else
381  {
382  res = std::fabs(1. / value - 1. / old_value);
383  conv = this->iteration_status(iter, res, x);
384  }
385  old_value = value;
386  }
387 
388  // in case of failure: throw
389  // exception
390  AssertThrow(conv == SolverControl::success,
391  SolverControl::NoConvergence(iter, res));
392  // otherwise exit as normal
393 }
