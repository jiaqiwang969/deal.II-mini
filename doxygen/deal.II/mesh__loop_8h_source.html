347  auto cell_action = [&](const CellIteratorBaseType &cell,
348  ScratchData & scratch,
349  CopyData & copy) {
350  // First reset the CopyData class to the empty copy_data given by the
351  // user.
352  copy = sample_copy_data;
353 
354  // Store the dimension in which we are working for later use
355  const auto dim = cell->get_triangulation().dimension;
356 
357  const bool ignore_subdomain =
358  (cell->get_triangulation().locally_owned_subdomain() ==
359  numbers::invalid_subdomain_id);
360 
361  types::subdomain_id current_subdomain_id =
362  (cell->is_level_cell() ? cell->level_subdomain_id() :
363  cell->subdomain_id());
364 
365  const bool own_cell =
366  ignore_subdomain ||
367  (current_subdomain_id ==
368  cell->get_triangulation().locally_owned_subdomain());
369 
370  if ((!ignore_subdomain) &&
371  (current_subdomain_id == numbers::artificial_subdomain_id))
372  return;
373 
374  if (!(flags & (cells_after_faces)) &&
375  (((flags & (assemble_own_cells)) && own_cell) ||
376  ((flags & assemble_ghost_cells) && !own_cell)))
377  cell_worker(cell, scratch, copy);
378 
379  if (flags & (work_on_faces | work_on_boundary))
380  for (const unsigned int face_no : cell->face_indices())
381  {
382  if (cell->at_boundary(face_no) &&
383  !cell->has_periodic_neighbor(face_no))
384  {
385  // only integrate boundary faces of own cells
386  if ((flags & assemble_boundary_faces) && own_cell)
387  boundary_worker(cell, face_no, scratch, copy);
388  }
389  else
390  {
391  // interior face, potentially assemble
392  TriaIterator<typename CellIteratorBaseType::AccessorType>
393  neighbor = cell->neighbor_or_periodic_neighbor(face_no);
394 
395  types::subdomain_id neighbor_subdomain_id =
396  numbers::artificial_subdomain_id;
397  if (neighbor->is_level_cell())
398  neighbor_subdomain_id = neighbor->level_subdomain_id();
399  // subdomain id is only valid for active cells
400  else if (neighbor->is_active())
401  neighbor_subdomain_id = neighbor->subdomain_id();
402 
403  const bool own_neighbor =
404  ignore_subdomain ||
405  (neighbor_subdomain_id ==
406  cell->get_triangulation().locally_owned_subdomain());
407 
408  // skip all faces between two ghost cells
409  if (!own_cell && !own_neighbor)
410  continue;
411 
412  // skip if the user doesn't want faces between own cells
413  if (own_cell && own_neighbor &&
414  !(flags & (assemble_own_interior_faces_both |
415  assemble_own_interior_faces_once)))
416  continue;
417 
418  // skip face to ghost
419  if (own_cell != own_neighbor &&
420  !(flags &
421  (assemble_ghost_faces_both | assemble_ghost_faces_once)))
422  continue;
423 
424  // Deal with refinement edges from the refined side. Assuming
425  // one-irregular meshes, this situation should only occur if
426  // both cells are active.
427  const bool periodic_neighbor =
428  cell->has_periodic_neighbor(face_no);
429 
430  if (dim > 1 && ((!periodic_neighbor &&
431  cell->neighbor_is_coarser(face_no) &&
432  neighbor->is_active()) ||
433  (periodic_neighbor &&
434  cell->periodic_neighbor_is_coarser(face_no) &&
435  neighbor->is_active())))
436  {
437  Assert(cell->is_active(), ExcInternalError());
438 
439  // skip if only one processor needs to assemble the face
440  // to a ghost cell and the fine cell is not ours.
441  if (!own_cell && (flags & assemble_ghost_faces_once))
442  continue;
443 
444  const std::pair<unsigned int, unsigned int>
445  neighbor_face_no =
446  periodic_neighbor ?
447  cell->periodic_neighbor_of_coarser_periodic_neighbor(
448  face_no) :
449  cell->neighbor_of_coarser_neighbor(face_no);
450 
451  face_worker(cell,
452  face_no,
453  numbers::invalid_unsigned_int,
454  neighbor,
455  neighbor_face_no.first,
456  neighbor_face_no.second,
457  scratch,
458  copy);
459 
460  if (flags & assemble_own_interior_faces_both)
461  {
462  // If own faces are to be assembled from both sides,
463  // call the faceworker again with swapped arguments.
464  // This is because we won't be looking at an adaptively
465  // refined edge coming from the other side.
466  face_worker(neighbor,
467  neighbor_face_no.first,
468  neighbor_face_no.second,
469  cell,
470  face_no,
471  numbers::invalid_unsigned_int,
472  scratch,
473  copy);
474  }
475  }
476  else if (dim == 1 && cell->level() > neighbor->level())
477  {
478  // In one dimension, there is no other check to do
479  const unsigned int neighbor_face_no =
480  periodic_neighbor ?
481  cell->periodic_neighbor_face_no(face_no) :
482  cell->neighbor_face_no(face_no);
483  Assert(periodic_neighbor ||
484  neighbor->face(neighbor_face_no) ==
485  cell->face(face_no),
486  ExcInternalError());
487 
488  face_worker(cell,
489  face_no,
490  numbers::invalid_unsigned_int,
491  neighbor,
492  neighbor_face_no,
493  numbers::invalid_unsigned_int,
494  scratch,
495  copy);
496 
497  if (flags & assemble_own_interior_faces_both)
498  {
499  // If own faces are to be assembled from both sides,
500  // call the faceworker again with swapped arguments.
501  face_worker(neighbor,
502  neighbor_face_no,
503  numbers::invalid_unsigned_int,
504  cell,
505  face_no,
506  numbers::invalid_unsigned_int,
507  scratch,
508  copy);
509  }
510  }
511  else
512  {
513  // If iterator is active and neighbor is refined, skip
514  // internal face.
515  if (::internal::is_active_iterator(cell) &&
516  neighbor->has_children())
517  continue;
518 
519  // Now neighbor is on the same refinement level.
520  // Double check.
521  Assert(!cell->neighbor_is_coarser(face_no),
522  ExcInternalError());
523 
524  // If we own both cells only do faces from one side (unless
525  // AssembleFlags says otherwise). Here, we rely on cell
526  // comparison that will look at cell->index().
527  if (own_cell && own_neighbor &&
528  (flags & assemble_own_interior_faces_once) &&
529  (neighbor < cell))
530  continue;
531 
532  // We only look at faces to ghost on the same level once
533  // (only where own_cell=true and own_neighbor=false)
534  if (!own_cell)
535  continue;
536 
537  // now only one processor assembles faces_to_ghost. We let
538  // the processor with the smaller (level-)subdomain id
539  // assemble the face.
540  if (own_cell && !own_neighbor &&
541  (flags & assemble_ghost_faces_once) &&
542  (neighbor_subdomain_id < current_subdomain_id))
543  continue;
544 
545  const unsigned int neighbor_face_no =
546  periodic_neighbor ?
547  cell->periodic_neighbor_face_no(face_no) :
548  cell->neighbor_face_no(face_no);
549  Assert(periodic_neighbor ||
550  neighbor->face(neighbor_face_no) ==
551  cell->face(face_no),
552  ExcInternalError());
553 
554  face_worker(cell,
555  face_no,
556  numbers::invalid_unsigned_int,
557  neighbor,
558  neighbor_face_no,
559  numbers::invalid_unsigned_int,
560  scratch,
561  copy);
562  }
563  }
564  } // faces
565 
566  // Execute the cell_worker if faces are handled before cells
567  if ((flags & cells_after_faces) &&
568  (((flags & assemble_own_cells) && own_cell) ||
569  ((flags & assemble_ghost_cells) && !own_cell)))
570  cell_worker(cell, scratch, copy);
571  };
572 
573  // Submit to workstream
574  WorkStream::run(begin,
575  end,
576  cell_action,
577  copier,
578  sample_scratch_data,
579  sample_copy_data,
580  queue_length,
581  chunk_size);
582  }
