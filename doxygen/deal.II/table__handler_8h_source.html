764  template <class Archive>
765  void
766  TableEntry::save(Archive &ar, const unsigned int) const
767  {
768  // write first an identifier for the kind
769  // of data stored and then the actual
770  // data, in its correct data type
771  if (const int *p = boost::get<int>(&value))
772  {
773  char c = 'i';
774  ar &c &*p;
775  }
776  else if (const unsigned int *p = boost::get<unsigned int>(&value))
777  {
778  char c = 'u';
779  ar &c &*p;
780  }
781  else if (const double *p = boost::get<double>(&value))
782  {
783  char c = 'd';
784  ar &c &*p;
785  }
786  else if (const std::string *p = boost::get<std::string>(&value))
787  {
788  char c = 's';
789  ar &c &*p;
790  }
791  else if (const std::uint64_t *p = boost::get<std::uint64_t>(&value))

863 template <typename T>
864 void
865 TableHandler::add_value(const std::string &key, const T value)
866 {
867  // see if the column already exists
868  if (columns.find(key) == columns.end())
869  declare_column(key);
870 
871  if (auto_fill_mode == true)
872  {
873  // follow the algorithm given in the introduction to this class
874  // of padding columns as necessary
875  unsigned int max_col_length = 0;
876  for (const auto &column : columns)
877  max_col_length =
878  std::max(max_col_length,
879  static_cast<unsigned int>(column.second.entries.size()));
880 
881  while (columns[key].entries.size() + 1 < max_col_length)
882  {
883  columns[key].entries.push_back(internal::TableEntry(T()));
884  internal::TableEntry &entry = columns[key].entries.back();
885  entry.cache_string(columns[key].scientific, columns[key].precision);
886  columns[key].max_length =
887  std::max(columns[key].max_length,
888  static_cast<unsigned int>(
889  entry.get_cached_string().length()));
890  }
891  }
892 
893  // now push the value given to this function
894  columns[key].entries.push_back(internal::TableEntry(value));
895  internal::TableEntry &entry = columns[key].entries.back();
896  entry.cache_string(columns[key].scientific, columns[key].precision);
897  columns[key].max_length =
898  std::max(columns[key].max_length,
899  static_cast<unsigned int>(entry.get_cached_string().length()));
900 }
