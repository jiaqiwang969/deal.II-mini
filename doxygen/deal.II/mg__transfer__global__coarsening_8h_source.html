653  AssertDimension(min_level, dst.min_level());

656  for (unsigned int level = min_level; level <= max_level; ++level)
657  initialize_dof_vector(level, dst[level]);
658 
659  dst[transfer.max_level()].copy_locally_owned_data_from(src);
660 
661  for (unsigned int l = max_level; l > min_level; --l)
662  this->transfer[l].interpolate(dst[l - 1], dst[l]);
