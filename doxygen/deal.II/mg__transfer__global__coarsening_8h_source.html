645 
646  AssertDimension(min_level, dst.min_level());
647  AssertDimension(max_level, dst.max_level());
648 
649  for (unsigned int level = min_level; level <= max_level; ++level)
650  initialize_dof_vector(level, dst[level]);
651 
652  dst[transfer.max_level()].copy_locally_owned_data_from(src);
653 
654  for (unsigned int l = max_level; l > min_level; --l)
655  this->transfer[l].interpolate(dst[l - 1], dst[l]);
