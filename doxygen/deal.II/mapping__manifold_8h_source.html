383 template <int dim, int spacedim>
384 inline void
385 MappingManifold<dim, spacedim>::InternalData::
386  compute_manifold_quadrature_weights(const Quadrature<dim> &quad)
387 {
388  cell_manifold_quadrature_weights.resize(
389  quad.size(), std::vector<double>(GeometryInfo<dim>::vertices_per_cell));
390  for (unsigned int q = 0; q < quad.size(); ++q)
391  {
392  for (const unsigned int i : GeometryInfo<dim>::vertex_indices())
393  {
394  cell_manifold_quadrature_weights[q][i] =
395  GeometryInfo<dim>::d_linear_shape_function(quad.point(q), i);
396  }
397  }
398 }
