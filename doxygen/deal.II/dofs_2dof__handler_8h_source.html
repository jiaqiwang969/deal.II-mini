1591 template <int dim, int spacedim>
1592 template <typename number>
1593 types::global_dof_index
1594 DoFHandler<dim, spacedim>::n_boundary_dofs(
1595  const std::map<types::boundary_id, const Function<spacedim, number> *>
1596  &boundary_ids) const
1597 {
1598  Assert(!(dim == 2 && spacedim == 3) || this->hp_capability_enabled == false,
1599  ExcNotImplementedWithHP());
1600 
1601  // extract the set of boundary ids and forget about the function object
1602  // pointers
1603  std::set<types::boundary_id> boundary_ids_only;
1604  for (typename std::map<types::boundary_id,
1605  const Function<spacedim, number> *>::const_iterator p =
1606  boundary_ids.begin();
1607  p != boundary_ids.end();
1608  ++p)
1609  boundary_ids_only.insert(p->first);
1610 
1611  // then just hand everything over to the other function that does the work
1612  return n_boundary_dofs(boundary_ids_only);
1613 }
