53  template <int dim>
54  void
55  cell_matrix(FullMatrix<double> & M,
56  const FEValuesBase<dim> &fe,
57  const FEValuesBase<dim> &fetest,
58  double factor = 1.)
59  {
60  const unsigned int n_dofs = fe.dofs_per_cell;
61  const unsigned int t_dofs = fetest.dofs_per_cell;
62  AssertDimension(fe.get_fe().n_components(), dim);
63  AssertDimension(fetest.get_fe().n_components(), 1);
64  AssertDimension(M.m(), t_dofs);
65  AssertDimension(M.n(), n_dofs);
66 
67  for (unsigned int k = 0; k < fe.n_quadrature_points; ++k)
68  {
69  const double dx = fe.JxW(k) * factor;
70  for (unsigned int i = 0; i < t_dofs; ++i)
71  {
72  const double vv = fetest.shape_value(i, k);
73  for (unsigned int d = 0; d < dim; ++d)
74  for (unsigned int j = 0; j < n_dofs; ++j)
75  {
76  const double du = fe.shape_grad_component(j, k, d)[d];
77  M(i, j) += dx * du * vv;
78  }
79  }
80  }

460  template <int dim>
461  double
462  norm(const FEValuesBase<dim> & fe,
463  const ArrayView<const std::vector<Tensor<1, dim>>> &Du)
464  {
465  AssertDimension(fe.get_fe().n_components(), dim);
466  AssertVectorVectorDimension(Du, dim, fe.n_quadrature_points);
467 
468  double result = 0;
469  for (unsigned int k = 0; k < fe.n_quadrature_points; ++k)
470  {
471  double div = Du[0][k][0];
472  for (unsigned int d = 1; d < dim; ++d)
473  div += Du[d][k][d];
474  result += div * div * fe.JxW(k);
475  }
476  return result;
