208  DerivativeForm<1, chartdim, spacedim> DF_DG;
209

211  for (unsigned int c = 0; c < chartdim; ++c)
212  for (unsigned int s = 0; s < intermediate_dim; ++s)
213  DF_DG[d][c] += DG[d][s] * DF[s][c];
214
