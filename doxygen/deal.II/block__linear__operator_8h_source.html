329  // A helper function to apply a given vmult, or Tvmult to a vector with
330  // intermediate storage, similar to the corresponding helper
331  // function for LinearOperator. Here, two operators are used.
332  // The first one takes care of the first "column" and typically doesn't add.
333  // On the other hand, the second operator is normally an adding one.
334  template <typename Function1,
335  typename Function2,
336  typename Range,
337  typename Domain>
338  void
339  apply_with_intermediate_storage(const Function1 &first_op,
340  const Function2 &loop_op,
341  Range & v,
342  const Domain & u,
343  bool add)
344  {
345  GrowingVectorMemory<Range> vector_memory;
346 
347  typename VectorMemory<Range>::Pointer tmp(vector_memory);
348  tmp->reinit(v, /*bool omit_zeroing_entries =*/ true);
349 
350  const unsigned int n = u.n_blocks();
351  const unsigned int m = v.n_blocks();
352 
353  for (unsigned int i = 0; i < m; ++i)
354  {
355  first_op(*tmp, u, i, 0);
356  for (unsigned int j = 1; j < n; ++j)
357  loop_op(*tmp, u, i, j);
358  }
359 
360  if (add)
361  v += *tmp;
362  else
363  v = *tmp;
364  }

777 template <std::size_t m, typename Range, typename Domain, typename BlockPayload>
778 BlockLinearOperator<Range, Domain, BlockPayload>
779 block_diagonal_operator(
780  const std::array<LinearOperator<typename Range::BlockType,
781  typename Domain::BlockType,
782  typename BlockPayload::BlockType>,
783  m> &ops)
784 {
785  static_assert(
786  m > 0, "a blockdiagonal LinearOperator must consist of at least one block");
787 
788  using BlockType =
789  typename BlockLinearOperator<Range, Domain, BlockPayload>::BlockType;
790 
791  std::array<std::array<BlockType, m>, m> new_ops;
792 
793  // This is a bit tricky. We have to make sure that the off-diagonal
794  // elements of return_op.ops are populated correctly. They must be
795  // null_operators, but with correct reinit_domain_vector and
796  // reinit_range_vector functions.
797  for (unsigned int i = 0; i < m; ++i)
798  for (unsigned int j = 0; j < m; ++j)
799  if (i == j)
800  {
801  // diagonal elements are easy:
802  new_ops[i][j] = ops[i];
803  }
804  else
805  {
806  // create a null-operator...
807  new_ops[i][j] = null_operator(ops[i]);
808  // ... and fix up reinit_domain_vector:
809  new_ops[i][j].reinit_domain_vector = ops[j].reinit_domain_vector;
810  }
811 
812  return block_operator<m, m, Range, Domain>(new_ops);
813 }
