573  template <int dim, int spacedim>
574  template <class... FETypes>
575  FECollection<dim, spacedim>::FECollection(const FETypes &... fes)
576  {
577  static_assert(
578  is_base_of_all<FiniteElement<dim, spacedim>, FETypes...>::value,
579  "Not all of the input arguments of this function "
580  "are derived from FiniteElement<dim,spacedim>!");
581 
582  // loop over all of the given arguments and add the finite elements to
583  // this collection. Inlining the definition of fe_pointers causes internal
584  // compiler errors on GCC 7.1.1 so we define it separately:
585  const auto fe_pointers = {
586  (static_cast<const FiniteElement<dim, spacedim> *>(&fes))...};
587  for (const auto p : fe_pointers)
588  push_back(*p);
589  }
