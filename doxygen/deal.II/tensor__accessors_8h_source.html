395  // OK. This is utterly brutal template magic. Therefore, we will not
396  // comment on the individual internal helper classes, because this is
397  // of not much value, but explain the general recursion procedure.
398  //
399  // (In order of appearance)
400  //
401  // Our task is to reorder access to a tensor object where a specified
402  // index is moved to the end. Thus we want to construct an object
403  // <code>reordered</code> out of a <code>tensor</code> where the
404  // following access patterns are equivalent:
405  // @code
406  // tensor [i_0]...[i_index-1][i_index][i_index+1]...[i_n]
407  // reordered [i_0]...[i_index_1][i_index+1]...[i_n][i_index]
408  // @endcode
409  //
410  // The first task is to get rid of the application of
411  // [i_0]...[i_index-1]. This is a classical recursion pattern - relay
412  // the task from <index, rank> to <index-1, rank-1> by accessing the
413  // subtensor object:
414 
415  template <int index, int rank, typename T>
416  class ReorderedIndexView
417  {
418  public:
419  constexpr ReorderedIndexView(typename ReferenceType<T>::type t)
420  : t_(t)
421  {}
422 
423  using value_type = ReorderedIndexView<index - 1,
424  rank - 1,
425  typename ValueType<T>::value_type>;
426 
427  // Recurse by applying index j directly:
428  constexpr DEAL_II_ALWAYS_INLINE value_type
429  operator[](unsigned int j) const
430  {
431  return value_type(t_[j]);
432  }
433 
434  private:
435  typename ReferenceType<T>::type t_;
436  };
437 
438  // At some point we hit the condition index == 0 and rank > 1, i.e.,
439  // the first index should be reordered to the end.
440  //
441  // At this point we cannot be lazy any more and have to start storing
442  // indices because we get them in the wrong order. The user supplies
443  // [i_0][i_1]...[i_{rank - 1}]
444  // but we have to call the subtensor object with
445  // [i_{rank - 1}[i_0][i_1]...[i_{rank-2}]
446  //
447  // So give up and relay the task to the StoreIndex class:
448 
449  template <int rank, typename T>
450  class ReorderedIndexView<0, rank, T>
451  {
452  public:
453  constexpr ReorderedIndexView(typename ReferenceType<T>::type t)
454  : t_(t)
455  {}
456 
457  using value_type = StoreIndex<rank - 1, internal::Identity<T>>;
458 
459  constexpr DEAL_II_ALWAYS_INLINE value_type
460  operator[](unsigned int j) const
461  {
462  return value_type(Identity<T>(t_), j);
463  }
464 
465  private:
466  typename ReferenceType<T>::type t_;
467  };

494  // StoreIndex. Its implementation is easy - we haven't stored any
495  // indices yet. So, we just provide a function apply that returns the
496  // application of an index j to the stored tensor t_:

518  // StoreIndex is a class that stores an index recursively with every
519  // invocation of operator[](unsigned int j): We do this by recursively
520  // creating a new StoreIndex class of lower rank that stores the
521  // supplied index j and holds a copy of the current class (with all
522  // other stored indices). Again, we provide an apply member function
523  // that knows how to apply an index on the highest rank and all
524  // subsequently stored indices:
525 
526  template <int rank, typename S>
527  class StoreIndex
528  {
529  public:
530  constexpr StoreIndex(S s, int i)
531  : s_(s)
532  , i_(i)
533  {}
534 
535  using value_type = StoreIndex<rank - 1, StoreIndex<rank, S>>;
536 
537  constexpr DEAL_II_ALWAYS_INLINE value_type
538  operator[](unsigned int j) const
539  {
540  return value_type(*this, j);
541  }
542 
543  using return_type =
544  typename ValueType<typename S::return_type>::value_type;
545 
546  constexpr typename ReferenceType<return_type>::type
547  apply(unsigned int j) const
548  {
549  return s_.apply(j)[i_];
550  }
551 
552  private:
553  const S s_;
554  const int i_;
555  };

624  // Straightforward recursive pattern:
625  //
626  // As long as rank_1 > no_contr, assign indices from the left tensor to
627  // result. This builds up the first part of the nested outer loops:
628  //
629  // for(unsigned int i_0; i_0 < dim; ++i_0)
630  // ...
631  // for(i_; i_ < dim; ++i_)
632  // [...]
633  // result[i_0]..[i_] ... left[i_0]..[i_] ...
634 
635  template <int no_contr, int rank_1, int rank_2, int dim>
636  class Contract
637  {
638  public:
639  template <typename T1, typename T2, typename T3>
640  constexpr inline DEAL_II_ALWAYS_INLINE static void
641  contract(T1 &result, const T2 &left, const T3 &right)
642  {
643  for (unsigned int i = 0; i < dim; ++i)
644  Contract<no_contr, rank_1 - 1, rank_2, dim>::contract(result[i],
645  left[i],
646  right);
647  }
648  };
649 
650  // If rank_1 == no_contr leave out the remaining no_contr indices for
651  // the contraction and assign indices from the right tensor to the
652  // result. This builds up the second part of the nested loops:
653  //
654  // for(unsigned int i_0 = 0; i_0 < dim; ++i_0)
655  // ...
656  // for(unsigned int i_ = 0; i_ < dim; ++i_)
657  // for(unsigned int j_0 = 0; j_0 < dim; ++j_0)
658  // ...
659  // for(unsigned int j_ = 0; j_ < dim; ++j_)
660  // [...]
661  // result[i_0]..[i_][j_0]..[j_] ... left[i_0]..[i_] ...
662  // right[j_0]..[j_]
663  //
664 
665  template <int no_contr, int rank_2, int dim>
666  class Contract<no_contr, no_contr, rank_2, dim>
667  {
668  public:
669  template <typename T1, typename T2, typename T3>
670  constexpr inline DEAL_II_ALWAYS_INLINE static void
671  contract(T1 &result, const T2 &left, const T3 &right)
672  {
673  for (unsigned int i = 0; i < dim; ++i)
674  Contract<no_contr, no_contr, rank_2 - 1, dim>::contract(result[i],
675  left,
676  right[i]);
677  }
678  };
679 
680  // If rank_1 == rank_2 == no_contr we have built up all of the outer
681  // loop. Now, it is time to do the actual contraction:
682  //
683  // [...]
684  // {
685  // result[i_0]..[i_][j_0]..[j_] = 0.;
686  // for(unsigned int k_0 = 0; k_0 < dim; ++k_0)
687  // ...
688  // for(unsigned int k_ = 0; k_ < dim; ++k_)
689  // result[i_0]..[i_][j_0]..[j_] += left[i_0]..[i_][k_0]..[k_] *
690  // right[j_0]..[j_][k_0]..[k_];
691  // }
692  //
693  // Relay this summation to another helper class.
694 
695  template <int no_contr, int dim>
696  class Contract<no_contr, no_contr, no_contr, dim>
697  {
698  public:
699  template <typename T1, typename T2, typename T3>
700  constexpr inline DEAL_II_ALWAYS_INLINE static void
701  contract(T1 &result, const T2 &left, const T3 &right)
702  {
703  result = Contract2<no_contr, dim>::template contract2<T1>(left, right);
704  }
705  };
706 
707  // Straightforward recursion:
708  //
709  // Contract leftmost index and recurse one down.
710 
711  template <int no_contr, int dim>
712  class Contract2
713  {
714  public:
715  template <typename T1, typename T2, typename T3>
716  constexpr inline DEAL_II_ALWAYS_INLINE static T1
717  contract2(const T2 &left, const T3 &right)
718  {
719  // Some auto-differentiable numbers need explicit
720  // zero initialization.
721  if (dim == 0)
722  {
723  T1 result = ::internal::NumberType<T1>::value(0.0);
724  return result;
725  }
726  else
727  {
728  T1 result =
729  Contract2<no_contr - 1, dim>::template contract2<T1>(left[0],
730  right[0]);
731  for (unsigned int i = 1; i < dim; ++i)
732  result +=
733  Contract2<no_contr - 1, dim>::template contract2<T1>(left[i],
734  right[i]);
735  return result;
736  }
737  }
738  };

760  // Fully contract three tensorial objects
761  //
762  // As long as rank_1 > 0, recurse over left and middle:
763  //
764  // for(unsigned int i_0; i_0 < dim; ++i_0)
765  // ...
766  // for(i_; i_ < dim; ++i_)
767  // [...]
768  // left[i_0]..[i_] ... middle[i_0]..[i_] ... right
769 
770  template <int rank_1, int rank_2, int dim>
771  class Contract3
772  {
773  public:
774  template <typename T1, typename T2, typename T3, typename T4>
775  constexpr static inline T1
776  contract3(const T2 &left, const T3 &middle, const T4 &right)
777  {
778  // Some auto-differentiable numbers need explicit
779  // zero initialization.
780  T1 result = ::internal::NumberType<T1>::value(0.0);
781  for (unsigned int i = 0; i < dim; ++i)
782  result += Contract3<rank_1 - 1, rank_2, dim>::template contract3<T1>(
783  left[i], middle[i], right);
784  return result;
785  }
786  };

788  // If rank_1 ==0, continue to recurse over middle and right:
789  //
790  // for(unsigned int i_0; i_0 < dim; ++i_0)
791  // ...
792  // for(i_; i_ < dim; ++i_)
793  // for(unsigned int j_0; j_0 < dim; ++j_0)
794  // ...
795  // for(j_; j_ < dim; ++j_)
796  // [...]
797  // left[i_0]..[i_] ... middle[i_0]..[i_][j_0]..[j_] ...
798  // right[j_0]..[j_]
799

811  for (unsigned int i = 0; i < dim; ++i)
812  result +=
813  Contract3<0, rank_2 - 1, dim>::template contract3<T1>(left,
