388 template <int spacedim, typename Number>
389 inline BoundingBox<spacedim, Number>::BoundingBox(
390  const std::pair<Point<spacedim, Number>, Point<spacedim, Number>>
391  &boundary_points)
392 {
393  // We check the Bounding Box is not degenerate
394  for (unsigned int i = 0; i < spacedim; ++i)
395  Assert(boundary_points.first[i] <= boundary_points.second[i],
396  ExcMessage("Bounding Box can't be created: the points' "
397  "order should be bottom left, top right!"));
398 
399  this->boundary_points = boundary_points;
400 }

404 template <int spacedim, typename Number>
405 template <class Container>
406 inline BoundingBox<spacedim, Number>::BoundingBox(const Container &points)
407 {
408  // Use the default constructor in case points is empty instead of setting
409  // things to +oo and -oo
410  if (points.size() > 0)
411  {
412  auto &min = boundary_points.first;
413  auto &max = boundary_points.second;
414  std::fill(min.begin_raw(),
415  min.end_raw(),
416  std::numeric_limits<Number>::infinity());
417  std::fill(max.begin_raw(),
418  max.end_raw(),
419  -std::numeric_limits<Number>::infinity());
420 
421  for (const Point<spacedim, Number> &point : points)
422  for (unsigned int d = 0; d < spacedim; ++d)
423  {
424  min[d] = std::min(min[d], point[d]);
425  max[d] = std::max(max[d], point[d]);
426  }
427  }
428 }
