153  template <int dim, typename Number, typename VectorizedArrayType>
154  inline MappingDataOnTheFly<dim, Number, VectorizedArrayType>::
155  MappingDataOnTheFly(const Mapping<dim> & mapping,
156  const Quadrature<1> &quadrature,
157  const UpdateFlags update_flags)
158  : fe_values(
159  mapping,
160  fe_dummy,
161  Quadrature<dim>(quadrature),
162  MappingInfo<dim, Number, VectorizedArrayType>::compute_update_flags(
163  update_flags))
164  , quadrature_1d(quadrature)
165  {
166  mapping_info_storage.descriptor.resize(1);
167  mapping_info_storage.descriptor[0].initialize(quadrature);
168  mapping_info_storage.data_index_offsets.resize(1);
169  mapping_info_storage.JxW_values.resize(fe_values.n_quadrature_points);
170  mapping_info_storage.jacobians[0].resize(fe_values.n_quadrature_points);
171  if (update_flags & update_quadrature_points)
172  {
173  mapping_info_storage.quadrature_point_offsets.resize(1, 0);
174  mapping_info_storage.quadrature_points.resize(
175  fe_values.n_quadrature_points);
176  }
177  if (fe_values.get_update_flags() & update_normal_vectors)
178  {
179  mapping_info_storage.normal_vectors.resize(
180  fe_values.n_quadrature_points);
181  mapping_info_storage.normals_times_jacobians[0].resize(
182  fe_values.n_quadrature_points);
183  }
184  Assert(!(fe_values.get_update_flags() & update_jacobian_grads),
185  ExcNotImplemented());
186  }

201  template <int dim, typename Number, typename VectorizedArrayType>
202  inline void
203  MappingDataOnTheFly<dim, Number, VectorizedArrayType>::reinit(
204  typename ::Triangulation<dim>::cell_iterator cell)
205  {
206  if (present_cell == cell)
207  return;
208  present_cell = cell;
209  fe_values.reinit(present_cell);
210  for (unsigned int q = 0; q < fe_values.get_quadrature().size(); ++q)
211  {
212  if (fe_values.get_update_flags() & update_JxW_values)
213  mapping_info_storage.JxW_values[q] = fe_values.JxW(q);
214  if (fe_values.get_update_flags() & update_jacobians)
215  {
216  Tensor<2, dim> jac = fe_values.jacobian(q);
217  jac = invert(transpose(jac));
218  for (unsigned int d = 0; d < dim; ++d)
219  for (unsigned int e = 0; e < dim; ++e)
220  mapping_info_storage.jacobians[0][q][d][e] = jac[d][e];
221  }
222  if (fe_values.get_update_flags() & update_quadrature_points)
223  for (unsigned int d = 0; d < dim; ++d)
224  mapping_info_storage.quadrature_points[q][d] =
225  fe_values.quadrature_point(q)[d];
226  if (fe_values.get_update_flags() & update_normal_vectors)
227  {
228  for (unsigned int d = 0; d < dim; ++d)
229  mapping_info_storage.normal_vectors[q][d] =
230  fe_values.normal_vector(q)[d];
231  mapping_info_storage.normals_times_jacobians[0][q] =
232  mapping_info_storage.normal_vectors[q] *
233  mapping_info_storage.jacobians[0][q];
234  }
235  }
236  }
