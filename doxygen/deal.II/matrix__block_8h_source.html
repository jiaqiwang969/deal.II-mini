632 MatrixBlock<MatrixType>::add(const size_type gi,
633  const size_type gj,
634  const typename MatrixType::value_type value)
635 {
636  Assert(row_indices.size() != 0, ExcNotInitialized());
637  Assert(column_indices.size() != 0, ExcNotInitialized());
638 
639  const std::pair<unsigned int, size_type> bi = row_indices.global_to_local(gi);
640  const std::pair<unsigned int, size_type> bj =
641  column_indices.global_to_local(gj);
642 
643  Assert(bi.first == row, ExcBlockIndexMismatch(bi.first, row));
644  Assert(bj.first == column, ExcBlockIndexMismatch(bj.first, column));
645 
646  matrix.add(bi.second, bj.second, value);

674 template <typename number>
675 inline void
676 MatrixBlock<MatrixType>::add(const size_type b_row,
677  const size_type n_cols,
678  const size_type *col_indices,
679  const number * values,
680  const bool,
681  const bool)
682 {
683  Assert(row_indices.size() != 0, ExcNotInitialized());
684  Assert(column_indices.size() != 0, ExcNotInitialized());
685 
686  const std::pair<unsigned int, size_type> bi =
687  row_indices.global_to_local(b_row);
688 
689  // In debug mode, we check whether
690  // all indices are in the correct
691  // block.
692 
693  // Actually, for the time being, we
694  // leave it at this. While it may
695  // not be the most efficient way,
696  // it is at least thread safe.
697  //#ifdef DEBUG
698  Assert(bi.first == row, ExcBlockIndexMismatch(bi.first, row));
699 
700  for (size_type j = 0; j < n_cols; ++j)
701  {
702  const std::pair<unsigned int, size_type> bj =
703  column_indices.global_to_local(col_indices[j]);
704  Assert(bj.first == column, ExcBlockIndexMismatch(bj.first, column));
705 
706  matrix.add(bi.second, bj.second, values[j]);
707  }
708  //#endif
709 }
