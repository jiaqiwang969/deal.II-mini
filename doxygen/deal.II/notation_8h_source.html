678  matrix_component_factor(const unsigned int component_i,
679  const unsigned int component_j,
680  const bool symmetric)
681  {
682  if (symmetric == false)
683  return 1.0;
684

902  {
903  const std::pair<unsigned int, unsigned int> indices_ij =
904  internal::indices_from_component<dim>(

906  Assert(indices_ij.first < dim, ExcInternalError());
907  Assert(indices_ij.second < dim, ExcInternalError());

909  {
910  Assert(indices_ij.second >= indices_ij.first,
911  ExcInternalError());

922  internal::indices_from_component<dim>(c, false);
