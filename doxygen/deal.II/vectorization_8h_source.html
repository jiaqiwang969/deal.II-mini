1221 template <>
1222 inline DEAL_II_ALWAYS_INLINE void
1223 vectorized_load_and_transpose(const unsigned int n_entries,
1224  const double * in,
1225  const unsigned int * offsets,
1226  VectorizedArray<double, 8> *out)
1227 {
1228  // do not do full transpose because the code is long and will most
1229  // likely not pay off because many processors have two load units
1230  // (for the top 8 instructions) but only 1 permute unit (for the 8
1231  // shuffle/unpack instructions). rather start the transposition on the

1771  const float * in,

1774 {
1775  // Similar to the double case, we perform the work on smaller entities. In
1776  // this case, we start from 128 bit arrays and insert them into a full 512
1777  // bit index. This reduces the code size and register pressure because we do
1778  // shuffles on 4 numbers rather than 16.
1779  const unsigned int n_chunks = n_entries / 4;
1780

1782  // variable to a pre-exisiting value in out, which will never get used in
1783  // the end. Keep the initialization outside the loop because of a bug in
1784  // gcc-9.1 which generates a "vmovapd" instruction instead of "vmovupd" in
1785  // case t3 is initialized to zero (inside/outside of loop), see
1786  // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=90991
1787  __m512 t0, t1, t2, t3;
1788  if (n_chunks > 0)
1789  t3 = out[0].data;
1790  for (unsigned int i = 0; i < n_chunks; ++i)
