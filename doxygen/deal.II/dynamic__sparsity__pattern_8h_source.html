797  {
798  Assert(sparsity_pattern != nullptr, DummyAccessor());
799  Assert(other.sparsity_pattern != nullptr, DummyAccessor());
800  // compare the sparsity pattern the iterator points into, the

812  inline bool
813  Accessor::operator<(const Accessor &other) const
814  {
815  Assert(sparsity_pattern != nullptr, DummyAccessor());
816  Assert(other.sparsity_pattern != nullptr, DummyAccessor());
817  Assert(sparsity_pattern == other.sparsity_pattern, ExcInternalError());
818 
819  // if *this is past-the-end, then it is less than no one
820  if (current_row == numbers::invalid_size_type)
821  return (false);
822  // now *this should be an valid value
823  Assert(current_row < sparsity_pattern->n_rows(), ExcInternalError());
824 
825  // if other is past-the-end
826  if (other.current_row == numbers::invalid_size_type)
827  return (true);
828  // now other should be an valid value
829  Assert(other.current_row < sparsity_pattern->n_rows(), ExcInternalError());
830 
831  // both iterators are not one-past-the-end
832  return ((current_row < other.current_row) ||
833  ((current_row == other.current_row) &&
834  (current_entry < other.current_entry)));
835  }

838  inline void
839  Accessor::advance()
840  {
841  Assert(sparsity_pattern != nullptr, DummyAccessor());
842  Assert(current_row < sparsity_pattern->n_rows(), ExcInternalError());
843 
844  // move to the next element in this row
845  ++current_entry;
846 
847  // if this moves us beyond the end of the row, go to the next row
848  // if possible, or set the iterator to an invalid state if not.
849  //
850  // going to the next row is a bit complicated because we may have
851  // to skip over empty rows, and because we also have to avoid rows
852  // that aren't listed in a possibly passed IndexSet argument of
853  // the sparsity pattern. consequently, rather than trying to
854  // duplicate code here, just call the begin() function of the
855  // sparsity pattern itself
856  if (current_entry == end_of_row)
857  {
858  if (current_row + 1 < sparsity_pattern->n_rows())
859  *this = *sparsity_pattern->begin(current_row + 1);
860  else
861  *this = Accessor(sparsity_pattern); // invalid object
862  }
863  }

948 inline void
949 DynamicSparsityPattern::Line::add(const size_type j)
950 {
951  // first check the last element (or if line is still empty)
952  if ((entries.size() == 0) || (entries.back() < j))
953  {
954  entries.push_back(j);
955  return;
956  }
957 
958  // do a binary search to find the place where to insert:
959  std::vector<size_type>::iterator it =
960  Utilities::lower_bound(entries.begin(), entries.end(), j);
961 
962  // If this entry is a duplicate, exit immediately
963  if (*it == j)
964  return;
965 
966  // Insert at the right place in the vector. Vector grows automatically to
967  // fit elements. Always doubles its size.
968  entries.insert(it, j);
969 }

1080 inline DynamicSparsityPattern::iterator
1081 DynamicSparsityPattern::begin(const size_type r) const
1082 {
1083  AssertIndexRange(r, n_rows());
1084 
1085  if (!have_entries)
1086  return {this};
1087 
1088  if (rowset.size() > 0)
1089  {
1090  // We have an IndexSet that describes the locally owned set. For
1091  // performance reasons we need to make sure that we don't do a
1092  // linear search over 0..n_rows(). Instead, find the first entry
1093  // >= row r in the locally owned set (this is done in log
1094  // n_ranges time inside at()). From there, we move forward until
1095  // we find a non-empty row. By iterating over the IndexSet instead
1096  // of incrementing the row index, we potentially skip over entries
1097  // not in the rowset.
1098  IndexSet::ElementIterator it = rowset.at(r);
1099  if (it == rowset.end())
1100  return end(); // we don't own any row between r and the end
1101 
1102  // Instead of using row_length(*it)==0 in the while loop below,
1103  // which involves an expensive index_within_set() call, we
1104  // look at the lines vector directly. This works, because we are
1105  // walking over this vector entry by entry anyways.
1106  size_type rowindex = rowset.index_within_set(*it);
1107 
1108  while (it != rowset.end() && lines[rowindex].entries.size() == 0)
1109  {
1110  ++it;
1111  ++rowindex;
1112  }
1113 
1114  if (it == rowset.end())
1115  return end();
1116  else
1117  return {this, *it, 0};
1118  }
1119 
1120  // Without an index set we have to do a linear search starting at
1121  // row r until we find a non-empty one. We will check the lines vector
1122  // directly instead of going through the slower row_length() function
1123  size_type row = r;
1124 
1125  while (row < n_rows() && lines[row].entries.size() == 0)
1126  {
1127  ++row;
1128  }
1129 
1130  if (row == n_rows())
1131  return {this};
1132  else
1133  return {this, row, 0};
1134 }
