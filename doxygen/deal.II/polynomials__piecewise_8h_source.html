174  template <typename number>
175  inline number
176  PiecewisePolynomial<number>::value(const number x) const
177  {
178  AssertIndexRange(interval, n_intervals);
179  number y = x;
180  // shift polynomial if necessary
181  if (n_intervals > 1)
182  {
183  const number step = 1. / n_intervals;
184 
185  // polynomial spans over two intervals
186  if (spans_two_intervals == true)
187  {
188  const number offset = step * interval;
189  if (x < offset)
190  return 0;
191  else if (x > offset + step + step)
192  return 0;
193  else if (x < offset + step)
194  y = x - offset;
195  else
196  y = offset + step + step - x;
197  }
198  else
199  {
200  const number offset = step * interval;
201  if (x < offset || x > offset + step)
202  return 0;
203  else
204  y = x - offset;
205  }
206 
207  return polynomial.value(y);
208  }
209  else
210  return polynomial.value(x);
211  }
