496  template <int dim, int sdim>
497  inline IntegrationInfo<dim, sdim>::IntegrationInfo(
498  const IntegrationInfo<dim, sdim> &other)
499  : multigrid(other.multigrid)
500  , values(other.values)
501  , gradients(other.gradients)
502  , hessians(other.hessians)
503  , global_data(other.global_data)
504  , fe_pointer(other.fe_pointer)
505  , n_components(other.n_components)
506  {
507  fevalv.resize(other.fevalv.size());
508  for (unsigned int i = 0; i < other.fevalv.size(); ++i)
509  {
510  const FEValuesBase<dim, sdim> &p = *other.fevalv[i];
511  const FEValues<dim, sdim> * pc =
512  dynamic_cast<const FEValues<dim, sdim> *>(&p);
513  const FEFaceValues<dim, sdim> *pf =
514  dynamic_cast<const FEFaceValues<dim, sdim> *>(&p);
515  const FESubfaceValues<dim, sdim> *ps =
516  dynamic_cast<const FESubfaceValues<dim, sdim> *>(&p);
517 
518  if (pc != nullptr)
519  fevalv[i] =
520  std::make_shared<FEValues<dim, sdim>>(pc->get_mapping(),
521  pc->get_fe(),
522  pc->get_quadrature(),
523  pc->get_update_flags());
524  else if (pf != nullptr)
525  fevalv[i] =
526  std::make_shared<FEFaceValues<dim, sdim>>(pf->get_mapping(),
527  pf->get_fe(),
528  pf->get_quadrature(),
529  pf->get_update_flags());
530  else if (ps != nullptr)
531  fevalv[i] = std::make_shared<FESubfaceValues<dim, sdim>>(
532  ps->get_mapping(),
533  ps->get_fe(),
534  ps->get_quadrature(),
535  ps->get_update_flags());
536  else
537  Assert(false, ExcInternalError());
538  }
539  }

543  template <int dim, int sdim>
544  template <class FEVALUES>
545  inline void
546  IntegrationInfo<dim, sdim>::initialize(
547  const FiniteElement<dim, sdim> & el,
548  const Mapping<dim, sdim> & mapping,
549  const Quadrature<FEVALUES::integral_dimension> &quadrature,
550  const UpdateFlags flags,
551  const BlockInfo * block_info)
552  {
553  fe_pointer = &el;
554  if (block_info == nullptr || block_info->local().size() == 0)
555  {
556  fevalv.resize(1);
557  fevalv[0] = std::make_shared<FEVALUES>(mapping, el, quadrature, flags);
558  }
559  else
560  {
561  fevalv.resize(el.n_base_elements());
562  for (unsigned int i = 0; i < fevalv.size(); ++i)
563  fevalv[i] = std::make_shared<FEVALUES>(mapping,
564  el.base_element(i),
565  quadrature,
566  flags);
567  }
568  n_components = el.n_components();
569  }

598  template <int dim, int spacedim>
599  template <typename number>
600  inline void
601  IntegrationInfo<dim, spacedim>::reinit(
602  const DoFInfo<dim, spacedim, number> &info)
603  {
604  for (unsigned int i = 0; i < fevalv.size(); ++i)
605  {
606  FEValuesBase<dim, spacedim> &febase = *fevalv[i];
607  if (info.sub_number != numbers::invalid_unsigned_int)
608  {
609  // This is a subface
610  FESubfaceValues<dim, spacedim> &fe =
611  dynamic_cast<FESubfaceValues<dim, spacedim> &>(febase);
612  fe.reinit(info.cell, info.face_number, info.sub_number);
613  }
614  else if (info.face_number != numbers::invalid_unsigned_int)
615  {
616  // This is a face
617  FEFaceValues<dim, spacedim> &fe =
618  dynamic_cast<FEFaceValues<dim, spacedim> &>(febase);
619  fe.reinit(info.cell, info.face_number);
620  }
621  else
622  {
623  // This is a cell
624  FEValues<dim, spacedim> &fe =
625  dynamic_cast<FEValues<dim, spacedim> &>(febase);
626  fe.reinit(info.cell);
627  }
628  }
629 
630  const bool split_fevalues = info.block_info != nullptr;
631  if (!global_data->empty())
632  fill_local_data(info, split_fevalues);
633  }
