3790  template <typename Iterator, int spacedim, int structdim>
3791  Point<spacedim>
3792  project_to_d_linear_object(const Iterator & object,
3793  const Point<spacedim> &trial_point)
3794  {
3795  // let's look at this for simplicity for a quad (structdim==2) in a
3796  // space with spacedim>2 (notate trial_point by y): all points on the
3797  // surface are given by
3798  // x(\xi) = sum_i v_i phi_x(\xi)
3799  // where v_i are the vertices of the quad, and \xi=(\xi_1,\xi_2) are the
3800  // reference coordinates of the quad. so what we are trying to do is
3801  // find a point x on the surface that is closest to the point y. there
3802  // are different ways to solve this problem, but in the end it's a
3803  // nonlinear problem and we have to find reference coordinates \xi so
3804  // that J(\xi) = 1/2 || x(\xi)-y ||^2 is minimal. x(\xi) is a function
3805  // that is structdim-linear in \xi, so J(\xi) is a polynomial of degree
3806  // 2*structdim that we'd like to minimize. unless structdim==1, we'll
3807  // have to use a Newton method to find the answer. This leads to the
3808  // following formulation of Newton steps:
3809  //
3810  // Given \xi_k, find \delta\xi_k so that
3811  // H_k \delta\xi_k = - F_k
3812  // where H_k is an approximation to the second derivatives of J at
3813  // \xi_k, and F_k is the first derivative of J. We'll iterate this a
3814  // number of times until the right hand side is small enough. As a
3815  // stopping criterion, we terminate if ||\delta\xi||<eps.
3816  //
3817  // As for the Hessian, the best choice would be
3818  // H_k = J''(\xi_k)
3819  // but we'll opt for the simpler Gauss-Newton form
3820  // H_k = A^T A
3821  // i.e.
3822  // (H_k)_{nm} = \sum_{i,j} v_i*v_j *
3823  // \partial_n phi_i *
3824  // \partial_m phi_j
3825  // we start at xi=(0.5, 0.5).
3826  Point<structdim> xi;
3827  for (unsigned int d = 0; d < structdim; ++d)
3828  xi[d] = 0.5;
3829 
3830  Point<spacedim> x_k;
3831  for (const unsigned int i : GeometryInfo<structdim>::vertex_indices())
3832  x_k += object->vertex(i) *
3833  GeometryInfo<structdim>::d_linear_shape_function(xi, i);
3834 
3835  do
3836  {
3837  Tensor<1, structdim> F_k;
3838  for (const unsigned int i :
3839  GeometryInfo<structdim>::vertex_indices())
3840  F_k +=
3841  (x_k - trial_point) * object->vertex(i) *
3842  GeometryInfo<structdim>::d_linear_shape_function_gradient(xi,
3843  i);
3844 
3845  Tensor<2, structdim> H_k;
3846  for (const unsigned int i :
3847  GeometryInfo<structdim>::vertex_indices())
3848  for (const unsigned int j :
3849  GeometryInfo<structdim>::vertex_indices())
3850  {
3851  Tensor<2, structdim> tmp = outer_product(
3852  GeometryInfo<structdim>::d_linear_shape_function_gradient(
3853  xi, i),
3854  GeometryInfo<structdim>::d_linear_shape_function_gradient(
3855  xi, j));
3856  H_k += (object->vertex(i) * object->vertex(j)) * tmp;
3857  }
3858 
3859  const Tensor<1, structdim> delta_xi = -invert(H_k) * F_k;
3860  xi += delta_xi;
3861 
3862  x_k = Point<spacedim>();
3863  for (const unsigned int i :
3864  GeometryInfo<structdim>::vertex_indices())
3865  x_k += object->vertex(i) *
3866  GeometryInfo<structdim>::d_linear_shape_function(xi, i);
3867 
3868  if (delta_xi.norm() < 1e-7)
3869  break;
3870  }
3871  while (true);
3872 
3873  return x_k;
3874  }

3911  Point<Iterator::AccessorType::space_dimension>
3912  project_to_object(
3913  const Iterator & object,
3914  const Point<Iterator::AccessorType::space_dimension> &trial_point)
3915  {

3917  const int structdim = Iterator::AccessorType::structure_dimension;
3918 
3919  Point<spacedim> projected_point = trial_point;
3920 
3921  if (structdim >= spacedim)
3922  return projected_point;
3923  else if (structdim == 1 || structdim == 2)
3924  {
3925  using namespace internal::ProjectToObject;
3926  // Try to use the special flat algorithm for quads (this is better
3927  // than the general algorithm in 3D). This does not take into account
3928  // whether projected_point is outside the quad, but we optimize along
3929  // lines below anyway:
3930  const int dim = Iterator::AccessorType::dimension;
3931  const Manifold<dim, spacedim> &manifold = object->get_manifold();
3932  if (structdim == 2 && dynamic_cast<const FlatManifold<dim, spacedim> *>(
3933  &manifold) != nullptr)
3934  {
3935  projected_point =
3936  project_to_d_linear_object<Iterator, spacedim, structdim>(
3937  object, trial_point);
3938  }
3939  else
3940  {
3941  // We want to find a point on the convex hull (defined by the
3942  // vertices of the object and the manifold description) that is
3943  // relatively close to the trial point. This has a few issues:
3944  //
3945  // 1. For a general convex hull we are not guaranteed that a unique
3946  // minimum exists.
3947  // 2. The independent variables in the optimization process are the
3948  // weights given to Manifold::get_new_point, which must sum to 1,
3949  // so we cannot use standard finite differences to approximate a
3950  // gradient.
3951  //
3952  // There is not much we can do about 1., but for 2. we can derive
3953  // finite difference stencils that work on a structdim-dimensional
3954  // simplex and rewrite the optimization problem to use those
3955  // instead. Consider the structdim 2 case and let
3956  //
3957  // F(c0, c1, c2, c3) = Manifold::get_new_point(vertices, {c0, c1,
3958  // c2, c3})
3959  //
3960  // where {c0, c1, c2, c3} are the weights for the four vertices on
3961  // the quadrilateral. We seek to minimize the Euclidean distance
3962  // between F(...) and trial_point. We can solve for c3 in terms of
3963  // the other weights and get, for one coordinate direction
3964  //
3965  // d/dc0 ((x0 - F(c0, c1, c2, 1 - c0 - c1 - c2))^2)
3966  // = -2(x0 - F(...)) (d/dc0 F(...) - d/dc3 F(...))
3967  //
3968  // where we substitute back in for c3 after taking the
3969  // derivative. We can compute a stencil for the cross derivative
3970  // d/dc0 - d/dc3: this is exactly what cross_stencil approximates
3971  // (and gradient_entry computes the sum over the independent
3972  // variables). Below, we somewhat arbitrarily pick the last
3973  // component as the dependent one.
3974  //
3975  // Since we can now calculate derivatives of the objective
3976  // function we can use gradient descent to minimize it.
3977  //
3978  // Of course, this is much simpler in the structdim = 1 case (we
3979  // could rewrite the projection as a 1D optimization problem), but
3980  // to reduce the potential for bugs we use the same code in both
3981  // cases.
3982  const double step_size = object->diameter() / 64.0;
3983 
3984  constexpr unsigned int n_vertices_per_cell =
3985  GeometryInfo<structdim>::vertices_per_cell;
3986 
3987  std::array<Point<spacedim>, n_vertices_per_cell> vertices;
3988  for (unsigned int vertex_n = 0; vertex_n < n_vertices_per_cell;
3989  ++vertex_n)
3990  vertices[vertex_n] = object->vertex(vertex_n);
3991

3993  [&](const Tensor<1, n_vertices_per_cell> &weights)
3994  -> Point<spacedim> {
3995  return object->get_manifold().get_new_point(
3996  make_array_view(vertices.begin(), vertices.end()),
3997  make_array_view(weights.begin_raw(), weights.end_raw()));
3998  };
3999 
4000  // pick the initial weights as (normalized) inverse distances from
4001  // the trial point:
4002  Tensor<1, n_vertices_per_cell> guess_weights;
4003  double guess_weights_sum = 0.0;
4004  for (unsigned int vertex_n = 0; vertex_n < n_vertices_per_cell;
4005  ++vertex_n)
4006  {
4007  const double distance =
4008  vertices[vertex_n].distance(trial_point);
4009  if (distance == 0.0)
4010  {
4011  guess_weights = 0.0;
4012  guess_weights[vertex_n] = 1.0;
4013  guess_weights_sum = 1.0;
4014  break;
4015  }
4016  else
4017  {
4018  guess_weights[vertex_n] = 1.0 / distance;
4019  guess_weights_sum += guess_weights[vertex_n];

4021  }
4022  guess_weights /= guess_weights_sum;
4023  Assert(internal::weights_are_ok<structdim>(guess_weights),
4024  ExcInternalError());

4027  //
4028  // 1. An outer loop where we apply the gradient descent algorithm.
4029  // 2. An inner loop where we do a line search to find the optimal
4030  // length of the step one should take in the gradient direction.
4031  //
4032  for (unsigned int outer_n = 0; outer_n < 40; ++outer_n)
4033  {
4034  const unsigned int dependent_direction =
4035  n_vertices_per_cell - 1;
4036  Tensor<1, n_vertices_per_cell> current_gradient;
4037  for (unsigned int row_n = 0; row_n < n_vertices_per_cell;
4038  ++row_n)

4049  get_point_from_weights);
4050 
4051  current_gradient[dependent_direction] -=
4052  current_gradient[row_n];
4053  }
4054  }

4057  // above, but we may not want to take a full step in that
4058  // direction; instead, guess that we will go -0.5*gradient and
4059  // do quasi-Newton iteration to pick the best multiplier. The
4060  // goal is to find a scalar alpha such that
4061  //
4062  // F(x - alpha g)
4063  //
4064  // is minimized, where g is the gradient and F is the
4065  // objective function. To find the optimal value we find roots
4066  // of the derivative of the objective function with respect to
4067  // alpha by Newton iteration, where we approximate the first
4068  // and second derivatives of F(x - alpha g) with centered
4069  // finite differences.

4072  [&](const double gradient_weight_guess) -> double {

4076  current_gradient))

4081  {

4084  step_size,

4087  centered_second_difference(
4088  gradient_weight,
4089  step_size,
4090  gradient_weight_objective_function);

4093  // weight below

4095  break;

4097  gradient_weight - update_numerator / update_denominator;

4100  // gradient (things tend to be locally flat, so the gradient
4101  // itself is usually small)
4102  if (std::abs(gradient_weight) > 10)
4103  {
4104  gradient_weight = -10.0;
4105  break;
4106  }
4107  }
4108

4110  // between zero and one. If the update takes us outside of this
4111  // region then rescale the update to stay within the region and
4112  // try again
4113  Tensor<1, n_vertices_per_cell> tentative_weights =
4114  guess_weights + gradient_weight * current_gradient;
4115 
4116  double new_gradient_weight = gradient_weight;
4117  for (unsigned int iteration_count = 0; iteration_count < 40;
4118  ++iteration_count)
4119  {
4120  if (internal::weights_are_ok<structdim>(tentative_weights))
4121  break;
