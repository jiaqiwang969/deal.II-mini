1193  template <typename number>
1194  std::shared_ptr<hid_t>
1195  get_hdf5_datatype()
1196  {
1197  static_assert(std::is_same<number, float>::value ||
1198  std::is_same<number, double>::value ||
1199  std::is_same<number, int>::value ||
1200  std::is_same<number, bool>::value ||
1201  std::is_same<number, unsigned int>::value ||
1202  std::is_same<number, std::complex<float>>::value ||
1203  std::is_same<number, std::complex<double>>::value,
1204  "The data type you are trying to get the HDF5 tag for "
1205  "is not supported by this function.");
1206 
1207  if (std::is_same<number, float>::value)
1208  {
1209  return std::make_shared<hid_t>(H5T_NATIVE_FLOAT);
1210  }
1211  else if (std::is_same<number, double>::value)
1212  {
1213  return std::make_shared<hid_t>(H5T_NATIVE_DOUBLE);
1214  }
1215  else if (std::is_same<number, int>::value)
1216  {
1217  return std::make_shared<hid_t>(H5T_NATIVE_INT);
1218  }
1219  else if (std::is_same<number, unsigned int>::value)
1220  {
1221  return std::make_shared<hid_t>(H5T_NATIVE_UINT);
1222  }
1223  else if (std::is_same<number, bool>::value)
1224  {
1225  return std::make_shared<hid_t>(H5T_NATIVE_HBOOL);
1226  }
1227  else if (std::is_same<number, std::complex<float>>::value)
1228  {
1229  std::shared_ptr<hid_t> t_type =
1230  std::shared_ptr<hid_t>(new hid_t, [](hid_t *pointer) {
1231  // Release the HDF5 resource
1232  const herr_t ret = H5Tclose(*pointer);
1233  AssertNothrow(ret >= 0, ExcInternalError());
1234  (void)ret;
1235  delete pointer;
1236  });
1237 
1238  *t_type = H5Tcreate(H5T_COMPOUND, sizeof(std::complex<float>));
1239  // The C++ standards committee agreed to mandate that the storage
1240  // format used for the std::complex type be binary-compatible with
1241  // the C99 type, i.e. an array T[2] with consecutive real [0] and
1242  // imaginary [1] parts.
1243  herr_t ret = H5Tinsert(*t_type, "r", 0, H5T_NATIVE_FLOAT);
1244  Assert(ret >= 0, ExcInternalError());
1245  ret = H5Tinsert(*t_type, "i", sizeof(float), H5T_NATIVE_FLOAT);
1246  Assert(ret >= 0, ExcInternalError());
1247  (void)ret;
1248  return t_type;
1249  }
1250  else if (std::is_same<number, std::complex<double>>::value)
1251  {
1252  std::shared_ptr<hid_t> t_type =
1253  std::shared_ptr<hid_t>(new hid_t, [](hid_t *pointer) {
1254  // Release the HDF5 resource
1255  const herr_t ret = H5Tclose(*pointer);
1256  AssertNothrow(ret >= 0, ExcInternalError());
1257  (void)ret;
1258  delete pointer;
1259  });
1260  *t_type = H5Tcreate(H5T_COMPOUND, sizeof(std::complex<double>));
1261  // The C++ standards committee agreed to mandate that the storage
1262  // format used for the std::complex type be binary-compatible with
1263  // the C99 type, i.e. an array T[2] with consecutive real [0] and
1264  // imaginary [1] parts.
1265  herr_t ret = H5Tinsert(*t_type, "r", 0, H5T_NATIVE_DOUBLE);
1266  Assert(ret >= 0, ExcInternalError());
1267  ret = H5Tinsert(*t_type, "i", sizeof(double), H5T_NATIVE_DOUBLE);
1268  Assert(ret >= 0, ExcInternalError());
1269  (void)ret;
1270  return t_type;
1271  }
1272 
1273  // The function should not reach this point
1274  Assert(false, ExcInternalError());
1275  return {};
1276  }

1457  inline std::string
1458  no_collective_cause_to_string(const uint32_t no_collective_cause)
1459  {
1460  std::string message;
1461 
1462  auto append_to_message = [&message](const char *p) {
1463  if (message.length() > 0)
1464  message += ", ";
1465  message += p;
1466  };
1467 
1468  // The first is not a bitmask comparison, the rest are bitmask
1469  // comparisons.
1470  // https://support.hdfgroup.org/HDF5/doc/RM/RM_H5P.html#Property-GetMpioNoCollectiveCause
1471  // See H5Ppublic.h
1472  // Hex codes are used because the HDF5 Group can deprecate some of the
1473  // enum codes. For example the enum code H5D_MPIO_FILTERS is not defined
1474  // in 1.10.2 because it is possible to use compressed datasets with the
1475  // MPI/IO driver.
1476 
1477  // H5D_MPIO_COLLECTIVE
1478  if (no_collective_cause == 0x00)
1479  {
1480  append_to_message("H5D_MPIO_COLLECTIVE");
1481  }
1482  // H5D_MPIO_SET_INDEPENDENT
1483  if ((no_collective_cause & 0x01) == 0x01)
1484  {
1485  append_to_message("H5D_MPIO_SET_INDEPENDENT");
1486  }
1487  // H5D_MPIO_DATATYPE_CONVERSION
1488  if ((no_collective_cause & 0x02) == 0x02)
1489  {
1490  append_to_message("H5D_MPIO_DATATYPE_CONVERSION");
1491  }
1492  // H5D_MPIO_DATA_TRANSFORMS
1493  if ((no_collective_cause & 0x04) == 0x04)
1494  {
1495  append_to_message("H5D_MPIO_DATA_TRANSFORMS");
1496  }
1497  // H5D_MPIO_NOT_SIMPLE_OR_SCALAR_DATASPACES
1498  if ((no_collective_cause & 0x10) == 0x10)
1499  {
1500  append_to_message("H5D_MPIO_NOT_SIMPLE_OR_SCALAR_DATASPACES");
1501  }
1502  // H5D_MPIO_NOT_CONTIGUOUS_OR_CHUNKED_DATASET
1503  if ((no_collective_cause & 0x20) == 0x20)
1504  {
1505  append_to_message("H5D_MPIO_NOT_CONTIGUOUS_OR_CHUNKED_DATASET");
1506  }
1507  // H5D_MPIO_FILTERS
1508  if ((no_collective_cause & 0x40) == 0x40)
1509  {
1510  append_to_message("H5D_MPIO_FILTERS");
1511  }
1512  return message;
1513  }

1540  template <>
1541  inline std::string
1542  HDF5Object::get_attribute(const std::string &attr_name) const
1543  {
1544  // Reads a UTF8 variable string
1545  //
1546  // code inspired from
1547  // https://support.hdfgroup.org/ftp/HDF5/examples/misc-examples/vlstratt.c
1548  //
1549  // In the case of a variable length string the user does not have to reserve
1550  // memory for string_out. H5Aread will reserve the memory and the
1551  // user has to free the memory.
1552  //
1553  // Todo:
1554  // - Use H5Dvlen_reclaim instead of free
1555 
1556  char * string_out;
1557  hid_t attr;
1558  hid_t type;
1559  herr_t ret;
1560 
1561  /* Create a datatype to refer to. */
1562  type = H5Tcopy(H5T_C_S1);
1563  Assert(type >= 0, ExcInternalError());
1564 
1565  // Python strings are encoded in UTF8
1566  ret = H5Tset_cset(type, H5T_CSET_UTF8);
1567  Assert(type >= 0, ExcInternalError());
1568 
1569  ret = H5Tset_size(type, H5T_VARIABLE);
