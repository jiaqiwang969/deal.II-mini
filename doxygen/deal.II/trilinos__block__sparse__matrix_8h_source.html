373  Assert(d == 0, ExcScalarAssignmentOnlyForZeroValue());
374 
375  for (size_type r = 0; r < this->n_block_rows(); ++r)
376  for (size_type c = 0; c < this->n_block_cols(); ++c)

378 
379  return *this;
