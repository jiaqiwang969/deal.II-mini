926 template <typename Accessor>
927 inline bool
928 TriaRawIterator<Accessor>::
929 operator<(const TriaRawIterator<Accessor> &other) const
930 {
931  Assert(state() != IteratorState::invalid,
932  ExcDereferenceInvalidObject(accessor));
933  Assert(other.state() != IteratorState::invalid,
934  ExcDereferenceInvalidObject(other.accessor));
935 
936  Assert(&accessor.get_triangulation() == &other.accessor.get_triangulation(),
937  ExcInvalidComparison());
938 
939  // Deal with iterators past end
940  if (state() == IteratorState::past_the_end)
941  return false;
942  if (other.state() == IteratorState::past_the_end)
943  return true;
944 
945  return ((**this) < (*other));
946 }
