182  template <typename T>
183  T
184  reduce(const EvaluationFlags::EvaluationFlags &flags,
185  const ArrayView<const T> & values)
186  {
187  switch (flags)
188  {
189  case EvaluationFlags::avg:
190  {
191  return std::accumulate(values.begin(), values.end(), T{}) /
192  (T(1.0) * values.size());
193  }
194  case EvaluationFlags::max:
195  return *std::max_element(values.begin(), values.end());
196  case EvaluationFlags::min:
197  return *std::min_element(values.begin(), values.end());
198  case EvaluationFlags::insert:
199  return values[0];
200  default:
201  Assert(false, ExcNotImplemented());
202  return values[0];
203  }
204  }

236  template <int n_components,
237  int dim,
238  int spacedim,
239  typename VectorType,
240  typename value_type>
241  inline std::vector<value_type>
242  evaluate_at_points(
243  const Utilities::MPI::RemotePointEvaluation<dim, spacedim> &cache,
244  const DoFHandler<dim, spacedim> & dof_handler,
245  const VectorType & vector,
246  const EvaluationFlags::EvaluationFlags flags,
247  const UpdateFlags update_flags,
248  const ::EvaluationFlags::EvaluationFlags evaluation_flags,
249  const std::function<
250  value_type(const FEPointEvaluation<n_components, dim> &,
251  const unsigned int &)> process_quadrature_point)
252  {
253  Assert(cache.is_ready(),
254  ExcMessage(
255  "Utilities::MPI::RemotePointEvaluation is not ready yet! "
256  "Please call Utilities::MPI::RemotePointEvaluation::reinit() "
257  "yourself or another function that does this for you."));
258 
259  Assert(
260  &dof_handler.get_triangulation() == &cache.get_triangulation(),
261  ExcMessage(
262  "The provided Utilities::MPI::RemotePointEvaluation and DoFHandler "
263  "object have been set up with different Triangulation objects, "
264  "a scenario not supported!"));
265 
266  // evaluate values at points if possible
267  const auto evaluation_point_results = [&]() {
268  // helper function for accessing the global vector and interpolating
269  // the results onto the points
270  const auto evaluation_function = [&](auto & values,
271  const auto &cell_data) {
272  std::vector<typename VectorType::value_type> solution_values;
273 
274  std::vector<std::unique_ptr<FEPointEvaluation<n_components, dim>>>
275  evaluators(dof_handler.get_fe_collection().size());
276 
277  const auto get_evaluator = [&](const unsigned int active_fe_index)
278  -> FEPointEvaluation<n_components, dim> & {
279  if (evaluators[active_fe_index] == nullptr)
280  evaluators[active_fe_index] =
281  std::make_unique<FEPointEvaluation<n_components, dim>>(
282  cache.get_mapping(),
283  dof_handler.get_fe(active_fe_index),
284  update_flags);
285 
286  return *evaluators[active_fe_index];
287  };
288 
289  for (unsigned int i = 0; i < cell_data.cells.size(); ++i)
290  {
291  typename DoFHandler<dim>::active_cell_iterator cell = {
292  &cache.get_triangulation(),
293  cell_data.cells[i].first,
294  cell_data.cells[i].second,
295  &dof_handler};
296 
297  const ArrayView<const Point<dim>> unit_points(
298  cell_data.reference_point_values.data() +
299  cell_data.reference_point_ptrs[i],
300  cell_data.reference_point_ptrs[i + 1] -
301  cell_data.reference_point_ptrs[i]);
302 
303  solution_values.resize(
304  dof_handler.get_fe(cell->active_fe_index()).n_dofs_per_cell());
305  cell->get_dof_values(vector,
306  solution_values.begin(),
307  solution_values.end());
308 
309  auto &evaluator = get_evaluator(cell->active_fe_index());
310 
311  evaluator.reinit(cell, unit_points);
312  evaluator.evaluate(solution_values, evaluation_flags);
313 
314  for (unsigned int q = 0; q < unit_points.size(); ++q)
315  values[q + cell_data.reference_point_ptrs[i]] =
316  process_quadrature_point(evaluator, q);
317  }
318  };
319 
320  std::vector<value_type> evaluation_point_results;
321  std::vector<value_type> buffer;
322 
323  cache.template evaluate_and_process<value_type>(
324  evaluation_point_results, buffer, evaluation_function);
325 
326  return evaluation_point_results;
327  }();
328 
329  if (cache.is_map_unique())
330  {
331  // each point has exactly one result (unique map)
332  return evaluation_point_results;
333  }
334  else
335  {
336  // map is not unique (multiple or no results): postprocessing is
337  // needed
338  std::vector<value_type> unique_evaluation_point_results(
339  cache.get_point_ptrs().size() - 1);
340 
341  const auto &ptr = cache.get_point_ptrs();
342 
343  for (unsigned int i = 0; i < ptr.size() - 1; ++i)
344  {
345  const auto n_entries = ptr[i + 1] - ptr[i];
346  if (n_entries == 0)
347  continue;
348 
349  unique_evaluation_point_results[i] =
350  reduce(flags,
351  ArrayView<const value_type>(
352  evaluation_point_results.data() + ptr[i], n_entries));
353  }
354 
355  return unique_evaluation_point_results;
356  }
357  }
