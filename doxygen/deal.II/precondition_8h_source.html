1548 PreconditionSSOR<MatrixType>::initialize(
1549  const MatrixType & rA,
1550  const typename BaseClass::AdditionalData &parameters)
1551 {

1553 
1554  // in case we have a SparseMatrix class, we can extract information about
1555  // the diagonal.
1556  const SparseMatrix<typename MatrixType::value_type> *mat =
1557  dynamic_cast<const SparseMatrix<typename MatrixType::value_type> *>(
1558  &*this->A);
1559 
1560  // calculate the positions first after the diagonal.
1561  if (mat != nullptr)
1562  {
1563  const size_type n = this->A->n();
1564  pos_right_of_diagonal.resize(n, static_cast<std::size_t>(-1));

1768  const PreconditionerType &preconditioner,
1769  const unsigned int iteration_index,
1770  const double factor1,
1771  const double factor2,
1772  VectorType & solution_old,

1774  VectorType & temp_vector2,
1775  VectorType & solution)
1776  {
1777  if (iteration_index == 0)
1778  {
1779  solution.equ(factor2, rhs);
1780  preconditioner.vmult(solution_old, solution);
1781  }
1782  else if (iteration_index == 1)
1783  {
1784  // compute t = P^{-1} * (b-A*x^{n})
1785  temp_vector1.sadd(-1.0, 1.0, rhs);
1786  preconditioner.vmult(solution_old, temp_vector1);
1787 
1788  // compute x^{n+1} = x^{n} + f_1 * x^{n} + f_2 * t
1789  solution_old.sadd(factor2, 1 + factor1, solution);
1790  }
1791  else
1792  {
1793  // compute t = P^{-1} * (b-A*x^{n})
1794  temp_vector1.sadd(-1.0, 1.0, rhs);
1795  preconditioner.vmult(temp_vector2, temp_vector1);
1796 
1797  // compute x^{n+1} = x^{n} + f_1 * (x^{n}-x^{n-1}) + f_2 * t
1798  solution_old.sadd(-factor1, factor2, temp_vector2);
1799  solution_old.add(1 + factor1, solution);
1800  }
1801

1803  }

1829  void
1830  apply_to_subrange(const std::size_t begin, const std::size_t end) const
1831  {
1832  // To circumvent a bug in gcc
1833  // (https://gcc.gnu.org/bugzilla/show_bug.cgi?id=63945), we create
1834  // copies of the variables factor1 and factor2 and do not check based on
1835  // factor1.
1836  const Number factor1 = this->factor1;
1837  const Number factor1_plus_1 = 1. + this->factor1;
1838  const Number factor2 = this->factor2;
1839  if (iteration_index == 0)
1840  {
1841  DEAL_II_OPENMP_SIMD_PRAGMA
1842  for (std::size_t i = begin; i < end; ++i)
1843  solution[i] = factor2 * matrix_diagonal_inverse[i] * rhs[i];
1844  }
1845  else if (iteration_index == 1)
1846  {
1847  // x^{n+1} = x^{n} + f_1 * x^{n} + f_2 * P^{-1} * (b-A*x^{n})
1848  DEAL_II_OPENMP_SIMD_PRAGMA
1849  for (std::size_t i = begin; i < end; ++i)
1850  // for efficiency reason, write back to temp_vector that is
1851  // already read (avoid read-for-ownership)
1852  tmp_vector[i] =
1853  factor1_plus_1 * solution[i] +
1854  factor2 * matrix_diagonal_inverse[i] * (rhs[i] - tmp_vector[i]);
1855  }
1856  else
1857  {
1858  // x^{n+1} = x^{n} + f_1 * (x^{n}-x^{n-1})
1859  // + f_2 * P^{-1} * (b-A*x^{n})
1860  DEAL_II_OPENMP_SIMD_PRAGMA
1861  for (std::size_t i = begin; i < end; ++i)
1862  solution_old[i] =
1863  factor1_plus_1 * solution[i] - factor1 * solution_old[i] +
1864  factor2 * matrix_diagonal_inverse[i] * (rhs[i] - tmp_vector[i]);
1865  }
1866  }
1867 
1868  const Number * rhs;
1869  const Number * matrix_diagonal_inverse;
1870  const unsigned int iteration_index;
1871  const Number factor1;
1872  const Number factor2;
1873  mutable Number * solution_old;
1874  mutable Number * tmp_vector;
1875  mutable Number * solution;
1876  };
