238 template <typename MatrixType, typename PreconditionerType>
239 void
240 SolverQMRS<VectorType>::solve(const MatrixType & A,
241  VectorType & x,
242  const VectorType & b,
243  const PreconditionerType &preconditioner)
244 {
245  LogStream::Prefix prefix("SQMR");
246 
247 
248  // temporary vectors, allocated trough the @p VectorMemory object at the
249  // start of the actual solution process and deallocated at the end.
250  typename VectorMemory<VectorType>::Pointer Vr(this->memory);
251  typename VectorMemory<VectorType>::Pointer Vu(this->memory);
252  typename VectorMemory<VectorType>::Pointer Vq(this->memory);
253  typename VectorMemory<VectorType>::Pointer Vt(this->memory);
254  typename VectorMemory<VectorType>::Pointer Vd(this->memory);
255 
256 
257  // resize the vectors, but do not set
258  // the values since they'd be overwritten
259  // soon anyway.
260  Vr->reinit(x, true);
261  Vu->reinit(x, true);
262  Vq->reinit(x, true);
263  Vt->reinit(x, true);
264  Vd->reinit(x, true);
265 
266  step = 0;
267 
268  IterationResult state(SolverControl::failure, 0);
269 
270  do
271  {
272  if (step > 0)
273  deallog << "Restart step " << step << std::endl;
274  state = iterate(A, x, b, preconditioner, *Vr, *Vu, *Vq, *Vt, *Vd);
275  }
276  while (state.state == SolverControl::iterate);
277 
278 
279  // in case of failure: throw exception
280  AssertThrow(state.state == SolverControl::success,
281  SolverControl::NoConvergence(step, state.last_residual));
282  // otherwise exit as normal
283 }

285 template <class VectorType>
286 template <typename MatrixType, typename PreconditionerType>
287 typename SolverQMRS<VectorType>::IterationResult
288 SolverQMRS<VectorType>::iterate(const MatrixType & A,
289  VectorType & x,
290  const VectorType & b,
291  const PreconditionerType &preconditioner,
292  VectorType & r,
293  VectorType & u,
294  VectorType & q,
295  VectorType & t,
296  VectorType & d)
297 {
298  SolverControl::State state = SolverControl::iterate;
299 
300  int it = 0;
301 
302  double tau, rho, theta = 0;
303  double res;
304 
305  // Compute the start residual
306  A.vmult(r, x);
307  r.sadd(-1., 1., b);
308 
309  // Doing the initial preconditioning
310  if (additional_data.left_preconditioning)
311  {
312  // Left preconditioning
313  preconditioner.vmult(t, r);
314  q = t;
315  }
316  else
317  {
318  // Right preconditioning
319  t = r;
320  preconditioner.vmult(q, t);
321  }
322 
323  tau = t.norm_sqr();
324  res = std::sqrt(tau);
325 
326  if (this->iteration_status(step, res, x) == SolverControl::success)
327  return IterationResult(SolverControl::success, res);
328 
329  rho = q * r;
330 
331  while (state == SolverControl::iterate)
332  {
333  step++;
334  it++;
335  //--------------------------------------------------------------
336  // Step 1: apply the system matrix and compute one inner product
337  //--------------------------------------------------------------
338  A.vmult(t, q);
339  const double sigma = q * t;
340 
341  // Check the breakdown criterion
342  if (additional_data.breakdown_testing == true &&
343  std::fabs(sigma) < additional_data.breakdown_threshold)
344  return IterationResult(SolverControl::iterate, res);
345  // Update the residual
346  const double alpha = rho / sigma;
347  r.add(-alpha, t);
348 
349  //--------------------------------------------------------------
350  // Step 2: update the solution vector
351  //--------------------------------------------------------------
352  const double theta_old = theta;
353 
354  // Apply the preconditioner
355  if (additional_data.left_preconditioning)
356  {
357  // Left Preconditioning
358  preconditioner.vmult(t, r);
359  }
360  else
361  {
362  // Right Preconditioning
363  t = r;
364  }
365 
366  // Double updates
367  theta = t * t / tau;
368  const double psi = 1. / (1. + theta);
369  tau *= theta * psi;
370 
371  // Actual update of the solution vector
372  d.sadd(psi * theta_old, psi * alpha, q);
373  x += d;
374 
375  print_vectors(step, x, r, d);
376 
377  // Check for convergence
378  // Compute a simple and cheap upper bound of the norm of the residual
379  // vector b-Ax
380  res = std::sqrt((it + 1) * tau);
381  // If res lies close enough, within the desired tolerance, calculate the
382  // exact residual
383  if (res < additional_data.solver_tolerance)
384  {
385  A.vmult(u, x);
386  u.sadd(-1., 1., b);
387  res = u.l2_norm();
388  }
389  state = this->iteration_status(step, res, x);
390  if ((state == SolverControl::success) ||
391  (state == SolverControl::failure))
392  return IterationResult(state, res);
393 
394  //--------------------------------------------------------------
395  // Step 3: check breakdown criterion and update the vectors
396  //--------------------------------------------------------------
397  if (additional_data.breakdown_testing == true &&
398  std::fabs(sigma) < additional_data.breakdown_threshold)
399  return IterationResult(SolverControl::iterate, res);
400 
401  const double rho_old = rho;
402 
403  // Applying the preconditioner
404  if (additional_data.left_preconditioning)
405  {
406  // Left preconditioning
407  u = t;
408  }
409  else
410  {
411  // Right preconditioning
412  preconditioner.vmult(u, t);
413  }
414 
415  // Double and vector updates
416  rho = u * r;
417  const double beta = rho / rho_old;
418  q.sadd(beta, 1., u);
419  }
420  return IterationResult(SolverControl::success, res);
421 }
