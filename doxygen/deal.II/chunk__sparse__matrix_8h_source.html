1278 template <typename number>

1280 ChunkSparseMatrix<number>::compute_location(const size_type i,
1281  const size_type j) const
1282 {
1283  const size_type chunk_size = cols->get_chunk_size();
1284  const size_type chunk_index =
1285  cols->sparsity_pattern(i / chunk_size, j / chunk_size);
1286 
1287  if (chunk_index == ChunkSparsityPattern::invalid_entry)
1288  return ChunkSparsityPattern::invalid_entry;

1357 template <typename number>
1358 inline ChunkSparseMatrix<number> &
1359 ChunkSparseMatrix<number>::operator*=(const number factor)
1360 {
1361  Assert(cols != nullptr, ExcNeedsSparsityPattern());
1362  Assert(val != nullptr, ExcNotInitialized());
1363 
1364  const size_type chunk_size = cols->get_chunk_size();
1365 
1366  // multiply all elements of the matrix with the given factor. this includes
1367  // the padding elements in chunks that overlap the boundaries of the actual
1368  // matrix -- but since multiplication with a number does not violate the
1369  // invariant of keeping these elements at zero nothing can happen
1370  number * val_ptr = val.get();
1371  const number *const end_ptr =
1372  val.get() +
1373  cols->sparsity_pattern.n_nonzero_elements() * chunk_size * chunk_size;
1374  while (val_ptr != end_ptr)
1375  *val_ptr++ *= factor;
1376 
1377  return *this;
1378 }
