366 template <int order, int dim, int spacedim, typename Number>
367 inline Number
368 DerivativeForm<order, dim, spacedim, Number>::determinant() const
369 {
370  Assert(order == 1, ExcMessage("Only for order == 1."));
371  if (dim == spacedim)
372  {
373  const Tensor<2, dim, Number> T =
374  static_cast<Tensor<2, dim, Number>>(*this);
375  return ::determinant(T);
376  }
377  else
378  {
379  Assert(spacedim > dim, ExcMessage("Only for spacedim>dim."));
380  const DerivativeForm<1, spacedim, dim, Number> DF_t = this->transpose();
381  Tensor<2, dim, Number> G; // First fundamental form
382  for (unsigned int i = 0; i < dim; ++i)
383  for (unsigned int j = 0; j < dim; ++j)
384  G[i][j] = DF_t[i] * DF_t[j];
385 
386  return (std::sqrt(::determinant(G)));
387  }
