233  template <int dim, int spacedim>
234  inline const Point<spacedim> &
235  PropertyPool<dim, spacedim>::get_location(const Handle handle) const
236  {
237  const std::vector<double>::size_type data_index =
238  (handle != invalid_handle) ? handle : 0;
239 
240  // Ideally we would need to assert that 'handle' has not been deallocated
241  // by searching through 'currently_available_handles'. However, this
242  // is expensive and this function is performance critical, so instead
243  // just check against the array range, and rely on the fact
244  // that handles are invalidated when handed over to
245  // deallocate_properties_array().
246  Assert(data_index <= locations.size() - 1,
247  ExcMessage("Invalid location handle. This can happen if the "
248  "handle was duplicated and then one copy was deallocated "
249  "before trying to access the properties."));
250 
251  return locations[data_index];
252  }
