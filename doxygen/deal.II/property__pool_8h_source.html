279  template <int dim, int spacedim>
280  inline const Point<spacedim> &
281  PropertyPool<dim, spacedim>::get_location(const Handle handle) const
282  {
283  const std::vector<double>::size_type data_index =
284  (handle != invalid_handle) ? handle : 0;
285 
286  // Ideally we would need to assert that 'handle' has not been deallocated
287  // by searching through 'currently_available_handles'. However, this
288  // is expensive and this function is performance critical, so instead
289  // just check against the array range, and rely on the fact
290  // that handles are invalidated when handed over to
291  // deallocate_properties_array().
292  Assert(data_index <= locations.size() - 1,
293  ExcMessage("Invalid location handle. This can happen if the "
294  "handle was duplicated and then one copy was deallocated "
295  "before trying to access the properties."));
296 
297  return locations[data_index];
298  }
