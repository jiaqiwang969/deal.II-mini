512 ParsedConvergenceTable::error_from_exact(const Mapping<dim, spacedim> &mapping,
513  const DoFHandler<dim, spacedim> &dh,
514  const VectorType & solution,
515  const Function<spacedim> &exact,
516  const Function<spacedim> *weight)
517 {
518  const auto n_components = component_names.size();
519 
520  if (compute_error)
521  {
522  AssertDimension(exact.n_components, n_components);
523  AssertDimension(dh.get_fe().n_components(), n_components);
524 
525  const types::global_cell_index n_active_cells =
526  dh.get_triangulation().n_global_active_cells();
527  const unsigned int n_dofs = dh.n_dofs();
528 
529  for (const auto &col : extra_columns)
530  if (col == "cells")
531  {
532  table.add_value("cells", n_active_cells);
533  table.set_tex_caption("cells", "\\# cells");
534  table.set_tex_format("cells", "r");
535  }
536  else if (col == "dofs")
537  {
538  table.add_value("dofs", n_dofs);
539  table.set_tex_caption("dofs", "\\# dofs");
540  table.set_tex_format("dofs", "r");
541  }
542 
543  // A vector of zero std::functions with n_components components
544  const std::vector<std::function<double(const Point<spacedim> &)>>
545  zero_components(n_components,
546  [](const Point<spacedim> &) { return 0.0; });
547 
548  // The default weight function, with n_components components
549  std::vector<std::function<double(const Point<spacedim> &)>>
550  weight_components(n_components,
551  [](const Point<spacedim> &) { return 1.0; });
552 
553  if (weight != nullptr)
554  {
555  if (weight->n_components == 1)
556  {
557  for (auto &f : weight_components)
558  f = [&](const Point<spacedim> &p) { return weight->value(p); };
559  }
560  else
561  {
562  AssertDimension(weight->n_components, n_components);
563  for (unsigned int i = 0; i < n_components; ++i)
564  weight_components[i] = [&](const Point<spacedim> &p) {
565  return weight->value(p, i);
566  };
567  }
568  }
569 
570  for (unsigned int i = 0; i < norms_per_unique_component.size(); ++i)
571  {
572  std::map<VectorTools::NormType, double> errors;
573 
574  const auto &norms = norms_per_unique_component[i];
575  const auto &mask = unique_component_masks[i];
576 
577  // Simple case first
578  if (norms.empty())
579  continue;
580 
581  auto components_expr = zero_components;
582  for (unsigned int i = 0; i < n_components; ++i)
583  if (mask[i] == true)
584  components_expr[i] = weight_components[i];
585 
586  FunctionFromFunctionObjects<spacedim> select_component(
587  components_expr);
588 
589  Vector<float> difference_per_cell(
590  dh.get_triangulation().n_global_active_cells());
591 
592  QGauss<dim> q_gauss((dh.get_fe().degree + 1) * 2);
593 
594  for (const auto &norm : norms)
595  {
596  difference_per_cell = 0;
597  VectorTools::integrate_difference(mapping,
598  dh,
599  solution,
600  exact,
601  difference_per_cell,
602  q_gauss,
603  norm,
604  &select_component,
605  exponent);
606 
607  errors[norm] = VectorTools::compute_global_error(
608  dh.get_triangulation(), difference_per_cell, norm, exponent);
609 
610  std::string name = unique_component_names[i] + "_" +
611  Patterns::Tools::to_string(norm);
612  std::string latex_name = "@f$\\| " + unique_component_names[i] +
613  " - " + unique_component_names[i] +
614  "_h \\|_{" +
615  Patterns::Tools::to_string(norm) + "}@f$";
616 
617  table.add_value(name, errors[norm]);
618  table.set_precision(name, precision);
619  table.set_scientific(name, true);
620  table.set_tex_caption(name, latex_name);
621  }
622  }
623 
624  for (const auto &extra_col : extra_column_functions)
625  {
626  const double custom_error = extra_col.second.first();
627 
628  std::string name = extra_col.first;
629  table.add_value(name, custom_error);
630  table.set_precision(name, precision);
631  table.set_scientific(name, true);
632  }
633  }
634 }
