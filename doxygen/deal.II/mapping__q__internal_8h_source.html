74  template <int spacedim>
75  inline Point<2>
76  transform_real_to_unit_cell(
77  const std::array<Point<spacedim>, GeometryInfo<2>::vertices_per_cell>
78  & vertices,
79  const Point<spacedim> &p)
80  {
81  Assert(spacedim == 2, ExcInternalError());
82 
83  // For accuracy reasons, we do all arithmetic in extended precision
84  // (long double). This has a noticeable effect on the hit rate for
85  // borderline cases and thus makes the algorithm more robust.
86  const long double x = p(0);
87  const long double y = p(1);
88 
89  const long double x0 = vertices[0](0);
90  const long double x1 = vertices[1](0);
91  const long double x2 = vertices[2](0);
92  const long double x3 = vertices[3](0);
93 
94  const long double y0 = vertices[0](1);
95  const long double y1 = vertices[1](1);
96  const long double y2 = vertices[2](1);
97  const long double y3 = vertices[3](1);
98 
99  const long double a = (x1 - x3) * (y0 - y2) - (x0 - x2) * (y1 - y3);
100  const long double b = -(x0 - x1 - x2 + x3) * y + (x - 2 * x1 + x3) * y0 -
101  (x - 2 * x0 + x2) * y1 - (x - x1) * y2 +
102  (x - x0) * y3;
103  const long double c = (x0 - x1) * y - (x - x1) * y0 + (x - x0) * y1;
104 
105  const long double discriminant = b * b - 4 * a * c;
106  // exit if the point is not in the cell (this is the only case where the
107  // discriminant is negative)
108  AssertThrow(
109  discriminant > 0.0,
110  (typename Mapping<spacedim, spacedim>::ExcTransformationFailed()));
111 
112  long double eta1;
113  long double eta2;
114  const long double sqrt_discriminant = std::sqrt(discriminant);
115  // special case #1: if a is near-zero to make the discriminant exactly
116  // equal b, then use the linear formula
117  if (b != 0.0 && std::abs(b) == sqrt_discriminant)
118  {
119  eta1 = -c / b;
120  eta2 = -c / b;
121  }
122  // special case #2: a is zero for parallelograms and very small for
123  // near-parallelograms:
124  else if (std::abs(a) < 1e-8 * std::abs(b))
125  {
126  // if both a and c are very small then the root should be near
127  // zero: this first case will capture that
128  eta1 = 2 * c / (-b - sqrt_discriminant);
129  eta2 = 2 * c / (-b + sqrt_discriminant);
130  }
131  // finally, use the plain version:
132  else
133  {
134  eta1 = (-b - sqrt_discriminant) / (2 * a);
135  eta2 = (-b + sqrt_discriminant) / (2 * a);
136  }
137  // pick the one closer to the center of the cell.
138  const long double eta =
139  (std::abs(eta1 - 0.5) < std::abs(eta2 - 0.5)) ? eta1 : eta2;
140 
141  /*有两种方法可以从eta计算xi，但任何一种都可能有一个零分母。
142 *
143 */
144  const long double subexpr0 = -eta * x2 + x0 * (eta - 1);
145  const long double xi_denominator0 = eta * x3 - x1 * (eta - 1) + subexpr0;
146  const long double max_x = std::max(std::max(std::abs(x0), std::abs(x1)),
147  std::max(std::abs(x2), std::abs(x3)));
148 
149  if (std::abs(xi_denominator0) > 1e-10 * max_x)
150  {
151  const double xi = (x + subexpr0) / xi_denominator0;
152  return {xi, static_cast<double>(eta)};
153  }
154  else
155  {
156  const long double max_y =
157  std::max(std::max(std::abs(y0), std::abs(y1)),
158  std::max(std::abs(y2), std::abs(y3)));
159  const long double subexpr1 = -eta * y2 + y0 * (eta - 1);
160  const long double xi_denominator1 =
161  eta * y3 - y1 * (eta - 1) + subexpr1;
162  if (std::abs(xi_denominator1) > 1e-10 * max_y)
163  {
164  const double xi = (subexpr1 + y) / xi_denominator1;
165  return {xi, static_cast<double>(eta)};
166  }
167  else // give up and try Newton iteration
168  {
169  AssertThrow(
170  false,
171  (typename Mapping<spacedim,
172  spacedim>::ExcTransformationFailed()));
173  }
174  }
175  // bogus return to placate compiler. It should not be possible to get
176  // here.
177  Assert(false, ExcInternalError());
178  return {std::numeric_limits<double>::quiet_NaN(),
179  std::numeric_limits<double>::quiet_NaN()};
180  }

241  inline ::Table<2, double>
242  compute_support_point_weights_on_quad(const unsigned int polynomial_degree)
243  {
244  ::Table<2, double> loqvs;
245 
246  // we are asked to compute weights for interior support points, but
247  // there are no interior points if degree==1
248  if (polynomial_degree == 1)
249  return loqvs;
250 
251  const unsigned int M = polynomial_degree - 1;
252  const unsigned int n_inner_2d = M * M;
253  const unsigned int n_outer_2d = 4 + 4 * M;
254 
255  // set the weights of transfinite interpolation
256  loqvs.reinit(n_inner_2d, n_outer_2d);
257  QGaussLobatto<2> gl(polynomial_degree + 1);
258  for (unsigned int i = 0; i < M; ++i)
259  for (unsigned int j = 0; j < M; ++j)
260  {
261  const Point<2> &p =
262  gl.point((i + 1) * (polynomial_degree + 1) + (j + 1));
263  const unsigned int index_table = i * M + j;
264  for (unsigned int v = 0; v < 4; ++v)
265  loqvs(index_table, v) =
266  -GeometryInfo<2>::d_linear_shape_function(p, v);
267  loqvs(index_table, 4 + i) = 1. - p[0];
268  loqvs(index_table, 4 + i + M) = p[0];
269  loqvs(index_table, 4 + j + 2 * M) = 1. - p[1];
270  loqvs(index_table, 4 + j + 3 * M) = p[1];
271  }
272 
273  // the sum of weights of the points at the outer rim should be one.
274  // check this
275  for (unsigned int unit_point = 0; unit_point < n_inner_2d; ++unit_point)
276  Assert(std::fabs(std::accumulate(loqvs[unit_point].begin(),
277  loqvs[unit_point].end(),
278  0.) -
279  1) < 1e-13 * polynomial_degree,
280  ExcInternalError());
281 
282  return loqvs;
283  }

468  template <int dim, int spacedim, typename Number>
469  inline Point<dim, Number>
470  do_transform_real_to_unit_cell_internal(
471  const Point<spacedim, Number> & p,
472  const Point<dim, Number> & initial_p_unit,
473  const std::vector<Point<spacedim>> & points,
474  const std::vector<Polynomials::Polynomial<double>> &polynomials_1d,
475  const std::vector<unsigned int> & renumber,
476  const bool print_iterations_to_deallog = false)
477  {
478  AssertDimension(points.size(),
479  Utilities::pow(polynomials_1d.size(), dim));
480 
481  // Newton iteration to solve
482  // f(x)=p(x)-p=0
483  // where we are looking for 'x' and p(x) is the forward transformation
484  // from unit to real cell. We solve this using a Newton iteration
485  // x_{n+1}=x_n-[f'(x)]^{-1}f(x)
486  // The start value is set to be the linear approximation to the cell
487 
488  // The shape values and derivatives of the mapping at this point are
489  // previously computed.
490 
491  Point<dim, Number> p_unit = initial_p_unit;
492  auto p_real = internal::evaluate_tensor_product_value_and_gradient(
493  polynomials_1d, points, p_unit, polynomials_1d.size() == 2, renumber);
494 
495  Tensor<1, spacedim, Number> f = p_real.first - p;
496 
497  // early out if we already have our point in all SIMD lanes, i.e.,
498  // f.norm_square() < 1e-24 * p_real.second[0].norm_square(). To enable
499  // this step also for VectorizedArray where we do not have operator <,
500  // compare the result to zero which is defined for SIMD types
501  if (std::max(Number(0.),
502  f.norm_square() - 1e-24 * p_real.second[0].norm_square()) ==
503  Number(0.))
504  return p_unit;
505 
506  // we need to compare the position of the computed p(x) against the
507  // given point 'p'. We will terminate the iteration and return 'x' if
508  // they are less than eps apart. The question is how to choose eps --
509  // or, put maybe more generally: in which norm we want these 'p' and
510  // 'p(x)' to be eps apart.
511  //
512  // the question is difficult since we may have to deal with very
513  // elongated cells where we may achieve 1e-12*h for the distance of
514  // these two points in the 'long' direction, but achieving this
515  // tolerance in the 'short' direction of the cell may not be possible
516  //
517  // what we do instead is then to terminate iterations if
518  // \| p(x) - p \|_A < eps
519  // where the A-norm is somehow induced by the transformation of the
520  // cell. in particular, we want to measure distances relative to the
521  // sizes of the cell in its principal directions.
522  //
523  // to define what exactly A should be, note that to first order we have
524  // the following (assuming that x* is the solution of the problem, i.e.,
525  // p(x*)=p):
526  // p(x) - p = p(x) - p(x*)
527  // = -grad p(x) * (x*-x) + higher order terms
528  // This suggest to measure with a norm that corresponds to
529  // A = {[grad p(x)]^T [grad p(x)]}^{-1}
530  // because then
531  // \| p(x) - p \|_A \approx \| x - x* \|
532  // Consequently, we will try to enforce that
533  // \| p(x) - p \|_A = \| f \| <= eps
534  //
535  // Note that using this norm is a bit dangerous since the norm changes
536  // in every iteration (A isn't fixed by depending on xk). However, if
537  // the cell is not too deformed (it may be stretched, but not twisted)
538  // then the mapping is almost linear and A is indeed constant or
539  // nearly so.
540  const double eps = 1.e-11;
541  const unsigned int newton_iteration_limit = 20;
542 
543  Point<dim, Number> invalid_point;
544  invalid_point[0] = std::numeric_limits<double>::infinity();
545  bool try_project_to_unit_cell = false;
546 
547  unsigned int newton_iteration = 0;
548  Number f_weighted_norm_square = 1.;
549  Number last_f_weighted_norm_square = 1.;
550 
551  do
552  {
553  if (print_iterations_to_deallog)
554  deallog << "Newton iteration " << newton_iteration
555  << " for unit point guess " << p_unit << std::endl;
556 
557  // f'(x)
558  Tensor<2, spacedim, Number> df;
559  for (unsigned int d = 0; d < spacedim; ++d)
560  for (unsigned int e = 0; e < dim; ++e)
561  df[d][e] = p_real.second[e][d];
562 
563  // check determinand(df) > 0 on all SIMD lanes
564  if (!(std::min(determinant(df),
565  Number(std::numeric_limits<double>::min())) ==
566  Number(std::numeric_limits<double>::min())))
567  {
568  // We allow to enter this function with an initial guess
569  // outside the unit cell. We might have run into invalid
570  // Jacobians due to that, so we should at least try once to go
571  // back to the unit cell and go on with the Newton iteration
572  // from there. Since the outside case is unlikely, we can
573  // afford spending the extra effort at this place.
574  if (try_project_to_unit_cell == false)
575  {
576  p_unit = GeometryInfo<dim>::project_to_unit_cell(p_unit);
577  p_real = internal::evaluate_tensor_product_value_and_gradient(
578  polynomials_1d,
579  points,
580  p_unit,
581  polynomials_1d.size() == 2,
582  renumber);
583  f = p_real.first - p;
584  f_weighted_norm_square = 1.;
585  last_f_weighted_norm_square = 1;
586  try_project_to_unit_cell = true;
587  continue;
588  }
589  else
590  return invalid_point;
591  }
592 
593  // Solve [f'(x)]d=f(x)
594  const Tensor<2, spacedim, Number> df_inverse = invert(df);
595  const Tensor<1, spacedim, Number> delta = df_inverse * f;
596  last_f_weighted_norm_square = delta.norm_square();
597 
598  if (print_iterations_to_deallog)
599  deallog << " delta=" << delta << std::endl;
600 
601  // do a line search
602  double step_length = 1;
603  do
604  {
605  // update of p_unit. The spacedim-th component of transformed
606  // point is simply ignored in codimension one case. When this
607  // component is not zero, then we are projecting the point to
608  // the surface or curve identified by the cell.
609  Point<dim, Number> p_unit_trial = p_unit;
610  for (unsigned int i = 0; i < dim; ++i)
611  p_unit_trial[i] -= step_length * delta[i];
612 
613  // shape values and derivatives at new p_unit point
614  const auto p_real_trial =
615  internal::evaluate_tensor_product_value_and_gradient(
616  polynomials_1d,
617  points,
618  p_unit_trial,
619  polynomials_1d.size() == 2,
620  renumber);
621  const Tensor<1, spacedim, Number> f_trial =
622  p_real_trial.first - p;
623  f_weighted_norm_square = (df_inverse * f_trial).norm_square();
624 
625  if (print_iterations_to_deallog)
626  deallog << " step_length=" << step_length << std::endl
627  << " ||f || =" << f.norm() << std::endl
628  << " ||f*|| =" << f_trial.norm() << std::endl
629  << " ||f*||_A ="
630  << std::sqrt(f_weighted_norm_square) << std::endl;
631 
632  // See if we are making progress with the current step length
633  // and if not, reduce it by a factor of two and try again.
634  //
635  // Strictly speaking, we should probably use the same norm as we
636  // use for the outer algorithm. In practice, line search is just
637  // a crutch to find a "reasonable" step length, and so using the
638  // l2 norm is probably just fine.
639  //
640  // check f_trial.norm() < f.norm() in SIMD form. This is a bit
641  // more complicated because some SIMD lanes might not be doing
642  // any progress any more as they have already reached roundoff
643  // accuracy. We define that as the case of updates less than
644  // 1e-6. The tolerance might seem coarse but since we are
645  // dealing with a Newton iteration of a polynomial function we
646  // either converge quadratically or not any more. Thus, our
647  // selection is to terminate if either the norm of f is
648  // decreasing or that threshold of 1e-6 is reached.
649  if (std::max(f_weighted_norm_square - 1e-6 * 1e-6, Number(0.)) *
650  std::max(f_trial.norm_square() - f.norm_square(),
651  Number(0.)) ==
652  Number(0.))
653  {
654  p_real = p_real_trial;
655  p_unit = p_unit_trial;
656  f = f_trial;
657  break;
658  }
659  else if (step_length > 0.05)
660  step_length *= 0.5;
661  else
662  break;
663  }
664  while (true);
665 
666  // In case we terminated the line search due to the step becoming
667  // too small, we give the iteration another try with the
668  // projection of the initial guess to the unit cell before we give
669  // up, just like for the negative determinant case.
670  if (step_length <= 0.05 && try_project_to_unit_cell == false)
671  {
672  p_unit = GeometryInfo<dim>::project_to_unit_cell(p_unit);
673  p_real = internal::evaluate_tensor_product_value_and_gradient(
674  polynomials_1d,
675  points,
676  p_unit,
677  polynomials_1d.size() == 2,
678  renumber);
679  f = p_real.first - p;
680  f_weighted_norm_square = 1.;
681  last_f_weighted_norm_square = 1;
682  try_project_to_unit_cell = true;
683  continue;
684  }
685  else if (step_length <= 0.05)
686  return invalid_point;
687 
688  ++newton_iteration;
689  if (newton_iteration > newton_iteration_limit)
690  return invalid_point;
691  }
692  // Stop if f_weighted_norm_square <= eps^2 on all SIMD lanes or if the
693  // weighted norm is less than 1e-6 and the improvement against the
694  // previous step was less than a factor of 10 (in that regime, we
695  // either have quadratic convergence or roundoff errors due to a bad
696  // mapping)
697  while (
698  !(std::max(f_weighted_norm_square - eps * eps, Number(0.)) *
699  std::max(last_f_weighted_norm_square -
700  std::min(f_weighted_norm_square, Number(1e-6 * 1e-6)) *
701  100.,
702  Number(0.)) ==
703  Number(0.)));
704 
705  if (print_iterations_to_deallog)
706  deallog << "Iteration converged for p_unit = [ " << p_unit
707  << " ] and iteration error "
708  << std::sqrt(f_weighted_norm_square) << std::endl;
709 
710  return p_unit;
711  }

847  template <int dim, int spacedim>
848  class InverseQuadraticApproximation
849  {
850  public:
855  static constexpr unsigned int n_functions =
856  (spacedim == 1 ? 3 : (spacedim == 2 ? 6 : 10));
857 
866  InverseQuadraticApproximation(
867  const std::vector<Point<spacedim>> &real_support_points,
868  const std::vector<Point<dim>> & unit_support_points)
869  : normalization_shift(real_support_points[0])
870  , normalization_length(
871  1. / real_support_points[0].distance(real_support_points[1]))
872  , is_affine(true)
873  {
874  AssertDimension(real_support_points.size(), unit_support_points.size());
875 
876  // For the bi-/trilinear approximation, we cannot build a quadratic
877  // polynomial due to a lack of points (interpolation matrix would get
878  // singular), so pick the affine approximation. Similarly, it is not
879  // entirely clear how to gather enough information for the case dim <
880  // spacedim
881  if (real_support_points.size() ==
882  GeometryInfo<dim>::vertices_per_cell ||
883  dim < spacedim)
884  {
885  const auto affine = GridTools::affine_cell_approximation<dim>(
886  make_array_view(real_support_points));
887  DerivativeForm<1, spacedim, dim> A_inv =
888  affine.first.covariant_form().transpose();
889  coefficients[0] = apply_transformation(A_inv, affine.second);
890  for (unsigned int d = 0; d < spacedim; ++d)
891  for (unsigned int e = 0; e < dim; ++e)
892  coefficients[1 + d][e] = A_inv[e][d];
893  is_affine = true;
894  return;
895  }
896 
897  Tensor<2, n_functions> matrix;
898  std::array<double, n_functions> shape_values;
899  for (unsigned int q = 0; q < unit_support_points.size(); ++q)
900  {
901  // Evaluate quadratic shape functions in point, with the
902  // normalization applied in order to avoid roundoff issues with
903  // scaling far away from 1.
904  shape_values[0] = 1.;
905  const Tensor<1, spacedim> p_scaled =
906  normalization_length *
907  (real_support_points[q] - normalization_shift);
908  for (unsigned int d = 0; d < spacedim; ++d)
909  shape_values[1 + d] = p_scaled[d];
910  for (unsigned int d = 0, c = 0; d < spacedim; ++d)
911  for (unsigned int e = 0; e <= d; ++e, ++c)
912  shape_values[1 + spacedim + c] = p_scaled[d] * p_scaled[e];
913 
914  // Build lower diagonal of least squares matrix and rhs, the
915  // essential part being that we construct the matrix with the
916  // real points and the right hand side by comparing to the
917  // reference point positions which sets up an inverse
918  // interpolation.
919  for (unsigned int i = 0; i < n_functions; ++i)
920  for (unsigned int j = 0; j < n_functions; ++j)
921  matrix[i][j] += shape_values[i] * shape_values[j];
922  for (unsigned int i = 0; i < n_functions; ++i)
923  coefficients[i] += shape_values[i] * unit_support_points[q];
924  }
925 
926  // Factorize the matrix A = L * L^T in-place with the
927  // Cholesky-Banachiewicz algorithm. The implementation is similar to
928  // FullMatrix::cholesky() but re-implemented to avoid memory
929  // allocations and some unnecessary divisions which we can do here as
930  // we only need to solve with dim right hand sides.
931  for (unsigned int i = 0; i < n_functions; ++i)
932  {
933  double Lij_sum = 0;
934  for (unsigned int j = 0; j < i; ++j)
935  {
936  double Lik_Ljk_sum = 0;
937  for (unsigned int k = 0; k < j; ++k)
938  Lik_Ljk_sum += matrix[i][k] * matrix[j][k];
939  matrix[i][j] = matrix[j][j] * (matrix[i][j] - Lik_Ljk_sum);
940  Lij_sum += matrix[i][j] * matrix[i][j];
941  }
942  AssertThrow(matrix[i][i] - Lij_sum >= 0,
943  ExcMessage("Matrix not positive definite"));
944 
945  // Store the inverse in the diagonal since that is the quantity
946  // needed later in the factorization as well as the forward and
947  // backward substitution, minimizing the number of divisions.
948  matrix[i][i] = 1. / std::sqrt(matrix[i][i] - Lij_sum);
949  }
950 
951  // Solve lower triangular part, L * y = rhs.
952  for (unsigned int i = 0; i < n_functions; ++i)
953  {
954  Point<dim> sum = coefficients[i];
955  for (unsigned int j = 0; j < i; ++j)
956  sum -= matrix[i][j] * coefficients[j];
957  coefficients[i] = sum * matrix[i][i];
958  }
959 
960  // Solve upper triangular part, L^T * x = y (i.e., x = A^{-1} * rhs)
961  for (unsigned int i = n_functions; i > 0;)
962  {
963  --i;
964  Point<dim> sum = coefficients[i];
965  for (unsigned int j = i + 1; j < n_functions; ++j)
966  sum -= matrix[j][i] * coefficients[j];
967  coefficients[i] = sum * matrix[i][i];
968  }
969 
970  // Check whether the approximation is indeed affine, allowing to
971  // skip the quadratic terms.
972  is_affine = true;
973  for (unsigned int i = dim + 1; i < n_functions; ++i)
974  if (coefficients[i].norm_square() > 1e-20)
975  {
976  is_affine = false;
977  break;
978  }
979  }
980 
985  InverseQuadraticApproximation(const InverseQuadraticApproximation &) =
986  default;

992  template <typename Number>
993  Point<dim, Number>
994  compute(const Point<spacedim, Number> &p) const
995  {
996  Point<dim, Number> result;
997  for (unsigned int d = 0; d < dim; ++d)
998  result[d] = coefficients[0][d];
999 
1000  // Apply the normalization to ensure a good conditioning. Since Number
1001  // might be a vectorized array whereas the normalization is a point of
1002  // doubles, we cannot use the overload of operator- and must instead
1003  // loop over the components of the point.
1004  Point<spacedim, Number> p_scaled;
1005  for (unsigned int d = 0; d < spacedim; ++d)
1006  p_scaled[d] = (p[d] - normalization_shift[d]) * normalization_length;
1007 
1008  for (unsigned int d = 0; d < spacedim; ++d)
1009  result += coefficients[1 + d] * p_scaled[d];
1010 
1011  if (!is_affine)
1012  {
1013  for (unsigned int d = 0, c = 0; d < spacedim; ++d)
1014  for (unsigned int e = 0; e <= d; ++e, ++c)
1015  result +=
1016  coefficients[1 + spacedim + c] * (p_scaled[d] * p_scaled[e]);
1017  }
1018  return result;
1019  }

1027

1033

1039 
1044  bool is_affine;
1045  };

1817 
1818  if (update_flags &
1819  (update_boundary_forms | update_normal_vectors | update_jacobians |
1820  update_JxW_values | update_inverse_jacobians))
1821  {
1822  if (update_flags & update_boundary_forms)
1823  AssertDimension(output_data.boundary_forms.size(), n_q_points);

1827  AssertDimension(output_data.JxW_values.size(), n_q_points);
1828

1830

1832  // all of the flags below
1833 
1834  // map the unit tangentials to the real cell. checking for d!=dim-1
1835  // eliminates compiler warnings regarding unsigned int expressions <
1836  // 0.
1837  for (unsigned int d = 0; d != dim - 1; ++d)
1838  {
1839  Assert(face_no + GeometryInfo<dim>::faces_per_cell * d <
1840  data.unit_tangentials.size(),
1841  ExcInternalError());
1842  Assert(
1843  data.aux[d].size() <=
1844  data
1845  .unit_tangentials[face_no +
1846  GeometryInfo<dim>::faces_per_cell * d]
1847  .size(),
1848  ExcInternalError());

1852  data.unit_tangentials[face_no +
1853  GeometryInfo<dim>::faces_per_cell * d]),
1854  mapping_contravariant,
1855  data,
1856  make_array_view(data.aux[d]));
1857  }
1858

1860  {
1861  // if dim==spacedim, we can use the unit tangentials to compute
1862  // the boundary form by simply taking the cross product
1863  if (dim == spacedim)
1864  {
1865  for (unsigned int i = 0; i < n_q_points; ++i)
1866  switch (dim)
1867  {
1868  case 1:
1869  // in 1d, we don't have access to any of the
1870  // data.aux fields (because it has only dim-1
1871  // components), but we can still compute the
1872  // boundary form by simply looking at the number of
1873  // the face
1874  output_data.boundary_forms[i][0] =

1881  case 3:

1893  // cell normal vector
1894  //
1895  // to compute the cell normal, use the same method used in
1896  // fill_fe_values for cells above
1897  AssertDimension(data.contravariant.size(), n_q_points);

1904  output_data.boundary_forms[point] =
1905  data.contravariant[point].transpose()[0];
1906  output_data.boundary_forms[point] /=
1907  (face_no == 0 ? -1. : +1.) *
1908  output_data.boundary_forms[point].norm();

1910

1912  {

1914  data.contravariant[point].transpose();

1917  cross_product_3d(DX_t[0], DX_t[1]);
1918  cell_normal /= cell_normal.norm();

1921  // tangent and the cell normal:
1922  output_data.boundary_forms[point] =
1923  cross_product_3d(data.aux[0][point], cell_normal);
1924  }
1925  }
1926  }
1927  }
1928 
1929  if (update_flags & update_JxW_values)
1930  for (unsigned int i = 0; i < output_data.boundary_forms.size(); ++i)
1931  {

1936  {
1937  const double area_ratio = GeometryInfo<dim>::subface_ratio(
1938  cell->subface_case(face_no), subface_no);
1939  output_data.JxW_values[i] *= area_ratio;
1940  }
1941  }
1942 
1943  if (update_flags & update_normal_vectors)
1944  for (unsigned int i = 0; i < output_data.normal_vectors.size(); ++i)
1945  output_data.normal_vectors[i] =
1946  Point<spacedim>(output_data.boundary_forms[i] /
1947  output_data.boundary_forms[i].norm());
1948 
1949  if (update_flags & update_jacobians)
1950  for (unsigned int point = 0; point < n_q_points; ++point)
1951  output_data.jacobians[point] = data.contravariant[point];
1952 
1953  if (update_flags & update_inverse_jacobians)
1954  for (unsigned int point = 0; point < n_q_points; ++point)
1955  output_data.inverse_jacobians[point] =
1956  data.covariant[point].transpose();
1957  }
1958  }
