662  template <typename T>
663  struct NumberType
664  {
665  static constexpr DEAL_II_ALWAYS_INLINE DEAL_II_CUDA_HOST_DEV const T &
666  value(const T &t)
667  {
668  return t;
669  }
670 
671  // Below are generic functions that allows an overload for any
672  // type U that is transformable to type T. This is particularly
673  // useful when needing to cast exotic number types
674  // (e.g. auto-differentiable or symbolic numbers) to a floating
675  // point one, such as might happen when converting between tensor
676  // types.
677 
678  // Type T is constructible from F.
679  template <typename F>
680  static constexpr DEAL_II_ALWAYS_INLINE DEAL_II_CUDA_HOST_DEV T
681  value(const F &f,
682  typename std::enable_if<
683  !std::is_same<typename std::decay<T>::type,
684  typename std::decay<F>::type>::value &&
685  std::is_constructible<T, F>::value>::type * = nullptr)
686  {
687  return T(f);
688  }
689 
690  // Type T is explicitly convertible (but not constructible) from F.
691  template <typename F>
692  static constexpr DEAL_II_ALWAYS_INLINE T
693  value(const F &f,
694  typename std::enable_if<
695  !std::is_same<typename std::decay<T>::type,
696  typename std::decay<F>::type>::value &&
697  !std::is_constructible<T, F>::value &&
698  is_explicitly_convertible<const F, T>::value>::type * = nullptr)
699  {
700  return static_cast<T>(f);
701  }
702 
703  // Sacado doesn't provide any conversion operators, so we have
704  // to extract the value and perform further conversions from there.
705  // To be safe, we extend this to other possible AD numbers that
706  // might fall into the same category.
707  template <typename F>
708  static T
709  value(const F &f,
710  typename std::enable_if<
711  !std::is_same<typename std::decay<T>::type,
712  typename std::decay<F>::type>::value &&
713  !std::is_constructible<T, F>::value &&
714  !is_explicitly_convertible<const F, T>::value &&
715  Differentiation::AD::is_ad_number<F>::value>::type * = nullptr)
716  {
717  return Differentiation::AD::internal::NumberType<T>::value(f);
718  }
719  };
