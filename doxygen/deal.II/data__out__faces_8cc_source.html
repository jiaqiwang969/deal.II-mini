95 template <int dim, int spacedim>
96 void
97 DataOutFaces<dim, spacedim>::build_one_patch(
98  const FaceDescriptor *cell_and_face,
99  internal::DataOutFacesImplementation::ParallelData<dim, spacedim> &data,
100  DataOutBase::Patch<patch_dim, patch_spacedim> & patch)
101 {
102  Assert(cell_and_face->first->is_locally_owned(), ExcNotImplemented());
103 
104  // we use the mapping to transform the vertices. However, the mapping works
105  // on cells, not faces, so transform the face vertex to a cell vertex, that
106  // to a unit cell vertex and then, finally, that to the mapped vertex. In
107  // most cases this complicated procedure will be the identity.
108  for (unsigned int vertex = 0;
109  vertex < GeometryInfo<dim - 1>::vertices_per_cell;
110  ++vertex)
111  patch.vertices[vertex] =
112  data.mapping_collection[0].transform_unit_to_real_cell(
113  cell_and_face->first,
114  GeometryInfo<dim>::unit_cell_vertex(
115  GeometryInfo<dim>::face_to_cell_vertices(
116  cell_and_face->second,
117  vertex,
118  cell_and_face->first->face_orientation(cell_and_face->second),
119  cell_and_face->first->face_flip(cell_and_face->second),
120  cell_and_face->first->face_rotation(cell_and_face->second))));
121 
122  if (data.n_datasets > 0)
123  {
124  data.reinit_all_fe_values(this->dof_data,
125  cell_and_face->first,
126  cell_and_face->second);
127  const FEValuesBase<dim> &fe_patch_values = data.get_present_fe_values(0);
128 
129  const unsigned int n_q_points = fe_patch_values.n_quadrature_points;
130 
131  // store the intermediate points
132  Assert(patch.space_dim == dim, ExcInternalError());
133  const std::vector<Point<dim>> &q_points =
134  fe_patch_values.get_quadrature_points();
135  // resize the patch.data member in order to have enough memory for the
136  // quadrature points as well
137  patch.data.reinit(data.n_datasets + dim, patch.data.size(1));
138  // set the flag indicating that for this cell the points are explicitly
139  // given
140  patch.points_are_available = true;
141  // copy points to patch.data
142  for (unsigned int i = 0; i < dim; ++i)
143  for (unsigned int q = 0; q < n_q_points; ++q)
144  patch.data(patch.data.size(0) - dim + i, q) = q_points[q][i];
145 
146  // counter for data records
147  unsigned int offset = 0;
148 
149  // first fill dof_data
150  for (unsigned int dataset = 0; dataset < this->dof_data.size(); ++dataset)
151  {
152  const FEValuesBase<dim> &this_fe_patch_values =
153  data.get_present_fe_values(dataset);
154  const unsigned int n_components =
155  this_fe_patch_values.get_fe().n_components();
156  const DataPostprocessor<dim> *postprocessor =
157  this->dof_data[dataset]->postprocessor;
158  if (postprocessor != nullptr)
159  {
160  // we have to postprocess the data, so determine, which fields
161  // have to be updated
162  const UpdateFlags update_flags =
163  postprocessor->get_needed_update_flags();
164 
165  if (n_components == 1)
166  {
167  // at each point there is only one component of value,
168  // gradient etc.
169  if (update_flags & update_values)
170  this->dof_data[dataset]->get_function_values(
171  this_fe_patch_values,
172  internal::DataOutImplementation::ComponentExtractor::
173  real_part,
174  data.patch_values_scalar.solution_values);
175  if (update_flags & update_gradients)
176  this->dof_data[dataset]->get_function_gradients(
177  this_fe_patch_values,
178  internal::DataOutImplementation::ComponentExtractor::
179  real_part,
180  data.patch_values_scalar.solution_gradients);
181  if (update_flags & update_hessians)
182  this->dof_data[dataset]->get_function_hessians(
183  this_fe_patch_values,
184  internal::DataOutImplementation::ComponentExtractor::
185  real_part,
186  data.patch_values_scalar.solution_hessians);
187 
188  if (update_flags & update_quadrature_points)
189  data.patch_values_scalar.evaluation_points =
190  this_fe_patch_values.get_quadrature_points();
191 
192  if (update_flags & update_normal_vectors)
193  data.patch_values_scalar.normals =
194  this_fe_patch_values.get_normal_vectors();
195 
196  const typename DoFHandler<dim, spacedim>::active_cell_iterator
197  dh_cell(&cell_and_face->first->get_triangulation(),
198  cell_and_face->first->level(),
199  cell_and_face->first->index(),
200  this->dof_data[dataset]->dof_handler);
201  data.patch_values_scalar.template set_cell<dim>(dh_cell);
202 
203  postprocessor->evaluate_scalar_field(
204  data.patch_values_scalar,
205  data.postprocessed_values[dataset]);
206  }
207  else
208  {
209  // at each point there is a vector valued function and its
210  // derivative...
211  data.resize_system_vectors(n_components);
212  if (update_flags & update_values)
213  this->dof_data[dataset]->get_function_values(
214  this_fe_patch_values,
215  internal::DataOutImplementation::ComponentExtractor::
216  real_part,
217  data.patch_values_system.solution_values);
218  if (update_flags & update_gradients)
219  this->dof_data[dataset]->get_function_gradients(
220  this_fe_patch_values,
221  internal::DataOutImplementation::ComponentExtractor::
222  real_part,
223  data.patch_values_system.solution_gradients);
224  if (update_flags & update_hessians)
225  this->dof_data[dataset]->get_function_hessians(
226  this_fe_patch_values,
227  internal::DataOutImplementation::ComponentExtractor::
228  real_part,
229  data.patch_values_system.solution_hessians);
230 
231  if (update_flags & update_quadrature_points)
232  data.patch_values_system.evaluation_points =
233  this_fe_patch_values.get_quadrature_points();
234 
235  if (update_flags & update_normal_vectors)
236  data.patch_values_system.normals =
237  this_fe_patch_values.get_normal_vectors();
238 
239  const typename DoFHandler<dim, spacedim>::active_cell_iterator
240  dh_cell(&cell_and_face->first->get_triangulation(),
241  cell_and_face->first->level(),
242  cell_and_face->first->index(),
243  this->dof_data[dataset]->dof_handler);
244  data.patch_values_system.template set_cell<dim>(dh_cell);
245 
246  postprocessor->evaluate_vector_field(
247  data.patch_values_system,
248  data.postprocessed_values[dataset]);
249  }
250 
251  for (unsigned int q = 0; q < n_q_points; ++q)
252  for (unsigned int component = 0;
253  component < this->dof_data[dataset]->n_output_variables;
254  ++component)
255  patch.data(offset + component, q) =
256  data.postprocessed_values[dataset][q](component);
257  }
258  else
259  // now we use the given data vector without modifications. again,
260  // we treat single component functions separately for efficiency
261  // reasons.
262  if (n_components == 1)
263  {
264  this->dof_data[dataset]->get_function_values(
265  this_fe_patch_values,
266  internal::DataOutImplementation::ComponentExtractor::real_part,
267  data.patch_values_scalar.solution_values);
268  for (unsigned int q = 0; q < n_q_points; ++q)
269  patch.data(offset, q) =
270  data.patch_values_scalar.solution_values[q];
271  }
272  else
273  {
274  data.resize_system_vectors(n_components);
275  this->dof_data[dataset]->get_function_values(
276  this_fe_patch_values,
277  internal::DataOutImplementation::ComponentExtractor::real_part,
278  data.patch_values_system.solution_values);
279  for (unsigned int component = 0; component < n_components;
280  ++component)
281  for (unsigned int q = 0; q < n_q_points; ++q)
282  patch.data(offset + component, q) =
283  data.patch_values_system.solution_values[q](component);
284  }
285  // increment the counter for the actual data record
286  offset += this->dof_data[dataset]->n_output_variables;
287  }
288 
289  // then do the cell data
290  for (unsigned int dataset = 0; dataset < this->cell_data.size();
291  ++dataset)
292  {
293  // we need to get at the number of the cell to which this face
294  // belongs in order to access the cell data. this is not readily
295  // available, so choose the following rather inefficient way:
296  Assert(
297  cell_and_face->first->is_active(),
298  ExcMessage(
299  "The current function is trying to generate cell-data output "
300  "for a face that does not belong to an active cell. This is "
301  "not supported."));
302  const unsigned int cell_number = std::distance(
303  this->triangulation->begin_active(),
304  typename Triangulation<dim, spacedim>::active_cell_iterator(
305  cell_and_face->first));
306 
307  const double value = this->cell_data[dataset]->get_cell_data_value(
308  cell_number,
309  internal::DataOutImplementation::ComponentExtractor::real_part);
310  for (unsigned int q = 0; q < n_q_points; ++q)
311  patch.data(dataset + offset, q) = value;
312  }
313  }
314 }

327 template <int dim, int spacedim>
328 void
329 DataOutFaces<dim, spacedim>::build_patches(
330  const Mapping<dim, spacedim> &mapping,
331  const unsigned int n_subdivisions_)
332 {
333  const unsigned int n_subdivisions =
334  (n_subdivisions_ != 0) ? n_subdivisions_ : this->default_subdivisions;
335 
336  Assert(n_subdivisions >= 1,
337  Exceptions::DataOutImplementation::ExcInvalidNumberOfSubdivisions(
338  n_subdivisions));
339 
340  Assert(this->triangulation != nullptr,
341  Exceptions::DataOutImplementation::ExcNoTriangulationSelected());
342 
343  this->validate_dataset_names();
344 
345  unsigned int n_datasets = this->cell_data.size();
346  for (unsigned int i = 0; i < this->dof_data.size(); ++i)
347  n_datasets += this->dof_data[i]->n_output_variables;
348 
349  // first collect the cells we want to create patches of; we will
350  // then iterate over them. the end-condition of the loop needs to
351  // test that next_face() returns an end iterator, as well as for the
352  // case that first_face() returns an invalid FaceDescriptor object
353  std::vector<FaceDescriptor> all_faces;
354  for (FaceDescriptor face = first_face();
355  ((face.first != this->triangulation->end()) &&
356  (face != FaceDescriptor()));
357  face = next_face(face))
358  all_faces.push_back(face);
359 
360  // clear the patches array and allocate the right number of elements
361  this->patches.clear();
362  this->patches.reserve(all_faces.size());
363  Assert(this->patches.size() == 0, ExcInternalError());
364 
365 
366  std::vector<unsigned int> n_postprocessor_outputs(this->dof_data.size());
367  for (unsigned int dataset = 0; dataset < this->dof_data.size(); ++dataset)
368  if (this->dof_data[dataset]->postprocessor)
369  n_postprocessor_outputs[dataset] =
370  this->dof_data[dataset]->n_output_variables;
371  else
372  n_postprocessor_outputs[dataset] = 0;
373 
374  UpdateFlags update_flags = update_values;
375  for (unsigned int i = 0; i < this->dof_data.size(); ++i)
376  if (this->dof_data[i]->postprocessor)
377  update_flags |=
378  this->dof_data[i]->postprocessor->get_needed_update_flags();
379  update_flags |= update_quadrature_points;
380 
381  internal::DataOutFacesImplementation::ParallelData<dim, spacedim> thread_data(
382  n_datasets,
383  n_subdivisions,
384  n_postprocessor_outputs,
385  mapping,
386  this->get_fes(),
387  update_flags);
388  DataOutBase::Patch<patch_dim, patch_spacedim> sample_patch;
389  sample_patch.n_subdivisions = n_subdivisions;
390  sample_patch.data.reinit(n_datasets,
391  Utilities::fixed_power<patch_dim>(n_subdivisions +
392  1));
393 
394  // now build the patches in parallel
395  WorkStream::run(
396  all_faces.data(),
397  all_faces.data() + all_faces.size(),
398  [this](
399  const FaceDescriptor *cell_and_face,
400  internal::DataOutFacesImplementation::ParallelData<dim, spacedim> &data,
401  DataOutBase::Patch<patch_dim, patch_spacedim> &patch) {
402  this->build_one_patch(cell_and_face, data, patch);
403  },
404  [this](const DataOutBase::Patch<patch_dim, patch_spacedim> &patch) {
405  internal::DataOutFacesImplementation::append_patch_to_list<dim, spacedim>(
406  patch, this->patches);
407  },
408  thread_data,
409  sample_patch);
410 }

414 template <int dim, int spacedim>
415 typename DataOutFaces<dim, spacedim>::FaceDescriptor
416 DataOutFaces<dim, spacedim>::first_face()
417 {
418  // simply find first active cell with a face on the boundary
419  for (const auto &cell : this->triangulation->active_cell_iterators())
420  if (cell->is_locally_owned())
421  for (const unsigned int f : GeometryInfo<dim>::face_indices())
422  if (!surface_only || cell->face(f)->at_boundary())
423  return FaceDescriptor(cell, f);
424 
425  // just return an invalid descriptor if we haven't found a locally
426  // owned face. this can happen in parallel where all boundary
427  // faces are owned by other processors
428  return FaceDescriptor();
429 }

433 template <int dim, int spacedim>
434 typename DataOutFaces<dim, spacedim>::FaceDescriptor
435 DataOutFaces<dim, spacedim>::next_face(const FaceDescriptor &old_face)
436 {
437  FaceDescriptor face = old_face;
438 
439  // first check whether the present cell has more faces on the boundary. since
440  // we started with this face, its cell must clearly be locally owned
441  Assert(face.first->is_locally_owned(), ExcInternalError());
442  for (unsigned int f = face.second + 1; f < GeometryInfo<dim>::faces_per_cell;
443  ++f)
444  if (!surface_only || face.first->face(f)->at_boundary())
445  // yup, that is so, so return it
446  {
447  face.second = f;
448  return face;
449  }
450 
451  // otherwise find the next active cell that has a face on the boundary
452 
453  // convert the iterator to an active_iterator and advance this to the next
454  // active cell
455  typename Triangulation<dim, spacedim>::active_cell_iterator active_cell =
456  face.first;
457 
458  // increase face pointer by one
459  ++active_cell;
460 
461  // while there are active cells
462  while (active_cell != this->triangulation->end())
463  {
464  // check all the faces of this active cell. but skip it altogether
465  // if it isn't locally owned
466  if (active_cell->is_locally_owned())
467  for (const unsigned int f : GeometryInfo<dim>::face_indices())
468  if (!surface_only || active_cell->face(f)->at_boundary())
469  {
470  face.first = active_cell;
471  face.second = f;
472  return face;
473  }
474 
475  // the present cell had no faces on the boundary (or was not locally
476  // owned), so check next cell
477  ++active_cell;
478  }
479 
480  // we fell off the edge, so return with invalid pointer
481  face.first = this->triangulation->end();
482  face.second = 0;
483  return face;
484 }
