210 template <typename MatrixType, class RelaxationType, typename number>
211 inline void
212 MGSmootherBlock<MatrixType, RelaxationType, number>::smooth(
213  const unsigned int level,
214  BlockVector<number> & u,
215  const BlockVector<number> &rhs) const
216 {
217  LogStream::Prefix prefix("Smooth");
218 
219  unsigned int maxlevel = matrices.max_level();
220  unsigned int steps2 = this->steps;
221 
222  if (this->variable)
223  steps2 *= (1 << (maxlevel - level));
224 
225  typename VectorMemory<BlockVector<number>>::Pointer r(*this->mem);
226  typename VectorMemory<BlockVector<number>>::Pointer d(*this->mem);
227  r->reinit(u);
228  d->reinit(u);
229 
230  bool T = this->transpose;
231  if (this->symmetric && (steps2 % 2 == 0))
232  T = false;
233 
234  for (unsigned int i = 0; i < steps2; ++i)
235  {
236  if (T)
237  {
238  matrices[level].vmult(*r, u);
239  r->sadd(-1., 1., rhs);
240  smoothers[level].Tvmult(*d, *r);
241  }
242  else
243  {
244  matrices[level].vmult(*r, u);
245  r->sadd(-1., 1., rhs);
246  smoothers[level].vmult(*d, *r);
247  }
248  u += *d;
249  if (this->symmetric)
250  T = !T;
251  }
252 }
