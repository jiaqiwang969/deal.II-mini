406 template <typename Type>
407 Type &
408 GeneralDataStorage::get_object_with_name(const std::string &name)
409 {
410  AssertThrow(stores_object_with_name(name), ExcNameNotFound(name));
411 
412  Type *p = nullptr;
413 
414  if (any_data[name].type() == typeid(Type *))
415  {
416  p = boost::any_cast<Type *>(any_data[name]);
417  }
418  else if (any_data[name].type() == typeid(Type))
419  {
420  p = boost::any_cast<Type>(&any_data[name]);
421  }
422  else
423  {
424  AssertThrow(false,
425  ExcTypeMismatch(name,
426  any_data[name].type().name(),
427  typeid(Type).name()));
428  }
429 
430  return *p;
431 }
