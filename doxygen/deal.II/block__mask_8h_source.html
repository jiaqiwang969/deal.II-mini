297 inline BlockMask
298 BlockMask::operator|(const BlockMask &mask) const
299 {
300  // if one of the two masks denotes the all-block mask,
301  // then return the other one
302  if (block_mask.size() == 0)
303  return mask;
304  else if (mask.block_mask.size() == 0)
305  return *this;
306  else
307  {
308  // if both masks have individual entries set, form
309  // the combination of the two
310  AssertDimension(block_mask.size(), mask.block_mask.size());
311  std::vector<bool> new_mask(block_mask.size());
312  for (unsigned int i = 0; i < block_mask.size(); ++i)
313  new_mask[i] = (block_mask[i] || mask.block_mask[i]);
314 
315  return new_mask;
316  }
317 }
