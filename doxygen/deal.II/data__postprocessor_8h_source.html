897  void
898  CommonInputs<spacedim>::set_cell(
899  const typename DoFHandler<dim, spacedim>::cell_iterator &new_cell)
900  {
901  // see if we had previously already stored a cell that has the same
902  // data type; if so, reuse the memory location and avoid calling 'new'
903  // inside boost::any
904  if (typename DoFHandler<dim, spacedim>::cell_iterator *storage_location =
905  boost::any_cast<typename DoFHandler<dim, spacedim>::cell_iterator>(
906  &cell))
907  *storage_location = new_cell;
908  else
909  // if we had nothing stored before, or if we had stored a different
910  // data type, just let boost::any replace things
911  cell = new_cell;
912  }
