321 inline std::pair<unsigned int, BlockIndices::size_type>
322 BlockIndices::global_to_local(const size_type i) const
323 {
324  AssertIndexRange(i, total_size());
325  Assert(n_blocks > 0, ExcLowerRangeType<size_type>(i, size_type(1)));
326 
327  // start_indices[0] == 0 so we might as well start from the next one
328  const auto it =
329  --std::upper_bound(++start_indices.begin(), start_indices.end(), i);

331  return {std::distance(start_indices.begin(), it), i - *it};
332 }
