445  template <int dim>
446  void
447  upwind_value_matrix(FullMatrix<double> & M11,
448  FullMatrix<double> & M12,
449  FullMatrix<double> & M21,
450  FullMatrix<double> & M22,
451  const FEValuesBase<dim> & fe1,
452  const FEValuesBase<dim> & fe2,
453  const FEValuesBase<dim> & fetest1,
454  const FEValuesBase<dim> & fetest2,
455  const ArrayView<const std::vector<double>> &velocity,
456  const double factor = 1.)
457  {
458  const unsigned int n1 = fe1.dofs_per_cell;
459  // Multiply the quadrature point
460  // index below with this factor to
461  // have simpler data for constant
462  // velocities.
463  AssertDimension(velocity.size(), dim);
464  const unsigned int v_increment = (velocity[0].size() == 1) ? 0 : 1;
465  if (v_increment == 1)
466  {
467  AssertVectorVectorDimension(velocity, dim, fe1.n_quadrature_points);
468  }
469 
470  for (unsigned k = 0; k < fe1.n_quadrature_points; ++k)
471  {
472  double nbeta = fe1.normal_vector(k)[0] * velocity[0][k * v_increment];
473  for (unsigned int d = 1; d < dim; ++d)
474  nbeta += fe1.normal_vector(k)[d] * velocity[d][k * v_increment];
475  const double dx_nbeta = factor * std::abs(nbeta) * fe1.JxW(k);
476  FullMatrix<double> &M1 = nbeta > 0. ? M11 : M22;
477  FullMatrix<double> &M2 = nbeta > 0. ? M21 : M12;
478  const FEValuesBase<dim> &fe = nbeta > 0. ? fe1 : fe2;
479  const FEValuesBase<dim> &fetest = nbeta > 0. ? fetest1 : fetest2;
480  const FEValuesBase<dim> &fetestn = nbeta > 0. ? fetest2 : fetest1;
481  for (unsigned i = 0; i < n1; ++i)
482  for (unsigned j = 0; j < n1; ++j)
483  {
484  if (fe1.get_fe().is_primitive())
485  {
486  M1(i, j) += dx_nbeta * fe.shape_value(j, k) *
487  fetest.shape_value(i, k);
488  M2(i, j) -= dx_nbeta * fe.shape_value(j, k) *
489  fetestn.shape_value(i, k);
490  }
491  else
492  {
493  for (unsigned int d = 0; d < fe1.get_fe().n_components();
494  ++d)
495  {
496  M1(i, j) += dx_nbeta *
497  fe.shape_value_component(j, k, d) *
498  fetest.shape_value_component(i, k, d);
499  M2(i, j) -= dx_nbeta *
500  fe.shape_value_component(j, k, d) *
501  fetestn.shape_value_component(i, k, d);
502  }
503  }
504  }
505  }
