110  int iter = 0;
111  // Main loop
112  for (; conv == SolverControl::iterate; iter++)
113  {
114  // Compute residual
115  A.vmult(r, x);
116  r.sadd(-1., 1., b);
117 
118  // The required norm of the

121  // criterion() and stored

123  conv = this->iteration_status(iter, r.l2_norm(), x);
124  if (conv != SolverControl::iterate)
125  break;
126  R.step(x, b);
127  }
128 
129  // in case of failure: throw exception
130  AssertThrow(conv == SolverControl::success,
131  SolverControl::NoConvergence(iter, r.l2_norm()));
132  // otherwise exit as normal
