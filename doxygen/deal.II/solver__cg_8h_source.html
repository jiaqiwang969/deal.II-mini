252 template <typename VectorType>
253 inline void
254 SolverCG<VectorType>::compute_eigs_and_cond(
255  const std::vector<typename VectorType::value_type> &diagonal,
256  const std::vector<typename VectorType::value_type> &offdiagonal,
257  const boost::signals2::signal<void(const std::vector<double> &)>
258  & eigenvalues_signal,
259  const boost::signals2::signal<void(double)> &cond_signal)
260 {
261  // Avoid computing eigenvalues unless they are needed.
262  if (!cond_signal.empty() || !eigenvalues_signal.empty())
263  {
264  TridiagonalMatrix<typename VectorType::value_type> T(diagonal.size(),
265  true);
266  for (size_type i = 0; i < diagonal.size(); ++i)
267  {
268  T(i, i) = diagonal[i];
269  if (i < diagonal.size() - 1)
270  T(i, i + 1) = offdiagonal[i];
271  }
272  T.compute_eigenvalues();
273  // Need two eigenvalues to estimate the condition number.
274  if (diagonal.size() > 1)
275  {
276  auto condition_number = T.eigenvalue(T.n() - 1) / T.eigenvalue(0);
277  // Condition number is real valued and nonnegative; simply take
278  // the absolute value:
279  cond_signal(std::abs(condition_number));
280  }
281  // Avoid copying the eigenvalues of T to a vector unless a signal is
282  // connected.
283  if (!eigenvalues_signal.empty())
284  {
285  std::vector<double> eigenvalues(T.n());
286  for (unsigned int j = 0; j < T.n(); ++j)
287  {
288  // for a hermitian matrix, all eigenvalues are real-valued
289  // and non-negative, simply return the absolute value:
290  eigenvalues[j] = std::abs(T.eigenvalue(j));
291  }
292  eigenvalues_signal(eigenvalues);
293  }
294  }
295 }

299 template <typename VectorType>
300 template <typename MatrixType, typename PreconditionerType>
301 void
302 SolverCG<VectorType>::solve(const MatrixType & A,
303  VectorType & x,
304  const VectorType & b,
305  const PreconditionerType &preconditioner)
306 {
307  using number = typename VectorType::value_type;
308 
309  SolverControl::State conv = SolverControl::iterate;
310 
311  LogStream::Prefix prefix("cg");
312 
313  // Memory allocation
314  typename VectorMemory<VectorType>::Pointer g_pointer(this->memory);
315  typename VectorMemory<VectorType>::Pointer d_pointer(this->memory);
316  typename VectorMemory<VectorType>::Pointer h_pointer(this->memory);
317 
318  // define some aliases for simpler access
319  VectorType &g = *g_pointer;
320  VectorType &d = *d_pointer;
321  VectorType &h = *h_pointer;
322 
323  // Should we build the matrix for eigenvalue computations?
324  const bool do_eigenvalues =
325  !condition_number_signal.empty() || !all_condition_numbers_signal.empty() ||
326  !eigenvalues_signal.empty() || !all_eigenvalues_signal.empty();
327 
328  // vectors used for eigenvalue computations
329  std::vector<typename VectorType::value_type> diagonal;
330  std::vector<typename VectorType::value_type> offdiagonal;
331 
332  typename VectorType::value_type eigen_beta_alpha = 0;
333 
334  // resize the vectors, but do not set the values since they'd be overwritten
335  // soon anyway.
336  g.reinit(x, true);
337  d.reinit(x, true);
338  h.reinit(x, true);
339 
340  int it = 0;
341  number gh = number();
342  number beta = number();
343  number alpha = number();
344  number old_alpha = number();
345 
346  // compute residual. if vector is zero, then short-circuit the full
347  // computation
348  if (!x.all_zero())
349  {
350  A.vmult(g, x);
351  g.add(-1., b);
352  }
353  else
354  g.equ(-1., b);
355 
356  double res = g.l2_norm();
357  conv = this->iteration_status(0, res, x);
358  if (conv != SolverControl::iterate)
359  return;
360 
361  while (conv == SolverControl::iterate)
362  {
363  it++;
364  old_alpha = alpha;
365 
366  if (it > 1)
367  {
368  if (std::is_same<PreconditionerType, PreconditionIdentity>::value ==
369  false)
370  {
371  preconditioner.vmult(h, g);
372  beta = gh;
373  Assert(std::abs(beta) != 0., ExcDivideByZero());
374  gh = g * h;
375  beta = gh / beta;
376  d.sadd(beta, -1., h);
377  }
378  else
379  {
380  beta = gh;
381  gh = res * res;
382  beta = gh / beta;
383  d.sadd(beta, -1., g);
384  }
385  }
386  else
387  {
388  if (std::is_same<PreconditionerType, PreconditionIdentity>::value ==
389  false)
390  {
391  preconditioner.vmult(h, g);
392  d.equ(-1., h);
393  gh = g * h;
394  }
395  else
396  {
397  d.equ(-1., g);
398  gh = res * res;
399  }
400  }
401 
402  A.vmult(h, d);
403 
404  alpha = d * h;
405  Assert(std::abs(alpha) != 0., ExcDivideByZero());
406  alpha = gh / alpha;
407 
408  x.add(alpha, d);
409  res = std::sqrt(std::abs(g.add_and_dot(alpha, h, g)));
410 
411  print_vectors(it, x, g, d);
412 
413  if (it > 1)
414  {
415  this->coefficients_signal(old_alpha, beta);
416  // set up the vectors containing the diagonal and the off diagonal of
417  // the projected matrix.
418  if (do_eigenvalues)
419  {
420  diagonal.push_back(number(1.) / old_alpha + eigen_beta_alpha);
421  eigen_beta_alpha = beta / old_alpha;
422  offdiagonal.push_back(std::sqrt(beta) / old_alpha);
423  }
424  compute_eigs_and_cond(diagonal,
425  offdiagonal,
426  all_eigenvalues_signal,
427  all_condition_numbers_signal);
428  }
429 
430  conv = this->iteration_status(it, res, x);
431  }
432 
433  compute_eigs_and_cond(diagonal,
434  offdiagonal,
435  eigenvalues_signal,
436  condition_number_signal);
437 
438  // in case of failure: throw exception
439  if (conv != SolverControl::success)
440  AssertThrow(false, SolverControl::NoConvergence(it, res));
441  // otherwise exit as normal
442 }
