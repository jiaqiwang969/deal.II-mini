428  template <int dim, int spacedim>
429  inline void
430  Particle<dim, spacedim>::set_property_pool(
431  PropertyPool<dim, spacedim> &new_property_pool)
432  {
433  // First, we do want to save any properties that may
434  // have previously been set, and copy them over to the memory allocated
435  // on the new pool.
436  //
437  // It is possible that a particle currently has no properties -- for
438  // example if it has been created without an associated property
439  // pool (i.e., uses the default global pool which does not store any
440  // properties) but that the new pool has properties. In that case,
441  // there is simply nothing to transfer -- but the register_particle()
442  // call here will make sure that the newly allocated properties are
443  // zero-initialized.
444  const typename PropertyPool<dim, spacedim>::Handle new_handle =
445  new_property_pool.register_particle();
446 
447  const Point<spacedim> location = get_location();
448  const Point<dim> reference_location = get_reference_location();
449  const types::particle_index id = get_id();
450 
451  if ( /* old pool */ has_properties())
452  {
453  ArrayView<const double> old_properties = this->get_properties();
454  ArrayView<double> new_properties =
455  new_property_pool.get_properties(new_handle);
456  std::copy(old_properties.cbegin(),
457  old_properties.cend(),
458  new_properties.begin());
459  }
460 
461  // Now release the old memory handle
462  property_pool->deregister_particle(property_pool_handle);
463 
464 
465  // Then set the pointer to the property pool we want to use. Also set the
466  // handle to any properties.
467  property_pool = &new_property_pool;
468  property_pool_handle = new_handle;
469 
470  // Now also store the saved locations
471  set_location(location);
472  set_reference_location(reference_location);
473  set_id(id);
474  }

490  template <int dim, int spacedim>
491  inline bool
492  Particle<dim, spacedim>::has_properties() const
493  {
494  // Particles always have a property pool associated with them,
495  // but we can access properties only if there is a valid handle.
496  // The only way a particle can have no valid handle if it has
497  // been moved-from -- but that leaves an object in an invalid
498  // state, and so we can just assert that that can't be the case.
499  Assert((property_pool_handle !=
500  PropertyPool<dim, spacedim>::invalid_handle),
501  ExcInternalError());
502  return (property_pool->n_properties_per_slot() > 0);
503  }
