60  template <int dim, int spacedim>
61  void
62  RemotePointEvaluation<dim, spacedim>::reinit(
63  const std::vector<Point<spacedim>> &points,
64  const Triangulation<dim, spacedim> &tria,
65  const Mapping<dim, spacedim> & mapping)
66  {
67 #ifndef DEAL_II_WITH_MPI
68  Assert(false, ExcNeedsMPI());
69  (void)points;
70  (void)tria;
71  (void)mapping;
72 #else
73  if (tria_signal.connected())
74  tria_signal.disconnect();
75 
76  tria_signal =
77  tria.signals.any_change.connect([&]() { this->ready_flag = false; });
78 
79  this->tria = &tria;
80  this->mapping = &mapping;
81 
82  std::vector<BoundingBox<spacedim>> local_boxes;
83  for (const auto &cell : tria.active_cell_iterators())
84  if (cell->is_locally_owned())
85  local_boxes.push_back(mapping.get_bounding_box(cell));
86 
87  // create r-tree of bounding boxes
88  const auto local_tree = pack_rtree(local_boxes);
89 
90  // compress r-tree to a minimal set of bounding boxes
91  const auto local_reduced_box =
92  extract_rtree_level(local_tree, rtree_level);
93 
94  // gather bounding boxes of other processes
95  const auto global_bboxes =
96  Utilities::MPI::all_gather(tria.get_communicator(), local_reduced_box);
97 
98  const GridTools::Cache<dim, spacedim> cache(tria, mapping);
99 
100  const auto data =
101  GridTools::internal::distributed_compute_point_locations(
102  cache,
103  points,
104  global_bboxes,
105  tolerance,
106  true,
107  enforce_unique_mapping);
108 
109  this->recv_ranks = data.recv_ranks;
110  this->recv_ptrs = data.recv_ptrs;
111 
112  this->send_ranks = data.send_ranks;
113  this->send_ptrs = data.send_ptrs;
114 
115  this->recv_permutation = {};
116  this->recv_permutation.resize(data.recv_components.size());
117  this->point_ptrs.assign(points.size() + 1, 0);
118  for (unsigned int i = 0; i < data.recv_components.size(); ++i)
119  {
120  AssertIndexRange(std::get<2>(data.recv_components[i]),
121  this->recv_permutation.size());
122  this->recv_permutation[std::get<2>(data.recv_components[i])] = i;
123 
124  AssertIndexRange(std::get<1>(data.recv_components[i]) + 1,
125  this->point_ptrs.size());
126  this->point_ptrs[std::get<1>(data.recv_components[i]) + 1]++;
127  }
128 
129  unique_mapping = true;
130  for (unsigned int i = 0; i < points.size(); ++i)
131  {
132  if (unique_mapping && this->point_ptrs[i + 1] != 1)
133  unique_mapping = false;
134  this->point_ptrs[i + 1] += this->point_ptrs[i];
135  }
136 
137  cell_data = {};
138  send_permutation = {};
139 
140  std::pair<int, int> dummy{-1, -1};
141  for (const auto &i : data.send_components)
142  {
143  if (dummy != std::get<0>(i))
144  {
145  dummy = std::get<0>(i);
146  cell_data.cells.emplace_back(dummy);
147  cell_data.reference_point_ptrs.emplace_back(
148  cell_data.reference_point_values.size());
149  }
150 
151  cell_data.reference_point_values.emplace_back(std::get<3>(i));
152  send_permutation.emplace_back(std::get<5>(i));
153  }
154 
155  cell_data.reference_point_ptrs.emplace_back(
156  cell_data.reference_point_values.size());
157 
158  this->ready_flag = true;
159 #endif
160  }
