536  template <int structdim_, int dim, int spacedim>
537  ::TriaRawIterator<::TriaAccessor<structdim_, dim, spacedim>>
538  TriaObjects::next_free_single_object(
539  const Triangulation<dim, spacedim> &tria)
540  {
541  // TODO: Think of a way to ensure that we are using the correct
542  // triangulation, i.e. the one containing *this.
543 
544  AssertDimension(structdim_, this->structdim);
545 
546  int pos = next_free_single, last = used.size() - 1;
547  if (!reverse_order_next_free_single)
548  {
549  // first sweep forward, only use really single slots, do not use
550  // pair slots
551  for (; pos < last; ++pos)
552  if (!used[pos])
553  if (used[++pos])
554  {
555  // this was a single slot
556  pos -= 1;
557  break;
558  }
559  if (pos >= last)
560  {
561  reverse_order_next_free_single = true;
562  next_free_single = used.size() - 1;
563  pos = used.size() - 1;
564  }
565  else
566  next_free_single = pos + 1;
567  }
568 
569  if (reverse_order_next_free_single)
570  {
571  // second sweep, use all slots, even
572  // in pairs
573  for (; pos >= 0; --pos)
574  if (!used[pos])
575  break;
576  if (pos > 0)
577  next_free_single = pos - 1;
578  else
579  // no valid single object anymore
580  return ::TriaRawIterator<
581  ::TriaAccessor<structdim_, dim, spacedim>>(&tria, -1, -1);
582  }
583 
584  return ::TriaRawIterator<
585  ::TriaAccessor<structdim_, dim, spacedim>>(&tria, 0, pos);
586  }

590  template <int structdim_, int dim, int spacedim>
591  ::TriaRawIterator<::TriaAccessor<structdim_, dim, spacedim>>
592  TriaObjects::next_free_pair_object(const Triangulation<dim, spacedim> &tria)
593  {
594  // TODO: Think of a way to ensure that we are using the correct
595  // triangulation, i.e. the one containing *this.
596 
597  AssertDimension(structdim_, this->structdim);
598 
599  int pos = next_free_pair, last = used.size() - 1;
600  for (; pos < last; ++pos)
601  if (!used[pos])
602  if (!used[++pos])
603  {

612  else
