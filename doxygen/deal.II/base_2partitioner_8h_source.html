638  inline unsigned int
639  Partitioner::global_to_local(
640  const types::global_dof_index global_index) const
641  {
642  Assert(in_local_range(global_index) || is_ghost_entry(global_index),
643  ExcIndexNotPresent(global_index, my_pid));
644  if (in_local_range(global_index))
645  return static_cast<unsigned int>(global_index - local_range_data.first);
646  else if (is_ghost_entry(global_index))
647  return (locally_owned_size() +
648  static_cast<unsigned int>(
649  ghost_indices_data.index_within_set(global_index)));
650  else
651  // should only end up here in optimized mode, when we use this large
652  // number to trigger a segfault when using this method for array
653  // access
654  return numbers::invalid_unsigned_int;
655  }
