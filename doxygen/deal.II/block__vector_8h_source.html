340 template <typename Number>
341 template <typename InputIterator>
342 BlockVector<Number>::BlockVector(const std::vector<size_type> &block_sizes,
343  const InputIterator first,
344  const InputIterator end)
345 {
346  // first set sizes of blocks, but
347  // don't initialize them as we will
348  // copy elements soon
349  (void)end;
350  reinit(block_sizes, true);
351  InputIterator start = first;
352  for (size_type b = 0; b < block_sizes.size(); ++b)
353  {
354  InputIterator end = start;
355  std::advance(end, static_cast<signed int>(block_sizes[b]));
356  std::copy(start, end, this->block(b).begin());
357  start = end;
358  };
359  Assert(start == end, ExcIteratorRangeDoesNotMatchVectorSize());
360 }
