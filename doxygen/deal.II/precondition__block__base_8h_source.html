343 template <typename number>
344 inline void
345 PreconditionBlockBase<number>::reinit(unsigned int n,
346  size_type b,
347  bool compress,
348  Inversion method)
349 {
350  inversion = method;
351  var_same_diagonal = compress;
352  var_inverses_ready = false;
353  n_diagonal_blocks = n;
354 
355  if (compress)
356  {
357  switch (inversion)
358  {
359  case gauss_jordan:
360  var_inverse_full.resize(1);
361  var_inverse_full[0].reinit(b, b);
362  break;
363  case householder:
364  var_inverse_householder.resize(1);
365  break;
366  case svd:
367  var_inverse_svd.resize(1);
368  var_inverse_svd[0].reinit(b, b);
369  break;
370  default:
371  Assert(false, ExcNotImplemented());
372  }
373 
374  if (store_diagonals())
375  {
376  var_diagonal.resize(1);
377  var_diagonal[0].reinit(b, b);
378  }
379  }
380  else
381  {
382  // set the arrays to the right
383  // size. we could do it like this:
384  // var_inverse = vector<>(nblocks,FullMatrix<>())
385  // but this would involve copying many
386  // FullMatrix objects.
387  //
388  // the following is a neat trick which
389  // avoids copying
390  if (store_diagonals())
391  {
392  std::vector<FullMatrix<number>> tmp(n, FullMatrix<number>(b));
393  var_diagonal.swap(tmp);
394  }
395 
396  switch (inversion)
397  {
398  case gauss_jordan:
399  {
400  std::vector<FullMatrix<number>> tmp(n, FullMatrix<number>(b));
401  var_inverse_full.swap(tmp);
402  break;
403  }
404  case householder:
405  var_inverse_householder.resize(n);
406  break;
407  case svd:
408  {
409  std::vector<LAPACKFullMatrix<number>> tmp(
410  n, LAPACKFullMatrix<number>(b));
411  var_inverse_svd.swap(tmp);
412  break;
413  }
414  default:
415  Assert(false, ExcNotImplemented());
416  }
417  }
418 }

624 template <typename number>
625 inline void
626 PreconditionBlockBase<number>::log_statistics() const
627 {
628  deallog << "PreconditionBlockBase: " << size() << " blocks; ";
629 
630  if (inversion == svd)
631  {
632  unsigned int kermin = 100000000, kermax = 0;
633  double sigmin = 1.e300, sigmax = -1.e300;
634  double kappamin = 1.e300, kappamax = -1.e300;
635 
636  for (size_type b = 0; b < size(); ++b)
637  {
638  const LAPACKFullMatrix<number> &matrix = inverse_svd(b);
639  size_type k = 1;
640  while (k <= matrix.n_cols() &&
641  matrix.singular_value(matrix.n_cols() - k) == 0)
642  ++k;
643  const double s0 = matrix.singular_value(0);
644  const double sm = matrix.singular_value(matrix.n_cols() - k);
645  const double co = sm / s0;
646 
647  if (kermin > k)
648  kermin = k - 1;
649  if (kermax < k)
650  kermax = k - 1;
651  if (s0 < sigmin)
652  sigmin = s0;
653  if (sm > sigmax)
654  sigmax = sm;
655  if (co < kappamin)
656  kappamin = co;
657  if (co > kappamax)
658  kappamax = co;
659  }
660  deallog << "dim ker [" << kermin << ':' << kermax << "] sigma [" << sigmin
661  << ':' << sigmax << "] kappa [" << kappamin << ':' << kappamax
662  << ']' << std::endl;
663  }
664  else if (inversion == householder)
665  {}
666  else if (inversion == gauss_jordan)
667  {}
668  else
669  {
670  Assert(false, ExcNotImplemented());
671  }
672 }
