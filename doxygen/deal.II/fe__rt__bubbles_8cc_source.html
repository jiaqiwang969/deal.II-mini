41 template <int dim>
42 FE_RT_Bubbles<dim>::FE_RT_Bubbles(const unsigned int deg)
43  : FE_PolyTensor<dim>(
44  PolynomialsRT_Bubbles<dim>(deg),
45  FiniteElementData<dim>(get_dpo_vector(deg),
46  dim,
47  deg + 1,
48  FiniteElementData<dim>::Hdiv),
49  get_ria_vector(deg),
50  std::vector<ComponentMask>(PolynomialsRT_Bubbles<dim>::n_polynomials(deg),
51  std::vector<bool>(dim, true)))
52 {
53  Assert(dim >= 2, ExcImpossibleInDim(dim));
54  Assert(
55  deg >= 1,
56  ExcMessage(
57  "Lowest order RT_Bubbles element is degree 1, but you requested for degree 0"));
58  const unsigned int n_dofs = this->n_dofs_per_cell();
59 
60  this->mapping_kind = {mapping_raviart_thomas};
61  // Initialize support points and quadrature weights
62  initialize_support_points(deg);
63  // Compute the inverse node matrix to get
64  // the correct basis functions
65  FullMatrix<double> M = FETools::compute_node_matrix(*this);
66  this->inverse_node_matrix.reinit(n_dofs, n_dofs);
67  this->inverse_node_matrix.invert(M);
68 
69  // Reinit the vectors of prolongation matrices to the
70  // right sizes. There are no restriction matrices implemented
71  for (unsigned int ref_case = RefinementCase<dim>::cut_x;
72  ref_case < RefinementCase<dim>::isotropic_refinement + 1;
73  ++ref_case)
74  {
75  const unsigned int nc =
76  GeometryInfo<dim>::n_children(RefinementCase<dim>(ref_case));
77 
78  for (unsigned int i = 0; i < nc; ++i)
79  this->prolongation[ref_case - 1][i].reinit(n_dofs, n_dofs);
80  }
81 
82  // TODO: the implementation makes the assumption that all faces have the
83  // same number of dofs
84  AssertDimension(this->n_unique_faces(), 1);
85  const unsigned int face_no = 0;
86 
87  // Fill prolongation matrices with embedding operators
88  // set tolerance to 1, as embedding error accumulate quickly
89  FETools::compute_embedding_matrices(*this, this->prolongation, true, 1.0);
90  FullMatrix<double> face_embeddings[GeometryInfo<dim>::max_children_per_face];
91  for (unsigned int i = 0; i < GeometryInfo<dim>::max_children_per_face; ++i)
92  face_embeddings[i].reinit(this->n_dofs_per_face(face_no),
93  this->n_dofs_per_face(face_no));
94  FETools::compute_face_embedding_matrices<dim, double>(*this,
95  face_embeddings,
96  0,
97  0);
98  this->interface_constraints.reinit((1 << (dim - 1)) *
99  this->n_dofs_per_face(face_no),
100  this->n_dofs_per_face(face_no));
101  unsigned int target_row = 0;
102  for (unsigned int d = 0; d < GeometryInfo<dim>::max_children_per_face; ++d)
103  for (unsigned int i = 0; i < face_embeddings[d].m(); ++i)
104  {
105  for (unsigned int j = 0; j < face_embeddings[d].n(); ++j)
106  this->interface_constraints(target_row, j) = face_embeddings[d](i, j);
107  ++target_row;
108  }
109 
110  // We need to initialize the dof permutation table and the one for the sign
111  // change.
112  initialize_quad_dof_index_permutation_and_sign_change();
113 }

158 template <int dim>
159 void
160 FE_RT_Bubbles<dim>::initialize_support_points(const unsigned int deg)
161 {
162  // TODO: the implementation makes the assumption that all faces have the
163  // same number of dofs
164  AssertDimension(this->n_unique_faces(), 1);
165  const unsigned int face_no = 0;
166 
167  this->generalized_support_points.resize(this->n_dofs_per_cell());
168  this->generalized_face_support_points[face_no].resize(
169  this->n_dofs_per_face(face_no));
170 
171  // Index of the point being entered
172  unsigned int current = 0;
173 
174  // On the faces, we choose as many Gauss-Lobatto points
175  // as required to determine the normal component uniquely.
176  // This is the deg of the RT_Bubble element plus one.
177  if (dim > 1)
178  {
179  QGaussLobatto<dim - 1> face_points(deg + 1);
180  Assert(face_points.size() == this->n_dofs_per_face(face_no),
181  ExcInternalError());
182  for (unsigned int k = 0; k < this->n_dofs_per_face(face_no); ++k)
183  this->generalized_face_support_points[face_no][k] =
184  face_points.point(k);
185  Quadrature<dim> faces =
186  QProjector<dim>::project_to_all_faces(this->reference_cell(),
187  face_points);
188  for (unsigned int k = 0; k < this->n_dofs_per_face(face_no) *
189  GeometryInfo<dim>::faces_per_cell;
190  ++k)
191  this->generalized_support_points[k] = faces.point(
192  k + QProjector<dim>::DataSetDescriptor::face(this->reference_cell(),
193  0,
194  true,
195  false,
196  false,
197  this->n_dofs_per_face(
198  face_no)));
199 
200  current =
201  this->n_dofs_per_face(face_no) * GeometryInfo<dim>::faces_per_cell;
202  }
203 
204  if (deg == 1)
205  return;
206 
207  // In the interior, we need anisotropic Gauss-Lobatto quadratures,
208  // one for each direction
209  QGaussLobatto<1> high(deg + 1);
210  std::vector<Point<1>> pts = high.get_points();
211  pts.erase(pts.begin());
212  pts.erase(pts.end() - 1);
213 
214  std::vector<double> wts(pts.size(), 1);
215  Quadrature<1> low(pts, wts);
216 
217  for (unsigned int d = 0; d < dim; ++d)
218  {
219  std::unique_ptr<QAnisotropic<dim>> quadrature;
220  switch (dim)
221  {
222  case 1:
223  quadrature = std::make_unique<QAnisotropic<dim>>(high);
224  break;
225  case 2:
226  quadrature =
227  std::make_unique<QAnisotropic<dim>>(((d == 0) ? low : high),
228  ((d == 1) ? low : high));
229  break;
230  case 3:
231  quadrature =
232  std::make_unique<QAnisotropic<dim>>(((d == 0) ? low : high),
233  ((d == 1) ? low : high),
234  ((d == 2) ? low : high));
235  break;
236  default:
237  Assert(false, ExcNotImplemented());
238  }
239 
240  for (unsigned int k = 0; k < quadrature->size(); ++k)
241  this->generalized_support_points[current++] = quadrature->point(k);
242  }
243  Assert(current == this->n_dofs_per_cell(), ExcInternalError());
244 }

248 template <int dim>
249 std::vector<unsigned int>
250 FE_RT_Bubbles<dim>::get_dpo_vector(const unsigned int deg)
251 {
252  // We have (deg+1)^(dim-1) DoFs per face...
253  unsigned int dofs_per_face = 1;
254  for (unsigned int d = 1; d < dim; ++d)
255  dofs_per_face *= deg + 1;
256 
257  // ...plus the interior DoFs for the total of dim*(deg+1)^dim
258  const unsigned int interior_dofs =
259  dim * (deg - 1) * Utilities::pow(deg + 1, dim - 1);
260 
261  std::vector<unsigned int> dpo(dim + 1);
262  dpo[dim - 1] = dofs_per_face;
263  dpo[dim] = interior_dofs;
264 
265  return dpo;
266 }

280 template <int dim>
281 std::vector<bool>
282 FE_RT_Bubbles<dim>::get_ria_vector(const unsigned int deg)
283 {
284  const unsigned int dofs_per_cell =
285  PolynomialsRT_Bubbles<dim>::n_polynomials(deg);
286  unsigned int dofs_per_face = deg + 1;
287  for (unsigned int d = 2; d < dim; ++d)
288  dofs_per_face *= deg + 1;
289  // All face dofs need to be non-additive, since they have
290  // continuity requirements. The interior dofs are
291  // made additive.
292  std::vector<bool> ret_val(dofs_per_cell, false);
293  for (unsigned int i = GeometryInfo<dim>::faces_per_cell * dofs_per_face;
294  i < dofs_per_cell;
295  ++i)
296  ret_val[i] = true;
297 
298  return ret_val;
299 }

303 template <int dim>
304 void
305 FE_RT_Bubbles<dim>::convert_generalized_support_point_values_to_dof_values(
306  const std::vector<Vector<double>> &support_point_values,
307  std::vector<double> & nodal_values) const
308 {
309  Assert(support_point_values.size() == this->generalized_support_points.size(),
310  ExcDimensionMismatch(support_point_values.size(),
311  this->generalized_support_points.size()));
312  Assert(nodal_values.size() == this->n_dofs_per_cell(),
313  ExcDimensionMismatch(nodal_values.size(), this->n_dofs_per_cell()));
314  Assert(support_point_values[0].size() == this->n_components(),
315  ExcDimensionMismatch(support_point_values[0].size(),
316  this->n_components()));
317 
318  // First do interpolation on faces. There, the component
319  // evaluated depends on the face direction and orientation.
320  unsigned int fbase = 0;
321  unsigned int f = 0;
322  for (; f < GeometryInfo<dim>::faces_per_cell;
323  ++f, fbase += this->n_dofs_per_face(f))
324  {
325  for (unsigned int i = 0; i < this->n_dofs_per_face(f); ++i)
326  {
327  nodal_values[fbase + i] = support_point_values[fbase + i](
328  GeometryInfo<dim>::unit_normal_direction[f]);
329  }
330  }
331 
332  // The remaining points form dim chunks, one for each component.
333  const unsigned int istep = (this->n_dofs_per_cell() - fbase) / dim;
334  Assert((this->n_dofs_per_cell() - fbase) % dim == 0, ExcInternalError());
335 
336  f = 0;
337  while (fbase < this->n_dofs_per_cell())
338  {
339  for (unsigned int i = 0; i < istep; ++i)
340  {
341  nodal_values[fbase + i] = support_point_values[fbase + i](f);
342  }
343  fbase += istep;
344  ++f;
345  }
346  Assert(fbase == this->n_dofs_per_cell(), ExcInternalError());
347 }
