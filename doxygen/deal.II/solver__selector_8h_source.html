284 template <class Matrix, class Preconditioner>
285 void
286 SolverSelector<VectorType>::solve(const Matrix & A,
287  VectorType & x,
288  const VectorType & b,
289  const Preconditioner &precond) const
290 {
291  if (solver_name == "richardson")
292  {
293  SolverRichardson<VectorType> solver(*control, richardson_data);
294  solver.solve(A, x, b, precond);
295  }
296  else if (solver_name == "cg")
297  {
298  SolverCG<VectorType> solver(*control, cg_data);
299  solver.solve(A, x, b, precond);
300  }
301  else if (solver_name == "minres")
302  {
303  SolverMinRes<VectorType> solver(*control, minres_data);
304  solver.solve(A, x, b, precond);
305  }
306  else if (solver_name == "bicgstab")
307  {
308  SolverBicgstab<VectorType> solver(*control, bicgstab_data);
309  solver.solve(A, x, b, precond);
310  }
311  else if (solver_name == "gmres")
312  {
313  SolverGMRES<VectorType> solver(*control, gmres_data);
314  solver.solve(A, x, b, precond);
315  }
316  else if (solver_name == "fgmres")
317  {
318  SolverFGMRES<VectorType> solver(*control, fgmres_data);
319  solver.solve(A, x, b, precond);
320  }
321  else
322  Assert(false, ExcSolverDoesNotExist(solver_name));
323 }
