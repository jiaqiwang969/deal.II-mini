282 SmartPointer<T, P>::operator=(T *tt)
283 {

285  // requested

301 inline SmartPointer<T, P> &
302 SmartPointer<T, P>::operator=(const SmartPointer<T, Q> &tt)
303 {
304  // if objects on the left and right
305  // hand side of the operator= are
306  // the same, then this is a no-op
307  if (&tt == this)
308  return *this;
