297 template <typename T, typename P>
298 inline SmartPointer<T, P> &
299 SmartPointer<T, P>::operator=(T *tt)
300 {
301  // optimize if no real action is
302  // requested
303  if (t == tt)
304  return *this;

318 inline SmartPointer<T, P> &
319 SmartPointer<T, P>::operator=(const SmartPointer<T, Q> &tt)
320 {
321  // if objects on the left and right
322  // hand side of the operator= are
323  // the same, then this is a no-op
324  if (&tt == this)
325  return *this;
