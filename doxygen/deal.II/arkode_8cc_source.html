449  template <typename VectorType>
450  int
451  ARKode<VectorType>::do_evolve_time(VectorType & solution,
452  DiscreteTime &time,
453  const bool do_reset)
454  {
455  if (do_reset)
456  {
457  reset(time.get_current_time(), time.get_next_step_size(), solution);
458  if (output_step)
459  output_step(time.get_current_time(),
460  solution,
461  time.get_step_number());
462  }
463 
464  auto solution_nvector = internal::make_nvector_view(solution);
465 
466  while (!time.is_at_end())
467  {
468  time.set_desired_next_step_size(data.output_period);
469  double actual_next_time;
470 # if DEAL_II_SUNDIALS_VERSION_LT(4, 0, 0)
471  const auto status = SundialsARKode(arkode_mem,
472  time.get_next_time(),
473  solution_nvector,
474  &actual_next_time,
475  ARK_NORMAL);
476 # else
477  const auto status = ARKStepEvolve(arkode_mem,
478  time.get_next_time(),
479  solution_nvector,
480  &actual_next_time,
481  ARK_NORMAL);
482 # endif
483  (void)status;
484  AssertARKode(status);
485 
486  time.set_next_step_size(actual_next_time - time.get_current_time());
487  time.advance_time();
488 
489  while (solver_should_restart(time.get_current_time(), solution))
490  reset(time.get_current_time(),
491  time.get_previous_step_size(),
492  solution);
493 
494  if (output_step)
495  output_step(time.get_current_time(),
496  solution,
497  time.get_step_number());
498  }
499  last_end_time = time.get_current_time();
500  return time.get_step_number();
501  }

505 # if DEAL_II_SUNDIALS_VERSION_LT(4, 0, 0)
506  template <typename VectorType>
507  void
508  ARKode<VectorType>::reset(const double current_time,
509  const double current_time_step,
510  const VectorType &solution)
511  {
512  last_end_time = current_time;
513  if (arkode_mem)
514  ARKodeFree(&arkode_mem);
515 
516  arkode_mem = ARKodeCreate();
517 
518  int status;
519  (void)status;
520 
521  Assert(explicit_function || implicit_function,
522  ExcFunctionNotProvided("explicit_function || implicit_function"));
523 
524  // just a view on the memory in solution, all write operations on yy by
525  // ARKODE will automatically be mirrored to solution
526  auto initial_condition_nvector = internal::make_nvector_view(solution);
527 
528  status = ARKodeInit(
529  arkode_mem,
530  explicit_function ? &explicit_function_callback<VectorType> : nullptr,
531  implicit_function ? &implicit_function_callback<VectorType> : nullptr,
532  current_time,
533  initial_condition_nvector);
534  AssertARKode(status);
535 
536  if (get_local_tolerances)
537  {
538  const auto abs_tols = make_nvector_view(get_local_tolerances());
539  status =
540  ARKodeSVtolerances(arkode_mem, data.relative_tolerance, abs_tols);
541  AssertARKode(status);
542  }
543  else
544  {
545  status = ARKodeSStolerances(arkode_mem,
546  data.relative_tolerance,
547  data.absolute_tolerance);
548  AssertARKode(status);
549  }
550 
551  status = ARKodeSetInitStep(arkode_mem, current_time_step);
552  AssertARKode(status);
553 
554  status = ARKodeSetUserData(arkode_mem, this);
555  AssertARKode(status);
556 
557  status = ARKodeSetStopTime(arkode_mem, data.final_time);
558  AssertARKode(status);
559 
560  status =
561  ARKodeSetMaxNonlinIters(arkode_mem, data.maximum_non_linear_iterations);
562  AssertARKode(status);
563 
564  // Initialize solver
565  auto ARKode_mem = static_cast<ARKodeMem>(arkode_mem);
566 
567  if (solve_jacobian_system)
568  {
569  status = ARKodeSetNewton(arkode_mem);
570  AssertARKode(status);
571  if (data.implicit_function_is_linear)
572  {
573  status = ARKodeSetLinear(
574  arkode_mem, data.implicit_function_is_time_independent ? 0 : 1);
575  AssertARKode(status);
576  }
577 
578 
579  ARKode_mem->ark_lsolve = solve_with_jacobian_callback<VectorType>;
580  if (setup_jacobian)
581  {
582  ARKode_mem->ark_lsetup = setup_jacobian_callback<VectorType>;
583 # if DEAL_II_SUNDIALS_VERSION_LT(3, 0, 0)
584  ARKode_mem->ark_setupNonNull = true;
585 # endif
586  }
587  }
588  else
589  {
590  status =
591  ARKodeSetFixedPoint(arkode_mem, data.maximum_non_linear_iterations);
592  AssertARKode(status);
593  }
594 
595 
596  if (solve_mass_system)
597  {
598  ARKode_mem->ark_msolve = solve_with_mass_matrix_callback<VectorType>;
599 
600  if (setup_mass)
601  {
602  ARKode_mem->ark_msetup = setup_mass_matrix_callback<VectorType>;
603 # if DEAL_II_SUNDIALS_VERSION_LT(3, 0, 0)
604  ARKode_mem->ark_MassSetupNonNull = true;
605 # endif
606  }
607  }
608 
609  status = ARKodeSetOrder(arkode_mem, data.maximum_order);
610  AssertARKode(status);
611 
612  if (custom_setup)
613  custom_setup(arkode_mem);
614  }

685  template <typename VectorType>
686  void
687  ARKode<VectorType>::setup_system_solver(const VectorType &solution)
688  {
689  // no point in setting up a solver when there is no linear system
690  if (!implicit_function)
691  return;
692 
693  int status;
694  (void)status;
695  // Initialize solver
696  // currently only iterative linear solvers are supported
697  if (jacobian_times_vector)
698  {
699  SUNLinearSolver sun_linear_solver;
700  if (solve_linearized_system)
701  {
702  linear_solver =
703  std::make_unique<internal::LinearSolverWrapper<VectorType>>(
704  solve_linearized_system);
705  sun_linear_solver = *linear_solver;
706  }
707  else
