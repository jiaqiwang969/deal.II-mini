884 template <typename MatrixType, typename inverse_type>
885 inline inverse_type
886 PreconditionBlock<MatrixType, inverse_type>::el(size_type i, size_type j) const
887 {
888  const size_type bs = blocksize;
889  const unsigned int nb = i / bs;
890 
891  const FullMatrix<inverse_type> &B = this->inverse(nb);
892 
893  const size_type ib = i % bs;
894  const size_type jb = j % bs;
895 
896  if (jb + nb * bs != j)
897  {
898  return 0.;
899  }
900 
901  return B(ib, jb);
902 }

907 inline PreconditionBlockJacobi<MatrixType, inverse_type>::const_iterator::
908  Accessor::Accessor(
909  const PreconditionBlockJacobi<MatrixType, inverse_type> *matrix,
910  const size_type row)
911  : matrix(matrix)
912  , bs(matrix->block_size())
913  , a_block(row / bs)
914  , b_iterator(&matrix->inverse(0), 0, 0)
915  , b_end(&matrix->inverse(0), 0, 0)
916 {
917  // This is the end accessor, which
918  // does not have a valid block.
919  if (a_block == matrix->size())
920  return;
921 
922  const size_type r = row % bs;
923 
924  b_iterator = matrix->inverse(a_block).begin(r);
925  b_end = matrix->inverse(a_block).end();
926 
927  AssertIndexRange(a_block, matrix->size());
928 }
