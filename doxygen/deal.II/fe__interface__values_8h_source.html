893 template <int dim, int spacedim>
894 template <class CellIteratorType>
895 void
896 FEInterfaceValues<dim, spacedim>::reinit(
897  const CellIteratorType & cell,
898  const unsigned int face_no,
899  const unsigned int sub_face_no,
900  const typename identity<CellIteratorType>::type &cell_neighbor,
901  const unsigned int face_no_neighbor,
902  const unsigned int sub_face_no_neighbor)
903 {
904  if (sub_face_no == numbers::invalid_unsigned_int)
905  {
906  internal_fe_face_values.reinit(cell, face_no);
907  fe_face_values = &internal_fe_face_values;
908  }
909  else
910  {
911  internal_fe_subface_values.reinit(cell, face_no, sub_face_no);
912  fe_face_values = &internal_fe_subface_values;
913  }
914  if (sub_face_no_neighbor == numbers::invalid_unsigned_int)
915  {
916  internal_fe_face_values_neighbor.reinit(cell_neighbor, face_no_neighbor);
917  fe_face_values_neighbor = &internal_fe_face_values_neighbor;
918  }
919  else
920  {
921  internal_fe_subface_values_neighbor.reinit(cell_neighbor,
922  face_no_neighbor,
923  sub_face_no_neighbor);
924  fe_face_values_neighbor = &internal_fe_subface_values_neighbor;
925  }
926 
927  AssertDimension(fe_face_values->n_quadrature_points,
928  fe_face_values_neighbor->n_quadrature_points);
929 
930  const_cast<unsigned int &>(this->n_quadrature_points) =
931  fe_face_values->n_quadrature_points;
932 
933  // Set up dof mapping and remove duplicates (for continuous elements).
934  {
935  // Get dof indices first:
936  std::vector<types::global_dof_index> v(
937  fe_face_values->get_fe().n_dofs_per_cell());
938  cell->get_active_or_mg_dof_indices(v);
939  std::vector<types::global_dof_index> v2(
940  fe_face_values_neighbor->get_fe().n_dofs_per_cell());
941  cell_neighbor->get_active_or_mg_dof_indices(v2);
942 
943  // Fill a map from the global dof index to the left and right
944  // local index.
945  std::map<types::global_dof_index, std::pair<unsigned int, unsigned int>>
946  tempmap;
947  std::pair<unsigned int, unsigned int> invalid_entry(
948  numbers::invalid_unsigned_int, numbers::invalid_unsigned_int);
949 
950  for (unsigned int i = 0; i < v.size(); ++i)
951  {
952  // If not already existing, add an invalid entry:
953  auto result = tempmap.insert(std::make_pair(v[i], invalid_entry));
954  result.first->second.first = i;
955  }
956 
957  for (unsigned int i = 0; i < v2.size(); ++i)
958  {
959  // If not already existing, add an invalid entry:
960  auto result = tempmap.insert(std::make_pair(v2[i], invalid_entry));
961  result.first->second.second = i;
962  }
963 
964  // Transfer from the map to the sorted std::vectors.
965  dofmap.resize(tempmap.size());
966  interface_dof_indices.resize(tempmap.size());
967  unsigned int idx = 0;
968  for (auto &x : tempmap)
969  {
970  interface_dof_indices[idx] = x.first;
971  dofmap[idx] = {{x.second.first, x.second.second}};
972  ++idx;
973  }
974  }
975 }
