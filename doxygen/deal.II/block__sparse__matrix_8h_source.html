352 BlockSparseMatrix<number>::operator=(const double d)

354  Assert(d == 0, ExcScalarAssignmentOnlyForZeroValue());
355 
356  for (size_type r = 0; r < this->n_block_rows(); ++r)
357  for (size_type c = 0; c < this->n_block_cols(); ++c)
358  this->block(r, c) = d;
359 
360  return *this;

453 template <typename number>
454 template <class BlockVectorType>
455 inline void
456 BlockSparseMatrix<number>::precondition_Jacobi(BlockVectorType & dst,
457  const BlockVectorType &src,
458  const number omega) const
459 {
460  Assert(this->n_block_rows() == this->n_block_cols(), ExcNotQuadratic());
461  Assert(dst.n_blocks() == this->n_block_rows(),
462  ExcDimensionMismatch(dst.n_blocks(), this->n_block_rows()));
463  Assert(src.n_blocks() == this->n_block_cols(),
464  ExcDimensionMismatch(src.n_blocks(), this->n_block_cols()));
465 
466  // do a diagonal preconditioning. uses only
467  // the diagonal blocks of the matrix
468  for (size_type i = 0; i < this->n_block_rows(); ++i)
469  this->block(i, i).precondition_Jacobi(dst.block(i), src.block(i), omega);
470 }

477 BlockSparseMatrix<number>::precondition_Jacobi(Vector<number2> & dst,

488  ExcMessage("This function only works if the matrix has "
489  "a single block"));
490 
491  // do a diagonal preconditioning. uses only
492  // the diagonal blocks of the matrix
493  this->block(0, 0).precondition_Jacobi(dst, src, omega);
