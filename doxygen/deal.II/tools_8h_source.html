166  template <int dim, typename AdditionalData>
167  void
168  categorize_by_boundary_ids(const Triangulation<dim> &tria,
169  AdditionalData & additional_data)
170  {
171  // ... determine if we are on an active or a multigrid level
172  const unsigned int level = additional_data.mg_level;
173  const bool is_mg = (level != numbers::invalid_unsigned_int);
174 
175  // ... create empty list for the category of each cell
176  if (is_mg)
177  additional_data.cell_vectorization_category.assign(
178  std::distance(tria.begin(level), tria.end(level)), 0);
179  else
180  additional_data.cell_vectorization_category.assign(tria.n_active_cells(),
181  0);
182 
183  // ... set up scaling factor
184  std::vector<unsigned int> factors(GeometryInfo<dim>::faces_per_cell);
185 
186  auto bids = tria.get_boundary_ids();
187  std::sort(bids.begin(), bids.end());
188 
189  {
190  unsigned int n_bids = bids.size() + 1;
191  int offset = 1;
192  for (unsigned int i = 0; i < GeometryInfo<dim>::faces_per_cell;
193  i++, offset = offset * n_bids)
194  factors[i] = offset;
195  }
196 
197  const auto to_category = [&](const auto &cell) {
198  unsigned int c_num = 0;
199  for (unsigned int i = 0; i < GeometryInfo<dim>::faces_per_cell; i++)
200  {
201  auto &face = *cell->face(i);
202  if (face.at_boundary() && !cell->has_periodic_neighbor(i))
203  c_num +=
204  factors[i] * (1 + std::distance(bids.begin(),
205  std::find(bids.begin(),
206  bids.end(),
207  face.boundary_id())));
208  }
209  return c_num;
210  };
211 
212  if (!is_mg)
213  {
214  for (auto cell = tria.begin_active(); cell != tria.end(); ++cell)
215  {
216  if (cell->is_locally_owned())
217  additional_data
218  .cell_vectorization_category[cell->active_cell_index()] =
219  to_category(cell);
220  }
221  }
222  else
223  {
224  for (auto cell = tria.begin(level); cell != tria.end(level); ++cell)
225  {
226  if (cell->is_locally_owned_on_level())
227  additional_data.cell_vectorization_category[cell->index()] =
228  to_category(cell);
229  }
230  }
231 
232  // ... finalize set up of matrix_free
233  additional_data.hold_all_faces_to_owned_cells = true;
234  additional_data.cell_vectorization_categories_strict = true;
235  additional_data.mapping_update_flags_faces_by_cells =
236  additional_data.mapping_update_flags_inner_faces |
237  additional_data.mapping_update_flags_boundary_faces;
238  }

275  void
276  reinit(const unsigned int cell)
277  {
278  this->phi.reinit(cell);
279  // STEP 1: get relevant information from FEEvaluation
280  const unsigned int first_selected_component =
281  phi.get_first_selected_component();
282  const auto & dof_info = phi.get_dof_info();
283  const unsigned int n_fe_components = dof_info.start_components.back();
284  const unsigned int dofs_per_component = phi.dofs_per_component;
285  const auto & matrix_free = phi.get_matrix_free();
286 
287  const unsigned int n_lanes_filled =
288  matrix_free.n_active_entries_per_cell_batch(cell);
289 
290  std::array<const unsigned int *, n_lanes> dof_indices{};
291  {
292  for (unsigned int v = 0; v < n_lanes_filled; ++v)
293  dof_indices[v] =
294  dof_info.dof_indices.data() +
295  dof_info
296  .row_starts[(cell * n_lanes + v) * n_fe_components +
297  first_selected_component]
298  .first;
299  }
300 
301  // STEP 2: setup CSR storage of transposed locally-relevant
302  // constraint matrix
303  c_pools = std::array<internal::LocalCSR<Number>, n_lanes>();
304 
305  for (unsigned int v = 0; v < n_lanes_filled; ++v)
306  {
307  unsigned int index_indicators, next_index_indicators;
308 
309  index_indicators =
310  dof_info
311  .row_starts[(cell * n_lanes + v) * n_fe_components +
312  first_selected_component]
313  .second;
314  next_index_indicators =
315  dof_info
316  .row_starts[(cell * n_lanes + v) * n_fe_components +
317  first_selected_component + 1]
318  .second;
319 
320  // STEP 2a: setup locally-relevant constraint matrix in a
321  // coordinate list (COO)
322  std::vector<std::tuple<unsigned int, unsigned int, Number>>
323  locally_relevant_constrains; // (constrained local index,
324  // global index of dof which
325  // constrains, weight)
326 
327  if (n_components == 1 || n_fe_components == 1)
328  {
329  AssertDimension(n_components,
330  1); // TODO: currently no block vector supported
331 
332  unsigned int ind_local = 0;
333  for (; index_indicators != next_index_indicators;
334  ++index_indicators, ++ind_local)
335  {
336  const std::pair<unsigned short, unsigned short> indicator =
337  dof_info.constraint_indicator[index_indicators];
338 
339  for (unsigned int j = 0; j < indicator.first;
340  ++j, ++ind_local)
341  locally_relevant_constrains.emplace_back(
342  ind_local, dof_indices[v][j], 1.0);
343 
344  dof_indices[v] += indicator.first;
345 
346  const Number *data_val =
347  matrix_free.constraint_pool_begin(indicator.second);
348  const Number *end_pool =
349  matrix_free.constraint_pool_end(indicator.second);
350 
351  for (; data_val != end_pool; ++data_val, ++dof_indices[v])
352  locally_relevant_constrains.emplace_back(ind_local,
353  *dof_indices[v],
354  *data_val);
355  }
356 
357  AssertIndexRange(ind_local, dofs_per_component + 1);
358 
359  for (; ind_local < dofs_per_component;
360  ++dof_indices[v], ++ind_local)
361  locally_relevant_constrains.emplace_back(ind_local,
362  *dof_indices[v],
363  1.0);
364  }
365  else
366  {
367  // case with vector-valued finite elements where all
368  // components are included in one single vector. Assumption:
369  // first come all entries to the first component, then all
370  // entries to the second one, and so on. This is ensured by
371  // the way MatrixFree reads out the indices.
372  for (unsigned int comp = 0; comp < n_components; ++comp)
373  {
374  unsigned int ind_local = 0;
375 
376  // check whether there is any constraint on the current
377  // cell
378  for (; index_indicators != next_index_indicators;
379  ++index_indicators, ++ind_local)
380  {
381  const std::pair<unsigned short, unsigned short>
382  indicator =
383  dof_info.constraint_indicator[index_indicators];
384 
385  // run through values up to next constraint
386  for (unsigned int j = 0; j < indicator.first;
387  ++j, ++ind_local)
388  locally_relevant_constrains.emplace_back(
389  comp * dofs_per_component + ind_local,
390  dof_indices[v][j],
391  1.0);
392  dof_indices[v] += indicator.first;
393 
394  const Number *data_val =
395  matrix_free.constraint_pool_begin(indicator.second);
396  const Number *end_pool =
397  matrix_free.constraint_pool_end(indicator.second);
398 
399  for (; data_val != end_pool;
400  ++data_val, ++dof_indices[v])
401  locally_relevant_constrains.emplace_back(
402  comp * dofs_per_component + ind_local,
403  *dof_indices[v],
404  *data_val);
405  }
406 
407  AssertIndexRange(ind_local, dofs_per_component + 1);
408 
409  // get the dof values past the last constraint
410  for (; ind_local < dofs_per_component;
411  ++dof_indices[v], ++ind_local)
412  locally_relevant_constrains.emplace_back(
413  comp * dofs_per_component + ind_local,
414  *dof_indices[v],
415  1.0);
416 
417  if (comp + 1 < n_components)
418  {
419  next_index_indicators =
420  dof_info
421  .row_starts[(cell * n_lanes + v) * n_fe_components +
422  first_selected_component + comp + 2]
423  .second;
424  }
425  }
426  }
427 
428  // STEP 2b: transpose COO
429 
430  // presort vector for transposed access
431  std::sort(locally_relevant_constrains.begin(),
432  locally_relevant_constrains.end(),
433  [](const auto &a, const auto &b) {
434  if (std::get<1>(a) < std::get<1>(b))
435  return true;
436  return (std::get<1>(a) == std::get<1>(b)) &&
437  (std::get<0>(a) < std::get<0>(b));
438  });
439 
440  // make sure that all entries are unique
441  locally_relevant_constrains.erase(
442  unique(locally_relevant_constrains.begin(),
443  locally_relevant_constrains.end(),
444  [](const auto &a, const auto &b) {
445  return (std::get<1>(a) == std::get<1>(b)) &&
446  (std::get<0>(a) == std::get<0>(b));
447  }),
448  locally_relevant_constrains.end());
449 
450  // STEP 2c: translate COO to CRS
451  auto &c_pool = c_pools[v];
452  {
453  if (locally_relevant_constrains.size() > 0)
454  c_pool.row_lid_to_gid.emplace_back(
455  std::get<1>(locally_relevant_constrains.front()));
456  for (const auto &j : locally_relevant_constrains)
457  {
458  if (c_pool.row_lid_to_gid.back() != std::get<1>(j))
459  {
460  c_pool.row_lid_to_gid.push_back(std::get<1>(j));
461  c_pool.row.push_back(c_pool.val.size());
462  }
463 
464  c_pool.col.emplace_back(std::get<0>(j));
465  c_pool.val.emplace_back(std::get<2>(j));
466  }
467 
468  if (c_pool.val.size() > 0)
469  c_pool.row.push_back(c_pool.val.size());
470  }
471  }
472  // STEP 3: compute element matrix A_e, apply
473  // locally-relevant constraints C_e^T * A_e * C_e, and get the
474  // the diagonal entry
475  // (C_e^T * A_e * C_e)(i,i)
476  // or
477  // C_e^T(i,:) * A_e * C_e(:,i).
478  //
479  // Since, we compute the element matrix column-by-column and as a
480  // result never actually have the full element matrix, we actually
481  // perform following steps:
482  // 1) loop over all columns of the element matrix
483  // a) compute column i
484  // b) compute for each j (rows of C_e^T):
485  // (C_e^T(j,:) * A_e(:,i)) * C_e(i,j)
486  // or
487  // (C_e^T(j,:) * A_e(:,i)) * C_e^T(j,i)
488  // This gives a contribution the j-th entry of the
489  // locally-relevant diagonal and comprises the multiplication
490  // by the locally-relevant constraint matrix from the left and
491  // the right. There is no contribution to the j-th vector
492  // entry if the j-th row of C_e^T is empty or C_e^T(j,i) is
493  // zero.
494 
495  // set size locally-relevant diagonal
496  for (unsigned int v = 0; v < n_lanes_filled; ++v)
497  diagonals_local_constrained[v].assign(
498  c_pools[v].row_lid_to_gid.size(), Number(0.0));
499  }

501  void
502  prepare_basis_vector(const unsigned int i)
503  {
504  this->i = i;
505 
506  // compute i-th column of element stiffness matrix:
507  // this could be simply performed as done at the moment with
508  // matrix-free operator evaluation applied to a ith-basis vector
509  for (unsigned int j = 0; j < phi.dofs_per_cell; ++j)
510  phi.begin_dof_values()[j] = static_cast<Number>(i == j);
511  }
512 
513  void
514  submit()
515  {
516  const auto ith_column = phi.begin_dof_values();
517 
518  // apply local constraint matrix from left and from right:
519  // loop over all rows of transposed constrained matrix
520  for (unsigned int v = 0;
521  v < phi.get_matrix_free().n_active_entries_per_cell_batch(
522  phi.get_current_cell_index());
523  ++v)
524  {
525  const auto &c_pool = c_pools[v];
526 
527  for (unsigned int j = 0; j < c_pool.row.size() - 1; ++j)
528  {
529  // check if the result will be zero, so that we can skip
530  // the following computations -> binary search
531  const auto scale_iterator =
532  std::lower_bound(c_pool.col.begin() + c_pool.row[j],
533  c_pool.col.begin() + c_pool.row[j + 1],
534  i);
535 
536  // explanation: j-th row of C_e^T is empty (see above)
537  if (scale_iterator == c_pool.col.begin() + c_pool.row[j + 1])
538  continue;
539 
540  // explanation: C_e^T(j,i) is zero (see above)
541  if (*scale_iterator != i)
542  continue;
543 
544  // apply constraint matrix from the left
545  Number temp = 0.0;
546  for (unsigned int k = c_pool.row[j]; k < c_pool.row[j + 1]; ++k)
547  temp += c_pool.val[k] * ith_column[c_pool.col[k]][v];
548 
549  // apply constraint matrix from the right
550  diagonals_local_constrained[v][j] +=
551  temp *
552  c_pool.val[std::distance(c_pool.col.begin(), scale_iterator)];
553  }
554  }
555  }

557  void
558  distribute_local_to_global(
559  LinearAlgebra::distributed::Vector<Number> &diagonal_global)
560  {
561  // STEP 4: assembly results: add into global vector
562  for (unsigned int v = 0;
563  v < phi.get_matrix_free().n_active_entries_per_cell_batch(
564  phi.get_current_cell_index());
565  ++v)
566  for (unsigned int j = 0; j < c_pools[v].row.size() - 1; ++j)
567  ::internal::vector_access_add(
568  diagonal_global,
569  c_pools[v].row_lid_to_gid[j],
570  diagonals_local_constrained[v][j]);
571  }
