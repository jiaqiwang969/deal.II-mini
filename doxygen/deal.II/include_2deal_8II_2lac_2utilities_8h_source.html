224  template <typename NumberType>
225  std::array<NumberType, 3>
226  hyperbolic_rotation(const NumberType &f, const NumberType &g)
227  {
228  Assert(f != 0, ExcDivideByZero());
229  const NumberType tau = g / f;
230  AssertThrow(std::abs(tau) < 1.,
231  ExcMessage(
232  "real-valued Hyperbolic rotation does not exist for (" +
233  std::to_string(f) + "," + std::to_string(g) + ")"));
234  const NumberType u =
235  std::copysign(std::sqrt((1. - tau) * (1. + tau)),
236  f); // <-- more stable than std::sqrt(1.-tau*tau)
237  std::array<NumberType, 3> csr;
238  csr[0] = 1. / u; // c
239  csr[1] = csr[0] * tau; // s
240  csr[2] = f * u; // r
241  return csr;
242  }

258  template <typename NumberType>
259  std::array<NumberType, 3>
260  givens_rotation(const NumberType &f, const NumberType &g)
261  {
262  std::array<NumberType, 3> res;
263  // naive calculation for "r" may overflow or underflow:
264  // c = x / \sqrt{x^2+y^2}
265  // s = -y / \sqrt{x^2+y^2}
266 
267  // See Golub 2013, Matrix computations, Chapter 5.1.8
268  // Algorithm 5.1.3
269  // and
270  // Anderson (2000),
271  // Discontinuous Plane Rotations and the Symmetric Eigenvalue Problem.
272  // LAPACK Working Note 150, University of Tennessee, UT-CS-00-454,
273  // December 4, 2000.
274  // Algorithm 4
275  // We implement the latter below:
276  if (g == NumberType())
277  {
278  res[0] = std::copysign(1., f);
279  res[1] = NumberType();
280  res[2] = std::abs(f);
281  }
282  else if (f == NumberType())
283  {
284  res[0] = NumberType();
285  res[1] = std::copysign(1., g);
286  res[2] = std::abs(g);
287  }
288  else if (std::abs(f) > std::abs(g))
289  {
290  const NumberType tau = g / f;
291  const NumberType u = std::copysign(std::sqrt(1. + tau * tau), f);
292  res[0] = 1. / u; // c
293  res[1] = res[0] * tau; // s
294  res[2] = f * u; // r
295  }
296  else
297  {
298  const NumberType tau = f / g;
299  const NumberType u = std::copysign(std::sqrt(1. + tau * tau), g);
300  res[1] = 1. / u; // s
301  res[0] = res[1] * tau; // c
302  res[2] = g * u; // r
303  }
304 
305  return res;
306  }

310  template <typename OperatorType, typename VectorType>
311  double
312  lanczos_largest_eigenvalue(const OperatorType & H,
313  const VectorType & v0_,
314  const unsigned int k,
315  VectorMemory<VectorType> &vector_memory,
316  std::vector<double> * eigenvalues)
317  {
318  // Do k-step Lanczos:
319 
320  typename VectorMemory<VectorType>::Pointer v(vector_memory);
321  typename VectorMemory<VectorType>::Pointer v0(vector_memory);
322  typename VectorMemory<VectorType>::Pointer f(vector_memory);
323 
324  v->reinit(v0_);
325  v0->reinit(v0_);
326  f->reinit(v0_);
327 
328  // two vectors to store diagonal and subdiagonal of the Lanczos
329  // matrix
330  std::vector<double> diagonal;
331  std::vector<double> subdiagonal;
332 
333  // 1. Normalize input vector
334  (*v) = v0_;
335  double a = v->l2_norm();
336  Assert(a != 0, ExcDivideByZero());
337  (*v) *= 1. / a;
338 
339  // 2. Compute f = Hv; a = f*v; f <- f - av; T(0,0)=a;
340  H.vmult(*f, *v);
341  a = (*f) * (*v);
342  f->add(-a, *v);
343  diagonal.push_back(a);
344 
345  // 3. Loop over steps
346  for (unsigned int i = 1; i < k; ++i)
347  {
348  // 4. L2 norm of f
349  const double b = f->l2_norm();
350  Assert(b != 0, ExcDivideByZero());
351  // 5. v0 <- v; v <- f/b
352  *v0 = *v;
353  *v = *f;
354  (*v) *= 1. / b;
355  // 6. f = Hv; f <- f - b v0;
356  H.vmult(*f, *v);
357  f->add(-b, *v0);
358  // 7. a = f*v; f <- f - a v;
359  a = (*f) * (*v);
360  f->add(-a, *v);
361  // 8. T(i,i-1) = T(i-1,i) = b; T(i,i) = a;
362  diagonal.push_back(a);
363  subdiagonal.push_back(b);
364  }
365 
366  Assert(diagonal.size() == k, ExcInternalError());
367  Assert(subdiagonal.size() == k - 1, ExcInternalError());
368 
369  // Use Lapack dstev to get ||T||_2 norm, i.e. the largest eigenvalue
370  // of T
371  const types::blas_int n = k;
372  std::vector<double> Z; // unused for eigenvalues-only ("N") job
373  const types::blas_int ldz = 1; // ^^ (>=1)
374  std::vector<double> work; // ^^
375  types::blas_int info;
376  // call lapack_templates.h wrapper:
377  internal::UtilitiesImplementation::call_stev('N',
378  n,
379  diagonal.data(),
380  subdiagonal.data(),
381  Z.data(),
382  ldz,
383  work.data(),
384  &info);
385 
386  Assert(info == 0, LAPACKSupport::ExcErrorCode("dstev", info));
387 
388  if (eigenvalues != nullptr)
389  {
390  eigenvalues->resize(diagonal.size());
391  std::copy(diagonal.begin(), diagonal.end(), eigenvalues->begin());
392  }
393 
394  // note that the largest eigenvalue of T is below the largest
395  // eigenvalue of the operator.
396  // return ||T||_2 + ||f||_2, although it is not guaranteed to be an upper
397  // bound.
398  return diagonal[k - 1] + f->l2_norm();
399  }

402  template <typename OperatorType, typename VectorType>
403  void
404  chebyshev_filter(VectorType & x,
405  const OperatorType & op,
406  const unsigned int degree,
407  const std::pair<double, double> unwanted_spectrum,
408  const double a_L,
409  VectorMemory<VectorType> & vector_memory)
410  {
411  const double a = unwanted_spectrum.first;
412  const double b = unwanted_spectrum.second;
413  Assert(degree > 0, ExcMessage("Only positive degrees make sense."));
414 
415  const bool scale = (a_L < std::numeric_limits<double>::infinity());
416  Assert(
417  a < b,
418  ExcMessage(
419  "Lower bound of the unwanted spectrum should be smaller than the upper bound."));
420 
421  Assert(a_L <= a || a_L >= b || !scale,
422  ExcMessage(
423  "Scaling point should be outside of the unwanted spectrum."));
424 
425  // Setup auxiliary vectors:
426  typename VectorMemory<VectorType>::Pointer p_y(vector_memory);
427  typename VectorMemory<VectorType>::Pointer p_yn(vector_memory);
428 
429  p_y->reinit(x);
430  p_yn->reinit(x);
431 
432  // convenience to avoid pointers
433  VectorType &y = *p_y;
434  VectorType &yn = *p_yn;
435 
436  // Below is an implementation of
437  // Algorithm 3.2 in Zhou et al, Journal of Computational Physics 274
438  // (2014) 770-782 with **a bugfix for sigma1**. Here is the original
439  // algorithm verbatim:
440  //
441  // [Y]=chebyshev_filter_scaled(X, m, a, b, aL).
442  // e=(b-a)/2; c=(a+b)/2; =e/(c-aL); =2/;
443  // Y=(HX-cX)(/e);
444  // for i=2 to m do
445  // new =1/( - );
446  // Yt =(HY - cY)(2new/e)-(new)X;
447  // X =Y; Y =Yt; =new;
448 
449  const double e = (b - a) / 2.;
450  const double c = (a + b) / 2.;
451  const double alpha = 1. / e;
452  const double beta = -c / e;
453 
454  const double sigma1 =
455  e / (a_L - c); // BUGFIX which is relevant for odd degrees
456  double sigma = scale ? sigma1 : 1.;
457  const double tau = 2. / sigma;
458  op.vmult(y, x);
459  y.sadd(alpha * sigma, beta * sigma, x);
460 
461  for (unsigned int i = 2; i <= degree; ++i)
462  {
463  const double sigma_new = scale ? 1. / (tau - sigma) : 1.;
464  op.vmult(yn, y);
465  yn.sadd(2. * alpha * sigma_new, 2. * beta * sigma_new, y);
466  yn.add(-sigma * sigma_new, x);
467  x.swap(y);
468  y.swap(yn);
469  sigma = sigma_new;
470  }
471 
472  x.swap(y);
473  }
