1207 template <typename InitFunctionObject, typename LoopFunctionObject>
1208 void
1209 TimeDependent::do_loop(InitFunctionObject init_function,
1210  LoopFunctionObject loop_function,
1211  const TimeSteppingData &timestepping_data,
1212  const Direction direction)
1213 {
1214  // the following functions looks quite
1215  // disrupted due to the recurring switches
1216  // for forward and backward running loops.
1217  //
1218  // I chose to switch at every place where
1219  // it is needed, since it is so easy
1220  // to overlook something when you change
1221  // some code at one place when it needs
1222  // to be changed at a second place, here
1223  // for the other direction, also.
1224 
1225  const unsigned int n_timesteps = timesteps.size();
1226 
1227  // initialize the time steps for
1228  // a round of this loop
1229  for (unsigned int step = 0; step < n_timesteps; ++step)
1230  switch (direction)
1231  {
1232  case forward:
1233  init_function((&*timesteps[step]));
1234  break;
1235  case backward:
1236  init_function((&*timesteps[n_timesteps - step - 1]));
1237  break;
1238  };
1239 
1240 
1241  // wake up the first few time levels
1242  for (int step = -static_cast<int>(timestepping_data.look_ahead); step < 0;
1243  ++step)
1244  for (int look_ahead = 0;
1245  look_ahead <= static_cast<int>(timestepping_data.look_ahead);
1246  ++look_ahead)
1247  switch (direction)
1248  {
1249  case forward:
1250  if (step + look_ahead >= 0)
1251  timesteps[step + look_ahead]->wake_up(look_ahead);
1252  break;
1253  case backward:
1254  if (n_timesteps - (step + look_ahead) < n_timesteps)
1255  timesteps[n_timesteps - (step + look_ahead)]->wake_up(look_ahead);
1256  break;
1257  };
1258 
1259 
1260  for (unsigned int step = 0; step < n_timesteps; ++step)
1261  {
1262  // first thing: wake up the
1263  // timesteps ahead as necessary
1264  for (unsigned int look_ahead = 0;
1265  look_ahead <= timestepping_data.look_ahead;
1266  ++look_ahead)
1267  switch (direction)
1268  {
1269  case forward:
1270  if (step + look_ahead < n_timesteps)
1271  timesteps[step + look_ahead]->wake_up(look_ahead);
1272  break;
1273  case backward:
1274  if (n_timesteps > (step + look_ahead))
1275  timesteps[n_timesteps - (step + look_ahead) - 1]->wake_up(
1276  look_ahead);
1277  break;
1278  };
1279 
1280 
1281  // actually do the work
1282  switch (direction)
1283  {
1284  case forward:
1285  loop_function((&*timesteps[step]));
1286  break;
1287  case backward:
1288  loop_function((&*timesteps[n_timesteps - step - 1]));
1289  break;
1290  };
1291 
1292  // let the timesteps behind sleep
1293  for (unsigned int look_back = 0; look_back <= timestepping_data.look_back;
1294  ++look_back)
1295  switch (direction)
1296  {
1297  case forward:
1298  if (step >= look_back)
1299  timesteps[step - look_back]->sleep(look_back);
1300  break;
1301  case backward:
1302  if (n_timesteps - (step - look_back) <= n_timesteps)
1303  timesteps[n_timesteps - (step - look_back) - 1]->sleep(
1304  look_back);
1305  break;
1306  }
1307  }
1308 
1309  // make the last few timesteps sleep
1310  for (int step = n_timesteps;
1311  step < static_cast<int>(n_timesteps + timestepping_data.look_back);
1312  ++step)
1313  for (int look_back = 0;
1314  look_back <= static_cast<int>(timestepping_data.look_back);
1315  ++look_back)
1316  switch (direction)
1317  {
1318  case forward:
1319  if ((step - look_back >= 0) &&
1320  (step - look_back < static_cast<int>(n_timesteps)))
1321  timesteps[step - look_back]->sleep(look_back);
1322  break;
1323  case backward:
1324  if ((step - look_back >= 0) &&
1325  (step - look_back < static_cast<int>(n_timesteps)))
1326  timesteps[n_timesteps - (step - look_back) - 1]->sleep(look_back);
1327  break;
1328  };
1329 }
