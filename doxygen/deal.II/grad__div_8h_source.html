158  template <int dim>
159  inline void
160  nitsche_matrix(FullMatrix<double> & M,
161  const FEValuesBase<dim> &fe,
162  double penalty,
163  double factor = 1.)
164  {
165  const unsigned int n_dofs = fe.dofs_per_cell;
166 
167  AssertDimension(fe.get_fe().n_components(), dim);
168  AssertDimension(M.m(), n_dofs);
169  AssertDimension(M.n(), n_dofs);
170 
171  for (unsigned int k = 0; k < fe.n_quadrature_points; ++k)
172  {
173  const double dx = factor * fe.JxW(k);
174  const Tensor<1, dim> n = fe.normal_vector(k);
175  for (unsigned int i = 0; i < n_dofs; ++i)
176  for (unsigned int j = 0; j < n_dofs; ++j)
177  {
178  const double divu =
179  fe[FEValuesExtractors::Vector(0)].divergence(j, k);
180  const double divv =
181  fe[FEValuesExtractors::Vector(0)].divergence(i, k);
182  double un = 0., vn = 0.;
183  for (unsigned int d = 0; d < dim; ++d)
184  {
185  un += fe.shape_value_component(j, k, d) * n[d];
186  vn += fe.shape_value_component(i, k, d) * n[d];
187  }
188 
189  M(i, j) += dx * 2. * penalty * un * vn;
190  M(i, j) -= dx * (divu * vn + divv * un);
191  }
192  }
193  }

246  template <int dim>
247  void
248  nitsche_residual(Vector<double> & result,
249  const FEValuesBase<dim> & fe,
250  const ArrayView<const std::vector<double>> & input,
251  const ArrayView<const std::vector<Tensor<1, dim>>> &Dinput,
252  const ArrayView<const std::vector<double>> & data,
253  double penalty,
254  double factor = 1.)
255  {
256  const unsigned int n_dofs = fe.dofs_per_cell;
257  AssertDimension(fe.get_fe().n_components(), dim)
258  AssertVectorVectorDimension(input, dim, fe.n_quadrature_points);
259  AssertVectorVectorDimension(Dinput, dim, fe.n_quadrature_points);
260  AssertVectorVectorDimension(data, dim, fe.n_quadrature_points);
261 
262  for (unsigned int k = 0; k < fe.n_quadrature_points; ++k)
263  {
264  const double dx = factor * fe.JxW(k);
265  const Tensor<1, dim> n = fe.normal_vector(k);
266 
267  double umgn = 0.;
268  double divu = 0.;
269  for (unsigned int d = 0; d < dim; ++d)
270  {
271  umgn += (input[d][k] - data[d][k]) * n[d];
272  divu += Dinput[d][k][d];
273  }
274 
275  for (unsigned int i = 0; i < n_dofs; ++i)
276  {
277  double vn = 0.;
278  const double divv =
279  fe[FEValuesExtractors::Vector(0)].divergence(i, k);
280  for (unsigned int d = 0; d < dim; ++d)
281  vn += fe.shape_value_component(i, k, d) * n[d];
282 
283  result(i) +=
284  dx * (2. * penalty * umgn * vn - divv * umgn - divu * vn);
285  }

296  ip_matrix(FullMatrix<double> & M11,
297  FullMatrix<double> & M12,
298  FullMatrix<double> & M21,
299  FullMatrix<double> & M22,
300  const FEValuesBase<dim> &fe1,

314  AssertDimension(M22.m(), n_dofs);

316  const double fi = factor1;
317  const double fe = (factor2 < 0) ? factor1 : factor2;
318  const double f = .5 * (fi + fe);
319 
320  for (unsigned int k = 0; k < fe1.n_quadrature_points; ++k)
321  {
322  const double dx = fe1.JxW(k);
323  const Tensor<1, dim> n = fe1.normal_vector(k);
324  for (unsigned int i = 0; i < n_dofs; ++i)
325  for (unsigned int j = 0; j < n_dofs; ++j)
326  {
327  double uni = 0.;
328  double une = 0.;
329  double vni = 0.;
330  double vne = 0.;
331  const double divui =
332  fe1[FEValuesExtractors::Vector(0)].divergence(j, k);
333  const double divue =
334  fe2[FEValuesExtractors::Vector(0)].divergence(j, k);
335  const double divvi =
336  fe1[FEValuesExtractors::Vector(0)].divergence(i, k);
337  const double divve =
338  fe2[FEValuesExtractors::Vector(0)].divergence(i, k);
339 
340  for (unsigned int d = 0; d < dim; ++d)
341  {
342  uni += fe1.shape_value_component(j, k, d) * n[d];
343  une += fe2.shape_value_component(j, k, d) * n[d];
344  vni += fe1.shape_value_component(i, k, d) * n[d];
345  vne += fe2.shape_value_component(i, k, d) * n[d];
346  }
347  M11(i, j) +=
348  dx * (-.5 * fi * divvi * uni - .5 * fi * divui * vni +
349  f * penalty * uni * vni);
350  M12(i, j) +=
351  dx * (.5 * fi * divvi * une - .5 * fe * divue * vni -
352  f * penalty * vni * une);
353  M21(i, j) +=
354  dx * (-.5 * fe * divve * uni + .5 * fi * divui * vne -
355  f * penalty * uni * vne);
356  M22(i, j) +=
357  dx * (.5 * fe * divve * une + .5 * fe * divue * vne +
358  f * penalty * une * vne);
359  }
360  }
