620  * <h3>The assemble_system function</h3>
621  * The assemble function here is similar to that in @ref step_12 "step-12" and @ref step_47 "step-47".
622  * Different from assembling by hand, we just need to focus
623  * on assembling on each cell, each boundary face, and each
624  * interior face. The loops over cells and faces are handled
625  * automatically by MeshWorker::mesh_loop().
626  *
627 
628  *
629  * The function starts by defining a local (lambda) function that is
630  * used to integrate the cell terms:
631  *
632  * @code
633  * template <int dim>
634  * void SIPGLaplace<dim>::assemble_system()
635  * {
636  * const auto cell_worker =
637  * [&](const auto &cell, auto &scratch_data, auto &copy_data) {
638  * const FEValues<dim> &fe_v = scratch_data.reinit(cell);
639  * const unsigned int dofs_per_cell = fe_v.dofs_per_cell;
640  * copy_data.reinit(cell, dofs_per_cell);
641  *
642  * const auto & q_points = scratch_data.get_quadrature_points();
643  * const unsigned int n_q_points = q_points.size();
644  * const std::vector<double> &JxW = scratch_data.get_JxW_values();
645  *
646  * std::vector<double> rhs(n_q_points);
647  * rhs_function->value_list(q_points, rhs);
648  *
649  * for (unsigned int point = 0; point < n_q_points; ++point)
650  * for (unsigned int i = 0; i < fe_v.dofs_per_cell; ++i)
651  * {
652  * for (unsigned int j = 0; j < fe_v.dofs_per_cell; ++j)
653  * copy_data.cell_matrix(i, j) +=
654  * diffusion_coefficient * // nu
655  * fe_v.shape_grad(i, point) * // grad v_h
656  * fe_v.shape_grad(j, point) * // grad u_h
657  * JxW[point]; // dx
658  *
659  * copy_data.cell_rhs(i) += fe_v.shape_value(i, point) * // v_h
660  * rhs[point] * // f
661  * JxW[point]; // dx
662  * }
663  * };
664  *
665  * @endcode
666  *
667  * Next, we need a function that assembles face integrals on the boundary:
668  *
669  * @code
670  * const auto boundary_worker = [&](const auto & cell,
671  * const unsigned int &face_no,
672  * auto & scratch_data,
673  * auto & copy_data) {

675  *
676  * const auto & q_points = scratch_data.get_quadrature_points();
677  * const unsigned int n_q_points = q_points.size();
678  * const unsigned int dofs_per_cell = fe_fv.dofs_per_cell;
679  *
680  * const std::vector<double> & JxW = scratch_data.get_JxW_values();
681  * const std::vector<Tensor<1, dim>> &normals =
682  * scratch_data.get_normal_vectors();

685  * exact_solution->value_list(q_points, g);
686  *
687  * const double extent1 = cell->measure() / cell->face(face_no)->measure();
688  * const double penalty = get_penalty_factor(degree, extent1, extent1);
689  *
690  * for (unsigned int point = 0; point < n_q_points; ++point)
691  * {
692  * for (unsigned int i = 0; i < dofs_per_cell; ++i)
693  * for (unsigned int j = 0; j < dofs_per_cell; ++j)
694  * copy_data.cell_matrix(i, j) +=
695  * (-diffusion_coefficient * // - nu
696  * fe_fv.shape_value(i, point) * // v_h
697  * (fe_fv.shape_grad(j, point) * // (grad u_h .
698  * normals[point]) // n)
699  *
700  * - diffusion_coefficient * // - nu
701  * (fe_fv.shape_grad(i, point) * // (grad v_h .
702  * normals[point]) * // n)
703  * fe_fv.shape_value(j, point) // u_h
704  *
705  * + diffusion_coefficient * penalty * // + nu sigma
706  * fe_fv.shape_value(i, point) * // v_h
707  * fe_fv.shape_value(j, point) // u_h
708  *
709  * ) *
710  * JxW[point]; // dx
711  *
712  * for (unsigned int i = 0; i < dofs_per_cell; ++i)
713  * copy_data.cell_rhs(i) +=
714  * (-diffusion_coefficient * // - nu
715  * (fe_fv.shape_grad(i, point) * // (grad v_h .
716  * normals[point]) * // n)
717  * g[point] // g
718  *
719  *
720  * + diffusion_coefficient * penalty * // + nu sigma

724  * JxW[point]; // dx
725  * }

895  * each of the local (lambda) functions is doing, recall first that
896  * the local cell residual is defined as
897  * @f$h_K^2 \left\| f + \nu \Delta u_h \right\|_K^2@f$:
898  *
899  * @code
900  * template <int dim>
901  * void SIPGLaplace<dim>::compute_error_estimate()
902  * {
903  * const auto cell_worker =
904  * [&](const auto &cell, auto &scratch_data, auto &copy_data) {
905  * const FEValues<dim> &fe_v = scratch_data.reinit(cell);
906  *
907  * copy_data.cell_index = cell->active_cell_index();
908  *
909  * const auto & q_points = fe_v.get_quadrature_points();
910  * const unsigned int n_q_points = q_points.size();
911  * const std::vector<double> &JxW = fe_v.get_JxW_values();
912  *
913  * std::vector<Tensor<2, dim>> hessians(n_q_points);
914  * fe_v.get_function_hessians(solution, hessians);
915  *
916  * std::vector<double> rhs(n_q_points);
917  * rhs_function->value_list(q_points, rhs);
918  *
919  * const double hk = cell->diameter();
920  * double residual_norm_square = 0;
921  *
922  * for (unsigned int point = 0; point < n_q_points; ++point)
923  * {
924  * const double residual =
925  * rhs[point] + diffusion_coefficient * trace(hessians[point]);
926  * residual_norm_square += residual * residual * JxW[point];
927  * }
928  * copy_data.value = hk * hk * residual_norm_square;
929  * };
930  *
931  * @endcode
932  *
933  * Next compute boundary terms @f$\sum_{f\in \partial K \cap \partial \Omega}
934  * \sigma \left\| [ u_h-g_D ] \right\|_f^2 @f$:
935  *
936  * @code
937  * const auto boundary_worker = [&](const auto & cell,
938  * const unsigned int &face_no,
939  * auto & scratch_data,
940  * auto & copy_data) {
941  * const FEFaceValuesBase<dim> &fe_fv = scratch_data.reinit(cell, face_no);
942  *
943  * const auto & q_points = fe_fv.get_quadrature_points();
944  * const unsigned n_q_points = q_points.size();
945  *
946  * const std::vector<double> &JxW = fe_fv.get_JxW_values();
947  *
948  * std::vector<double> g(n_q_points);
949  * exact_solution->value_list(q_points, g);
950  *
951  * std::vector<double> sol_u(n_q_points);
952  * fe_fv.get_function_values(solution, sol_u);
953  *
954  * const double extent1 = cell->measure() / cell->face(face_no)->measure();
955  * const double penalty = get_penalty_factor(degree, extent1, extent1);
956  *
957  * double difference_norm_square = 0.;
958  * for (unsigned int point = 0; point < q_points.size(); ++point)
959  * {
960  * const double diff = (g[point] - sol_u[point]);
961  * difference_norm_square += diff * diff * JxW[point];
962  * }
963  * copy_data.value += penalty * difference_norm_square;
964  * };
965  *
966  * @endcode
967  *
968  * And finally interior face terms @f$\sum_{f\in \partial K}\lbrace \sigma
969  * \left\| [u_h] \right\|_f^2 + h_f \left\| [\nu \nabla u_h \cdot
970  * \mathbf n ] \right\|_f^2 \rbrace@f$:
971  *
972  * @code
973  * const auto face_worker = [&](const auto & cell,
974  * const unsigned int &f,
975  * const unsigned int &sf,
976  * const auto & ncell,
977  * const unsigned int &nf,
978  * const unsigned int &nsf,
979  * auto & scratch_data,
980  * auto & copy_data) {
981  * const FEInterfaceValues<dim> &fe_iv =
982  * scratch_data.reinit(cell, f, sf, ncell, nf, nsf);
983  *
984  * copy_data.face_data.emplace_back();
985  * CopyDataFace &copy_data_face = copy_data.face_data.back();
986  *
987  * copy_data_face.cell_indices[0] = cell->active_cell_index();
988  * copy_data_face.cell_indices[1] = ncell->active_cell_index();
989  *
990  * const std::vector<double> & JxW = fe_iv.get_JxW_values();
991  * const std::vector<Tensor<1, dim>> &normals = fe_iv.get_normal_vectors();
992  *
993  * const auto & q_points = fe_iv.get_quadrature_points();
994  * const unsigned int n_q_points = q_points.size();
995  *
996  * std::vector<double> jump(n_q_points);
997  * get_function_jump(fe_iv, solution, jump);
998  *
999  * std::vector<Tensor<1, dim>> grad_jump(n_q_points);
1000  * get_function_gradient_jump(fe_iv, solution, grad_jump);
1001  *
1002  * const double h = cell->face(f)->diameter();
1003  *
1004  * const double extent1 = cell->measure() / cell->face(f)->measure();
1005  * const double extent2 = ncell->measure() / ncell->face(nf)->measure();
1006  * const double penalty = get_penalty_factor(degree, extent1, extent2);
1007  *
1008  * double flux_jump_square = 0;
1009  * double u_jump_square = 0;
1010  * for (unsigned int point = 0; point < n_q_points; ++point)
1011  * {
1012  * u_jump_square += jump[point] * jump[point] * JxW[point];
1013  * const double flux_jump = grad_jump[point] * normals[point];
1014  * flux_jump_square +=
1015  * diffusion_coefficient * flux_jump * flux_jump * JxW[point];
1016  * }
1017  * copy_data_face.values[0] =
1018  * 0.5 * h * (flux_jump_square + penalty * u_jump_square);
1019  * copy_data_face.values[1] = copy_data_face.values[0];
1020  * };
1021  *
1022  * @endcode
1023  *
1024  * Having computed local contributions for each cell, we still
1025  * need a way to copy these into the global vector that will hold
1026  * the error estimators for all cells:
1027  *
1028  * @code
1029  * const auto copier = [&](const auto &copy_data) {
1030  * if (copy_data.cell_index != numbers::invalid_unsigned_int)
1031  * estimated_error_square_per_cell[copy_data.cell_index] +=
1032  * copy_data.value;
1033  * for (auto &cdf : copy_data.face_data)
1034  * for (unsigned int j = 0; j < 2; ++j)
1035  * estimated_error_square_per_cell[cdf.cell_indices[j]] += cdf.values[j];
1036  * };
1037  *
1038  * @endcode
1039  *
1040  * After all of this set-up, let's do the actual work: We resize
1041  * the vector into which the results will be written, and then
1042  * drive the whole process using the MeshWorker::mesh_loop()
1043  * function.
1044  *
1045  * @code
1046  * estimated_error_square_per_cell.reinit(triangulation.n_active_cells());
1047  *

1049  * update_hessians | update_quadrature_points | update_JxW_values;
1050  * const UpdateFlags face_flags = update_values | update_gradients |
1051  * update_quadrature_points |
1052  * update_JxW_values | update_normal_vectors;
1053  *
1054  * ScratchData scratch_data(
1055  * mapping, fe, quadrature, cell_flags, face_quadrature, face_flags);
1056  *
1057  * CopyData copy_data;
1058  * MeshWorker::mesh_loop(dof_handler.begin_active(),
1059  * dof_handler.end(),

1061  * copier,

1063  * copy_data,

1068  * face_worker);
1069  * }
1070  *
1071  * @endcode

1074  * <a name="Thecompute_energy_norm_errorfunction"></a>
1075  * <h3>The compute_energy_norm_error() function</h3>
1076  * Next, we evaluate the accuracy in terms of the energy norm.
1077  * This function is similar to the assembling of the error estimator above.
1078  * Here we compute the square of the energy norm defined by
1079  * @f[
1080  * \|u \|_{1,h}^2 = \sum_{K \in \Gamma_h} \nu\|\nabla u \|_K^2 +
1081  * \sum_{f \in F_i} \sigma \| [ u ] \|_f^2 +
1082  * \sum_{f \in F_b} \sigma \|u\|_f^2.
1083  * @f]
1084  * Therefore the corresponding error is
1085  * @f[
1086  * \|u -u_h \|_{1,h}^2 = \sum_{K \in \Gamma_h} \nu\|\nabla (u_h - u) \|_K^2
1087  * + \sum_{f \in F_i} \sigma \|[ u_h ] \|_f^2 + \sum_{f \in F_b}\sigma
1088  * \|u_h-g_D\|_f^2.
1089  * @f]
1090  *
1091  * @code
1092  * template <int dim>
1093  * double SIPGLaplace<dim>::compute_energy_norm_error()
1094  * {
1095  * energy_norm_square_per_cell.reinit(triangulation.n_active_cells());
1096  *
1097  * @endcode
1098  *
1099  * Assemble @f$\sum_{K \in \Gamma_h} \nu\|\nabla (u_h - u) \|_K^2 @f$.
1100  *
1101  * @code
1102  * const auto cell_worker =
1103  * [&](const auto &cell, auto &scratch_data, auto &copy_data) {
1104  * const FEValues<dim> &fe_v = scratch_data.reinit(cell);
1105  *
1106  * copy_data.cell_index = cell->active_cell_index();
1107  *
1108  * const auto & q_points = fe_v.get_quadrature_points();
1109  * const unsigned int n_q_points = q_points.size();
1110  * const std::vector<double> &JxW = fe_v.get_JxW_values();

1112  * std::vector<Tensor<1, dim>> grad_u(n_q_points);
1113  * fe_v.get_function_gradients(solution, grad_u);
1114  *
1115  * std::vector<Tensor<1, dim>> grad_exact(n_q_points);
1116  * exact_solution->gradient_list(q_points, grad_exact);
1117  *

1119  * for (unsigned int point = 0; point < n_q_points; ++point)
1120  * {
1121  * norm_square +=
1122  * (grad_u[point] - grad_exact[point]).norm_square() * JxW[point];
1123  * }
1124  * copy_data.value = diffusion_coefficient * norm_square;
1125  * };
1126  *
1127  * @endcode
1128  *
1129  * Assemble @f$\sum_{f \in F_b}\sigma \|u_h-g_D\|_f^2@f$.
1130  *
1131  * @code

1133  * const unsigned int &face_no,
1134  * auto & scratch_data,
1135  * auto & copy_data) {

1137  *
1138  * const auto & q_points = fe_fv.get_quadrature_points();
1139  * const unsigned n_q_points = q_points.size();
1140  *
1141  * const std::vector<double> &JxW = fe_fv.get_JxW_values();
1142  *
1143  * std::vector<double> g(n_q_points);
1144  * exact_solution->value_list(q_points, g);
1145  *
1146  * std::vector<double> sol_u(n_q_points);
1147  * fe_fv.get_function_values(solution, sol_u);
1148  *
1149  * const double extent1 = cell->measure() / cell->face(face_no)->measure();
1150  * const double penalty = get_penalty_factor(degree, extent1, extent1);
1151  *
1152  * double difference_norm_square = 0.;
1153  * for (unsigned int point = 0; point < q_points.size(); ++point)
1154  * {
1155  * const double diff = (g[point] - sol_u[point]);
1156  * difference_norm_square += diff * diff * JxW[point];
1157  * }
1158  * copy_data.value += penalty * difference_norm_square;
1159  * };
1160  *
1161  * @endcode
1162  *
1163  * Assemble @f$\sum_{f \in F_i} \sigma \| [ u_h ] \|_f^2@f$.
1164  *
1165  * @code
1166  * const auto face_worker = [&](const auto & cell,
1167  * const unsigned int &f,

1171  * const unsigned int &nsf,
1172  * auto & scratch_data,
1173  * auto & copy_data) {

1175  * scratch_data.reinit(cell, f, sf, ncell, nf, nsf);
1176  *
1177  * copy_data.face_data.emplace_back();

1179  *
1180  * copy_data_face.cell_indices[0] = cell->active_cell_index();

1185  * const auto & q_points = fe_iv.get_quadrature_points();

1188  * std::vector<double> jump(n_q_points);
1189  * get_function_jump(fe_iv, solution, jump);
1190  *

1199  * }
1200  * copy_data_face.values[0] = 0.5 * penalty * u_jump_square;
1201  * copy_data_face.values[1] = copy_data_face.values[0];
1202  * };
1203  *
1204  * const auto copier = [&](const auto &copy_data) {
1205  * if (copy_data.cell_index != numbers::invalid_unsigned_int)
1206  * energy_norm_square_per_cell[copy_data.cell_index] += copy_data.value;

1208  * for (unsigned int j = 0; j < 2; ++j)

1210  * };

1213  * update_gradients | update_quadrature_points | update_JxW_values;
1214  * UpdateFlags face_flags =
1215  * update_values | update_quadrature_points | update_JxW_values;
