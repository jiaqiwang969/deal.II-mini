897  * @f$h_K^2 \left\| f + \nu \Delta u_h \right\|_K^2@f$:

906  *
907  * copy_data.cell_index = cell->active_cell_index();
908  *
909  * const auto & q_points = fe_v.get_quadrature_points();
910  * const unsigned int n_q_points = q_points.size();
911  * const std::vector<double> &JxW = fe_v.get_JxW_values();
912  *
913  * std::vector<Tensor<2, dim>> hessians(n_q_points);
914  * fe_v.get_function_hessians(solution, hessians);
915  *
916  * std::vector<double> rhs(n_q_points);
917  * rhs_function->value_list(q_points, rhs);

920  * double residual_norm_square = 0;
921  *

923  * {
924  * const double residual =
925  * rhs[point] + diffusion_coefficient * trace(hessians[point]);
926  * residual_norm_square += residual * residual * JxW[point];
927  * }
928  * copy_data.value = hk * hk * residual_norm_square;
929  * };

955  * const double penalty = get_penalty_factor(degree, extent1, extent1);
956  *
957  * double difference_norm_square = 0.;
958  * for (unsigned int point = 0; point < q_points.size(); ++point)
959  * {
960  * const double diff = (g[point] - sol_u[point]);

1076  * Next, we evaluate the accuracy in terms of the energy norm.
1077  * This function is similar to the assembling of the error estimator above.
1078  * Here we compute the square of the energy norm defined by
1079  * @f[

1085  * @f[
