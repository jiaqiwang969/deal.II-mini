440 {
441  size_type used_total = 0;

443  size_type reserved_total = 0;
444  std::vector<size_type> used_by_line_total;
445 
446  size_type used;
447  size_type allocated;
448  size_type reserved;

450 
451  for (size_type i = 0; i < n_block_rows(); ++i)

453  {
454  used_by_line.clear();
455  out << "block:\t" << i << '\t' << j << std::endl;
456  block(i, j).compute_statistics(
457  used, allocated, reserved, used_by_line, full);
458 
459  out << "used:" << used << std::endl
460  << "allocated:" << allocated << std::endl
461  << "reserved:" << reserved << std::endl;
462 
463  used_total += used;
464  allocated_total += allocated;
465  reserved_total += reserved;
466 
467  if (full)
468  {
469  used_by_line_total.resize(used_by_line.size());
470  for (size_type i = 0; i < used_by_line.size(); ++i)
471  if (used_by_line[i] != 0)
472  {
473  out << "row-entries\t" << i << "\trows\t" << used_by_line[i]
474  << std::endl;
475  used_by_line_total[i] += used_by_line[i];
476  }
477  }
478  }
479  out << "Total" << std::endl
480  << "used:" << used_total << std::endl
481  << "allocated:" << allocated_total << std::endl
482  << "reserved:" << reserved_total << std::endl;
483  for (size_type i = 0; i < used_by_line_total.size(); ++i)
484  if (used_by_line_total[i] != 0)
485  {
486  out << "row-entries\t" << i << "\trows\t" << used_by_line_total[i]
487  << std::endl;
488  }
