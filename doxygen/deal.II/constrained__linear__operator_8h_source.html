62 template <typename Range, typename Domain, typename Payload>
63 LinearOperator<Range, Domain, Payload>
64 distribute_constraints_linear_operator(
65  const AffineConstraints<typename Range::value_type> &constraints,
66  const LinearOperator<Range, Domain, Payload> & exemplar)
67 {
68  LinearOperator<Range, Domain, Payload> return_op = exemplar;
69 
70  return_op.vmult_add = [&constraints](Range &v, const Domain &u) {
71  Assert(!::PointerComparison::equal(&v, &u),
72  ::ExcMessage("The domain and range vectors must be different "
73  "storage locations"));
74 
75  // First, add vector u to v unconditionally and clean up constrained
76  // degrees of freedom later.
77  v += u;
78 
79  const auto &locally_owned_elements = v.locally_owned_elements();
80  for (const auto &line : constraints.get_lines())
81  {
82  const auto i = line.index;
83  if (locally_owned_elements.is_element(i))
84  {
85  v(i) -= u(i);
86  const auto &entries = line.entries;
87  for (types::global_dof_index j = 0; j < entries.size(); ++j)
88  {
89  const auto pos = entries[j].first;
90  v(i) += u(pos) * entries[j].second;
91  }
92  }
93  }
94 
95  v.compress(VectorOperation::add);
96  };

239 template <typename Range, typename Domain, typename Payload>
240 LinearOperator<Range, Domain, Payload>
241 constrained_linear_operator(
242  const AffineConstraints<typename Range::value_type> &constraints,
243  const LinearOperator<Range, Domain, Payload> & linop)
244 {
245  const auto C = distribute_constraints_linear_operator(constraints, linop);
246  const auto Ct = transpose_operator(C);

286 template <typename Range, typename Domain, typename Payload>
287 PackagedOperation<Range>
288 constrained_right_hand_side(
289  const AffineConstraints<typename Range::value_type> &constraints,
290  const LinearOperator<Range, Domain, Payload> & linop,
291  const Range & right_hand_side)
292 {
293  PackagedOperation<Range> return_comp;
294 
295  return_comp.reinit_vector = linop.reinit_range_vector;
296 
297  return_comp.apply_add = [&constraints, &linop, &right_hand_side](Range &v) {
298  const auto C = distribute_constraints_linear_operator(constraints, linop);
299  const auto Ct = transpose_operator(C);
300 
301  GrowingVectorMemory<Domain> vector_memory;
302  typename VectorMemory<Domain>::Pointer k(vector_memory);
303  linop.reinit_domain_vector(*k, /*bool fast=*/ false);
304  constraints.distribute(*k);
305 
306  v += Ct * (right_hand_side - linop * *k);
307  };
308 
309  return_comp.apply = [apply_add = return_comp.apply_add](Range &v) {
310  v = 0.;
311  apply_add(v);
312  };
313 
314  return return_comp;
315 }
