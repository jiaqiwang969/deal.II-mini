118  CellIDTranslator<dim>::CellIDTranslator(
119  const types::global_cell_index n_coarse_cells,
120  const types::global_cell_index n_global_levels)

146  "of indices."));

148  tree_sizes.push_back(0);

231  template <int dim>
232  types::global_cell_index
233  CellIDTranslator<dim>::convert_cell_id_binary_type_to_level_coarse_cell_id(
234  const typename CellId::binary_type &binary_representation)
235  {
236  // exploiting the structure of CellId::binary_type
237  // see also the documentation of CellId
238 
239  // actual coarse-grid id
240  const unsigned int coarse_cell_id = binary_representation[0];
241  const unsigned int n_child_indices = binary_representation[1] >> 2;
242 
243  const unsigned int children_per_value =
244  sizeof(CellId::binary_type::value_type) * 8 / dim;
245  unsigned int child_level = 0;
246  unsigned int binary_entry = 2;
247 
248  // path to the get to the cell
249  std::vector<unsigned int> cell_indices;
250  while (child_level < n_child_indices)
251  {
252  Assert(binary_entry < binary_representation.size(), ExcInternalError());
253 
254  for (unsigned int j = 0; j < children_per_value; ++j)
255  {
256  unsigned int cell_index =
257  (((binary_representation[binary_entry] >> (j * dim))) &
258  (GeometryInfo<dim>::max_children_per_cell - 1));
259  cell_indices.push_back(cell_index);
260  ++child_level;
261  if (child_level == n_child_indices)
262  break;
263  }
264  ++binary_entry;
265  }
266 
267  // compute new coarse-grid id: c_{i+1} = c_{i}*2^dim + q;
268  types::global_cell_index level_coarse_cell_id = coarse_cell_id;
269  for (auto i : cell_indices)
270  level_coarse_cell_id =
271  level_coarse_cell_id * GeometryInfo<dim>::max_children_per_cell + i;
272 
273  return level_coarse_cell_id;
274  }
