459 template <int dim, typename Number>
460 template <typename Number2, int spacedim>
461 void
462 MGTransferMatrixFree<dim, Number>::interpolate_to_mg(
463  const DoFHandler<dim, spacedim> & dof_handler,
464  MGLevelObject<LinearAlgebra::distributed::Vector<Number>> &dst,
465  const LinearAlgebra::distributed::Vector<Number2> & src) const
466 {
467  const unsigned int min_level = dst.min_level();
468  const unsigned int max_level = dst.max_level();
469 
470  Assert(max_level == dof_handler.get_triangulation().n_global_levels() - 1,
471  ExcDimensionMismatch(
472  max_level, dof_handler.get_triangulation().n_global_levels() - 1));
473 
474  const FiniteElement<dim, spacedim> &fe = dof_handler.get_fe();
475 
476  for (unsigned int level = min_level; level <= max_level; ++level)
477  if (dst[level].size() != dof_handler.n_dofs(level) ||
478  dst[level].locally_owned_size() !=
479  dof_handler.locally_owned_mg_dofs(level).n_elements())
480  dst[level].reinit(this->vector_partitioners[level]);
481 
482  // copy fine level vector to active cells in MG hierarchy
483  this->copy_to_mg(dof_handler, dst, src, true);
484 
485  // FIXME: maybe need to store hanging nodes constraints per level?
486  // MGConstrainedDoFs does NOT keep this info right now, only periodicity
487  // constraints...
488 
489  // do the transfer from level to level-1:
490  dst[max_level].update_ghost_values();
491  for (unsigned int level = max_level; level > min_level; --level)
492  {
493  // auxiliary vector which always has ghost elements
494  const LinearAlgebra::distributed::Vector<Number> *input = nullptr;
495  LinearAlgebra::distributed::Vector<Number> ghosted_fine;
496  if (dst[level].get_partitioner().get() ==
497  this->vector_partitioners[level].get())
498  input = &dst[level];
499  else
500  {
501  ghosted_fine.reinit(this->vector_partitioners[level]);
502  ghosted_fine.copy_locally_owned_data_from(dst[level]);
503  ghosted_fine.update_ghost_values();
504  input = &ghosted_fine;
505  }
506 
507  std::vector<Number> dof_values_coarse(fe.n_dofs_per_cell());
508  Vector<Number> dof_values_fine(fe.n_dofs_per_cell());
509  Vector<Number> tmp(fe.n_dofs_per_cell());
510  std::vector<types::global_dof_index> dof_indices(fe.n_dofs_per_cell());
511  for (const auto &cell : dof_handler.cell_iterators_on_level(level - 1))
512  if (cell->is_locally_owned_on_level())
513  {
514  // if we get to a cell without children (== active), we can
515  // skip it as there values should be already set by the
516  // equivalent of copy_to_mg()
517  if (cell->is_active())
518  continue;
519 
520  std::fill(dof_values_coarse.begin(), dof_values_coarse.end(), 0.);
521  for (unsigned int child = 0; child < cell->n_children(); ++child)
522  {
523  cell->child(child)->get_mg_dof_indices(dof_indices);
524  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
525  dof_values_fine(i) = (*input)(dof_indices[i]);
526  fe.get_restriction_matrix(child, cell->refinement_case())
527  .vmult(tmp, dof_values_fine);
528  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
529  if (fe.restriction_is_additive(i))
530  dof_values_coarse[i] += tmp[i];
531  else if (tmp(i) != 0.)
532  dof_values_coarse[i] = tmp[i];
533  }
534  cell->get_mg_dof_indices(dof_indices);
535  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
536  if (dof_handler.locally_owned_mg_dofs(level - 1).is_element(
537  dof_indices[i]))
538  dst[level - 1](dof_indices[i]) = dof_values_coarse[i];
539  }
540 
541  dst[level - 1].update_ghost_values();
542  }
543 }

569 template <int dim, typename Number>
570 template <typename Number2, int spacedim>
571 void
572 MGTransferBlockMatrixFree<dim, Number>::copy_to_mg(
573  const std::vector<const DoFHandler<dim, spacedim> *> & dof_handler,
574  MGLevelObject<LinearAlgebra::distributed::BlockVector<Number>> &dst,
575  const LinearAlgebra::distributed::BlockVector<Number2> & src) const
576 {
577  const unsigned int n_blocks = src.n_blocks();
578  AssertDimension(dof_handler.size(), n_blocks);
579 
580  if (n_blocks == 0)
581  return;
582 
583  const unsigned int min_level = dst.min_level();
584  const unsigned int max_level = dst.max_level();
585 
586  // this function is normally called within the Multigrid class with
587  // dst == defect level block vector. At first run this vector is not
588  // initialized. Do this below:
589  {
590  const parallel::TriangulationBase<dim, spacedim> *tria =
591  (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
592  &(dof_handler[0]->get_triangulation())));
593  for (unsigned int i = 1; i < n_blocks; ++i)
594  AssertThrow(
595  (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
596  &(dof_handler[0]->get_triangulation())) == tria),
597  ExcMessage("The DoFHandler use different Triangulations!"));
598 
599  MGLevelObject<bool> do_reinit;
600  do_reinit.resize(min_level, max_level);
601  for (unsigned int level = min_level; level <= max_level; ++level)
602  {
603  do_reinit[level] = false;
604  if (dst[level].n_blocks() != n_blocks)
605  {
606  do_reinit[level] = true;
607  continue; // level
608  }
609  for (unsigned int b = 0; b < n_blocks; ++b)
610  {
611  LinearAlgebra::distributed::Vector<Number> &v = dst[level].block(b);
612  if (v.size() !=
613  dof_handler[b]->locally_owned_mg_dofs(level).size() ||
614  v.locally_owned_size() !=
615  dof_handler[b]->locally_owned_mg_dofs(level).n_elements())
616  {
617  do_reinit[level] = true;
618  break; // b
619  }
620  }
621  }
622 
623  for (unsigned int level = min_level; level <= max_level; ++level)
624  {
625  if (do_reinit[level])
626  {
627  dst[level].reinit(n_blocks);
628  for (unsigned int b = 0; b < n_blocks; ++b)
629  {
630  LinearAlgebra::distributed::Vector<Number> &v =
631  dst[level].block(b);
632  v.reinit(dof_handler[b]->locally_owned_mg_dofs(level),
633  dof_handler[b]->get_communicator());
634  }
635  dst[level].collect_sizes();
636  }
637  else
638  dst[level] = 0;
639  }
640  }
641 
642  // FIXME: this a quite ugly as we need a temporary object:
643  MGLevelObject<LinearAlgebra::distributed::Vector<Number>> dst_non_block(
644  min_level, max_level);
645 
646  for (unsigned int b = 0; b < n_blocks; ++b)
647  {
648  for (unsigned int l = min_level; l <= max_level; ++l)
649  dst_non_block[l].reinit(dst[l].block(b));
650  const unsigned int data_block = same_for_all ? 0 : b;
651  matrix_free_transfer_vector[data_block].copy_to_mg(*dof_handler[b],
652  dst_non_block,
653  src.block(b));
654 
655  for (unsigned int l = min_level; l <= max_level; ++l)
656  dst[l].block(b) = dst_non_block[l];
657  }
658 }
