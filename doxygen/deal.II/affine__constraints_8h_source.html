1620 template <typename number>
1621 inline void
1622 AffineConstraints<number>::add_line(const size_type line_n)
1623 {
1624  Assert(sorted == false, ExcMatrixIsClosed());
1625 
1626  // the following can happen when we compute with distributed meshes and dof
1627  // handlers and we constrain a degree of freedom whose number we don't have
1628  // locally. if we don't abort here the program will try to allocate several
1629  // terabytes of memory to resize the various arrays below :-)
1630  Assert(line_n != numbers::invalid_size_type, ExcInternalError());
1631  const size_type line_index = calculate_line_index(line_n);
