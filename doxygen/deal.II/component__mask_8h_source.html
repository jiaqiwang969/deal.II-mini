328 inline ComponentMask
329 ComponentMask::operator|(const ComponentMask &mask) const
330 {
331  // if one of the two masks denotes the all-component mask,
332  // then return the other one
333  if (component_mask.size() == 0)
334  return mask;
335  else if (mask.component_mask.size() == 0)
336  return *this;
337  else
338  {
339  // if both masks have individual entries set, form
340  // the combination of the two
341  AssertDimension(component_mask.size(), mask.component_mask.size());
342  std::vector<bool> new_mask(component_mask.size());
343  for (unsigned int i = 0; i < component_mask.size(); ++i)
344  new_mask[i] = (component_mask[i] || mask.component_mask[i]);
345 
346  return new_mask;
347  }
348 }
