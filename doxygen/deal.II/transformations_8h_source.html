712 template <typename Number>
713 Tensor<2, 3, Number>
714 Physics::Transformations::Rotations::rotation_matrix_3d(
715  const Point<3, Number> &axis,
716  const Number & angle)
717 {
718  Assert(std::abs(axis.norm() - 1.0) < 1e-9,
719  ExcMessage("The supplied axial vector is not a unit vector."));
720  const Number c = std::cos(angle);
721  const Number s = std::sin(angle);
722  const Number t = 1. - c;
723  const Number rotation[3][3] = {{t * axis[0] * axis[0] + c,
724  t * axis[0] * axis[1] - s * axis[2],
725  t * axis[0] * axis[2] + s * axis[1]},
726  {t * axis[0] * axis[1] + s * axis[2],
727  t * axis[1] * axis[1] + c,
728  t * axis[1] * axis[2] - s * axis[0]},
729  {t * axis[0] * axis[2] - s * axis[1],
730  t * axis[1] * axis[2] + s * axis[0],
731  t * axis[2] * axis[2] + c}};
732  return Tensor<2, 3, Number>(rotation);
733 }

1092 template <int dim, typename Number>
1093 inline SymmetricTensor<2, dim, Number>
1094 Physics::Transformations::basis_transformation(
1095  const SymmetricTensor<2, dim, Number> &T,
1096  const Tensor<2, dim, Number> & B)
1097 {
1098  Tensor<2, dim, Number> tmp_1;
1099  for (unsigned int i = 0; i < dim; ++i)
1100  for (unsigned int J = 0; J < dim; ++J)
1101  // Loop over I but complex.h defines a macro I, so use I_ instead
1102  for (unsigned int I_ = 0; I_ < dim; ++I_)
1103  tmp_1[i][J] += B[i][I_] * T[I_][J];
1104 
1105  SymmetricTensor<2, dim, Number> out;
1106  for (unsigned int i = 0; i < dim; ++i)
1107  for (unsigned int j = i; j < dim; ++j)
1108  for (unsigned int J = 0; J < dim; ++J)
1109  out[i][j] += B[j][J] * tmp_1[i][J];
1110 
1111  return out;
1112 }

1116 template <int dim, typename Number>
1117 inline Tensor<4, dim, Number>
1118 Physics::Transformations::basis_transformation(const Tensor<4, dim, Number> &H,
1119  const Tensor<2, dim, Number> &B)
1120 {
1121  // This contraction order and indexing might look a bit dubious, so a
1122  // quick explanation as to what's going on is probably in order:
1123  //
1124  // When the contract() function operates on the inner indices, the
1125  // result has the inner index and outer index transposed, i.e.
1126  // contract<2,1>(H,F) implies
1127  // T_{IJLk} = (H_{IJMN} F_{mM}) \delta_{mL} \delta_{Nk}
1128  // rather than T_{IJkL} (the desired result).
1129  // So, in effect, contraction of the 3rd (inner) index with F as the
1130  // second argument results in its transposition with respect to its
1131  // adjacent neighbor. This is due to the position of the argument F,
1132  // leading to the free index being on the right hand side of the result.
1133  // However, given that we can do two transformations from the LHS of H
1134  // and two from the right we can undo the otherwise erroneous
1135  // swapping of the outer indices upon application of the second
1136  // sets of contractions.
1137  //
1138  // Note: Its significantly quicker (in 3d) to push forward
1139  // each index individually
1140  return contract<1, 1>(
1141  B, contract<1, 1>(B, contract<2, 1>(contract<2, 1>(H, B), B)));

1153  // break and recover the symmetry properties of the tensors.
1154  // We also want to perform a minimal number of operations here
1155  // and avoid some complications related to the transposition of
1156  // tensor indices when contracting inner indices using the contract()
1157  // function. (For an explanation of the contraction operations,
1158  // please see the note in the equivalent function for standard
1159  // Tensors.) So what we'll do here is manually perform the first
1160  // and last contractions that break/recover the tensor symmetries
1161  // on the inner indices, and use the contract() function only on

1165  // each index individually

1167  // Push forward (inner) index 1
1168  Tensor<4, dim, Number> tmp;

1170  for (unsigned int I_ = 0; I_ < dim; ++I_)

1172  for (unsigned int K = 0; K < dim; ++K)
1173  for (unsigned int L = 0; L < dim; ++L)
1174  for (unsigned int J = 0; J < dim; ++J)

1176 
1177  // Push forward (outer) indices 0 and 3
1178  tmp = contract<1, 0>(B, contract<3, 1>(tmp, B));
1179

1181  SymmetricTensor<4, dim, Number> out;

1183  for (unsigned int j = i; j < dim; ++j)
1184  for (unsigned int k = 0; k < dim; ++k)
1185  for (unsigned int l = k; l < dim; ++l)
1186  for (unsigned int K = 0; K < dim; ++K)
