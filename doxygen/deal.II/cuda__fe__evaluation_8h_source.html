481  n_components_,
482  Number>::gradient_type
483  FEEvaluation<dim, fe_degree, n_q_points_1d, n_components_, Number>::
484  get_gradient() const
485  {
486  static_assert(n_components_ == 1, "This function only supports FE with one \
487  components");
488 
489  // TODO optimize if the mesh is uniform
490  const unsigned int q_point = internal::compute_index<dim, n_q_points_1d>();
491  const Number * inv_jacobian = &inv_jac[q_point];
492  gradient_type grad;
493  for (int d_1 = 0; d_1 < dim; ++d_1)
494  {
495  Number tmp = 0.;
496  for (int d_2 = 0; d_2 < dim; ++d_2)
497  tmp += inv_jacobian[padding_length * n_cells * (dim * d_2 + d_1)] *
498  gradients[d_2][q_point];
499  grad[d_1] = tmp;
500  }
501 
502  return grad;
503  }
