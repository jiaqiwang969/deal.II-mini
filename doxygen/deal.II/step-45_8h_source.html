734  *
735  * const FEValuesExtractors::Vector velocities(0);
736  * const FEValuesExtractors::Scalar pressure(dim);
737  *
738  * std::vector<SymmetricTensor<2, dim>> symgrad_phi_u(dofs_per_cell);
739  * std::vector<double> div_phi_u(dofs_per_cell);
740  * std::vector<double> phi_p(dofs_per_cell);
741  *
742  * for (const auto &cell : dof_handler.active_cell_iterators())
743  * if (cell->is_locally_owned())
744  * {

746  * local_matrix = 0;
747  * local_preconditioner_matrix = 0;
748  * local_rhs = 0;
749  *
750  * right_hand_side.vector_value_list(fe_values.get_quadrature_points(),
751  * rhs_values);
752  *
753  * for (unsigned int q = 0; q < n_q_points; ++q)
754  * {
755  * for (unsigned int k = 0; k < dofs_per_cell; ++k)
756  * {
757  * symgrad_phi_u[k] =

761  * }

764  * {
765  * for (unsigned int j = 0; j <= i; ++j)
766  * {
767  * local_matrix(i, j) +=
768  * (symgrad_phi_u[i] * symgrad_phi_u[j] // diffusion
769  * - div_phi_u[i] * phi_p[j] // pressure force
770  * - phi_p[i] * div_phi_u[j]) // divergence
771  * * fe_values.JxW(q);
772  *
773  * local_preconditioner_matrix(i, j) +=

783  * }

786  * for (unsigned int j = i + 1; j < dofs_per_cell; ++j)
787  * {

800  * local_dof_indices,
