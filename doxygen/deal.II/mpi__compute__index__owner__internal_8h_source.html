504  void
505  partition(const IndexSet &owned_indices, const MPI_Comm &comm)
506  {
507 #ifdef DEAL_II_WITH_MPI
508  const unsigned int n_procs = n_mpi_processes(comm);
509  const unsigned int my_rank = this_mpi_process(comm);
510 
511  size = owned_indices.size();
512 
513  Assert(size > 0, ExcNotImplemented());
514 
515  dofs_per_process = (size + n_procs - 1) / n_procs;
516  if (dofs_per_process < range_minimum_grain_size)
517  {
518  dofs_per_process = range_minimum_grain_size;
519  stride_small_size = dofs_per_process * n_procs / size;
520  }
521  else
522  stride_small_size = 1;
523  local_range.first = get_index_offset(my_rank);
524  local_range.second = get_index_offset(my_rank + 1);
525 
526  locally_owned_size = local_range.second - local_range.first;
527 
528  actually_owning_ranks = {};
529  actually_owning_ranks.resize(locally_owned_size,
530  numbers::invalid_unsigned_int);
531 #else
532  (void)owned_indices;
533  (void)comm;
534 #endif
535  }

978  requesters[owner_index].back().second.back().second !=

980  requesters[owner_index].back().second.emplace_back(
981  index - dict.local_range.first,
982  index - dict.local_range.first + 1);
983  else
984  ++requesters[owner_index].back().second.back().second;
985  }
