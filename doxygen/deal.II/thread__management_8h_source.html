171  template <typename ForwardIterator>
172  std::vector<std::pair<ForwardIterator, ForwardIterator>>
173  split_range(const ForwardIterator &begin,
174  const ForwardIterator &end,
175  const unsigned int n_intervals)
176  {
177  using IteratorPair = std::pair<ForwardIterator, ForwardIterator>;
178 
179  // in non-multithreaded mode, we often have the case that this
180  // function is called with n_intervals==1, so have a shortcut here
181  // to handle that case efficiently
182 
183  if (n_intervals == 1)
184  return (std::vector<IteratorPair>(1, IteratorPair(begin, end)));
185 
186  // if more than one interval requested, do the full work
187  const unsigned int n_elements = std::distance(begin, end);
188  const unsigned int n_elements_per_interval = n_elements / n_intervals;
189  const unsigned int residual = n_elements % n_intervals;
190 
191  std::vector<IteratorPair> return_values(n_intervals);
192 
193  return_values[0].first = begin;
194  for (unsigned int i = 0; i < n_intervals; ++i)
195  {
196  if (i != n_intervals - 1)
197  {
198  return_values[i].second = return_values[i].first;
199  // note: the cast is performed to avoid a warning of gcc
200  // that in the library `dist>=0' is checked (dist has a
201  // template type, which here is unsigned if no cast is
202  // performed)
203  std::advance(return_values[i].second,
204  static_cast<signed int>(n_elements_per_interval));
205  // distribute residual in division equally among the first
206  // few subintervals
207  if (i < residual)
208  ++return_values[i].second;
209 
210  return_values[i + 1].first = return_values[i].second;
211  }
212  else
213  return_values[i].second = end;
214  }
215  return return_values;
216  }

1074  wait()
1075  {
1076  // If we have previously already moved the result, then we don't
1077  // need a lock and can just return.
1078  if (task_has_finished)
1079  return;
1080 
1081  // Else, we need to go under a lock and try again. A different thread
1082  // may have waited and finished the task since then, so we have to try
1083  // a second time. (This is Schmidt's double-checking pattern.)
1084  std::lock_guard<std::mutex> lock(mutex);
1085  if (task_has_finished)
1086  return;
1087  else

1090  // result. (We could have made the set_from() function
1091  // that we call here wait for the future to be ready --
