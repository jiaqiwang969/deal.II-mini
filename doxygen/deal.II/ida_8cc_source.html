226  template <typename VectorType>
227  unsigned int
228  IDA<VectorType>::solve_dae(VectorType &solution, VectorType &solution_dot)
229  {
230  double t = data.initial_time;
231  double h = data.initial_step_size;
232  unsigned int step_number = 0;
233 
234  int status;
235  (void)status;
236 
237  reset(data.initial_time, data.initial_step_size, solution, solution_dot);
238 
239  // The solution is stored in
240  // solution. Here we take only a
241  // view of it.
242 
243  double next_time = data.initial_time;
244 
245  output_step(0, solution, solution_dot, 0);
246 
247  while (t < data.final_time)
248  {
249  next_time += data.output_period;
250 
251  auto yy = internal::make_nvector_view(solution);
252  auto yp = internal::make_nvector_view(solution_dot);
253 
254  status = IDASolve(ida_mem, next_time, &t, yy, yp, IDA_NORMAL);
255  AssertIDA(status);
256 
257  status = IDAGetLastStep(ida_mem, &h);
258  AssertIDA(status);
259 
260  while (solver_should_restart(t, solution, solution_dot))
261  reset(t, h, solution, solution_dot);
262 
263  step_number++;
264 
265  output_step(t, solution, solution_dot, step_number);
266  }
267 
268  return step_number;
269  }

273  template <typename VectorType>
274  void
275  IDA<VectorType>::reset(const double current_time,
276  const double current_time_step,
277  VectorType & solution,
278  VectorType & solution_dot)
279  {
280  bool first_step = (current_time == data.initial_time);
281 
282  if (ida_mem)
283  IDAFree(&ida_mem);
284 
285  ida_mem = IDACreate();
286 
287  int status;
288  (void)status;
289 
290  auto yy = internal::make_nvector_view(solution);
291  auto yp = internal::make_nvector_view(solution_dot);
292 
293  status =
294  IDAInit(ida_mem, residual_callback<VectorType>, current_time, yy, yp);
295  AssertIDA(status);
296  if (get_local_tolerances)
297  {
298  const auto abs_tols =
299  internal::make_nvector_view(get_local_tolerances());
300  status = IDASVtolerances(ida_mem, data.relative_tolerance, abs_tols);
301  AssertIDA(status);
302  }
303  else
304  {
305  status = IDASStolerances(ida_mem,
306  data.relative_tolerance,
307  data.absolute_tolerance);
308  AssertIDA(status);
309  }
310 
311  status = IDASetInitStep(ida_mem, current_time_step);
312  AssertIDA(status);
313 
314  status = IDASetUserData(ida_mem, this);
315  AssertIDA(status);
316 
317  if (data.ic_type == AdditionalData::use_y_diff ||
318  data.reset_type == AdditionalData::use_y_diff ||
319  data.ignore_algebraic_terms_for_errors)
320  {
321  VectorType diff_comp_vector(solution);
322  diff_comp_vector = 0.0;
323  auto dc = differential_components();
324  for (auto i = dc.begin(); i != dc.end(); ++i)
325  diff_comp_vector[*i] = 1.0;
326  diff_comp_vector.compress(VectorOperation::insert);
327 
328  const auto diff_id = internal::make_nvector_view(diff_comp_vector);
329  status = IDASetId(ida_mem, diff_id);
330  AssertIDA(status);
331  }
332 
333  status = IDASetSuppressAlg(ida_mem, data.ignore_algebraic_terms_for_errors);
334  AssertIDA(status);
335 
336  // status = IDASetMaxNumSteps(ida_mem, max_steps);
337  status = IDASetStopTime(ida_mem, data.final_time);
338  AssertIDA(status);
339 
340  status = IDASetMaxNonlinIters(ida_mem, data.maximum_non_linear_iterations);
341  AssertIDA(status);
342 
343  // Initialize solver
344 # if DEAL_II_SUNDIALS_VERSION_LT(4, 0, 0)
345  auto IDA_mem = static_cast<IDAMem>(ida_mem);
346 
347  IDA_mem->ida_lsetup = t_dae_lsetup<VectorType>;
348 
349  if (solve_jacobian_system)
350  IDA_mem->ida_lsolve = solve_with_jacobian_callback<VectorType>;
351  else
352  AssertThrow(false, ExcFunctionNotProvided("solve_jacobian_system"));
353 # if DEAL_II_SUNDIALS_VERSION_LT(3, 0, 0)
354  IDA_mem->ida_setupNonNull = true;
355 # endif
356 # else
357  SUNMatrix J = nullptr;
358  SUNLinearSolver LS = nullptr;
359 
360  // and attach it to the SUNLinSol object. The functions that will get
361  // called do not actually receive the IDAMEM object, just the LS
362  // object, so we have to store a pointer to the current
363  // object in the LS object
364  LS = SUNLinSolNewEmpty();
365  LS->content = this;
366 
367  LS->ops->gettype = [](SUNLinearSolver /*ignored*/) -> SUNLinearSolver_Type {
368  return SUNLINEARSOLVER_MATRIX_ITERATIVE;
369  };
370 
371  LS->ops->free = [](SUNLinearSolver LS) -> int {
372  if (LS->content)
373  {
374  LS->content = nullptr;
375  }
376  if (LS->ops)
377  {
378  free(LS->ops);
379  LS->ops = nullptr;
380  }
381  free(LS);
382  LS = nullptr;
383  return 0;
384  };
385 
386  AssertThrow(solve_jacobian_system || solve_with_jacobian,
387  ExcFunctionNotProvided(
388  "solve_jacobian_system or solve_with_jacobian"));
389  LS->ops->solve = solve_with_jacobian_callback<VectorType>;
390 
391  // When we set an iterative solver IDA requires that resid is provided. From
392  // SUNDIALS docs If an iterative method computes the preconditioned initial
393  // residual and returns with a successful solve without performing any
394  // iterations (i.e., either the initial guess or the preconditioner is
395  // sufficiently accurate), then this optional routine may be called by the
396  // SUNDIALS package. This routine should return the N_Vector containing the
397  // preconditioned initial residual vector.
398  LS->ops->resid = [](SUNLinearSolver /*ignored*/) -> N_Vector {
399  return nullptr;
400  };
401  // When we set an iterative solver IDA requires that last number of
402  // iteration is provided. Since we can't know what kind of solver the user
403  // has provided we set 1. This is clearly suboptimal.
404  LS->ops->numiters = [](SUNLinearSolver /*ignored*/) -> int { return 1; };
405  // Even though we don't use it, IDA still wants us to set some
406  // kind of matrix object for the nonlinear solver. This is because
407  // if we don't set it, it won't call the functions that set up
408  // the matrix object (i.e., the argument to the 'IDASetJacFn'
409  // function below).
410  J = SUNMatNewEmpty();
411  J->content = this;
412 
413  J->ops->getid = [](SUNMatrix /*ignored*/) -> SUNMatrix_ID {
414  return SUNMATRIX_CUSTOM;
415  };
416 
417  J->ops->destroy = [](SUNMatrix A) {
418  if (A->content)
419  {
420  A->content = nullptr;
421  }
422  if (A->ops)
423  {
424  free(A->ops);
425  A->ops = nullptr;
426  }
427  free(A);
428  A = nullptr;
429  };
430 
431  // Now set the linear system and Jacobian objects in the solver:
432  status = IDASetLinearSolver(ida_mem, LS, J);
433  AssertIDA(status);
434 
435  status = IDASetLSNormFactor(ida_mem, data.ls_norm_factor);
436  AssertIDA(status);
437  // Finally tell IDA about
438  // it as well. The manual says that this must happen *after*
439  // calling IDASetLinearSolver
440  status = IDASetJacFn(ida_mem, &setup_jacobian_callback<VectorType>);
441  AssertIDA(status);
442 # endif
443  status = IDASetMaxOrd(ida_mem, data.maximum_order);
444  AssertIDA(status);
445 
446  typename AdditionalData::InitialConditionCorrection type;
447  if (first_step)
448  type = data.ic_type;
449  else
450  type = data.reset_type;
451 
452  status =
453  IDASetMaxNumItersIC(ida_mem, data.maximum_non_linear_iterations_ic);
454  AssertIDA(status);
455 
456  if (type == AdditionalData::use_y_dot)
457  {
458  // (re)initialization of the vectors
459  status =
460  IDACalcIC(ida_mem, IDA_Y_INIT, current_time + current_time_step);
461  AssertIDA(status);
462 
463  status = IDAGetConsistentIC(ida_mem, yy, yp);
464  AssertIDA(status);
465  }
466  else if (type == AdditionalData::use_y_diff)
467  {
468  status =
469  IDACalcIC(ida_mem, IDA_YA_YDP_INIT, current_time + current_time_step);
470  AssertIDA(status);
471 
472  status = IDAGetConsistentIC(ida_mem, yy, yp);
473  AssertIDA(status);
474  }
475  }
