114  // Here is the system for how we store constraint types in a binary mask.
115  // This is not a complete contradiction-free system, i.e., there are
116  // invalid states that we just assume that we never get.
117 
118  // If the mask is zero, there are no constraints. Then, there are three
119  // different fields with one bit per dimension. The first field determines
120  // the type, or the position of an element along each direction. The
121  // second field determines if there is a constrained face with that
122  // direction as normal. The last field determines if there is a
123  // constrained edge of a given pair of coordinate planes, but where
124  // neither of the corresponding faces are constrained (only valid in 3D).
125 
126  // The element is placed in the 'first position' along *-axis. These also
127  // determine which face is constrained. For example, in 2D, if
128  // constr_face_x and constr_type are set, then x = 0 is constrained.
129  constexpr unsigned int constr_type_x = 1 << 0;
130  constexpr unsigned int constr_type_y = 1 << 1;
131  constexpr unsigned int constr_type_z = 1 << 2;
132 
133  // Element has as a constraint at * = 0 or * = fe_degree face
134  constexpr unsigned int constr_face_x = 1 << 3;
135  constexpr unsigned int constr_face_y = 1 << 4;
136  constexpr unsigned int constr_face_z = 1 << 5;
137 
138  // Element has as a constraint at * = 0 or * = fe_degree edge
139  constexpr unsigned int constr_edge_xy = 1 << 6;
140  constexpr unsigned int constr_edge_yz = 1 << 7;
141  constexpr unsigned int constr_edge_zx = 1 << 8;

143  template <int dim>
144  void
145  setup_constraint_weigths(unsigned int fe_degree)
146  {
147  const unsigned int face_no =
148  0; // we assume that all faces have the same number of dofs
149 
150  FE_Q<2> fe_q(fe_degree);
151  FullMatrix<double> interpolation_matrix(fe_q.n_dofs_per_face(face_no),
152  fe_q.n_dofs_per_face(face_no));
153  fe_q.get_subface_interpolation_matrix(fe_q,
154  0,
155  interpolation_matrix,
156  face_no);
157 
158  std::vector<unsigned int> mapping =
159  FETools::lexicographic_to_hierarchic_numbering<1>(fe_degree);
160 
161  FullMatrix<double> mapped_matrix(fe_q.n_dofs_per_face(face_no),
162  fe_q.n_dofs_per_face(face_no));
163  for (unsigned int i = 0; i < fe_q.n_dofs_per_face(face_no); ++i)
164  for (unsigned int j = 0; j < fe_q.n_dofs_per_face(face_no); ++j)
165  mapped_matrix(i, j) = interpolation_matrix(mapping[i], mapping[j]);
166 
167  cudaError_t error_code =
168  cudaMemcpyToSymbol(internal::constraint_weights,
169  &mapped_matrix[0][0],
170  sizeof(double) * fe_q.n_dofs_per_face(face_no) *
171  fe_q.n_dofs_per_face(face_no));
172  AssertCuda(error_code);
173  }

204  template <>
205  inline void
206  HangingNodes<3>::setup_line_to_cell()
207  {
208  // In 3D, we can have DoFs on only an edge being constrained (e.g. in a
209  // cartesian 2x2x2 grid, where only the upper left 2 cells are refined).
210  // This sets up a helper data structure in the form of a mapping from
211  // edges (i.e. lines) to neighboring cells.
212 
213  // Mapping from an edge to which children that share that edge.
214  const unsigned int line_to_children[12][2] = {{0, 2},
215  {1, 3},
216  {0, 1},
217  {2, 3},
218  {4, 6},
219  {5, 7},
220  {4, 5},
221  {6, 7},
222  {0, 4},
223  {1, 5},
224  {2, 6},
225  {3, 7}};
226 
227  std::vector<std::vector<std::pair<cell_iterator, unsigned int>>>
228  line_to_inactive_cells(n_raw_lines);
229 
230  // First add active and inactive cells to their lines:
231  for (const auto &cell : dof_handler.cell_iterators())
232  {
233  for (unsigned int line = 0; line < GeometryInfo<3>::lines_per_cell;
234  ++line)
235  {
236  const unsigned int line_idx = cell->line(line)->index();
237  if (cell->is_active())
238  line_to_cells[line_idx].push_back(std::make_pair(cell, line));
239  else
240  line_to_inactive_cells[line_idx].push_back(
241  std::make_pair(cell, line));
242  }
243  }
244 
245  // Now, we can access edge-neighboring active cells on same level to also
246  // access of an edge to the edges "children". These are found from looking
247  // at the corresponding edge of children of inactive edge neighbors.
248  for (unsigned int line_idx = 0; line_idx < n_raw_lines; ++line_idx)
249  {
250  if ((line_to_cells[line_idx].size() > 0) &&
251  line_to_inactive_cells[line_idx].size() > 0)
252  {
253  // We now have cells to add (active ones) and edges to which they
254  // should be added (inactive cells).
255  const cell_iterator &inactive_cell =
256  line_to_inactive_cells[line_idx][0].first;
257  const unsigned int neighbor_line =
258  line_to_inactive_cells[line_idx][0].second;
259 
260  for (unsigned int c = 0; c < 2; ++c)
261  {
262  const cell_iterator &child =
263  inactive_cell->child(line_to_children[neighbor_line][c]);
264  const unsigned int child_line_idx =
265  child->line(neighbor_line)->index();
266 
267  // Now add all active cells
268  for (const auto cl : line_to_cells[line_idx])
269  line_to_cells[child_line_idx].push_back(cl);
270  }
271  }
272  }
273  }

277  template <int dim>
278  template <typename CellIterator>
279  inline void
280  HangingNodes<dim>::setup_constraints(
281  std::vector<types::global_dof_index> & dof_indices,
282  const CellIterator & cell,
283  const std::shared_ptr<const Utilities::MPI::Partitioner> &partitioner,
284  unsigned int & mask) const
285  {
286  mask = 0;
287  const unsigned int n_dofs_1d = fe_degree + 1;
288  const unsigned int dofs_per_face =
289  Utilities::fixed_power<dim - 1>(n_dofs_1d);
290 
291  std::vector<types::global_dof_index> neighbor_dofs(dofs_per_face);
292 
293  const auto lex_face_mapping =
294  FETools::lexicographic_to_hierarchic_numbering<dim - 1>(fe_degree);
295 
296  for (const unsigned int face : GeometryInfo<dim>::face_indices())
297  {
298  if ((!cell->at_boundary(face)) &&
299  (cell->neighbor(face)->has_children() == false))
300  {
301  const active_cell_iterator &neighbor = cell->neighbor(face);
302 
303  // Neighbor is coarser than us, i.e., face is constrained
304  if (neighbor->level() < cell->level())
305  {
306  const unsigned int neighbor_face =
307  cell->neighbor_face_no(face);
308 
309  // Find position of face on neighbor
310  unsigned int subface = 0;
311  for (; subface < GeometryInfo<dim>::max_children_per_face;
312  ++subface)
313  if (neighbor->neighbor_child_on_subface(neighbor_face,
314  subface) == cell)
315  break;
316 
317  // Get indices to read
318  neighbor->face(neighbor_face)->get_dof_indices(neighbor_dofs);
319  // If the vector is distributed, we need to transform the
320  // global indices to local ones.
321  if (partitioner)
322  for (auto &index : neighbor_dofs)
323  index = partitioner->global_to_local(index);
324 
325  if (dim == 2)
326  {
327  if (face < 2)
328  {
329  mask |= internal::constr_face_x;
330  if (face == 0)
331  mask |= internal::constr_type_x;
332  if (subface == 0)
333  mask |= internal::constr_type_y;
334  }
335  else
336  {
337  mask |= internal::constr_face_y;
338  if (face == 2)
339  mask |= internal::constr_type_y;
340  if (subface == 0)
341  mask |= internal::constr_type_x;
342  }
343 
344  // Reorder neighbor_dofs and copy into faceth face of
345  // dof_indices
346 
347  // Offset if upper/right face
348  unsigned int offset = (face % 2 == 1) ? fe_degree : 0;
349 
350  for (unsigned int i = 0; i < n_dofs_1d; ++i)
351  {
352  unsigned int idx = 0;
353  // If X-line, i.e., if y = 0 or y = fe_degree
354  if (face > 1)
355  idx = n_dofs_1d * offset + i;
356  // If Y-line, i.e., if x = 0 or x = fe_degree
357  else
358  idx = n_dofs_1d * i + offset;
359 
360  dof_indices[idx] = neighbor_dofs[lex_face_mapping[i]];
361  }
362  }
363  else if (dim == 3)
364  {
365  const bool transpose = !(cell->face_orientation(face));
366 
367  int rotate = 0;
368 
369  if (cell->face_rotation(face))
370  rotate -= 1;
371  if (cell->face_flip(face))
372  rotate -= 2;
373 
374  rotate_face(rotate, n_dofs_1d, neighbor_dofs);
375  rotate_subface_index(rotate, subface);
376 
377  if (transpose)
378  {
379  transpose_face(neighbor_dofs);
380  transpose_subface_index(subface);
381  }
382 
383  // YZ-plane
384  if (face < 2)
385  {
386  mask |= internal::constr_face_x;
387  if (face == 0)
388  mask |= internal::constr_type_x;
389  if (subface % 2 == 0)
390  mask |= internal::constr_type_y;
391  if (subface / 2 == 0)
392  mask |= internal::constr_type_z;
393  }
394  // XZ-plane
395  else if (face < 4)
396  {
397  mask |= internal::constr_face_y;
398  if (face == 2)
399  mask |= internal::constr_type_y;
400  if (subface % 2 == 0)
401  mask |= internal::constr_type_z;
402  if (subface / 2 == 0)
403  mask |= internal::constr_type_x;
404  }
405  // XY-plane
406  else
407  {
408  mask |= internal::constr_face_z;
409  if (face == 4)
410  mask |= internal::constr_type_z;
411  if (subface % 2 == 0)
412  mask |= internal::constr_type_x;
413  if (subface / 2 == 0)
414  mask |= internal::constr_type_y;
415  }
416 
417  // Offset if upper/right/back face
418  unsigned int offset = (face % 2 == 1) ? fe_degree : 0;
419 
420  for (unsigned int i = 0; i < n_dofs_1d; ++i)
421  {
422  for (unsigned int j = 0; j < n_dofs_1d; ++j)
423  {
424  unsigned int idx = 0;
425  // If YZ-plane, i.e., if x = 0 or x = fe_degree,
426  // and orientation standard
427  if (face < 2)
428  idx = n_dofs_1d * n_dofs_1d * i +
429  n_dofs_1d * j + offset;
430  // If XZ-plane, i.e., if y = 0 or y = fe_degree,
431  // and orientation standard
432  else if (face < 4)
433  idx = n_dofs_1d * n_dofs_1d * j +
434  n_dofs_1d * offset + i;
435  // If XY-plane, i.e., if z = 0 or z = fe_degree,
436  // and orientation standard
437  else
438  idx = n_dofs_1d * n_dofs_1d * offset +
439  n_dofs_1d * i + j;
440 
441  dof_indices[idx] =
442  neighbor_dofs[lex_face_mapping[n_dofs_1d * i +
443  j]];
444  }
445  }
446  }
447  else
448  ExcNotImplemented();
449  }
450  }
451  }
452 
453  // In 3D we can have a situation where only DoFs on an edge are
454  // constrained. Append these here.
455  if (dim == 3)
456  {
457  // For each line on cell, which faces does it belong to, what is the
458  // edge mask, what is the types of the faces it belong to, and what is
459  // the type along the edge.
460  const unsigned int line_to_edge[12][4] = {
461  {internal::constr_face_x | internal::constr_face_z,
462  internal::constr_edge_zx,
463  internal::constr_type_x | internal::constr_type_z,
464  internal::constr_type_y},
465  {internal::constr_face_x | internal::constr_face_z,
466  internal::constr_edge_zx,
467  internal::constr_type_z,
468  internal::constr_type_y},
469  {internal::constr_face_y | internal::constr_face_z,
470  internal::constr_edge_yz,
471  internal::constr_type_y | internal::constr_type_z,
472  internal::constr_type_x},
473  {internal::constr_face_y | internal::constr_face_z,
474  internal::constr_edge_yz,
475  internal::constr_type_z,
476  internal::constr_type_x},
477  {internal::constr_face_x | internal::constr_face_z,
478  internal::constr_edge_zx,
479  internal::constr_type_x,
480  internal::constr_type_y},
481  {internal::constr_face_x | internal::constr_face_z,
482  internal::constr_edge_zx,
483  0,
484  internal::constr_type_y},
485  {internal::constr_face_y | internal::constr_face_z,
486  internal::constr_edge_yz,
487  internal::constr_type_y,
488  internal::constr_type_x},
489  {internal::constr_face_y | internal::constr_face_z,
490  internal::constr_edge_yz,
491  0,
492  internal::constr_type_x},
493  {internal::constr_face_x | internal::constr_face_y,
494  internal::constr_edge_xy,
495  internal::constr_type_x | internal::constr_type_y,
496  internal::constr_type_z},
497  {internal::constr_face_x | internal::constr_face_y,
498  internal::constr_edge_xy,
499  internal::constr_type_y,
500  internal::constr_type_z},
501  {internal::constr_face_x | internal::constr_face_y,
502  internal::constr_edge_xy,
503  internal::constr_type_x,
504  internal::constr_type_z},
505  {internal::constr_face_x | internal::constr_face_y,
506  internal::constr_edge_xy,
507  0,
508  internal::constr_type_z}};
509 
510  for (unsigned int local_line = 0;
511  local_line < GeometryInfo<dim>::lines_per_cell;
512  ++local_line)
513  {
514  // If we don't already have a constraint for as part of a face
515  if (!(mask & line_to_edge[local_line][0]))
516  {
517  // For each cell which share that edge
518  const unsigned int line = cell->line(local_line)->index();
519  for (const auto edge_neighbor : line_to_cells[line])
520  {
521  // If one of them is coarser than us
522  const cell_iterator neighbor_cell = edge_neighbor.first;
523  if (neighbor_cell->level() < cell->level())
524  {
525  const unsigned int local_line_neighbor =
526  edge_neighbor.second;
527  mask |= line_to_edge[local_line][1] |
528  line_to_edge[local_line][2];
529 
530  bool flipped = false;
531  if (cell->line(local_line)->vertex_index(0) ==
532  neighbor_cell->line(local_line_neighbor)
533  ->vertex_index(0))
534  {
535  // Assuming line directions match axes directions,
536  // we have an unflipped edge of first type
537  mask |= line_to_edge[local_line][3];
538  }
539  else if (cell->line(local_line)->vertex_index(1) ==
540  neighbor_cell->line(local_line_neighbor)
541  ->vertex_index(1))
542  {
543  // We have an unflipped edge of second type
544  }
545  else if (cell->line(local_line)->vertex_index(1) ==
546  neighbor_cell->line(local_line_neighbor)
547  ->vertex_index(0))
548  {
549  // We have a flipped edge of second type
550  flipped = true;
551  }
552  else if (cell->line(local_line)->vertex_index(0) ==
553  neighbor_cell->line(local_line_neighbor)
554  ->vertex_index(1))
555  {
556  // We have a flipped edge of first type
557  mask |= line_to_edge[local_line][3];
558  flipped = true;
559  }
560  else
561  ExcInternalError();
562 
563  // Copy the unconstrained values
564  neighbor_dofs.resize(n_dofs_1d * n_dofs_1d *
565  n_dofs_1d);
566  neighbor_cell->get_dof_indices(neighbor_dofs);
567  // If the vector is distributed, we need to transform
568  // the global indices to local ones.
569  if (partitioner)
570  for (auto &index : neighbor_dofs)
571  index = partitioner->global_to_local(index);
572 
573  for (unsigned int i = 0; i < n_dofs_1d; ++i)
574  {
575  // Get local dof index along line
576  const unsigned int idx =
577  line_dof_idx(local_line, i, n_dofs_1d);
578  dof_indices[idx] = neighbor_dofs
579  [lexicographic_mapping[line_dof_idx(
580  local_line_neighbor,
581  flipped ? fe_degree - i : i,
582  n_dofs_1d)]];
583  }
584 
585  // Stop looping over edge neighbors
586  break;
587  }
588  }
589  }
590  }
591  }

611  template <int dim>
612  inline void
613  HangingNodes<dim>::rotate_face(
614  int times,
615  unsigned int n_dofs_1d,
616  std::vector<types::global_dof_index> &dofs) const
617  {
618  const unsigned int rot_mapping[4] = {2, 0, 3, 1};
619 
620  times = times % 4;
621  times = times < 0 ? times + 4 : times;
622 
623  std::vector<types::global_dof_index> copy(dofs.size());
624  for (int t = 0; t < times; ++t)
625  {
626  std::swap(copy, dofs);
627 
628  // Vertices
629  for (unsigned int i = 0; i < 4; ++i)
630  dofs[rot_mapping[i]] = copy[i];
631 
632  // Edges
633  const unsigned int n_int = n_dofs_1d - 2;
634  unsigned int offset = 4;
635  for (unsigned int i = 0; i < n_int; ++i)
636  {
637  // Left edge
638  dofs[offset + i] = copy[offset + 2 * n_int + (n_int - 1 - i)];
639  // Right edge
640  dofs[offset + n_int + i] =
641  copy[offset + 3 * n_int + (n_int - 1 - i)];
642  // Bottom edge
643  dofs[offset + 2 * n_int + i] = copy[offset + n_int + i];
644  // Top edge
645  dofs[offset + 3 * n_int + i] = copy[offset + i];
646  }
647 
648  // Interior points
649  offset += 4 * n_int;
650 
651  for (unsigned int i = 0; i < n_int; ++i)
652  for (unsigned int j = 0; j < n_int; ++j)
653  dofs[offset + i * n_int + j] =
654  copy[offset + j * n_int + (n_int - 1 - i)];
655  }
656  }

758  template <unsigned int fe_degree,
759  unsigned int direction,
760  bool transpose,
761  typename Number>
762  __device__ inline void
763  interpolate_boundary_2d(const unsigned int constraint_mask,
764  Number * values)
765  {
766  const unsigned int x_idx = threadIdx.x % (fe_degree + 1);
767  const unsigned int y_idx = threadIdx.y;
768 
769  const unsigned int this_type =
770  (direction == 0) ? internal::constr_type_x : internal::constr_type_y;
771 
772  const unsigned int interp_idx = (direction == 0) ? x_idx : y_idx;
773 
774  Number t = 0;
775  // Flag is true if dof is constrained for the given direction and the
776  // given face.
777  const bool constrained_face =
778  (constraint_mask &
779  (((direction == 0) ? internal::constr_face_y : 0) |
780  ((direction == 1) ? internal::constr_face_x : 0)));
781 
782  // Flag is true if for the given direction, the dof is constrained with
783  // the right type and is on the correct side (left (= 0) or right (=
784  // fe_degree))
785  const bool constrained_dof =
786  ((direction == 0) && ((constraint_mask & internal::constr_type_y) ?
787  (y_idx == 0) :
788  (y_idx == fe_degree))) ||
789  ((direction == 1) && ((constraint_mask & internal::constr_type_x) ?
790  (x_idx == 0) :
791  (x_idx == fe_degree)));
792 
793  if (constrained_face && constrained_dof)
794  {
795  const bool type = constraint_mask & this_type;
796 
797  if (type)
798  {
799  for (unsigned int i = 0; i <= fe_degree; ++i)
800  {
801  const unsigned int real_idx =
802  (direction == 0) ? index2<fe_degree + 1>(i, y_idx) :
803  index2<fe_degree + 1>(x_idx, i);
804 
805  const Number w =
806  transpose ?
807  internal::constraint_weights[i * (fe_degree + 1) +
808  interp_idx] :
809  internal::constraint_weights[interp_idx *
810  (fe_degree + 1) +
811  i];
812  t += w * values[real_idx];
813  }
814  }
815  else
816  {
817  for (unsigned int i = 0; i <= fe_degree; ++i)
818  {
819  const unsigned int real_idx =
820  (direction == 0) ? index2<fe_degree + 1>(i, y_idx) :
821  index2<fe_degree + 1>(x_idx, i);
822 
823  const Number w =
824  transpose ?
825  internal::constraint_weights[(fe_degree - i) *
826  (fe_degree + 1) +
827  fe_degree - interp_idx] :
828  internal::constraint_weights[(fe_degree - interp_idx) *
829  (fe_degree + 1) +
830  fe_degree - i];
831  t += w * values[real_idx];
832  }
833  }
834  }
835 
836  // The synchronization is done for all the threads in one block with
837  // each block being assigned to one element.
838  __syncthreads();
839  if (constrained_face && constrained_dof)
840  values[index2<fe_degree + 1>(x_idx, y_idx)] = t;
841 
842  __syncthreads();
843  }

848  unsigned int direction,
849  bool transpose,
850  typename Number>
851  __device__ inline void
852  interpolate_boundary_3d(const unsigned int constraint_mask,
853  Number * values)
854  {
855  const unsigned int x_idx = threadIdx.x % (fe_degree + 1);
856  const unsigned int y_idx = threadIdx.y;
857  const unsigned int z_idx = threadIdx.z;
858 
859  const unsigned int this_type =
860  (direction == 0) ? internal::constr_type_x :
861  (direction == 1) ? internal::constr_type_y :
862  internal::constr_type_z;
863  const unsigned int face1_type =
864  (direction == 0) ? internal::constr_type_y :
865  (direction == 1) ? internal::constr_type_z :
866  internal::constr_type_x;
867  const unsigned int face2_type =
868  (direction == 0) ? internal::constr_type_z :
869  (direction == 1) ? internal::constr_type_x :
870  internal::constr_type_y;
871 
872  // If computing in x-direction, need to match against constr_face_y or
873  // constr_face_z
874  const unsigned int face1 = (direction == 0) ? internal::constr_face_y :
875  (direction == 1) ?
876  internal::constr_face_z :
877  internal::constr_face_x;
878  const unsigned int face2 = (direction == 0) ? internal::constr_face_z :
879  (direction == 1) ?
880  internal::constr_face_x :
881  internal::constr_face_y;
882  const unsigned int edge = (direction == 0) ? internal::constr_edge_yz :
883  (direction == 1) ?
884  internal::constr_edge_zx :
885  internal::constr_edge_xy;
886  const unsigned int constrained_face =
887  constraint_mask & (face1 | face2 | edge);
888 
889  const unsigned int interp_idx =
890  (direction == 0) ? x_idx : (direction == 1) ? y_idx : z_idx;
891  const unsigned int face1_idx =
892  (direction == 0) ? y_idx : (direction == 1) ? z_idx : x_idx;
893  const unsigned int face2_idx =
894  (direction == 0) ? z_idx : (direction == 1) ? x_idx : y_idx;
895 
896  Number t = 0;
897  const bool on_face1 = (constraint_mask & face1_type) ?
898  (face1_idx == 0) :
899  (face1_idx == fe_degree);
900  const bool on_face2 = (constraint_mask & face2_type) ?
901  (face2_idx == 0) :
902  (face2_idx == fe_degree);
903  const bool constrained_dof =
904  (((constraint_mask & face1) && on_face1) ||
905  ((constraint_mask & face2) && on_face2) ||
906  ((constraint_mask & edge) && on_face1 && on_face2));
907 
908  if (constrained_face && constrained_dof)
909  {
910  const bool type = constraint_mask & this_type;
911  if (type)
912  {
913  for (unsigned int i = 0; i <= fe_degree; ++i)
914  {
915  const unsigned int real_idx =
916  (direction == 0) ?
917  index3<fe_degree + 1>(i, y_idx, z_idx) :
918  (direction == 1) ?
919  index3<fe_degree + 1>(x_idx, i, z_idx) :
920  index3<fe_degree + 1>(x_idx, y_idx, i);
921 
922  const Number w =
923  transpose ?
924  internal::constraint_weights[i * (fe_degree + 1) +
925  interp_idx] :
926  internal::constraint_weights[interp_idx *
927  (fe_degree + 1) +
928  i];
929  t += w * values[real_idx];
930  }
931  }
932  else
933  {
934  for (unsigned int i = 0; i <= fe_degree; ++i)
935  {
936  const unsigned int real_idx =
937  (direction == 0) ?
938  index3<fe_degree + 1>(i, y_idx, z_idx) :
939  (direction == 1) ?
940  index3<fe_degree + 1>(x_idx, i, z_idx) :
941  index3<fe_degree + 1>(x_idx, y_idx, i);
942 
943  const Number w =
944  transpose ?
945  internal::constraint_weights[(fe_degree - i) *
946  (fe_degree + 1) +
947  fe_degree - interp_idx] :
948  internal::constraint_weights[(fe_degree - interp_idx) *
949  (fe_degree + 1) +
950  fe_degree - i];
951  t += w * values[real_idx];
952  }
953  }
954  }
955 
956  // The synchronization is done for all the threads in one block with
957  // each block being assigned to one element.
958  __syncthreads();
959 
960  if (constrained_face && constrained_dof)
961  values[index3<fe_degree + 1>(x_idx, y_idx, z_idx)] = t;
962 
963  __syncthreads();
964  }
