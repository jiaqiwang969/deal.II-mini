972 template <int dim, int patch_dim, int spacedim, int patch_spacedim>
973 template <int dim2, int spacedim2>
974 void
975 DataOut_DoFData<dim, patch_dim, spacedim, patch_spacedim>::merge_patches(
976  const DataOut_DoFData<dim2, patch_dim, spacedim2, patch_spacedim> &source,
977  const Point<patch_spacedim> & shift)
978 {
979  const std::vector<Patch> &source_patches = source.get_patches();
980  Assert((patches.size() != 0) && (source_patches.size() != 0),
981  ExcMessage("When calling this function, both the current "
982  "object and the one being merged need to have a "
983  "nonzero number of patches associated with it. "
984  "Either you called this function on objects that "
985  "are empty, or you may have forgotten to call "
986  "the 'build_patches()' function."));
987  // Check equality of component names
988  Assert(get_dataset_names() == source.get_dataset_names(),
989  Exceptions::DataOutImplementation::ExcIncompatibleDatasetNames());
990 
991  // Make sure patches are compatible. Ideally, we would check that all input
992  // patches from both collections are all compatible, but we'll be content
993  // with checking that just the first ones from both sources are.
994  //
995  // We check compatibility by testing that both sets of patches result
996  // from the same number of subdivisions, and that they have the same
997  // number of source vectors (they really should, since we already checked
998  // that there are the same number of source components above, but you
999  // never know). This implies that the data should have the same number of
1000  // columns. They should really have the same number of rows as well,
1001  // but depending on whether a patch has points included or not, the
1002  // number of rows may or may not include coordinates for the points,
1003  // and the comparison has to account for that because in each source
1004  // stream, the patches may include some that have points included.
1005  Assert(patches[0].n_subdivisions == source_patches[0].n_subdivisions,
1006  Exceptions::DataOutImplementation::ExcIncompatiblePatchLists());
1007  Assert(patches[0].data.n_cols() == source_patches[0].data.n_cols(),
1008  Exceptions::DataOutImplementation::ExcIncompatiblePatchLists());
1009  Assert((patches[0].data.n_rows() +
1010  (patches[0].points_are_available ? 0 : patch_spacedim)) ==
1011  (source_patches[0].data.n_rows() +
1012  (source_patches[0].points_are_available ? 0 : patch_spacedim)),
1013  Exceptions::DataOutImplementation::ExcIncompatiblePatchLists());
1014 
1015  // check equality of the vector data
1016  // specifications
1017  Assert(get_nonscalar_data_ranges().size() ==
1018  source.get_nonscalar_data_ranges().size(),
1019  ExcMessage("Both sources need to declare the same components "
1020  "as vectors."));
1021  for (unsigned int i = 0; i < get_nonscalar_data_ranges().size(); ++i)
1022  {
1023  Assert(std::get<0>(get_nonscalar_data_ranges()[i]) ==
1024  std::get<0>(source.get_nonscalar_data_ranges()[i]),
1025  ExcMessage("Both sources need to declare the same components "
1026  "as vectors."));
1027  Assert(std::get<1>(get_nonscalar_data_ranges()[i]) ==
1028  std::get<1>(source.get_nonscalar_data_ranges()[i]),
1029  ExcMessage("Both sources need to declare the same components "
1030  "as vectors."));
1031  Assert(std::get<2>(get_nonscalar_data_ranges()[i]) ==
1032  std::get<2>(source.get_nonscalar_data_ranges()[i]),
1033  ExcMessage("Both sources need to declare the same components "
1034  "as vectors."));
1035  }
1036 
1037  // merge patches. store old number
1038  // of elements, since we need to
1039  // adjust patch numbers, etc
1040  // afterwards
1041  const unsigned int old_n_patches = patches.size();
1042  patches.insert(patches.end(), source_patches.begin(), source_patches.end());
1043 
1044  // perform shift, if so desired
1045  if (shift != Point<patch_spacedim>())
