1600  PreconditionBase::transpose()
1601  {
1602  // This only flips a flag that tells
1603  // Trilinos that any vmult operation
1604  // should be done with the
1605  // transpose. However, the matrix
1606  // structure is not reset.

1613  }
1614  else
1615  {
1616  ierr = preconditioner->SetUseTranspose(false);
1617  AssertThrow(ierr == 0, ExcTrilinosError(ierr));
1618  }
1619  }

1655  // For the implementation of the <code>vmult</code> function with deal.II
1656  // data structures we note that invoking a call of the Trilinos
1657  // preconditioner requires us to use Epetra vectors as well. We do this by
1658  // providing a view, i.e., feed Trilinos with a pointer to the data, so we
1659  // avoid copying the content of the vectors during the iteration (this
1660  // function is only useful when used in serial anyway). In the declaration
1661  // of the right hand side, we need to cast the source vector (that is
1662  // <code>const</code> in all deal.II calls) to non-constant value, as this
1663  // is the way Trilinos wants to have them.
1664  inline void
1665  PreconditionBase::vmult(::Vector<double> & dst,
1666  const ::Vector<double> &src) const
1667  {
1668  AssertDimension(dst.size(),
1669  preconditioner->OperatorDomainMap().NumMyElements());
1670  AssertDimension(src.size(),
1671  preconditioner->OperatorRangeMap().NumMyElements());
1672  Epetra_Vector tril_dst(View,
1673  preconditioner->OperatorDomainMap(),
1674  dst.begin());
1675  Epetra_Vector tril_src(View,
1676  preconditioner->OperatorRangeMap(),
1677  const_cast<double *>(src.begin()));
1678 
1679  const int ierr = preconditioner->ApplyInverse(tril_src, tril_dst);
1680  AssertThrow(ierr == 0, ExcTrilinosError(ierr));
1681  }
