237 template <typename VectorType>
238 void
239 SolverBFGS<VectorType>::solve(
240  const std::function<typename VectorType::value_type(const VectorType &x,
241  VectorType &f)> &compute,
242  VectorType & x)
243 {
244  // Also see scipy Fortran implementation
245  // https://github.com/scipy/scipy/blob/master/scipy/optimize/lbfgsb_src/lbfgsb.f
246  // and Octave-optim implementation:
247  // https://sourceforge.net/p/octave/optim/ci/default/tree/src/__bfgsmin.cc
248  LogStream::Prefix prefix("BFGS");
249 
250  // default line search:
251  bool first_step = true;
252  Number f_prev = 0.;
253  // provide default line search if no signal was attached
254  VectorType x0;
255  if (line_search_signal.empty())
256  {
257  x0.reinit(x);
258  const auto default_line_min =
259  [&](Number &f, VectorType &x, VectorType &g, const VectorType &p) {
260  const Number f0 = f;
261  const Number g0 = g * p;
262  Assert(g0 < 0,
263  ExcMessage(
264  "Function does not decrease along the current direction"));
265 
266  // save current solution value (to be used in line_search):
267  x0 = x;
268 
269  // see scipy implementation
270  // https://docs.scipy.org/doc/scipy/reference/generated/scipy.optimize.line_search.html#scipy.optimize.line_search
271  // and Eq. 2.6.8 in Fletcher 2013, Practical methods of optimization
272  Number df = f_prev - f;
273  Assert(first_step || df >= 0.,
274  ExcMessage("Function value is not decreasing"));
275  df = std::max(df, 100. * std::numeric_limits<Number>::epsilon());
276  // guess a reasonable first step:
277  const Number a1 =
278  (first_step ? 1. : std::min(1., -1.01 * 2. * df / g0));
279  Assert(a1 > 0., ExcInternalError());
280  f_prev = f;
281 
282  // 1D line-search function
283  const auto line_func =
284  [&](const Number &x_line) -> std::pair<Number, Number> {
285  x = x0;
286  x.add(x_line, p);
287  f = compute(x, g);
288  const Number g_line = g * p;
289  return std::make_pair(f, g_line);
290  };
291 
292  // loose line search:
293  const auto res = LineMinimization::line_search<Number>(
294  line_func,
295  f0,
296  g0,
297  LineMinimization::poly_fit<Number>,
298  a1,
299  0.9,
300  0.001);
301 
302  if (first_step)
303  first_step = false;
304 
305  return res.first;
306  };
307  this->connect_line_search_slot(default_line_min);
308  }
309 
310  // FIXME: Octave has convergence in terms of:
311  // function change tolerance, default 1e-12
312  // parameter change tolerance, default 1e-6
313  // gradient tolerance, default 1e-5
314  // SolverBase and/or SolverControl need extension
315 
316  VectorType g(x), p(x), y_k(x), s_k(x);
317 
318  std::vector<Number> c1;
319  c1.reserve(additional_data.max_history_size);
320 
321  // limited history
322  FiniteSizeHistory<VectorType> y(additional_data.max_history_size);
323  FiniteSizeHistory<VectorType> s(additional_data.max_history_size);
324  FiniteSizeHistory<Number> rho(additional_data.max_history_size);
325 
326  unsigned int m = 0;
327  Number f;
328 
329  SolverControl::State conv = SolverControl::iterate;
330  unsigned int k = 0;
331 
332  f = compute(x, g);
333 
334  conv = this->iteration_status(k, g.l2_norm(), x);
335  if (conv != SolverControl::iterate)
336  return;
337 
338  while (conv == SolverControl::iterate)
339  {
340  if (additional_data.debug_output)
341  deallog << "Iteration " << k << " history " << m << std::endl
342  << "f=" << f << std::endl;
343 
344  // 1. Two loop recursion to calculate p = - H*g
345  c1.resize(m);
346  p = g;
347  // first loop:
348  for (unsigned int i = 0; i < m; ++i)
349  {
350  c1[i] = rho[i] * (s[i] * p);
351  p.add(-c1[i], y[i]);
352  }
353  // H0
354  if (!preconditioner_signal.empty())
355  preconditioner_signal(p, s, y);
356 
357  // second loop:
358  for (int i = m - 1; i >= 0; --i)
359  {
360  Assert(i >= 0, ExcInternalError());
361  const Number c2 = rho[i] * (y[i] * p);
362  p.add(c1[i] - c2, s[i]);
363  }
364  p *= -1.;
365 
366  // 2. Line search
367  s_k = x;
368  y_k = g;
369  const Number alpha = line_search_signal(f, x, g, p)
370  .get(); // <-- signals return boost::optional
371  s_k.sadd(-1, 1, x);
372  y_k.sadd(-1, 1, g);
373 
374  if (additional_data.debug_output)
375  deallog << "Line search a=" << alpha << " f=" << f << std::endl;
376 
377  // 3. Check convergence
378  k++;
379  const Number g_l2 = g.l2_norm();
380  conv = this->iteration_status(k, g_l2, x);
381  if (conv != SolverControl::iterate)
382  break;
383 
384  // 4. Store s, y, rho
385  const Number curvature = s_k * y_k;
386  if (additional_data.debug_output)
387  deallog << "Curvature " << curvature << std::endl;
388 
389  if (curvature > 0. && additional_data.max_history_size > 0)
390  {
391  s.add(s_k);
392  y.add(y_k);
393  rho.add(1. / curvature);
394  m = s.size();
395 
396  Assert(y.size() == m, ExcInternalError());
397  Assert(rho.size() == m, ExcInternalError());
398  }
399 
400  Assert(m <= additional_data.max_history_size, ExcInternalError());
401  }
402 
403  // In the case of failure: throw exception.
404  AssertThrow(conv == SolverControl::success,
405  SolverControl::NoConvergence(k, g.l2_norm()));
406 }
