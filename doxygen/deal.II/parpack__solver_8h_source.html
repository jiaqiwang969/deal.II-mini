707 template <typename VectorType>
708 void
709 PArpackSolver<VectorType>::internal_reinit(const IndexSet &locally_owned_dofs)
710 {
711  // store local indices to write to vectors
712  locally_owned_dofs.fill_index_vector(local_indices);
713 
714  // scalars
715  nloc = locally_owned_dofs.n_elements();
716  ncv = additional_data.number_of_arnoldi_vectors;
717 
718  AssertDimension(local_indices.size(), nloc);
719 
720  // vectors
721  ldv = nloc;
722  v.resize(ldv * ncv, 0.0);
723 
724  resid.resize(nloc, 1.0);
725 
726  // work arrays for ARPACK
727  workd.resize(3 * nloc, 0.0);
728 
729  lworkl =
730  additional_data.symmetric ? ncv * ncv + 8 * ncv : 3 * ncv * ncv + 6 * ncv;
731  workl.resize(lworkl, 0.);
732 
733  ldz = nloc;
734  z.resize(ldz * ncv, 0.); // TODO we actually need only ldz*nev
735 
736  // WORKEV Double precision work array of dimension 3*NCV.
737  lworkev = additional_data.symmetric ? 0 /*not used in symmetric case*/
738  :
739  3 * ncv;
740  workev.resize(lworkev, 0.);
741 
742  select.resize(ncv, 0);
743 }

808 template <typename VectorType>
809 template <typename MatrixType1, typename MatrixType2, typename INVERSE>
810 void
811 PArpackSolver<VectorType>::solve(const MatrixType1 &system_matrix,
812  const MatrixType2 &mass_matrix,
813  const INVERSE & inverse,
814  std::vector<std::complex<double>> &eigenvalues,
815  std::vector<VectorType *> &eigenvectors,
816  const unsigned int n_eigenvalues)
817 {
818  if (additional_data.symmetric)
819  {
820  Assert(n_eigenvalues <= eigenvectors.size(),
821  PArpackExcInvalidEigenvectorSize(n_eigenvalues,
822  eigenvectors.size()));
823  }
824  else
825  Assert(n_eigenvalues + 1 <= eigenvectors.size(),
826  PArpackExcInvalidEigenvectorSizeNonsymmetric(n_eigenvalues,
827  eigenvectors.size()));
828 
829  Assert(n_eigenvalues <= eigenvalues.size(),
830  PArpackExcInvalidEigenvalueSize(n_eigenvalues, eigenvalues.size()));
831 
832 
833  // use eigenvectors to get the problem size so that it is possible to
834  // employ LinearOperator for mass_matrix.
835  Assert(n_eigenvalues < eigenvectors[0]->size(),
836  PArpackExcInvalidNumberofEigenvalues(n_eigenvalues,
837  eigenvectors[0]->size()));
838 
839  Assert(additional_data.number_of_arnoldi_vectors < eigenvectors[0]->size(),
840  PArpackExcInvalidNumberofArnoldiVectors(
841  additional_data.number_of_arnoldi_vectors, eigenvectors[0]->size()));
842 
843  Assert(additional_data.number_of_arnoldi_vectors > 2 * n_eigenvalues + 1,
844  PArpackExcSmallNumberofArnoldiVectors(
845  additional_data.number_of_arnoldi_vectors, n_eigenvalues));
846 
847  int mode = additional_data.mode;
848 
849  // reverse communication parameter
850  // must be zero on the first call to pdnaupd
851  int ido = 0;
852 
853  // 'G' generalized eigenvalue problem
854  // 'I' standard eigenvalue problem
855  char bmat[2];
856  bmat[0] = (mode == 1) ? 'I' : 'G';
857  bmat[1] = '\0';
858 
859  // Specify the eigenvalues of interest, possible parameters:
860  // "LA" algebraically largest
861  // "SA" algebraically smallest
862  // "LM" largest magnitude
863  // "SM" smallest magnitude
864  // "LR" largest real part
865  // "SR" smallest real part
866  // "LI" largest imaginary part
867  // "SI" smallest imaginary part
868  // "BE" both ends of spectrum simultaneous
869  char which[3];
870  switch (additional_data.eigenvalue_of_interest)
871  {
872  case algebraically_largest:
873  std::strcpy(which, "LA");
874  break;
875  case algebraically_smallest:
876  std::strcpy(which, "SA");
877  break;
878  case largest_magnitude:
879  std::strcpy(which, "LM");
880  break;
881  case smallest_magnitude:
882  std::strcpy(which, "SM");
883  break;
884  case largest_real_part:
885  std::strcpy(which, "LR");
886  break;
887  case smallest_real_part:
888  std::strcpy(which, "SR");
889  break;
890  case largest_imaginary_part:
891  std::strcpy(which, "LI");
892  break;
893  case smallest_imaginary_part:
894  std::strcpy(which, "SI");
895  break;
896  case both_ends:
897  std::strcpy(which, "BE");
898  break;
899  }
900 
901  // tolerance for ARPACK
902  double tol = control().tolerance();
903 
904  // information to the routines
905  std::vector<int> iparam(11, 0);
906 
907  iparam[0] = 1;
908  // shift strategy: exact shifts with respect to the current Hessenberg matrix
909  // H.
910 
911  // maximum number of iterations
912  iparam[2] = control().max_steps();
913 
914  // Parpack currently works only for NB = 1
915  iparam[3] = 1;
916 
917  // Sets the mode of dsaupd:
918  // 1 is A*x=lambda*x, OP = A, B = I
919  // 2 is A*x = lambda*M*x, OP = inv[M]*A, B = M
920  // 3 is shift-invert mode, OP = inv[A-sigma*M]*M, B = M
921  // 4 is buckling mode,
922  // 5 is Cayley mode.
923 
924  iparam[6] = mode;
925  std::vector<int> ipntr(14, 0);
926 
927  // information out of the iteration
928  // If INFO .EQ. 0, a random initial residual vector is used.
929  // If INFO .NE. 0, RESID contains the initial residual vector,
930  // possibly from a previous run.
931  // Typical choices in this situation might be to use the final value
932  // of the starting vector from the previous eigenvalue calculation
933  int info = initial_vector_provided ? 1 : 0;
934 
935  // Number of eigenvalues of OP to be computed. 0 < NEV < N.
936  int nev = n_eigenvalues;
937  int n_inside_arpack = nloc;
938 
939  // IDO = 99: done
940  while (ido != 99)
941  {
942  // call of ARPACK pdnaupd routine
943  if (additional_data.symmetric)
944  pdsaupd_(&mpi_communicator_fortran,
945  &ido,
946  bmat,
947  &n_inside_arpack,
948  which,
949  &nev,
950  &tol,
951  resid.data(),
952  &ncv,
953  v.data(),
954  &ldv,
955  iparam.data(),
956  ipntr.data(),
957  workd.data(),
958  workl.data(),
959  &lworkl,
960  &info);
961  else
962  pdnaupd_(&mpi_communicator_fortran,
963  &ido,
964  bmat,
965  &n_inside_arpack,
966  which,
967  &nev,
968  &tol,
969  resid.data(),
970  &ncv,
971  v.data(),
972  &ldv,
973  iparam.data(),
974  ipntr.data(),
975  workd.data(),
976  workl.data(),
977  &lworkl,
978  &info);
979 
980  AssertThrow(info == 0, PArpackExcInfoPdnaupd(info));
981 
982  // if we converge, we shall not modify anything in work arrays!
983  if (ido == 99)
984  break;
985 
986  // IPNTR(1) is the pointer into WORKD for X,
987  // IPNTR(2) is the pointer into WORKD for Y.
988  const int shift_x = ipntr[0] - 1;
989  const int shift_y = ipntr[1] - 1;
990  Assert(shift_x >= 0, ::ExcInternalError());
991  Assert(shift_x + nloc <= static_cast<int>(workd.size()),
992  ::ExcInternalError());
993  Assert(shift_y >= 0, ::ExcInternalError());
994  Assert(shift_y + nloc <= static_cast<int>(workd.size()),
995  ::ExcInternalError());
996 
997  src = 0.;
998 
999  // switch based on both ido and mode
1000  if ((ido == -1) || (ido == 1 && mode < 3))
1001  // compute Y = OP * X
1002  {
1003  src.add(nloc, local_indices.data(), workd.data() + shift_x);
1004  src.compress(VectorOperation::add);
1005 
1006  if (mode == 3)
