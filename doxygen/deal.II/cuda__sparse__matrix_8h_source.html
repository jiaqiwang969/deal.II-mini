416  template <class StreamType>
417  inline void
418  SparseMatrix<Number>::print(StreamType &out,
419  const bool across,
420  const bool diagonal_first) const
421  {
422  Assert(column_index_dev != nullptr, ExcNotInitialized());
423  Assert(val_dev != nullptr, ExcNotInitialized());
424  Assert(row_ptr_dev != nullptr, ExcNotInitialized());
425 
426  std::vector<int> rows(n_rows + 1);
427  std::vector<int> cols(nnz);
428  std::vector<double> val(nnz);
429  Utilities::CUDA::copy_to_host(row_ptr_dev.get(), rows);
430  Utilities::CUDA::copy_to_host(column_index_dev.get(), cols);
431  Utilities::CUDA::copy_to_host(val_dev.get(), val);
432 
433  bool has_diagonal = false;
434  Number diagonal = Number();
435 
436  for (size_type i = 0; i < n_rows; ++i)
437  {
438  if (diagonal_first)
439  {
440  // find the diagonal and print if it exists
441  for (size_type j = rows[i]; j < rows[i + 1] && cols[j] <= i; ++j)
442  {
443  if (i == cols[j])
444  {
445  diagonal = val[j];
446  has_diagonal = true;
447  if (across)
448  out << ' ' << i << ',' << i << ':' << diagonal;
449  else
450  out << '(' << i << ',' << i << ") " << diagonal
451  << std::endl;
452  break;
453  }
454  }
455  }
456  for (size_type j = rows[i]; j < rows[i + 1]; ++j)
457  {
458  if (has_diagonal && i == cols[j])
459  continue;
460  if (across)
461  out << ' ' << i << ',' << cols[j] << ':' << val[j];
462  else
463  out << "(" << i << "," << cols[j] << ") " << val[j] << std::endl;
464  }
465  }
466  if (across)
467  out << std::endl;
468  }

489  Utilities::CUDA::copy_to_host(column_index_dev.get(), cols);
490  Utilities::CUDA::copy_to_host(val_dev.get(), val);
491 
492  unsigned int width = width_;
493 
494  std::ios::fmtflags old_flags = out.flags();
495  unsigned int old_precision = out.precision(precision);
496 
497  if (scientific)
498  {
499  out.setf(std::ios::scientific, std::ios::floatfield);
500  if (!width)
501  width = precision + 7;
502  }
503  else
504  {
505  out.setf(std::ios::fixed, std::ios::floatfield);
506  if (!width)
507  width = precision + 2;
508  }
509 
510  for (size_type i = 0; i < n_rows; ++i)
511  {
512  size_type j = rows[i];
513  for (size_type k = 0; k < n_cols; ++k)
514  {
515  if (k == cols[j])
516  {
517  out << std::setw(width) << val[j] * Number(denominator) << ' ';
518  ++j;
519  }
520  else
521  out << std::setw(width) << zero_string << ' ';
522  }
523  out << std::endl;
524  };
525  AssertThrow(out, ExcIO());
526 
527  // reset output format
528  out.precision(old_precision);
529  out.flags(old_flags);
