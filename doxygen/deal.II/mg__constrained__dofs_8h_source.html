209 template <int dim, int spacedim>
210 inline void
211 MGConstrainedDoFs::initialize(
212  const DoFHandler<dim, spacedim> &dof,
213  const MGLevelObject<IndexSet> & level_relevant_dofs)
214 {
215  boundary_indices.clear();
216  refinement_edge_indices.clear();
217  level_constraints.clear();
218  user_constraints.clear();
219 
220  const unsigned int nlevels = dof.get_triangulation().n_global_levels();
221  const unsigned int min_level = level_relevant_dofs.min_level();
222  const unsigned int max_level = (level_relevant_dofs.max_level() == 0) ?
223  nlevels - 1 :
224  level_relevant_dofs.max_level();
225  const bool user_level_dofs =
226  (level_relevant_dofs.max_level() == 0) ? false : true;
227 
228  // At this point level_constraint and refinement_edge_indices are empty.
229  refinement_edge_indices.resize(nlevels);
230  level_constraints.resize(nlevels);
231  user_constraints.resize(nlevels);
232  for (unsigned int l = min_level; l <= max_level; ++l)
233  {
234  if (user_level_dofs)
235  {
236  level_constraints[l].reinit(level_relevant_dofs[l]);
237  }
238  else
239  {
240  IndexSet relevant_dofs;
241  DoFTools::extract_locally_relevant_level_dofs(dof, l, relevant_dofs);
242  level_constraints[l].reinit(relevant_dofs);
243  }
244 
245  // Loop through relevant cells and faces finding those which are periodic
246  // neighbors.
247  typename DoFHandler<dim, spacedim>::cell_iterator cell = dof.begin(l),
248  endc = dof.end(l);
249  for (; cell != endc; ++cell)
250  if (cell->level_subdomain_id() != numbers::artificial_subdomain_id)
251  {
252  for (auto f : cell->face_indices())
253  if (cell->has_periodic_neighbor(f) &&
254  cell->periodic_neighbor(f)->level() == cell->level())
255  {
256  if (cell->is_locally_owned_on_level())
257  {
258  Assert(
259  cell->periodic_neighbor(f)->level_subdomain_id() !=
260  numbers::artificial_subdomain_id,
261  ExcMessage(
262  "Periodic neighbor of a locally owned cell must either be owned or ghost."));
263  }
264  // Cell is a level-ghost and its neighbor is a
265  // level-artificial cell nothing to do here
266  else if (cell->periodic_neighbor(f)->level_subdomain_id() ==
267  numbers::artificial_subdomain_id)
268  {
269  Assert(cell->is_locally_owned_on_level() == false,
270  ExcInternalError());
271  continue;
272  }
273 
274  const unsigned int dofs_per_face =
275  dof.get_fe(0).n_dofs_per_face(f);
276  std::vector<types::global_dof_index> dofs_1(dofs_per_face);
277  std::vector<types::global_dof_index> dofs_2(dofs_per_face);
278 
279  cell->periodic_neighbor(f)
280  ->face(cell->periodic_neighbor_face_no(f))
281  ->get_mg_dof_indices(l, dofs_1, 0);
282  cell->face(f)->get_mg_dof_indices(l, dofs_2, 0);
283  // Store periodicity information in the level
284  // AffineConstraints object. Skip DoFs for which we've
285  // previously entered periodicity constraints already; this
286  // can happen, for example, for a vertex dof at a periodic
287  // boundary that we visit from more than one cell
288  for (unsigned int i = 0; i < dofs_per_face; ++i)
289  if (level_constraints[l].can_store_line(dofs_2[i]) &&
290  level_constraints[l].can_store_line(dofs_1[i]) &&
291  !level_constraints[l].is_constrained(dofs_2[i]) &&
292  !level_constraints[l].is_constrained(dofs_1[i]))
293  {
294  level_constraints[l].add_line(dofs_2[i]);
295  level_constraints[l].add_entry(dofs_2[i],
296  dofs_1[i],
297  1.);
298  }
299  }
300  }
301  level_constraints[l].close();
302 
303  // Initialize with empty IndexSet of correct size
304  refinement_edge_indices[l] = IndexSet(dof.n_dofs(l));
305  }
306 
307  MGTools::extract_inner_interface_dofs(dof, refinement_edge_indices);
308 }

333 template <int dim, int spacedim>
334 inline void
335 MGConstrainedDoFs::make_no_normal_flux_constraints(
336  const DoFHandler<dim, spacedim> &dof,
337  const types::boundary_id bid,
338  const unsigned int first_vector_component)
339 {
340  // For a given boundary id, find which vector component is on the boundary
341  // and set a zero boundary constraint for those degrees of freedom.
342  const unsigned int n_components = dof.get_fe_collection().n_components();
343  AssertIndexRange(first_vector_component + dim - 1, n_components);
344 
345  ComponentMask comp_mask(n_components, false);
346 
347 
348  typename Triangulation<dim>::face_iterator
349  face = dof.get_triangulation().begin_face(),
350  endf = dof.get_triangulation().end_face();
351  for (; face != endf; ++face)
352  if (face->at_boundary() && face->boundary_id() == bid)
353  for (unsigned int d = 0; d < dim; ++d)
354  {
355  Tensor<1, dim, double> unit_vec;
356  unit_vec[d] = 1.0;
357 
358  const Tensor<1, dim> normal_vec =
359  face->get_manifold().normal_vector(face, face->center());
360 
361  if (std::abs(std::abs(unit_vec * normal_vec) - 1.0) < 1e-10)
362  comp_mask.set(d + first_vector_component, true);
363  else
364  Assert(
365  std::abs(unit_vec * normal_vec) < 1e-10,
366  ExcMessage(
367  "We can currently only support no normal flux conditions "
368  "for a specific boundary id if all faces are normal to the "
369  "x, y, or z axis."));
370  }
371 
372  Assert(comp_mask.n_selected_components() == 1,
373  ExcMessage(
374  "We can currently only support no normal flux conditions "
375  "for a specific boundary id if all faces are facing in the "
376  "same direction, i.e., a boundary normal to the x-axis must "
377  "have a different boundary id than a boundary normal to the "
378  "y- or z-axis and so on. If the mesh here was produced using "
379  "GridGenerator::..., setting colorize=true during mesh generation "
380  "and calling make_no_normal_flux_constraints() for each no normal "
381  "flux boundary will fulfill the condition."));
382 
383  this->make_zero_boundary_constraints(dof, {bid}, comp_mask);
384 }

387 inline void
388 MGConstrainedDoFs::add_user_constraints(
389  const unsigned int level,
390  const AffineConstraints<double> &constraints_on_level)
391 {
392  AssertIndexRange(level, user_constraints.size());
393 
394  // Get the relevant DoFs from level_constraints if
395  // the user constraint matrix has not been initialized
396  if (user_constraints[level].get_local_lines().size() == 0)
397  user_constraints[level].reinit(level_constraints[level].get_local_lines());
398 
399  user_constraints[level].merge(
400  constraints_on_level,
401  AffineConstraints<double>::MergeConflictBehavior::right_object_wins);
402  user_constraints[level].close();
403 }

435 inline bool
436 MGConstrainedDoFs::is_interface_matrix_entry(
437  const unsigned int level,
438  const types::global_dof_index i,
439  const types::global_dof_index j) const
440 {
441  const IndexSet &interface_dofs_on_level =

445  && !interface_dofs_on_level.is_element(j) // !at_refinement_edge(j)
446  && !this->is_boundary_index(level, i) // !on_boundary(i)
447  && !this->is_boundary_index(level, j); // !on_boundary(j)
