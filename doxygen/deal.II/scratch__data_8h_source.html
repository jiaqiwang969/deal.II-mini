888  void
889  ScratchData<dim, spacedim>::extract_local_dof_values(
890  const std::string &global_vector_name,
891  const VectorType & input_vector,
892  const Number dummy)
893  {
894  const unsigned int n_dofs = local_dof_indices.size();
895 
896  const std::string name =
897  get_unique_dofs_name(global_vector_name, n_dofs, dummy);
898 
899  auto &independent_local_dofs =
900  internal_data_storage
901  .template get_or_add_object_with_name<std::vector<Number>>(name,
902  n_dofs);
903 
904  AssertDimension(independent_local_dofs.size(), n_dofs);
905 
906  if (Differentiation::AD::is_tapeless_ad_number<Number>::value == true)
907  for (unsigned int i = 0; i < n_dofs; ++i)
908  Differentiation::AD::internal::Marking<Number>::independent_variable(
909  input_vector(local_dof_indices[i]),
910  i,
911  n_dofs,
912  independent_local_dofs[i]);
913  else
914  for (unsigned int i = 0; i < n_dofs; ++i)
915  independent_local_dofs[i] = input_vector(local_dof_indices[i]);
916  }
