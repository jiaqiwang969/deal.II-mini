584 template <class VectorType>
585 inline void
586 SolverGMRES<VectorType>::givens_rotation(Vector<double> &h,
587  Vector<double> &b,
588  Vector<double> &ci,
589  Vector<double> &si,
590  int col) const
591 {
592  for (int i = 0; i < col; i++)
593  {
594  const double s = si(i);
595  const double c = ci(i);
596  const double dummy = h(i);
597  h(i) = c * dummy + s * h(i + 1);
598  h(i + 1) = -s * dummy + c * h(i + 1);
599  };
600 
601  const double r = 1. / std::sqrt(h(col) * h(col) + h(col + 1) * h(col + 1));
602  si(col) = h(col + 1) * r;
603  ci(col) = h(col) * r;
604  h(col) = ci(col) * h(col) + si(col) * h(col + 1);
605  b(col + 1) = -si(col) * b(col);
606  b(col) *= ci(col);
607 }

611 template <class VectorType>
612 inline double
613 SolverGMRES<VectorType>::modified_gram_schmidt(
614  const internal::SolverGMRESImplementation::TmpVectors<VectorType>
615  & orthogonal_vectors,
616  const unsigned int dim,
617  const unsigned int accumulated_iterations,
618  VectorType & vv,
619  Vector<double> & h,
620  bool & reorthogonalize,
621  const boost::signals2::signal<void(int)> &reorthogonalize_signal)
622 {
623  Assert(dim > 0, ExcInternalError());
624  const unsigned int inner_iteration = dim - 1;
625 
626  // need initial norm for detection of re-orthogonalization, see below
627  double norm_vv_start = 0;
628  const bool consider_reorthogonalize =
629  (reorthogonalize == false) && (inner_iteration % 5 == 4);
630  if (consider_reorthogonalize)
631  norm_vv_start = vv.l2_norm();
632 
633  // Orthogonalization
634  h(0) = vv * orthogonal_vectors[0];
635  for (unsigned int i = 1; i < dim; ++i)
636  h(i) = vv.add_and_dot(-h(i - 1),
637  orthogonal_vectors[i - 1],
638  orthogonal_vectors[i]);
639  double norm_vv =
640  std::sqrt(vv.add_and_dot(-h(dim - 1), orthogonal_vectors[dim - 1], vv));
641 
642  // Re-orthogonalization if loss of orthogonality detected. For the test, use
643  // a strategy discussed in C. T. Kelley, Iterative Methods for Linear and
644  // Nonlinear Equations, SIAM, Philadelphia, 1995: Compare the norm of vv
645  // after orthogonalization with its norm when starting the
646  // orthogonalization. If vv became very small (here: less than the square
647  // root of the machine precision times 10), it is almost in the span of the
648  // previous vectors, which indicates loss of precision.
649  if (consider_reorthogonalize)
650  {
651  if (norm_vv >
652  10. * norm_vv_start *
653  std::sqrt(
654  std::numeric_limits<typename VectorType::value_type>::epsilon()))
655  return norm_vv;
656 
657  else
658  {
659  reorthogonalize = true;
660  if (!reorthogonalize_signal.empty())
661  reorthogonalize_signal(accumulated_iterations);
662  }
663  }
664 
665  if (reorthogonalize == true)
666  {
667  double htmp = vv * orthogonal_vectors[0];
668  h(0) += htmp;
669  for (unsigned int i = 1; i < dim; ++i)
670  {
671  htmp = vv.add_and_dot(-htmp,
672  orthogonal_vectors[i - 1],
673  orthogonal_vectors[i]);
674  h(i) += htmp;
675  }
676  norm_vv =
677  std::sqrt(vv.add_and_dot(-htmp, orthogonal_vectors[dim - 1], vv));
678  }
679 
680  return norm_vv;
681 }

685 template <class VectorType>
686 inline void
687 SolverGMRES<VectorType>::compute_eigs_and_cond(
688  const FullMatrix<double> &H_orig,
689  const unsigned int dim,
690  const boost::signals2::signal<void(const std::vector<std::complex<double>> &)>
691  &eigenvalues_signal,
692  const boost::signals2::signal<void(const FullMatrix<double> &)>
693  & hessenberg_signal,
694  const boost::signals2::signal<void(double)> &cond_signal)
695 {
696  // Avoid copying the Hessenberg matrix if it isn't needed.
697  if ((!eigenvalues_signal.empty() || !hessenberg_signal.empty() ||
698  !cond_signal.empty()) &&
699  dim > 0)
700  {
701  LAPACKFullMatrix<double> mat(dim, dim);
702  for (unsigned int i = 0; i < dim; ++i)
703  for (unsigned int j = 0; j < dim; ++j)
704  mat(i, j) = H_orig(i, j);
705  hessenberg_signal(H_orig);
706  // Avoid computing eigenvalues if they are not needed.
707  if (!eigenvalues_signal.empty())
708  {
709  // Copy mat so that we can compute svd below. Necessary since
710  // compute_eigenvalues will leave mat in state
711  // LAPACKSupport::unusable.
712  LAPACKFullMatrix<double> mat_eig(mat);
713  mat_eig.compute_eigenvalues();
714  std::vector<std::complex<double>> eigenvalues(dim);
715  for (unsigned int i = 0; i < mat_eig.n(); ++i)
716  eigenvalues[i] = mat_eig.eigenvalue(i);
717  // Sort eigenvalues for nicer output.
718  std::sort(eigenvalues.begin(),
719  eigenvalues.end(),
720  internal::SolverGMRESImplementation::complex_less_pred);
721  eigenvalues_signal(eigenvalues);
722  }
723  // Calculate condition number, avoid calculating the svd if a slot
724  // isn't connected. Need at least a 2-by-2 matrix to do the estimate.
725  if (!cond_signal.empty() && (mat.n() > 1))
726  {
727  mat.compute_svd();
728  double condition_number =
729  mat.singular_value(0) / mat.singular_value(mat.n() - 1);
730  cond_signal(condition_number);
731  }
732  }

737 template <class VectorType>
738 template <typename MatrixType, typename PreconditionerType>
739 void
740 SolverGMRES<VectorType>::solve(const MatrixType & A,
741  VectorType & x,
742  const VectorType & b,
743  const PreconditionerType &preconditioner)
744 {
745  // TODO:[?] Check, why there are two different start residuals.
746  // TODO:[GK] Make sure the parameter in the constructor means maximum basis
747  // size
748 
749  LogStream::Prefix prefix("GMRES");
750 
751  // extra call to std::max to placate static analyzers: coverity rightfully
752  // complains that data.max_n_tmp_vectors - 2 may overflow
753  const unsigned int n_tmp_vectors =
754  std::max(additional_data.max_n_tmp_vectors, 3u);
755 
756  // Generate an object where basis vectors are stored.
757  internal::SolverGMRESImplementation::TmpVectors<VectorType> tmp_vectors(
758  n_tmp_vectors, this->memory);
759 
760  // number of the present iteration; this
761  // number is not reset to zero upon a
762  // restart
763  unsigned int accumulated_iterations = 0;
764 
765  const bool do_eigenvalues =
766  !condition_number_signal.empty() || !all_condition_numbers_signal.empty() ||
767  !eigenvalues_signal.empty() || !all_eigenvalues_signal.empty() ||
768  !hessenberg_signal.empty() || !all_hessenberg_signal.empty();
769  // for eigenvalue computation, need to collect the Hessenberg matrix (before
770  // applying Givens rotations)
771  FullMatrix<double> H_orig;
772  if (do_eigenvalues)
773  H_orig.reinit(n_tmp_vectors, n_tmp_vectors - 1);
774 
775  // matrix used for the orthogonalization process later
776  H.reinit(n_tmp_vectors, n_tmp_vectors - 1);
777 
778  // some additional vectors, also used in the orthogonalization
779  ::Vector<double> gamma(n_tmp_vectors), ci(n_tmp_vectors - 1),
780  si(n_tmp_vectors - 1), h(n_tmp_vectors - 1);
781 
782 
783  unsigned int dim = 0;
784 
785  SolverControl::State iteration_state = SolverControl::iterate;
786  double last_res = -std::numeric_limits<double>::max();
787 
788  // switch to determine whether we want a left or a right preconditioner. at
789  // present, left is default, but both ways are implemented
790  const bool left_precondition = !additional_data.right_preconditioning;
791 
792  // Per default the left preconditioned GMRes uses the preconditioned
793  // residual and the right preconditioned GMRes uses the unpreconditioned
794  // residual as stopping criterion.
795  const bool use_default_residual = additional_data.use_default_residual;
796 
797  // define two aliases
798  VectorType &v = tmp_vectors(0, x);
799  VectorType &p = tmp_vectors(n_tmp_vectors - 1, x);
800 
801  // Following vectors are needed when we are not using the default residuals
802  // as stopping criterion
803  typename VectorMemory<VectorType>::Pointer r;
804  typename VectorMemory<VectorType>::Pointer x_;
805  std::unique_ptr<::Vector<double>> gamma_;
806  if (!use_default_residual)
807  {
808  r = std::move(typename VectorMemory<VectorType>::Pointer(this->memory));
809  x_ = std::move(typename VectorMemory<VectorType>::Pointer(this->memory));
810  r->reinit(x);
811  x_->reinit(x);
812 
813  gamma_ = std::make_unique<::Vector<double>>(gamma.size());
814  }
815 
816  bool re_orthogonalize = additional_data.force_re_orthogonalization;
817 
819  // outer iteration: loop until we either reach convergence or the maximum
820  // number of iterations is exceeded. each cycle of this loop amounts to one
821  // restart
822  do
823  {
824  // reset this vector to the right size
825  h.reinit(n_tmp_vectors - 1);
826 
827  if (left_precondition)
828  {
829  A.vmult(p, x);
830  p.sadd(-1., 1., b);
831  preconditioner.vmult(v, p);
832  }
833  else
834  {
835  A.vmult(v, x);
836  v.sadd(-1., 1., b);
837  };
838 
839  double rho = v.l2_norm();
840 
841  // check the residual here as well since it may be that we got the exact
842  // (or an almost exact) solution vector at the outset. if we wouldn't
843  // check here, the next scaling operation would produce garbage
844  if (use_default_residual)
845  {
846  last_res = rho;
847  iteration_state =
848  this->iteration_status(accumulated_iterations, rho, x);
849 
850  if (iteration_state != SolverControl::iterate)
851  break;
852  }
853  else
854  {
855  deallog << "default_res=" << rho << std::endl;
856 
857  if (left_precondition)
858  {
859  A.vmult(*r, x);
860  r->sadd(-1., 1., b);
861  }
862  else
863  preconditioner.vmult(*r, v);
864 
865  double res = r->l2_norm();
866  last_res = res;
867  iteration_state =
868  this->iteration_status(accumulated_iterations, res, x);
869 
870  if (iteration_state != SolverControl::iterate)
871  break;
872  }
873 
874  gamma(0) = rho;
875 
876  v *= 1. / rho;
877 
878  // inner iteration doing at most as many steps as there are temporary
879  // vectors. the number of steps actually been done is propagated outside
880  // through the @p dim variable
881  for (unsigned int inner_iteration = 0;
882  ((inner_iteration < n_tmp_vectors - 2) &&
883  (iteration_state == SolverControl::iterate));
884  ++inner_iteration)
885  {
886  ++accumulated_iterations;
887  // yet another alias
888  VectorType &vv = tmp_vectors(inner_iteration + 1, x);
889 
890  if (left_precondition)
891  {
892  A.vmult(p, tmp_vectors[inner_iteration]);
893  preconditioner.vmult(vv, p);
894  }
895  else
896  {
897  preconditioner.vmult(p, tmp_vectors[inner_iteration]);
898  A.vmult(vv, p);
899  }
900 
901  dim = inner_iteration + 1;
902 
903  const double s = modified_gram_schmidt(tmp_vectors,
904  dim,
905  accumulated_iterations,
906  vv,
907  h,
908  re_orthogonalize,
909  re_orthogonalize_signal);
910  h(inner_iteration + 1) = s;
911 
912  // s=0 is a lucky breakdown, the solver will reach convergence,
913  // but we must not divide by zero here.
914  if (s != 0)
915  vv *= 1. / s;
916 
917  // for eigenvalues, get the resulting coefficients from the
918  // orthogonalization process
919  if (do_eigenvalues)
920  for (unsigned int i = 0; i < dim + 1; ++i)
921  H_orig(i, inner_iteration) = h(i);
922 
923  // Transformation into tridiagonal structure
924  givens_rotation(h, gamma, ci, si, inner_iteration);
925 
926  // append vector on matrix
927  for (unsigned int i = 0; i < dim; ++i)
928  H(i, inner_iteration) = h(i);
929 
930  // default residual
931  rho = std::fabs(gamma(dim));
932 
933  if (use_default_residual)
934  {
935  last_res = rho;
936  iteration_state =
937  this->iteration_status(accumulated_iterations, rho, x);
938  }
939  else
940  {
941  deallog << "default_res=" << rho << std::endl;
942 
943  ::Vector<double> h_(dim);
944  *x_ = x;
945  *gamma_ = gamma;
946  H1.reinit(dim + 1, dim);
947 
948  for (unsigned int i = 0; i < dim + 1; ++i)
949  for (unsigned int j = 0; j < dim; ++j)

986  }

999  dim,
1000  all_eigenvalues_signal,
1001  all_hessenberg_signal,
1002  condition_number_signal);
1003 
1004  H1.backward(h, gamma);
1005 
1006  if (left_precondition)
1007  for (unsigned int i = 0; i < dim; ++i)
1008  x.add(h(i), tmp_vectors[i]);
1009  else
1010  {
1011  p = 0.;
1012  for (unsigned int i = 0; i < dim; ++i)
1013  p.add(h(i), tmp_vectors[i]);
1014  preconditioner.vmult(v, p);
1015  x.add(1., v);
1016  };
1017  // end of outer iteration. restart if no convergence and the number of
1018  // iterations is not exceeded
1019  }
1020  while (iteration_state == SolverControl::iterate);
1021 
1022  compute_eigs_and_cond(H_orig,
1023  dim,
1024  eigenvalues_signal,
1025  hessenberg_signal,
1026  condition_number_signal);
1027 
1028  if (!krylov_space_signal.empty())
1029  krylov_space_signal(tmp_vectors);
1030 
1031  // in case of failure: throw exception
1032  AssertThrow(iteration_state == SolverControl::success,
1033  SolverControl::NoConvergence(accumulated_iterations, last_res));

1163  basis_size, this->memory);

1196  {
