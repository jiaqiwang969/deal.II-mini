82  virtual void
83  print_info(std::ostream &out) const override
84  {
85  out
86  << "Iterative method reported convergence failure in step " << last_step
87  << ". The residual in the last step was " << last_residual << ".\n\n"
88  << "This error message can indicate that you have simply not allowed "
89  << "a sufficiently large number of iterations for your iterative solver "
90  << "to converge. This often happens when you increase the size of your "
91  << "problem. In such cases, the last residual will likely still be very "
92  << "small, and you can make the error go away by increasing the allowed "
93  << "number of iterations when setting up the SolverControl object that "
94  << "determines the maximal number of iterations you allow."
95  << "\n\n"
96  << "The other situation where this error may occur is when your matrix "
97  << "is not invertible (e.g., your matrix has a null-space), or if you "
98  << "try to apply the wrong solver to a matrix (e.g., using CG for a "
99  << "matrix that is not symmetric or not positive definite). In these "
100  << "cases, the residual in the last iteration is likely going to be large."
101  << std::endl;
102  }
