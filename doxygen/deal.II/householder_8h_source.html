165 template <typename number>
166 template <typename number2>
167 void
168 Householder<number>::initialize(const FullMatrix<number2> &M)
169 {
170  const size_type m = M.n_rows(), n = M.n_cols();
171  storage.reinit(m, n);
172  storage.fill(M);
173  Assert(!storage.empty(), typename FullMatrix<number2>::ExcEmptyMatrix());
174  diagonal.resize(m);
175 
176  // m > n, src.n() = m
177  Assert(storage.n_cols() <= storage.n_rows(),
178  ExcDimensionMismatch(storage.n_cols(), storage.n_rows()));
179 
180  for (size_type j = 0; j < n; ++j)
181  {
182  number2 sigma = 0;
183  size_type i;
184  // sigma = ||v||^2
185  for (i = j; i < m; ++i)
186  sigma += storage(i, j) * storage(i, j);
187  // We are ready if the column is
188  // empty. Are we?
189  if (std::fabs(sigma) < 1.e-15)
190  return;
191 
192  number2 s = (storage(j, j) < 0) ? std::sqrt(sigma) : -std::sqrt(sigma);
193  //
194  number2 beta = std::sqrt(1. / (sigma - s * storage(j, j)));
195 
196  // Make column j the Householder
197  // vector, store first entry in
198  // diagonal
199  diagonal[j] = beta * (storage(j, j) - s);
200  storage(j, j) = s;
201 
202  for (i = j + 1; i < m; ++i)
203  storage(i, j) *= beta;
204 
205 
206  // For all subsequent columns do
207  // the Householder reflection
208  for (size_type k = j + 1; k < n; ++k)
209  {
210  number2 sum = diagonal[j] * storage(j, k);
211  for (i = j + 1; i < m; ++i)
212  sum += storage(i, j) * storage(i, k);
213 
214  storage(j, k) -= sum * this->diagonal[j];
215  for (i = j + 1; i < m; ++i)
216  storage(i, k) -= sum * storage(i, j);
217  }
218  }
219 }

232 template <typename number>
233 template <typename number2>
234 double
235 Householder<number>::least_squares(Vector<number2> & dst,
236  const Vector<number2> &src) const
237 {
238  Assert(!storage.empty(), typename FullMatrix<number2>::ExcEmptyMatrix());
239  AssertDimension(dst.size(), storage.n());
240  AssertDimension(src.size(), storage.m());
241 
242  const size_type m = storage.m(), n = storage.n();
243 
244  GrowingVectorMemory<Vector<number2>> mem;
245  typename VectorMemory<Vector<number2>>::Pointer aux(mem);
246  aux->reinit(src, true);
247  *aux = src;
248  // m > n, m = src.n, n = dst.n
249 
250  // Multiply Q_n ... Q_2 Q_1 src
251  // Where Q_i = I - v_i v_i^T
252  for (size_type j = 0; j < n; ++j)
253  {
254  // sum = v_i^T dst
255  number2 sum = diagonal[j] * (*aux)(j);
256  for (size_type i = j + 1; i < m; ++i)
257  sum += static_cast<number2>(storage(i, j)) * (*aux)(i);
258  // dst -= v * sum
259  (*aux)(j) -= sum * diagonal[j];
260  for (size_type i = j + 1; i < m; ++i)
261  (*aux)(i) -= sum * static_cast<number2>(storage(i, j));
262  }
263  // Compute norm of residual
264  number2 sum = 0.;
265  for (size_type i = n; i < m; ++i)
266  sum += (*aux)(i) * (*aux)(i);
267  AssertIsFinite(sum);
268 
269  // Compute solution
270  storage.backward(dst, *aux);
271 
272  return std::sqrt(sum);
273 }
