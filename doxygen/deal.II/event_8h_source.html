199  inline bool
200  Event::test(const Event &event) const
201  {
202  // First, test all_true in this
203  if (all_true)
204  return true;
205 
206  const unsigned int n = flags.size();
207  const unsigned int m = event.flags.size();
208  const unsigned int n_min = (n < m) ? n : m;
209 
210  // Now, if all_true set in the
211  // other, then all must be true
212  // in this
213  if (event.all_true)
214  {
215  // Non existing flags are
216  // always assumed false
217  if (m > n)
218  return false;
219 
220  // Test all flags separately
221  // and return false if one is
222  // not set
223  return std::find(flags.begin(), flags.end(), false) == flags.end();
224  }
225 
226  // Finally, compare each flag
227  // separately
228  for (unsigned int i = 0; i < n_min; ++i)
229  if (event.flags[i] && !flags[i])
230  return false;
231  for (unsigned int i = n_min; i < m; ++i)
232  if (event.flags[i])
233  return false;
234  return true;
235  }
