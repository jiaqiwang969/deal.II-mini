294 template <int dim>
295 template <int order>
296 Tensor<order, dim>
297 TensorProductPolynomialsBubbles<dim>::compute_derivative(
298  const unsigned int i,
299  const Point<dim> & p) const
300 {
301  const unsigned int q_degree = tensor_polys.polynomials.size() - 1;
302  const unsigned int max_q_indices = tensor_polys.n();
303  Assert(i < max_q_indices + /* n_bubbles= */ ((q_degree <= 1) ? 1 : dim),
304  ExcInternalError());
305 
306  // treat the regular basis functions
307  if (i < max_q_indices)
308  return tensor_polys.template compute_derivative<order>(i, p);
309 
310  const unsigned int comp = i - tensor_polys.n();
311 
312  Tensor<order, dim> derivative;
313  switch (order)
314  {
315  case 1:
316  {
317  Tensor<1, dim> &derivative_1 =
318  *reinterpret_cast<Tensor<1, dim> *>(&derivative);
319 
320  for (unsigned int d = 0; d < dim; ++d)
321  {
322  derivative_1[d] = 1.;
323  // compute grad(4*\prod_{i=1}^d (x_i(1-x_i)))(p)
324  for (unsigned j = 0; j < dim; ++j)
325  derivative_1[d] *=
326  (d == j ? 4 * (1 - 2 * p(j)) : 4 * p(j) * (1 - p(j)));
327  // and multiply with (2*x_i-1)^{r-1}
328  for (unsigned int i = 0; i < q_degree - 1; ++i)
329  derivative_1[d] *= 2 * p(comp) - 1;
330  }
331 
332  if (q_degree >= 2)
333  {
334  // add \prod_{i=1}^d 4*(x_i(1-x_i))(p)
335  double value = 1.;
336  for (unsigned int j = 0; j < dim; ++j)
337  value *= 4 * p(j) * (1 - p(j));
338  // and multiply with grad(2*x_i-1)^{r-1}
339  double tmp = value * 2 * (q_degree - 1);
340  for (unsigned int i = 0; i < q_degree - 2; ++i)
341  tmp *= 2 * p(comp) - 1;
342  derivative_1[comp] += tmp;
343  }
344 
345  return derivative;
346  }
347  case 2:
348  {
349  Tensor<2, dim> &derivative_2 =
350  *reinterpret_cast<Tensor<2, dim> *>(&derivative);
351 
352  double v[dim + 1][3];
353  {
354  for (unsigned int c = 0; c < dim; ++c)
355  {
356  v[c][0] = 4 * p(c) * (1 - p(c));
357  v[c][1] = 4 * (1 - 2 * p(c));
358  v[c][2] = -8;
359  }
360 
361  double tmp = 1.;
362  for (unsigned int i = 0; i < q_degree - 1; ++i)
363  tmp *= 2 * p(comp) - 1;
364  v[dim][0] = tmp;
365 
366  if (q_degree >= 2)
367  {
368  double tmp = 2 * (q_degree - 1);
369  for (unsigned int i = 0; i < q_degree - 2; ++i)
370  tmp *= 2 * p(comp) - 1;
371  v[dim][1] = tmp;
372  }
373  else
374  v[dim][1] = 0.;
375 
376  if (q_degree >= 3)
377  {
378  double tmp = 4 * (q_degree - 2) * (q_degree - 1);
379  for (unsigned int i = 0; i < q_degree - 3; ++i)
380  tmp *= 2 * p(comp) - 1;
381  v[dim][2] = tmp;
382  }
383  else
384  v[dim][2] = 0.;
385  }
386 
387  // calculate (\partial_j \partial_k \psi) * monomial
388  Tensor<2, dim> grad_grad_1;
389  for (unsigned int d1 = 0; d1 < dim; ++d1)
390  for (unsigned int d2 = 0; d2 < dim; ++d2)
391  {
392  grad_grad_1[d1][d2] = v[dim][0];
393  for (unsigned int x = 0; x < dim; ++x)
394  {
395  unsigned int derivative = 0;
396  if (d1 == x || d2 == x)
397  {
398  if (d1 == d2)
399  derivative = 2;
400  else
401  derivative = 1;
402  }
403  grad_grad_1[d1][d2] *= v[x][derivative];
404  }
405  }
406 
407  // calculate (\partial_j \psi) *(\partial_k monomial)
408  // and (\partial_k \psi) *(\partial_j monomial)
409  Tensor<2, dim> grad_grad_2;
410  Tensor<2, dim> grad_grad_3;
411  for (unsigned int d = 0; d < dim; ++d)
412  {
413  grad_grad_2[d][comp] = v[dim][1];
414  grad_grad_3[comp][d] = v[dim][1];
415  for (unsigned int x = 0; x < dim; ++x)
416  {
417  grad_grad_2[d][comp] *= v[x][d == x];
418  grad_grad_3[comp][d] *= v[x][d == x];
419  }
420  }
421 
422  // calculate \psi *(\partial j \partial_k monomial) and sum
423  double psi_value = 1.;
424  for (unsigned int x = 0; x < dim; ++x)
425  psi_value *= v[x][0];
426 
427  for (unsigned int d1 = 0; d1 < dim; ++d1)
428  for (unsigned int d2 = 0; d2 < dim; ++d2)
429  derivative_2[d1][d2] =
430  grad_grad_1[d1][d2] + grad_grad_2[d1][d2] + grad_grad_3[d1][d2];
431  derivative_2[comp][comp] += psi_value * v[dim][2];
432 
433  return derivative;
434  }
435  default:
436  {
437  Assert(false, ExcNotImplemented());
438  return derivative;
439  }
440  }
441 }
