660  template <class RelaxationType, typename VectorType>
661  inline void
662  SmootherRelaxation<RelaxationType, VectorType>::smooth(
663  const unsigned int level,
664  VectorType & u,
665  const VectorType & rhs) const
666  {
667  unsigned int maxlevel = this->max_level();
668  unsigned int steps2 = this->steps;
669 
670  if (this->variable)
671  steps2 *= (1 << (maxlevel - level));
672 
673  bool T = this->transpose;
674  if (this->symmetric && (steps2 % 2 == 0))
675  T = false;
676  if (this->debug > 0)
677  deallog << 'S' << level << ' ';
678 
679  for (unsigned int i = 0; i < steps2; ++i)
680  {
681  if (T)
682  (*this)[level].Tstep(u, rhs);
683  else
684  (*this)[level].step(u, rhs);
685  if (this->symmetric)
686  T = !T;
687  }
688  }
