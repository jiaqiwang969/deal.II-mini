212  virtual std::unique_ptr<
213  typename FiniteElement<dim, spacedim>::InternalDataBase>
214  get_data(
215  const UpdateFlags update_flags,
216  const Mapping<dim, spacedim> &mapping,
217  const Quadrature<dim> & quadrature,
218  ::internal::FEValuesImplementation::FiniteElementRelatedData<dim,
219  spacedim>
220  &output_data) const override
221  {
222  (void)mapping;
223  (void)output_data;
224  // generate a new data object and
225  // initialize some fields
226  std::unique_ptr<typename FiniteElement<dim, spacedim>::InternalDataBase>
227  data_ptr = std::make_unique<InternalData>();
228  auto &data = dynamic_cast<InternalData &>(*data_ptr);
229  data.update_each = requires_update_flags(update_flags);
230 
231  const unsigned int n_q_points = quadrature.size();
232 
233  // some scratch arrays
234  std::vector<Tensor<1, dim>> values(0);
235  std::vector<Tensor<2, dim>> grads(0);
236  std::vector<Tensor<3, dim>> grad_grads(0);
237  std::vector<Tensor<4, dim>> third_derivatives(0);
238  std::vector<Tensor<5, dim>> fourth_derivatives(0);
239 
240  if (update_flags & (update_values | update_gradients | update_hessians))
241  data.dof_sign_change.resize(this->dofs_per_cell);
242 
243  // initialize fields only if really
244  // necessary. otherwise, don't
245  // allocate memory
246 
247  const bool update_transformed_shape_values =
248  std::any_of(this->mapping_kind.begin(),
249  this->mapping_kind.end(),
250  [](const MappingKind t) { return t != mapping_none; });
251 
252  const bool update_transformed_shape_grads =
253  std::any_of(this->mapping_kind.begin(),
254  this->mapping_kind.end(),
255  [](const MappingKind t) {
256  return (t == mapping_raviart_thomas || t == mapping_piola ||
257  t == mapping_nedelec || t == mapping_contravariant);
258  });
259 
260  const bool update_transformed_shape_hessian_tensors =
261  update_transformed_shape_values;
262 
263  if (update_flags & update_values)
264  {
265  values.resize(this->n_dofs_per_cell());
266  data.shape_values.reinit(this->n_dofs_per_cell(), n_q_points);
267  if (update_transformed_shape_values)
268  data.transformed_shape_values.resize(n_q_points);
269  }
270 
271  if (update_flags & update_gradients)
272  {
273  grads.resize(this->n_dofs_per_cell());
274  data.shape_grads.reinit(this->n_dofs_per_cell(), n_q_points);
275  data.transformed_shape_grads.resize(n_q_points);
276 
277  if (update_transformed_shape_grads)
278  data.untransformed_shape_grads.resize(n_q_points);
279  }
280 
281  if (update_flags & update_hessians)
282  {
283  grad_grads.resize(this->n_dofs_per_cell());
284  data.shape_grad_grads.reinit(this->n_dofs_per_cell(), n_q_points);
285  data.transformed_shape_hessians.resize(n_q_points);
286  if (update_transformed_shape_hessian_tensors)
287  data.untransformed_shape_hessian_tensors.resize(n_q_points);
288  }
289 
290  // Compute shape function values
291  // and derivatives and hessians on
292  // the reference cell.
293  // Make sure, that for the
294  // node values N_i holds
295  // N_i(v_j)=\delta_ij for all basis
296  // functions v_j
297  if (update_flags & (update_values | update_gradients))
298  for (unsigned int k = 0; k < n_q_points; ++k)
299  {
300  poly_space->evaluate(quadrature.point(k),
301  values,
302  grads,
303  grad_grads,
304  third_derivatives,
305  fourth_derivatives);
306 
307  if (update_flags & update_values)
308  {
309  if (inverse_node_matrix.n_cols() == 0)
310  for (unsigned int i = 0; i < this->n_dofs_per_cell(); ++i)
311  data.shape_values[i][k] = values[i];
312  else
313  for (unsigned int i = 0; i < this->n_dofs_per_cell(); ++i)
314  {
315  Tensor<1, dim> add_values;
316  for (unsigned int j = 0; j < this->n_dofs_per_cell(); ++j)
317  add_values += inverse_node_matrix(j, i) * values[j];
318  data.shape_values[i][k] = add_values;
319  }
320  }
321 
322  if (update_flags & update_gradients)
323  {
324  if (inverse_node_matrix.n_cols() == 0)
325  for (unsigned int i = 0; i < this->n_dofs_per_cell(); ++i)
326  data.shape_grads[i][k] = grads[i];
327  else
328  for (unsigned int i = 0; i < this->n_dofs_per_cell(); ++i)
329  {
330  Tensor<2, dim> add_grads;
331  for (unsigned int j = 0; j < this->n_dofs_per_cell(); ++j)
332  add_grads += inverse_node_matrix(j, i) * grads[j];
333  data.shape_grads[i][k] = add_grads;
334  }
335  }
336 
337  if (update_flags & update_hessians)
338  {
339  if (inverse_node_matrix.n_cols() == 0)
340  for (unsigned int i = 0; i < this->n_dofs_per_cell(); ++i)
341  data.shape_grad_grads[i][k] = grad_grads[i];
342  else
343  for (unsigned int i = 0; i < this->n_dofs_per_cell(); ++i)
344  {
345  Tensor<3, dim> add_grad_grads;
346  for (unsigned int j = 0; j < this->n_dofs_per_cell(); ++j)
347  add_grad_grads +=
348  inverse_node_matrix(j, i) * grad_grads[j];
349  data.shape_grad_grads[i][k] = add_grad_grads;
350  }
351  }
352  }
353  return data_ptr;
354  }
