1098  template <typename ForwardIterator, typename OutputIterator>
1099  inline void
1100  VectorBase::extract_subvector_to(const ForwardIterator indices_begin,
1101  const ForwardIterator indices_end,
1102  OutputIterator values_begin) const
1103  {
1104  const PetscInt n_idx = static_cast<PetscInt>(indices_end - indices_begin);
1105  if (n_idx == 0)
1106  return;
1107 
1108  // if we are dealing
1109  // with a parallel vector
1110  if (ghosted)
1111  {
1112  // there is the possibility
1113  // that the vector has
1114  // ghost elements. in that
1115  // case, we first need to
1116  // figure out which
1117  // elements we own locally,
1118  // then get a pointer to
1119  // the elements that are
1120  // stored here (both the
1121  // ones we own as well as
1122  // the ghost elements). in
1123  // this array, the locally
1124  // owned elements come
1125  // first followed by the
1126  // ghost elements whose
1127  // position we can get from
1128  // an index set
1129  PetscInt begin, end;
1130  PetscErrorCode ierr = VecGetOwnershipRange(vector, &begin, &end);
1131  AssertThrow(ierr == 0, ExcPETScError(ierr));
1132 
1133  Vec locally_stored_elements = nullptr;
1134  ierr = VecGhostGetLocalForm(vector, &locally_stored_elements);
1135  AssertThrow(ierr == 0, ExcPETScError(ierr));
1136 
1137  PetscInt lsize;
1138  ierr = VecGetSize(locally_stored_elements, &lsize);
1139  AssertThrow(ierr == 0, ExcPETScError(ierr));
1140 
1141  PetscScalar *ptr;
1142  ierr = VecGetArray(locally_stored_elements, &ptr);
1143  AssertThrow(ierr == 0, ExcPETScError(ierr));
1144 
1145  for (PetscInt i = 0; i < n_idx; ++i)
1146  {
1147  const unsigned int index = *(indices_begin + i);
1148  if (index >= static_cast<unsigned int>(begin) &&
1149  index < static_cast<unsigned int>(end))
1150  {
1151  // local entry
1152  *(values_begin + i) = *(ptr + index - begin);
1153  }
1154  else
1155  {
1156  // ghost entry
1157  const unsigned int ghostidx =
1158  ghost_indices.index_within_set(index);
1159 
1160  AssertIndexRange(ghostidx + end - begin, lsize);
1161  *(values_begin + i) = *(ptr + ghostidx + end - begin);
1162  }
1163  }
1164 
1165  ierr = VecRestoreArray(locally_stored_elements, &ptr);
1166  AssertThrow(ierr == 0, ExcPETScError(ierr));
1167 
1168  ierr = VecGhostRestoreLocalForm(vector, &locally_stored_elements);
1169  AssertThrow(ierr == 0, ExcPETScError(ierr));
1170  }
1171  // if the vector is local or the
1172  // caller, then simply access the
1173  // element we are interested in
1174  else
1175  {
1176  PetscInt begin, end;
1177  PetscErrorCode ierr = VecGetOwnershipRange(vector, &begin, &end);
1178  AssertThrow(ierr == 0, ExcPETScError(ierr));
1179 
1180  PetscScalar *ptr;
1181  ierr = VecGetArray(vector, &ptr);
1182  AssertThrow(ierr == 0, ExcPETScError(ierr));
1183 
1184  for (PetscInt i = 0; i < n_idx; ++i)
1185  {
1186  const unsigned int index = *(indices_begin + i);
1187 
1188  Assert(index >= static_cast<unsigned int>(begin) &&
1189  index < static_cast<unsigned int>(end),
1190  ExcInternalError());
1191 
1192  *(values_begin + i) = *(ptr + index - begin);
1193  }
1194 
1195  ierr = VecRestoreArray(vector, &ptr);
1196  AssertThrow(ierr == 0, ExcPETScError(ierr));
1197  }
1198  }
