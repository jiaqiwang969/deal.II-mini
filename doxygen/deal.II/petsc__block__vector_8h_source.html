314  inline BlockVector &
315  BlockVector::operator=(const BlockVector &v)
316  {
317  // we only allow assignment to vectors with the same number of blocks
318  // or to an empty BlockVector
319  Assert(n_blocks() == 0 || n_blocks() == v.n_blocks(),
320  ExcDimensionMismatch(n_blocks(), v.n_blocks()));
321 
322  if (this->n_blocks() != v.n_blocks())
323  reinit(v.n_blocks());
324 
325  for (size_type i = 0; i < this->n_blocks(); ++i)
326  this->components[i] = v.block(i);
327 
328  collect_sizes();
329 
330  return *this;
331  }
