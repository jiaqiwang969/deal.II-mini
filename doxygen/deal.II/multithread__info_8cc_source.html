37 unsigned int
38 MultithreadInfo::n_cores()
39 {
40  // There is a slight semantic change between our n_cores() call and the
41  // std::thread alternative: in case of an error the latter one returns 0
42  // in contrast to a 1 that n_cores() used to do. For compatibility, let's
43  // translate to our numbering scheme:
44  const unsigned int n_cores = std::thread::hardware_concurrency();
45  return n_cores == 0 ? 1 : n_cores;
46 }

49 void
50 MultithreadInfo::set_thread_limit(const unsigned int max_threads)
51 {
52  // set the maximal number of threads to the given value as specified
53  n_max_threads = max_threads;
54 
55  // then also see if something was given in the environment
56  {
57  if (const char *penv = std::getenv("DEAL_II_NUM_THREADS"))
58  {
59  unsigned int max_threads_env = numbers::invalid_unsigned_int;
60  try
61  {
62  max_threads_env = Utilities::string_to_int(std::string(penv));
63  }
64  catch (...)
65  {
66  AssertThrow(
67  false,
68  ExcMessage(
69  std::string(
70  "When specifying the <DEAL_II_NUM_THREADS> environment "
71  "variable, it needs to be something that can be interpreted "
72  "as an integer. The text you have in the environment "
73  "variable is <") +
74  penv + ">"));
75  }
76 
77  AssertThrow(max_threads_env > 0,
78  ExcMessage(
79  "When specifying the <DEAL_II_NUM_THREADS> environment "
80  "variable, it needs to be a positive number."));
81 
82  if (n_max_threads != numbers::invalid_unsigned_int)
83  n_max_threads = std::min(n_max_threads, max_threads_env);
84  else
85  n_max_threads = max_threads_env;
86  }
87  }
88 
89  // If we have not set the number of allowed threads yet, just default to
90  // the number of available cores
91  if (n_max_threads == numbers::invalid_unsigned_int)
92  n_max_threads = n_cores();
93 
94 #ifdef DEAL_II_WITH_TBB
95  // Initialize the scheduler and destroy the old one before doing so
96  static tbb::task_scheduler_init dummy(tbb::task_scheduler_init::deferred);
97  if (dummy.is_active())
98  dummy.terminate();
99  dummy.initialize(n_max_threads);
100 #endif
101 
102 #ifdef DEAL_II_WITH_TASKFLOW
103  executor = std::make_unique<tf::Executor>(n_max_threads);
104 #endif
105 }
