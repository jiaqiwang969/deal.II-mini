683  // This function determines the number of cells per block, possibly at compile
684  // time (by virtue of being 'constexpr')
685  // TODO this function should be rewritten using meta-programming
686  __host__ __device__ constexpr unsigned int
687  cells_per_block_shmem(int dim, int fe_degree)
688  {
689  /* clang-format off */
690  // We are limiting the number of threads according to the
691  // following formulas:
692  // - in 2D: `threads = cells * (k+1)^d <= 4*CUDAWrappers::warp_size`
693  // - in 3D: `threads = cells * (k+1)^d <= 2*CUDAWrappers::warp_size`
694  return dim==2 ? (fe_degree==1 ? CUDAWrappers::warp_size : // 128
695  fe_degree==2 ? CUDAWrappers::warp_size/4 : // 72
696  fe_degree==3 ? CUDAWrappers::warp_size/8 : // 64
697  fe_degree==4 ? CUDAWrappers::warp_size/8 : // 100
698  1) :
699  dim==3 ? (fe_degree==1 ? CUDAWrappers::warp_size/4 : // 64
700  fe_degree==2 ? CUDAWrappers::warp_size/16 : // 54
701  1) : 1;
702  /* clang-format on */
703  }

712  template <int dim>
713  __device__ inline unsigned int
714  q_point_id_in_cell(const unsigned int n_q_points_1d)
715  {
716  return (dim == 1 ?
717  threadIdx.x % n_q_points_1d :
718  dim == 2 ?
719  threadIdx.x % n_q_points_1d + n_q_points_1d * threadIdx.y :
720  threadIdx.x % n_q_points_1d +
721  n_q_points_1d * (threadIdx.y + n_q_points_1d * threadIdx.z));
722  }

837  template <int dim, typename Number>
838  DataHost<dim, Number>
839  copy_mf_data_to_host(
840  const typename ::CUDAWrappers::MatrixFree<dim, Number>::Data &data,
841  const UpdateFlags &update_flags)
842  {
843  DataHost<dim, Number> data_host;
844 
845  data_host.id = data.id;
846  data_host.n_cells = data.n_cells;
847  data_host.padding_length = data.padding_length;
848  data_host.row_start = data.row_start;
849  data_host.use_coloring = data.use_coloring;
850 
851  const unsigned int n_elements =
852  data_host.n_cells * data_host.padding_length;
853  if (update_flags & update_quadrature_points)
854  {
855  data_host.q_points.resize(n_elements);
856  Utilities::CUDA::copy_to_host(data.q_points, data_host.q_points);
857  }
858 
859  data_host.local_to_global.resize(n_elements);
860  Utilities::CUDA::copy_to_host(data.local_to_global,
861  data_host.local_to_global);
862 
863  if (update_flags & update_gradients)
864  {
865  data_host.inv_jacobian.resize(n_elements * dim * dim);
866  Utilities::CUDA::copy_to_host(data.inv_jacobian,
867  data_host.inv_jacobian);
868  }
869 
870  if (update_flags & update_JxW_values)
871  {
872  data_host.JxW.resize(n_elements);
873  Utilities::CUDA::copy_to_host(data.JxW, data_host.JxW);
874  }
875 
876  data_host.constraint_mask.resize(data_host.n_cells);
877  Utilities::CUDA::copy_to_host(data.constraint_mask,
878  data_host.constraint_mask);
879 
880  return data_host;
881  }
