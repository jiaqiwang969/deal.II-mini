863  template <int dim, int spacedim>
864  template <class VectorType>
865  inline typename std::enable_if<
866  std::is_convertible<VectorType *, Function<spacedim> *>::value ==
867  false>::type
868  ParticleHandler<dim, spacedim>::set_particle_positions(
869  const VectorType &input_vector,
870  const bool displace_particles)
871  {
872  AssertDimension(input_vector.size(),
873  get_next_free_particle_index() * spacedim);
874  for (auto &p : *this)
875  {
876  auto new_point(displace_particles ? p.get_location() :
877  Point<spacedim>());
878  const auto id = p.get_id();
879  for (unsigned int i = 0; i < spacedim; ++i)
880  new_point[i] += input_vector[id * spacedim + i];
881  p.set_location(new_point);
882  }
883  sort_particles_into_subdomains_and_cells();
884  }

888  template <int dim, int spacedim>
889  template <class VectorType>
890  inline void
891  ParticleHandler<dim, spacedim>::get_particle_positions(
892  VectorType &output_vector,
893  const bool add_to_output_vector)
894  {
895  AssertDimension(output_vector.size(),
896  get_next_free_particle_index() * spacedim);
897  for (const auto &p : *this)
898  {
899  auto point = p.get_location();
900  const auto id = p.get_id();
901  if (add_to_output_vector)
902  for (unsigned int i = 0; i < spacedim; ++i)
903  output_vector[id * spacedim + i] += point[i];
904  else
905  for (unsigned int i = 0; i < spacedim; ++i)
906  output_vector[id * spacedim + i] = point[i];
907  }
908  if (add_to_output_vector)
909  output_vector.compress(VectorOperation::add);
910  else
911  output_vector.compress(VectorOperation::insert);
912  }
