231 inline std::ostream &
232 operator<<(std::ostream &os, const CellId &cid)
233 {
234  os << cid.coarse_cell_id << '_' << cid.n_child_indices << ':';
235  for (unsigned int i = 0; i < cid.n_child_indices; ++i)
236  // write the child indices. because they are between 0 and 2^dim-1, they all
237  // just have one digit, so we could write them as one character
238  // objects. it's probably clearer to write them as one-digit characters
239  // starting at '0'
240  os << static_cast<unsigned char>('0' + cid.child_indices[i]);
241  return os;
242 }

265 inline std::istream &
266 operator>>(std::istream &is, CellId &cid)
267 {
268  unsigned int cellid;
269  is >> cellid;
270  if (is.eof())
271  return is;
272 
273  cid.coarse_cell_id = cellid;
274  char dummy;
275  is >> dummy;
276  Assert(dummy == '_', ExcMessage("invalid CellId"));
277  is >> cid.n_child_indices;
278  is >> dummy;
279  Assert(dummy == ':', ExcMessage("invalid CellId"));
280 
281  unsigned char value;
282  for (unsigned int i = 0; i < cid.n_child_indices; ++i)
283  {
284  // read the one-digit child index (as an integer number) and
285  // convert it back into unsigned integer type
286  is >> value;
287  cid.child_indices[i] = value - '0';
288  }
289  return is;
290 }

320 CellId::operator<(const CellId &other) const
321 {

323  return this->coarse_cell_id < other.coarse_cell_id;
324 
325  unsigned int idx = 0;

329  return false;

332  return child_indices[idx] < other.child_indices[idx];
333 
334  ++idx;

338  return false;
