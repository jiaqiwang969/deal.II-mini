969 template <typename number>
970 inline SparseMatrixEZ<number>::const_iterator::const_iterator(
971  const SparseMatrixEZ<number> *matrix,
972  const size_type r,
973  const unsigned short i)
974  : accessor(matrix, r, i)
975 {
976  // Finish if this is the end()
977  if (r == accessor.matrix->m() && i == 0)
978  return;
979 
980  // Make sure we never construct an
981  // iterator pointing to a
982  // non-existing entry
983 
984  // If the index points beyond the
985  // end of the row, try the next
986  // row.
987  if (accessor.a_index >= accessor.matrix->row_info[accessor.a_row].length)
988  {
989  do
990  {
991  ++accessor.a_row;
992  }
993  // Beware! If the next row is
994  // empty, iterate until a
995  // non-empty row is found or we
996  // hit the end of the matrix.
997  while (accessor.a_row < accessor.matrix->m() &&
998  accessor.matrix->row_info[accessor.a_row].length == 0);
999  }
1000 }

1124 template <typename number>
1125 inline typename SparseMatrixEZ<number>::Entry *
1126 SparseMatrixEZ<number>::allocate(const size_type row, const size_type col)
1127 {
1128  AssertIndexRange(row, m());
1129  AssertIndexRange(col, n());
1130 
1131  RowInfo & r = row_info[row];
1132  const size_type end = r.start + r.length;
1133 
1134  size_type i = r.start;
1135  // If diagonal exists and this
1136  // column is higher, start only
1137  // after diagonal.
1138  if (r.diagonal != RowInfo::invalid_diagonal && col >= row)
1139  i += r.diagonal;
1140  // Find position of entry
1141  while (i < end && data[i].column < col)
1142  ++i;
1143 
1144  // entry found
1145  if (i != end && data[i].column == col)
1146  return &data[i];
1147 
1148  // Now, we must insert the new
1149  // entry and move all successive
1150  // entries back.
1151 
1152  // If no more space is available
1153  // for this row, insert new
1154  // elements into the vector.
1155  // TODO:[GK] We should not extend this row if i<end
1156  if (row != row_info.size() - 1)
1157  {
1158  if (end >= row_info[row + 1].start)
1159  {
1160  // Failure if increment 0
1161  Assert(increment != 0, ExcEntryAllocationFailure(row, col));
1162 
1163  // Insert new entries
1164  data.insert(data.begin() + end, increment, Entry());
1165  // Update starts of
1166  // following rows
1167  for (size_type rn = row + 1; rn < row_info.size(); ++rn)
1168  row_info[rn].start += increment;
1169  }
1170  }
1171  else
1172  {
1173  if (end >= data.size())
1174  {
1175  // Here, appending a block
1176  // does not increase
1177  // performance.
1178  data.push_back(Entry());
1179  }
1180  }
1181 
1182  Entry *entry = &data[i];
1183  // Save original entry
1184  Entry temp = *entry;
1185  // Insert new entry here to
1186  // make sure all entries
1187  // are ordered by column
1188  // index
1189  entry->column = col;
1190  entry->value = 0;
1191  // Update row_info
1192  ++r.length;
1193  if (col == row)
1194  r.diagonal = i - r.start;
1195  else if (col < row && r.diagonal != RowInfo::invalid_diagonal)
1196  ++r.diagonal;
1197 
1198  if (i == end)
1199  return entry;
1200 
1201  // Move all entries in this
1202  // row up by one
1203  for (size_type j = i + 1; j < end; ++j)
1204  {
1205  // There should be no invalid
1206  // entry below end
1207  Assert(data[j].column != Entry::invalid, ExcInternalError());
1208 
1209  // TODO[GK]: This could be done more efficiently by moving starting at the
1210  // top rather than swapping starting at the bottom
1211  std::swap(data[j], temp);
1212  }
1213  Assert(data[end].column == Entry::invalid, ExcInternalError());
1214 
1215  data[end] = temp;
1216 
1217  return entry;
1218 }

1443 template <typename number>
1444 template <typename MatrixTypeA, typename MatrixTypeB>
1445 inline void
1446 SparseMatrixEZ<number>::conjugate_add(const MatrixTypeA &A,
1447  const MatrixTypeB &B,
1448  const bool transpose)
1449 {
1450  // Compute the result
1451  // r_ij = \sum_kl b_ik b_jl a_kl
1452 
1453  // Assert (n() == B.m(), ExcDimensionMismatch(n(), B.m()));
1454  // Assert (m() == B.m(), ExcDimensionMismatch(m(), B.m()));
1455  // Assert (A.n() == B.n(), ExcDimensionMismatch(A.n(), B.n()));
1456  // Assert (A.m() == B.n(), ExcDimensionMismatch(A.m(), B.n()));
1457 
1458  // Somehow, we have to avoid making
1459  // this an operation of complexity
1460  // n^2. For the transpose case, we
1461  // can go through the non-zero
1462  // elements of A^-1 and use the
1463  // corresponding rows of B only.
1464  // For the non-transpose case, we
1465  // must find a trick.
1466  typename MatrixTypeB::const_iterator b1 = B.begin();
1467  const typename MatrixTypeB::const_iterator b_final = B.end();
1468  if (transpose)
1469  while (b1 != b_final)
1470  {
1471  const size_type i = b1->column();
1472  const size_type k = b1->row();
1473  typename MatrixTypeB::const_iterator b2 = B.begin();
1474  while (b2 != b_final)
1475  {
1476  const size_type j = b2->column();
1477  const size_type l = b2->row();
1478 
1479  const typename MatrixTypeA::value_type a = A.el(k, l);
1480 
1481  if (a != 0.)
1482  add(i, j, a * b1->value() * b2->value());
1483  ++b2;
1484  }
1485  ++b1;
1486  }
1487  else
1488  {
1489  // Determine minimal and
1490  // maximal row for a column in
1491  // advance.
1492 
1493  std::vector<size_type> minrow(B.n(), B.m());
1494  std::vector<size_type> maxrow(B.n(), 0);
1495  while (b1 != b_final)
1496  {
1497  const size_type r = b1->row();
1498  if (r < minrow[b1->column()])
1499  minrow[b1->column()] = r;
1500  if (r > maxrow[b1->column()])
1501  maxrow[b1->column()] = r;
1502  ++b1;
1503  }
1504 
1505  typename MatrixTypeA::const_iterator ai = A.begin();
1506  const typename MatrixTypeA::const_iterator ae = A.end();
1507 
1508  while (ai != ae)
1509  {
1510  const typename MatrixTypeA::value_type a = ai->value();
1511  // Don't do anything if
1512  // this entry is zero.
1513  if (a == 0.)
1514  continue;
1515 
1516  // Now, loop over all rows
1517  // having possibly a
1518  // nonzero entry in column
1519  // ai->row()
1520  b1 = B.begin(minrow[ai->row()]);
1521  const typename MatrixTypeB::const_iterator be1 =
1522  B.end(maxrow[ai->row()]);
1523  const typename MatrixTypeB::const_iterator be2 =
1524  B.end(maxrow[ai->column()]);
1525 
1526  while (b1 != be1)
1527  {
1528  const double b1v = b1->value();
1529  // We need the product
1530  // of both. If it is
1531  // zero, we can save
1532  // the work
1533  if (b1->column() == ai->row() && (b1v != 0.))
1534  {
1535  const size_type i = b1->row();
1536 
1537  typename MatrixTypeB::const_iterator b2 =
1538  B.begin(minrow[ai->column()]);
1539  while (b2 != be2)
1540  {
1541  if (b2->column() == ai->column())
1542  {
1543  const size_type j = b2->row();
1544  add(i, j, a * b1v * b2->value());
1545  }
1546  ++b2;
1547  }
1548  }
1549  ++b1;
1550  }
1551  ++ai;
1552  }
1553  }
1554 }
